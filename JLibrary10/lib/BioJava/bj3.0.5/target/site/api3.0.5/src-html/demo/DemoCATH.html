001package demo;
002
003
004import org.biojava.bio.structure.Structure;
005
006import org.biojava.bio.structure.align.util.AtomCache;
007import org.biojava.bio.structure.align.util.UserConfiguration;
008import org.biojava.bio.structure.cath.CathDatabase;
009import org.biojava.bio.structure.cath.CathDomain;
010import org.biojava.bio.structure.cath.CathInstallation;
011
012/** An example for how to access CATH data.
013 * 
014 * run with -Xmx512M
015 * 
016 * @author Andreas Prlic
017 *
018 */
019public class DemoCATH {
020        public static void main(String[] args){
021                
022                
023                
024                UserConfiguration config = new UserConfiguration();
025                config.setPdbFilePath("/tmp/");
026                
027                AtomCache cache  = new AtomCache(config);
028                                
029                CathDatabase database = new CathInstallation(config.getPdbFilePath());
030                
031                String domainID = "1cdgA01";
032                
033                CathDomain cathDomain = database.getDescriptionByCathId(domainID);
034                        
035                System.out.println(cathDomain);
036                
037                
038                Structure cathDomainStructure;
039                try {
040                        cathDomainStructure = cache.getStructure(domainID);
041                        System.out.println(cathDomainStructure);
042                } catch (Exception e) {
043                        // TODO Auto-generated catch block
044                        e.printStackTrace();
045                } 
046                
047                
048                
049                
050        }
051}
