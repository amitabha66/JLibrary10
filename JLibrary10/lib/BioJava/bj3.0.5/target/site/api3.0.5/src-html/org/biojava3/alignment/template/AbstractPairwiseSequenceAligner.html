024package org.biojava3.alignment.template;
025
026import java.util.ArrayList;
027import java.util.List;
028
029import org.biojava3.core.sequence.template.Compound;
030import org.biojava3.core.sequence.template.CompoundSet;
031import org.biojava3.core.sequence.template.Sequence;
032
033/**
034 * Implements common code for an {@link Aligner} for a pair of {@link Sequence}s.
035 *
036 * @author Mark Chapman
037 * @param <S> each {@link Sequence} of the alignment pair is of type S
038 * @param <C> each element of an {@link AlignedSequence} is a {@link Compound} of type C
039 */
040public abstract class AbstractPairwiseSequenceAligner<S extends Sequence<C>, C extends Compound>
041        extends AbstractMatrixAligner<S, C> implements PairwiseSequenceAligner<S, C> {
042
043    // additional input fields
044    private S query, target;
045
046    // additional output field
047    protected SequencePair<S, C> pair;
048
049    /**
050     * Before running a pairwise global sequence alignment, data must be sent in via calls to
051     * {@link #setQuery(Sequence)}, {@link #setTarget(Sequence)}, {@link #setGapPenalty(GapPenalty)}, and
052     * {@link #setSubstitutionMatrix(SubstitutionMatrix)}.
053     */
054    protected AbstractPairwiseSequenceAligner() {
055    }
056
057    /**
058     * Prepares for a pairwise global sequence alignment.
059     *
060     * @param query the first {@link Sequence} of the pair to align
061     * @param target the second {@link Sequence} of the pair to align
062     * @param gapPenalty the gap penalties used during alignment
063     * @param subMatrix the set of substitution scores used during alignment
064     */
065    protected AbstractPairwiseSequenceAligner(S query, S target, GapPenalty gapPenalty,
066            SubstitutionMatrix<C> subMatrix) {
067        this(query, target, gapPenalty, subMatrix, false);
068    }
069
070    /**
071     * Prepares for a pairwise sequence alignment.
072     *
073     * @param query the first {@link Sequence} of the pair to align
074     * @param target the second {@link Sequence} of the pair to align
075     * @param gapPenalty the gap penalties used during alignment
076     * @param subMatrix the set of substitution scores used during alignment
077     * @param local if true, find a region of similarity rather than aligning every compound
078     */
079    protected AbstractPairwiseSequenceAligner(S query, S target, GapPenalty gapPenalty,
080            SubstitutionMatrix<C> subMatrix, boolean local) {
081        super(gapPenalty, subMatrix, local);
082        this.query = query;
083        this.target = target;
084        reset();
085    }
086
087    /**
088     * Sets the query {@link Sequence}.
089     *
090     * @param query the first {@link Sequence} of the pair to align
091     */
092    public void setQuery(S query) {
093        this.query = query;
094        reset();
095    }
096
097    /**
098     * Sets the target {@link Sequence}.
099     *
100     * @param target the second {@link Sequence} of the pair to align
101     */
102    public void setTarget(S target) {
103        this.target = target;
104        reset();
105    }
106
107    // method for PairwiseSequenceAligner
108
109    @Override
110    public SequencePair<S, C> getPair() {
111        if (pair == null) {
112            align();
113        }
114        return pair;
115    }
116
117    // methods for PairwiseSequenceScorer
118
119    @Override
120    public S getQuery() {
121        return query;
122    }
123
124    @Override
125    public S getTarget() {
126        return target;
127    }
128
129    // methods for AbstractMatrixAligner
130
131    @Override
132    protected CompoundSet<C> getCompoundSet() {
133        return (query == null) ? null : query.getCompoundSet();
134    }
135
136    @Override
137    protected List<C> getCompoundsOfQuery() {
138        return (query == null) ? new ArrayList<C>() : query.getAsList();
139    }
140
141    @Override
142    protected List<C> getCompoundsOfTarget() {
143        return (target == null) ? new ArrayList<C>() : target.getAsList();
144    }
145
146    @Override
147    protected int[] getScoreMatrixDimensions() {
148        return new int[] { (query == null) ? 1 : query.getLength() + 1, (target == null) ? 1 : target.getLength() + 1,
149                (getGapPenalty() == null || getGapPenalty().getType() == GapPenalty.Type.LINEAR) ? 1 : 3 };
150    }
151
152    @Override
153    protected short getSubstitutionScore(int queryColumn, int targetColumn) {
154        return getSubstitutionMatrix().getValue(query.getCompoundAt(queryColumn), target.getCompoundAt(targetColumn));
155    }
156
157    @Override
158    protected boolean isReady() {
159        return query != null && target != null && getGapPenalty() != null && getSubstitutionMatrix() != null &&
160                query.getCompoundSet().equals(target.getCompoundSet());
161    }
162
163    @Override
164    protected void reset() {
165        super.reset();
166        pair = null;
167        if (query != null && target != null && getGapPenalty() != null && getSubstitutionMatrix() != null &&
168                query.getCompoundSet().equals(target.getCompoundSet())) {
169            int maxq = 0, maxt = 0;
170            for (C c : query) {
171                maxq += getSubstitutionMatrix().getValue(c, c);
172            }
173            for (C c : target) {
174                maxt += getSubstitutionMatrix().getValue(c, c);
175            }
176            max = (short) Math.max(maxq, maxt);
177            score = min = isLocal() ? 0 : (short) (2 * getGapPenalty().getOpenPenalty() + (query.getLength() +
178                    target.getLength()) * getGapPenalty().getExtensionPenalty());
179        }
180    }
181
182}
