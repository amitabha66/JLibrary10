001package org.biojava.bio.structure.scop;
002
003import java.util.List;
004
005/** General API how to interact with SCOP
006 * 
007 * @author Andreas Prlic
008 * @since 3.0.2
009 */
010public interface ScopDatabase {
011
012        /** Get all records of a particular classification.
013         * 
014         * @param category e.g. "superfamily"
015         * @return all records of this type
016         */
017        public abstract List<ScopDescription> getByCategory(ScopCategory category);
018
019        /** Get all scop descriptions that start with a classifcation ID, e.g. b.1.18
020         * 
021         * @param query
022         * @return list of scop descriptions
023         */
024        public abstract List<ScopDescription> filterByClassificationId(String query);
025
026        /** get the SCOP sub-tree for a particular domain. 
027         * 
028         * @param domain
029         * @return list of ScopNodes providing the path to this domain
030         */
031        public abstract List<ScopNode> getTree(ScopDomain domain);
032
033        
034        /** search through SCOP and filter based on domain name
035         * 
036         * @param query a (part) of a name
037         * @return list of matchin ScopDomains
038         */
039        public abstract List<ScopDomain> filterByDomainName(String query);
040
041        /** Get all scop descriptions that start with a certain name. e.g. Globin
042         * 
043         * @param query
044         * @return list of scop descriptions
045         */
046        public abstract List<ScopDescription> filterByDescription(String query);
047
048        /** Return the SCOP description for a node in the hierarchy by its "sunid" id.
049         * 
050         * @param sunid
051         * @return a ScopDescription object
052         */
053        public abstract ScopDescription getScopDescriptionBySunid(int sunid);
054
055        /** Get a list of ScopDomains that have been assigned to a PDB ID
056         * 
057         * @param pdbId the PDB entry
058         * @return a list of ScopDomains
059         */
060        public abstract List<ScopDomain> getDomainsForPDB(String pdbId);
061
062        /** get a ScopDomain by its SCOP ID (warning, they are not stable between releases!)
063         * 
064         *
065         * @param scopId e.g. d2bq6a1 
066         * @return a ScopDomain or null if no domain with the particular ID could be found
067         */
068        public abstract ScopDomain getDomainByScopID(String scopId);
069
070        /** Access a particular ScopNode. The scopNode then allows to traverse through the scop hierarchy...
071         * 
072         * @param sunid the scop unique id
073         * @return a ScopNode that matches this sunid
074         */
075        public abstract ScopNode getScopNode(int sunid);
076        
077        /** Returns the SCOP version
078         * 
079         * @return version of SCOP
080         */
081
082        public abstract String getScopVersion();
083
084        /** Get a SCOP domain by its sunid
085         * 
086         * @param sunid the scop unique id
087         * @return a list of scop domains that match this sunid
088         */
089        public abstract List<ScopDomain> getScopDomainsBySunid(Integer sunid);
090
091}
