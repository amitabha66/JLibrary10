025package org.biojava.bio.structure.scop;
026
027import java.io.InputStream;
028import java.net.URL;
029import java.util.List;
030
031import org.biojava.bio.structure.align.client.JFatCatClient;
032import org.biojava.bio.structure.align.util.HTTPConnectionTools;
033import org.biojava.bio.structure.domain.SerializableCache;
034import org.biojava.bio.structure.scop.server.ScopDomains;
035
036
037/** An extension of the RemoteScopInstallation that caches some of the data locally.
038 * 
039 * @author Andreas Prlic
040 *
041 */
042public class CachedRemoteScopInstallation extends SerializableCache<String,ScopDomain> implements ScopDatabase {
043
044        private static String CACHE_FILE_NAME = "remotescopinstallation.ser";
045        
046        RemoteScopInstallation proxy ;
047        
048        SerializableCache<Integer,ScopDescription> scopDescriptionCache ;
049        
050        public CachedRemoteScopInstallation() {
051                this(true);
052                
053                
054
055        }
056        
057        public CachedRemoteScopInstallation(boolean useCache) {
058
059                super(CACHE_FILE_NAME);
060
061                proxy = new RemoteScopInstallation();
062                
063                scopDescriptionCache = new SerializableCache<Integer,ScopDescription>("scopDescriptionCache.ser");
064                //scopDescriptionCache.setDebug(true);
065                
066                if ( ! useCache) {
067                        System.err.println("CachedRemoteScopInstallation disableing cache");
068                        disableCache();
069                        scopDescriptionCache.disableCache();
070                } else {
071                        
072                        if ( serializedCache.size() < 8000){
073                                loadRepresentativeDomains();
074                        }
075                }
076
077        }
078
079
080        /** get the ranges of representative domains from the centralized server
081         * 
082         */
083        private void loadRepresentativeDomains() {
084                
085                        ScopDomains results = null;
086                        try {
087                                URL u = new URL(RemoteScopInstallation.DEFAULT_SERVER + "getRepresentativeScopDomains");
088                                System.out.println(u);
089                                InputStream response = HTTPConnectionTools.getInputStream(u);
090                                String xml = JFatCatClient.convertStreamToString(response);                     
091                                //System.out.println(xml);
092                                results  = ScopDomains.fromXML(xml);
093
094                                
095                                System.out.println("got " + results.getScopDomain().size() + " domain ranges for Scop domains from server.");
096                                for (ScopDomain dom : results.getScopDomain()){
097                                        String scopId = dom.getScopId();
098                                        serializedCache.put(scopId, dom);
099                                }
100                                
101                                
102                        } catch (Exception e){
103                                e.printStackTrace();
104                        }
105                        return ;
106                        
107                
108
109                
110        }
111        
112        
113        
114        public List<ScopDescription> getByCategory(ScopCategory category) {
115                return proxy.getByCategory(category);
116        }
117
118        
119        public List<ScopDescription> filterByClassificationId(String query) {
120                return proxy.filterByClassificationId(query);
121        }
122
123        
124        public List<ScopNode> getTree(ScopDomain domain) {
125                return proxy.getTree(domain);
126        }
127
128        
129        public List<ScopDomain> filterByDomainName(String query) {
130                return proxy.filterByDomainName(query);
131        }
132
133        
134        public List<ScopDescription> filterByDescription(String query) {
135                return proxy.filterByClassificationId(query);
136        }
137
138        
139        public ScopDescription getScopDescriptionBySunid(int sunid) {
140                
141                ScopDescription desc = scopDescriptionCache.get(sunid);
142                if ( desc != null)
143                        return desc;
144                
145                
146                desc =  proxy.getScopDescriptionBySunid(sunid);
147                if ( desc != null)
148                        scopDescriptionCache.cache(sunid,desc);
149                return desc;
150        }
151
152        
153        public List<ScopDomain> getDomainsForPDB(String pdbId) {
154                
155                return proxy.getDomainsForPDB(pdbId);
156        }
157
158        
159        public ScopDomain getDomainByScopID(String scopId) {
160                ScopDomain dom;
161                
162                if ( serializedCache != null){                  
163                        if ( serializedCache.containsKey(scopId)) {
164                                dom = serializedCache.get(scopId);
165                                if ( dom != null) {
166                                        return dom;
167                                }
168                        }                       
169                }
170                
171                 dom = proxy.getDomainByScopID(scopId);
172                
173                if ( dom != null)
174                        cache(scopId, dom);
175                
176                
177                return dom;
178        }
179
180
181        public ScopNode getScopNode(int sunid) {
182                return proxy.getScopNode(sunid);
183        }
184
185
186        public String getScopVersion() {
187                return proxy.getScopVersion();
188        }
189
190
191        public List<ScopDomain> getScopDomainsBySunid(Integer sunid) {
192                return proxy.getScopDomainsBySunid(sunid);
193        }
194
195        @Override
196        public void flushCache() {
197                System.out.println("flushing CachedRemoteScopInstallation");
198                super.flushCache();
199                scopDescriptionCache.flushCache();
200        }
201
202        
203}
