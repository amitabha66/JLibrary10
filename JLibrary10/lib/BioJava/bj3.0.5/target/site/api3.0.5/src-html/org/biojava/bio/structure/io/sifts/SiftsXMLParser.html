056        public void parseXmlFile(InputStream is){
057                entities = new ArrayList<SiftsEntity>();
058
059                //get the factory
060                DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
061
062                try {
063
064                        //Using factory get an instance of document builder
065                        DocumentBuilder db = dbf.newDocumentBuilder();
066
067                        //parse using builder to get DOM representation of the XML file
068                        dom = db.parse(is);
069
070                        parseDocument();
071
072                }catch(ParserConfigurationException pce) {
073                        pce.printStackTrace();
074                }catch(SAXException se) {
075                        se.printStackTrace();
076                }catch(IOException ioe) {
077                        ioe.printStackTrace();
078                }
079        }

083                private void parseDocument(){
084                        //get the root element
085                        Element docEle = dom.getDocumentElement();
086
087                        //get a nodelist of  entities
088
089                        NodeList nl = docEle.getElementsByTagName("entity");
090                        if(nl != null && nl.getLength() > 0) {
091                                for(int i = 0 ; i < nl.getLength();i++) {
092
093                                        //get the entity element
094                                        Element el = (Element)nl.item(i);
095                                        //get the Employee object
096                                        SiftsEntity e = getSiftsEntity(el);
097
098                                        //add it to list
099                                        entities.add(e);
100                                }
101                        }
102                }
103
104                /**
105                 * <entity type="protein" entityId="A">
106                 */
107                private SiftsEntity getSiftsEntity(Element empEl) {
108
109                        //for each <employee> element get text or int values of
110                        //name ,id, age and name
111
112                        String type = empEl.getAttribute("type");
113                        String entityId = empEl.getAttribute("entityId");
114
115                        //Create a new Employee with the value read from the xml nodes
116                        SiftsEntity entity = new SiftsEntity(type,entityId);
117
118                        // get nodelist of segments...
119                        NodeList nl = empEl.getElementsByTagName("segment");
120                        if(nl != null && nl.getLength() > 0) {
121                                for(int i = 0 ; i < nl.getLength();i++) {
122
123                                        //get the entity element
124                                        Element el = (Element)nl.item(i);
125
126                                        SiftsSegment s = getSiftsSegment(el);
127
128                                        entity.addSegment(s);
129
130                                }
131                        }
132
133                        return entity;
134                }

136                /** segId="4hhb_A_1_140" start="1" end="140"
137                 * 
138                 * @param el
139                 * @return
140                 */
141                private SiftsSegment getSiftsSegment(Element el) {
142
143                        String segId = el.getAttribute("segId");
144                        String start = el.getAttribute("start");
145                        String end = el.getAttribute("end");
146                        SiftsSegment seg = new SiftsSegment(segId,start,end);
147
148                        // get nodelist of segments...
149                        NodeList nl = el.getElementsByTagName("listResidue");
150                        if(nl != null && nl.getLength() > 0) {
151                                for(int i = 0 ; i < nl.getLength();i++) {
152                                        //get the entity element
153                                        Element listResidueEl = (Element)nl.item(i);
154
155                                        NodeList residueNodes = listResidueEl.getElementsByTagName("residue");
156                                        if(residueNodes != null && residueNodes.getLength() > 0) {
157                                                for(int j = 0 ; j < residueNodes.getLength();j++) {
158                                                        Element residue = (Element) residueNodes.item(j);
159
160                                                        SiftsResidue pos = getResidue(residue);
161                                                        seg.addResidue(pos);
162                                                }
163                                        }
164
165                                }
166                        }
167
168
169                        return seg;
170                }

172                /**
173                 *  <residue dbResNum="1" dbResName="THR">
174          <crossRefDb dbSource="PDB" dbVersion="20101103"
175          dbCoordSys="PDBresnum" dbAccessionId="1a4w" dbResNum="1H"
176          dbResName="THR" dbChainId="L"></crossRefDb>
177          <crossRefDb dbSource="UniProt" dbVersion="157-2"
178          dbCoordSys="UniProt" dbAccessionId="P00734"
179          dbResNum="328" dbResName="T"></crossRefDb>
180          <crossRefDb dbSource="SCOP" dbVersion="1.75"
181          dbCoordSys="PDBresnum" dbAccessionId="26083"
182          dbResNum="1H" dbResName="THR" dbChainId="L"></crossRefDb>
183          <residueDetail dbSource="MSD" property="Annotation">
184          Not_Observed</residueDetail>
185        </residue>
186
187                 */
188                private SiftsResidue getResidue(Element residue) {
189
190                        SiftsResidue res = new SiftsResidue();
191                        
192                        String dbResNumS = residue.getAttribute("dbResNum");
193                        res.setNaturalPos(Integer.parseInt(dbResNumS));
194                        
195                        String seqResName = residue.getAttribute("dbResName");
196                        res.setSeqResName(seqResName);
197                        
198                        boolean observed = true;
199                        
200                        String detail = getTextValue(residue, "residueDetail");
201                        if ( detail != null)
202                        //System.out.println(">"+detail+"<");
203                        if ( detail != null && detail.trim().equalsIgnoreCase("Not_Observed")){
204                                observed = false;
205                        }
206                        res.setNotObserved(! observed);
207                        //else if ( detail != null && detail.trim().equalsIgnoreCase("Conflict")){
208                                //
209                        //}
210                        
211                        NodeList nl = residue.getElementsByTagName("crossRefDb");
212                        if(nl != null && nl.getLength() > 0) {
213                                for(int i = 0 ; i < nl.getLength();i++) {
214                                        //get the entity element
215                                        Element crossRefEl = (Element)nl.item(i);
216
217                                        String dbSource = crossRefEl.getAttribute("dbSource");
218                                        String dbCoordSys = crossRefEl.getAttribute("dbCoordSys");
219                                        String dbAccessionId = crossRefEl.getAttribute("dbAccessionId");
220                                        String dbResNum = crossRefEl.getAttribute("dbResNum");
221                                        String dbResName = crossRefEl.getAttribute("dbResName");
222                                        String dbChainId = crossRefEl.getAttribute("dbChainId");
223
224                                //      System.out.println(dbSource + " " + dbCoordSys + " " + dbAccessionId + " " + dbResNum + " " + dbResName + " " + dbChainId);
225                                        
226                                        if ( dbSource.equals("PDB") && ( dbCoordSys.equals("PDBresnum"))){
227                                                res.setPdbResNum(dbResNum);
228                                                res.setPdbResName(dbResName);
229                                                res.setChainId(dbChainId);
230                                                res.setPdbId(dbAccessionId);
231                                        } else if ( dbCoordSys.equals("UniProt")){
232                                                res.setUniProtPos(Integer.parseInt(dbResNum));
233                                                res.setUniProtResName(dbResName);
234                                                res.setUniProtAccessionId(dbAccessionId);
235                                        }
236                                }
237                        }
238                        return res;
239                }

243                /**
244                 * I take a xml element and the tag name, look for the tag and get
245                 * the text content
246                 * i.e for <employee><name>John</name></employee> xml snippet if
247                 * the Element points to employee node and tagName is 'name' I will return John
248                 */
249                private String getTextValue(Element ele, String tagName) {
250                        String textVal = null;
251                        NodeList nl = ele.getElementsByTagName(tagName);
252                        if(nl != null && nl.getLength() > 0) {
253                                Element el = (Element)nl.item(0);
254                                textVal = el.getFirstChild().getNodeValue();
255                        }
256
257                        return textVal;
258                }
