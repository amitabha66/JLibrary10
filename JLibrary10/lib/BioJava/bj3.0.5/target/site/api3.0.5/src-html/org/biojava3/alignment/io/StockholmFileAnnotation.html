345        public float[] stringToFloats(String string) {
346                String[] coublets= string.split(";");
347                float[] floats = new float[coublets.length*2];
348                int counter=0;
349                for (int i = 0; i < coublets.length; i++) {
350                        String[] subStrings = coublets[i].trim().split("\\s");
351                        float f = Float.parseFloat(subStrings[i]);
352                        floats[counter++]=f;
353                }
354                return floats;
355        }
