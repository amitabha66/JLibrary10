067        /**
068         * Holder for the ranges, contain pointers to starting and ending position 
069         * on the sequence which comprises a disordered region. Immutable. 
070         * @author pvtroshin
071         */
072        public static class Range {
073                /**
074                 * Range starting position counts from 1 (the first position on the sequence is 1)
075                 */
076                public final int from; 
077                /**
078                 * The range ending position includes the last residue. 
079                 */
080                public final int to; 
081        
082                public final float score;
083                public Range(int from, int to, float score) {
084                        assert from>=0; 
085                        assert from<to; 
086                        this.from = from; 
087                        this.to = to; 
088                        this.score = score;
089                }
090
091                @Override
092                public String toString() { 
093                        return "Range" + " From:" + from + "\t" + "to: " + to + "\n";
094                }
095                
096                @Override
097                public int hashCode() {
098                        final int prime = 31;
099                        int result = 1;
100                        result = prime * result + from;
101                        result = prime * result + to;
102                        return result;
103                }
104
105                @Override
106                public boolean equals(Object obj) {
107                        if (this == obj)
108                                return true;
109                        if (obj == null)
110                                return false;
111                        if (getClass() != obj.getClass())
112                                return false;
113                        Range other = (Range) obj;
114                        if (from != other.from)
115                                return false;
116                        if (to != other.to)
117                                return false;
118                        return true;
119                }
120                
121                
122        }

151        /** Utility method to convert a BioJava ProteinSequence object to the FastaSequence 
152         *  object used internally in JRonn.
153         * 
154         * @param sequence
155         * @return
156         */
157        public static FastaSequence convertProteinSequencetoFasta(ProteinSequence sequence){
158                StringBuffer buf = new StringBuffer();
159                for (AminoAcidCompound compound : sequence) {
160                        
161                        String c = compound.getShortName();
162                        
163                        if (! SequenceUtil.NON_AA.matcher(c).find()) {
164                                buf.append(c);
165                        } else {                                
166                                buf.append("X");
167                        }                                                                       
168                }
169                
170                return new FastaSequence(sequence.getAccession().getID(),buf.toString());
171        }

173        private static float[] predictSerial(FastaSequence fsequence) {
174                ORonn.validateSequenceForRonn(fsequence);
175                ORonn ronn;
176                float[] disorder = null; 
177                try {
178                        ronn = new ORonn(fsequence, loader);
179                        disorder = ronn.call().getMeanScores();
180                } catch (NumberFormatException e) {
181                        throw new RuntimeException("Jronn fails to load models " + e.getLocalizedMessage(), e);
182                } catch (IOException e) {
183                        throw new RuntimeException("Jronn fails to load models " + e.getLocalizedMessage(), e);
184                }
185                return disorder;  
186        }

188        /**
189         * Calculates the disordered regions of the sequence. More formally, the regions for which the 
190         * probability of disorder is greater then 0.50.  
191         *  
192         *   
193         * @param sequence an instance of FastaSequence object, holding the name and the sequence.
194         * @return the array of ranges if there are any residues predicted to have the 
195         * probability of disorder greater then 0.5, null otherwise. 
196         *
197         */
198        public static Range[] getDisorder(FastaSequence sequence) {
199                float[] scores = getDisorderScores(sequence);
200                return scoresToRanges(scores, RonnConstraint.DEFAULT_RANGE_PROBABILITY_THRESHOLD);
201        }

203        /**
204         * Convert raw scores to ranges. Gives ranges for given probability of disorder value 
205         * @param scores the raw probability of disorder scores for each residue in the sequence.  
206         * @param probability the cut off threshold. Include all residues with the probability of disorder greater then this value
207         * @return the array of ranges if there are any residues predicted to have the 
208         * probability of disorder greater then {@code probability}, null otherwise.
209         */
210        public static Range[] scoresToRanges(float[] scores, float probability)  {
211                assert scores!=null && scores.length>0;
212                assert probability>0 && probability<1;
213                
214                int count=0;
215                int regionLen=0;
216                List<Range> ranges = new ArrayList<Range>();
217                for(float score: scores) { 
218                        count++;
219                        // Round to 2 decimal points before comparison 
220                        score = (float) (Math.round(score*100.0)/100.0);
221                        if(score>probability) {
222                                regionLen++;
223                        } else {
224                                if(regionLen>0) {
225                                        ranges.add(new Range(count-regionLen, count-1,score));
226                                }
227                                regionLen=0;
228                        }
229                }
230                // In case of the range to boundary runs to the very end of the sequence 
231                if(regionLen>1) {
232                        ranges.add(new Range(count-regionLen+1, count,scores[scores.length-1]));
233                }
234                return ranges.toArray(new Range[ranges.size()]);                
235
236        }

238        /**
239         * Calculates the probability of disorder scores for each residue in the sequence for 
240         * many sequences in the input.
241         * 
242         * @param sequences the list of the FastaSequence objects 
243         * @return the Map with key->FastaSequence, value->probability of disorder for each residue
244         * @see #getDisorder(FastaSequence)
245         */
246        public static Map<FastaSequence,float[]> getDisorderScores(List<FastaSequence> sequences) {
247                Map<FastaSequence,float[]> results = new TreeMap<FastaSequence, float[]>();
248                for(FastaSequence fsequence : sequences) {
249                        results.put(fsequence, predictSerial(fsequence));
250                }
251                return results; 
252        }

254        /**
255         * Calculates the disordered regions of the sequence for many sequences in the input.
256         * 
257         * @param sequences sequences the list of the FastaSequence objects
258         * @return
259         * @see #getDisorder(FastaSequence)
260         */
261        public static Map<FastaSequence,Range[]> getDisorder(List<FastaSequence> sequences) {
262                Map<FastaSequence,Range[]> disorderRanges = new TreeMap<FastaSequence,Range[]>();
263                for(FastaSequence fs: sequences) {
264                        disorderRanges.put(fs, getDisorder(fs));
265                }
266                return disorderRanges; 
267        }
