001package org.biojava.bio.structure.io;
002
003import java.io.IOException;
004
005import org.biojava.bio.structure.Structure;
006import org.biojava.bio.structure.StructureException;
007
008/** A class that can provide a protein structure object from somewhere.
009 * 
010 * @author Andreas Prlic
011 * @since 3.2
012 */
013public interface StructureProvider {
014
015        /** get the structure for a PDB ID
016         * 
017         * @param pdbId
018         * @return
019         */
020        public Structure getStructureById(String pdbId) throws StructureException,IOException;
021        
022        /** get the biological unit for a file
023         * 
024         * @param pdbId
025         * @return
026         */
027        public Structure getBiologicalUnit(String pdbId) throws StructureException, IOException;
028
029        
030        /** Set the parameters that should be used for file parsing
031         * 
032         * @param params FileParsingParameters
033         */
034        public void setFileParsingParameters(FileParsingParameters params);
035           
036        
037        /** Get the parameters that should be used for file parsing
038     * 
039     * @return the FileParsingParameters that are configuring the behavior of the parser
040     */
041    public FileParsingParameters getFileParsingParameters();
042    
043    
044}
