001package demo;
002
003import org.biojava.bio.structure.scop.BerkeleyScopInstallation;
004import org.biojava.bio.structure.scop.ScopDatabase;
005import org.biojava.bio.structure.scop.ScopFactory;
006
007/** A demo for how to use the Berkeley version of SCOP instead of the default UK-SCOP
008 * 
009 * @author andreas
010 * @since 3.0.4
011 *
012 */
013public class DemoBerkeleyScop {
014        public static void main(String[]args){
015
016
017                ScopDatabase berkeley = new BerkeleyScopInstallation();
018
019                ScopFactory.setScopDatabase(berkeley);
020
021                // whenever you want to get access to SCOP now request it like this:
022                ScopDatabase scop = ScopFactory.getSCOP();
023                // ... and do something with it
024
025
026                // eg. you can run all the demos that work for the UK - SCOP (currently at version 1.75) 
027                // this demo no automatically picks up the Berkeley version (currently 1.75A)
028                DemoSCOP scopDemo = new DemoSCOP();
029                
030                scopDemo.getCategories();
031                scopDemo.printDomainsForPDB();
032                scopDemo.traverseHierarchy();
033                scopDemo.alignSuperfamily();
034
035
036        }
037}
