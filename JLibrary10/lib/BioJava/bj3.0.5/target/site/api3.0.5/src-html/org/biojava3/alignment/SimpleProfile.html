047/**
048 * Implements a data structure for the results of sequence alignment.  Every {@link List} returned is unmodifiable.
049 *
050 * @author Mark Chapman
051 * @param <S> each element of the alignment {@link Profile} is of type S
052 * @param <C> each element of an {@link AlignedSequence} is a {@link Compound} of type C
053 */
054public class SimpleProfile<S extends Sequence<C>, C extends Compound> implements Profile<S, C> {
055
056        private List<AlignedSequence<S, C>> list;
057        private List<S> originals;
058        private int length;
059
060        /**
061         * Creates a pair profile for the given already aligned sequences.
062         *
063         * @param query the first sequence of the pair
064         * @param target the second sequence of the pair
065         * @throws IllegalArgumentException if sequences differ in size
066         */
067        protected SimpleProfile(AlignedSequence<S, C> query, AlignedSequence<S, C> target) {
068                if (query.getLength() != target.getLength()) {
069                        throw new IllegalArgumentException("Aligned sequences differ in size");
070                }
071                list = new ArrayList<AlignedSequence<S, C>>();
072                list.add(query);
073                list.add(target);
074                list = Collections.unmodifiableList(list);
075                originals = new ArrayList<S>();
076                originals.add((S) query.getOriginalSequence());
077                originals.add((S) target.getOriginalSequence());
078                originals = Collections.unmodifiableList(originals);
079                length = query.getLength();
080        }

082        /**
083         * Creates a profile from a single sequence.
084         *
085         * @param sequence sequence to seed profile
086         */
087        public SimpleProfile(S sequence) {
088                List<Step> s = new ArrayList<Step>();
089                for (int i = 0; i < sequence.getLength(); i++) {
090                        s.add(Step.COMPOUND);
091                }
092                list = new ArrayList<AlignedSequence<S, C>>();
093                list.add(new SimpleAlignedSequence<S, C>(sequence, s));
094                list = Collections.unmodifiableList(list);
095                originals = new ArrayList<S>();
096                originals.add(sequence);
097                originals = Collections.unmodifiableList(originals);
098                length = sequence.getLength();
099        }

101        /**
102         * Creates a pair profile for the given sequences.
103         *
104         * @param query the first sequence of the pair
105         * @param target the second sequence of the pair
106         * @param sx lists whether the query sequence aligns a {@link Compound} or gap at each index of the alignment
107         * @param xb number of {@link Compound}s skipped in the query sequence before the aligned region
108         * @param xa number of {@link Compound}s skipped in the query sequence after the aligned region
109         * @param sy lists whether the target sequence aligns a {@link Compound} or gap at each index of the alignment
110         * @param yb number of {@link Compound}s skipped in the target sequence before the aligned region
111         * @param ya number of {@link Compound}s skipped in the target sequence after the aligned region
112         * @throws IllegalArgumentException if alignments differ in size or given sequences do not fit in alignments
113         */
114        protected SimpleProfile(S query, S target, List<Step> sx, int xb, int xa, List<Step> sy, int yb, int ya) {
115                if (sx.size() != sy.size()) {
116                        throw new IllegalArgumentException("Alignments differ in size");
117                }
118                list = new ArrayList<AlignedSequence<S, C>>();
119                list.add(new SimpleAlignedSequence<S, C>(query, sx, xb, xa));
120                list.add(new SimpleAlignedSequence<S, C>(target, sy, yb, ya));
121                list = Collections.unmodifiableList(list);
122                originals = new ArrayList<S>();
123                originals.add(query);
124                originals.add(target);
125                originals = Collections.unmodifiableList(originals);
126                length = sx.size();
127        }

129        /**
130         * Creates a pair profile for the given profiles.
131         *
132         * @param query the first profile of the pair
133         * @param target the second profile of the pair
134         * @param sx lists whether the query profile aligns a {@link Compound} or gap at each index of the alignment
135         * @param sy lists whether the target profile aligns a {@link Compound} or gap at each index of the alignment
136         * @throws IllegalArgumentException if alignments differ in size or given profiles do not fit in alignments
137         */
138        protected SimpleProfile(Profile<S, C> query, Profile<S, C> target, List<Step> sx, List<Step> sy) {
139                if (sx.size() != sy.size()) {
140                        throw new IllegalArgumentException("Alignments differ in size");
141                }
142                list = new ArrayList<AlignedSequence<S, C>>();
143                for (AlignedSequence<S, C> s : query) {
144                        list.add(new SimpleAlignedSequence<S, C>(s, sx));
145                }
146                for (AlignedSequence<S, C> s : target) {
147                        list.add(new SimpleAlignedSequence<S, C>(s, sy));
148                }
149                list = Collections.unmodifiableList(list);
150                originals = new ArrayList<S>();
151                originals.addAll(query.getOriginalSequences());
152                originals.addAll(target.getOriginalSequences());
153                originals = Collections.unmodifiableList(originals);
154                length = sx.size();
155        }

157         /**
158     * Creates a profile for the already aligned sequences.
159     * @param alignedSequences the already aligned sequences
160     * @throws IllegalArgument if aligned sequences differ in length or
161     * collection is empty.
162     */
163    public SimpleProfile(Collection<AlignedSequence<S,C>> alignedSequences) {
164        list = new ArrayList<AlignedSequence<S,C>>();
165        originals = new ArrayList<S>();
166        
167        Iterator<AlignedSequence<S,C>> itr = alignedSequences.iterator();
168        if(!itr.hasNext()) {
169            throw new IllegalArgumentException("alignedSequences must not be empty");
170        }
171        
172        AlignedSequence<S, C> curAlignedSeq = itr.next();
173        length = curAlignedSeq.getLength();
174        list.add(curAlignedSeq);
175        originals.add((S) curAlignedSeq.getOriginalSequence());
176        
177        while (itr.hasNext()) {
178            curAlignedSeq = itr.next();
179            if (curAlignedSeq.getLength() != length) {
180                throw new IllegalArgumentException("Aligned sequences differ in size");
181            }
182            list.add(curAlignedSeq);
183            originals.add((S) curAlignedSeq.getOriginalSequence());
184        }
185        list = Collections.unmodifiableList(list);
186        originals = Collections.unmodifiableList(originals);
187    }

246        @Override
247        public int[] getCompoundCountsAt(int alignmentIndex, List<C> compounds) {
248                int[] counts = new int[compounds.size()];
249                C gap = getCompoundSet().getCompoundForString("-");
250                int igap = compounds.indexOf(gap);
251                for (C compound : getCompoundsAt(alignmentIndex)) {
252                        int i = compounds.indexOf(compound);
253                        if (i >= 0 && i != igap && !getCompoundSet().compoundsEquivalent(compound, gap)) {
254                                counts[i]++;
255                        }
256                }
257                return counts;
258        }

280        @Override
281        public float[] getCompoundWeightsAt(int alignmentIndex, List<C> compounds) {
282                float[] weights = new float[compounds.size()];
283                int[] counts = getCompoundCountsAt(alignmentIndex, compounds);
284                float total = 0.0f;
285                for (int i : counts) {
286                        total += i;
287                }
288                if (total > 0.0f) {
289                        for (int i = 0; i < weights.length; i++) {
290                                weights[i] = counts[i]/total;
291                        }
292                }
293                return weights;
294        }

350        @Override
351        public boolean hasGap(int alignmentIndex) {
352                C gap = getCompoundSet().getCompoundForString("-");
353                for (C compound : getCompoundsAt(alignmentIndex)) {
354                        if (getCompoundSet().compoundsEquivalent(compound, gap)) {
355                                return true;
356                        }
357                }
358                return false;
359        }

377        public String toString(StringFormat format) {
378        switch (format) {
379        case ALN:
380        case CLUSTALW:
381        default:
382            return toString(60, String.format("CLUSTAL W MSA from BioJava%n%n"), IOUtils.getIDFormat(list) + "   ",
383                    false, true, true, false, true, false);
384        case FASTA:
385            return toString(60, null, ">%s%n", false, false, false, false, false, false);
386        case GCG:
387        case MSF:
388            return toString(50, IOUtils.getGCGHeader(list), IOUtils.getIDFormat(list), false, false, true, false,
389                    false, false);
390        case PDBWEB:
391            return toString(60, null, "%s", true, true, true, false, true, true);
392        }

411        // creates formatted String
412        private String toString(int width, String header, String idFormat, boolean seqIndexPre, boolean seqIndexPost,
413                        boolean interlaced, boolean aligIndices, boolean aligConservation, boolean webDisplay) {
414
415                // TODO handle circular alignments
416                StringBuilder s = (header == null) ? new StringBuilder() : new StringBuilder(header);
417                
418                if ( webDisplay && list.size() == 2){
419                        s.append("<div><pre>");
420                }
421
422                width = Math.max(1, width);
423                int seqIndexPad = (int) (Math.floor(Math.log10(getLength())) + 2);
424                String seqIndexFormatPre = "%" + seqIndexPad + "d ", seqIndexFormatPost = "%" + seqIndexPad + "d";
425                if (interlaced) {
426                        String aligIndFormat = "%-" + Math.max(1, width / 2) + "d %" + Math.max(1, width - (width / 2) - 1) +
427                        "d%n";
428                        for (int i = 0; i < getLength(); i += width) {
429                                int start = i + 1, end = Math.min(getLength(), i + width);
430                                if (i > 0) {
431                                        s.append(String.format("%n"));
432                                }
433                                if (aligIndices) {
434                                        if (end < i + width) {
435                                                int line = end - start + 1;
436                                                aligIndFormat = "%-" + Math.max(1, line / 2) + "d %" + Math.max(1, line - (line / 2) - 1) +
437                                                "d%n";
438                                        }
439                                        if (idFormat != null) {
440                                                s.append(String.format(idFormat, ""));
441                                        }
442                                        if (seqIndexPre) {
443                                                s.append(String.format("%" + (seqIndexPad + 1) + "s", ""));
444                                        }
445                                        s.append(String.format(aligIndFormat, start, end));
446                                }
447
448                                int counter = 0;
449                                for (AlignedSequence<S, C> as : list) {
450                                        counter++;
451
452                                        if ( webDisplay && list.size() == 2){
453                                                printSequenceAlignmentWeb(s, counter, idFormat, seqIndexPre, seqIndexFormatPre, seqIndexPost,
454                                                        seqIndexFormatPost, start, end);
455                                        } else {
456                                                if (idFormat != null) {
457                                                        s.append(String.format(idFormat, as.getAccession()));
458                                                }
459                                                if (seqIndexPre) {
460                                                        s.append(String.format(seqIndexFormatPre, as.getSequenceIndexAt(start)));
461                                                }
462
463                                                s.append(as.getSubSequence(start, end).getSequenceAsString());
464
465                                                if (seqIndexPost) {
466                                                        s.append(String.format(seqIndexFormatPost, as.getSequenceIndexAt(end)));
467                                                }
468                                                s.append(String.format("%n"));
469                                        }
470
471                                        if (aligConservation && list.size() == 2 && counter == 1) {
472                                            printConservation(s, idFormat, seqIndexPad, seqIndexPre, start, end, webDisplay);
473                                        }
474                                }
475
476                        }
477                } else {
478                        for (AlignedSequence<S, C> as : list) {
479                                if (idFormat != null) {
480                                        s.append(String.format(idFormat, as.getAccession()));
481                                }
482                                for (int i = 0; i < getLength(); i += width) {
483                                        int start = i + 1, end = Math.min(getLength(), i + width);
484                                        if (seqIndexPre) {
485                                                s.append(String.format(seqIndexFormatPre, as.getSequenceIndexAt(start)));
486                                        }
487                                        s.append(as.getSubSequence(start, end).getSequenceAsString());
488                                        if (seqIndexPost) {
489                                                s.append(String.format(seqIndexFormatPost, as.getSequenceIndexAt(end)));
490                                        }
491                                        s.append(String.format("%n"));
492                                }
493                        }
494                }
495
496                if (webDisplay && aligConservation && list.size() == 2) {
497                        s.append(IOUtils.getPDBLegend());
498                }
499                return s.toString();
500        }

502        private void printSequenceAlignmentWeb(StringBuilder s, int counter, String idFormat, boolean seqIndexPre,
503                String seqIndexFormatPre, boolean seqIndexPost, String seqIndexFormatPost, int start, int end) {
504                AlignedSequence<S,C> as1 = list.get(0), as2 = list.get(1), as = list.get(counter - 1);
505
506                if (idFormat != null) {
507                        s.append(String.format(idFormat, as.getAccession()));
508                }
509                if (seqIndexPre) {
510                        s.append(String.format(seqIndexFormatPre, as.getSequenceIndexAt(start)));
511                }
512
513        String mySeq = as.getSubSequence(start, end).getSequenceAsString();
514        String s1 = as1.getSubSequence(start, end).getSequenceAsString();
515        String s2 = as2.getSubSequence(start, end).getSequenceAsString();
516
517                for (int i = 0; i < s1.length(); i++) {
518                        if (i >= s2.length() || i >= mySeq.length())
519                                break;
520
521                        char c1 = s1.charAt(i);
522                        char c2 = s2.charAt(i);
523                        char c = mySeq.charAt(i);
524                        s.append(IOUtils.getPDBCharacter(true, c1, c2, isSimilar(c1, c2), c));
525                }
526
527                if (seqIndexPost) {
528                        s.append(String.format(seqIndexFormatPost, as.getSequenceIndexAt(end)));
529                }
530                s.append(String.format("%n"));
531
532        }

534        private void printConservation(StringBuilder s, String idFormat, int seqIndexPad, boolean seqIndexPre, int start,
535                int end, boolean webDisplay) {
536            AlignedSequence<S,C> as1 = list.get(0), as2 = list.get(1);
537
538            if (idFormat != null) {
539                AccessionID ac1 = as1.getAccession();
540                String id1 = (ac1 == null) ? "null" : ac1.getID();
541                id1 = id1.replaceAll(".", " ");
542                s.append(String.format(idFormat, id1));
543            }
544
545            if (seqIndexPre) {
546                s.append(String.format("%" + (seqIndexPad + 1) + "s", ""));
547            }
548
549            String subseq1 = as1.getSubSequence(start, end).getSequenceAsString();
550            String subseq2 = as2.getSubSequence(start, end).getSequenceAsString();
551
552            for ( int ii =0 ; ii < subseq1.length() ; ii++){
553                if ( ii >= subseq2.length())
554                    break;
555                char c1 = subseq1.charAt(ii);
556                char c2 = subseq2.charAt(ii);
557                s.append(IOUtils.getPDBConservation(webDisplay, c1, c2, isSimilar(c1, c2)));
558            }
559
560            s.append(String.format("%n"));
561        }

563        protected static final SubstitutionMatrix<AminoAcidCompound> matrix = SubstitutionMatrixHelper.getBlosum65();
564
565        private boolean isSimilar(char c1, char c2) {
566                AminoAcidCompoundSet set = AminoAcidCompoundSet.getAminoAcidCompoundSet();
567
568                AminoAcidCompound aa1 = set.getCompoundForString(""+c1);
569                AminoAcidCompound aa2 = set.getCompoundForString(""+c2);
570
571                short val = matrix.getValue(aa1,aa2);
572                return val > 0;
573        }
