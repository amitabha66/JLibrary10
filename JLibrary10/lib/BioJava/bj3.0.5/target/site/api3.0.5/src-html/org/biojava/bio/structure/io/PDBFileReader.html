053/**
054 * <p>
055 *  The wrapper class for parsing a PDB file.
056 *  </p>
057 *
058 *
059 *  <p>
060 *  Several flags can be set for this class
061 *  <ul>
062 *  
063 * <li> {@link #setAutoFetch(boolean)} - if the PDB file can not be found locally, should it be fetched
064 *  from the PDB ftp servers? (default:false)</li>
065 *  <li> Other parameters can be set using the {@link #setFileParsingParameters(FileParsingParameters)}</li>
066 *  </ul>
067 *  </p>
068 *
069 *
070 *
071 *<h2>Example</h2>
072 * <p>
073 * Q: How can I get a Structure object from a PDB file?
074 * </p>
075 * <p>
076 * A:
077 * <pre>
078 public {@link Structure} loadStructure(String pathToPDBFile){
079                {@link PDBFileReader} pdbreader = new {@link PDBFileReader}();
080
081                {@link Structure} structure = null;
082                try{
083                        structure = pdbreader.getStructure(pathToPDBFile);
084                        System.out.println(structure);
085                } catch (IOException e) {
086                        e.printStackTrace();
087                }
088                return structure;
089        }
090 </pre>
091 *
092 * Access PDB files from a directory, take care of compressed PDB files
093 * <pre>
094 * public {@link Structure} loadStructureById() {
095                String path = "/path/to/PDB/directory/";
096
097                {@link PDBFileReader} pdbreader = new {@link PDBFileReader}();
098                pdbreader.setPath(path);
099                {@link Structure} structure = null;
100                try {
101                        structure = pdbreader.getStructureById("5pti");
102                } catch (IOException e){
103                        e.printStackTrace();
104                }
105                return structure;
106
107        }
108        </pre>
109 *
110 *
111 * @author Andreas Prlic
112 *
113 */

278        /** Check the directory that contains the local cache of PDB and chemical component definiton files.
279         * By default, at startup we fall back to java.io.tmpdir
280         */
281        private  void checkPath(){
282
283                if ((path == null) || (path.equals("")) || path.equals("null")) {
284
285                        String syspath = System.getProperty(AbstractUserArgumentProcessor.PDB_DIR);
286
287                        if ((syspath != null) && (! syspath.equals("")) && (! syspath.equals("null"))){
288
289                                path = syspath;
290                                return;
291                        }
292
293                        // accessing temp. OS directory:         
294                        String property = "java.io.tmpdir";
295
296                        String tempdir = System.getProperty(property);
297
298                        if ( !(tempdir.endsWith(lineSplit) ) )
299                                tempdir = tempdir + lineSplit;
300
301                        // 
302                        System.err.println("you did not set the path in PDBFileReader, don't know where to write the downloaded file to");
303                        System.err.println("assuming default location is temp directory: " + tempdir);
304                        path = tempdir;
305                }
306        }

388        /** try to find the file in the filesystem and return a filestream in order to parse it
389         * rules how to find file
390         * - first check: if file is in path specified by PDBpath
391         * - secnd check: if not found check in PDBpath/xy/ where xy is second and third char of PDBcode.
392         * if autoFetch is set it will try to download missing PDB files automatically.
393         */
394
395        private InputStream getInputStream(String pdbId)
396                        throws IOException
397                        {
398
399                if ( pdbId.length() < 4)
400                        throw new IOException("the provided ID does not look like a PDB ID : " + pdbId);
401
402                checkPath();
403
404                InputStream inputStream =null;
405
406                String pdbFile = getLocalPDBFilePath(pdbId);
407
408                if ( pdbFile != null) {
409                        InputStreamProvider isp = new InputStreamProvider();
410
411                        try {
412                                inputStream = isp.getInputStream(pdbFile);
413                                return inputStream;
414                        } catch (Exception e){
415                                e.printStackTrace();
416                                // something is wrong with the file!
417                                // it probably should be downloaded again...
418                                pdbFile = null;
419                        }
420
421                }
422                if ( pdbFile == null ) {
423                        if (autoFetch){//from here we try our online search
424                                if(fetchCurrent && !fetchFileEvenIfObsolete) {
425                                        String current = PDBStatus.getCurrent(pdbId);
426
427                                        if(current == null) {
428                                                // either an error or there is not current entry
429                                                current = pdbId;
430                                        }
431                                        return downloadAndGetInputStream(current, CURRENT_FILES_PATH);
432                                } else if(fetchFileEvenIfObsolete && PDBStatus.getStatus(pdbId) == Status.OBSOLETE) {
433                                        return downloadAndGetInputStream(pdbId, OBSOLETE_FILES_PATH);
434                                } else {
435                                        return downloadAndGetInputStream(pdbId, CURRENT_FILES_PATH);
436                                }
437                        }else {
438                                String message = "no structure with PDB code " + pdbId + " found!" ;
439                                throw new IOException (message);
440                        }
441                }
442                return null ;
443                        }

445        /** Returns null if local PDB file does not exist or should be downloaded again...
446         * 
447         * @param pdbId
448         * @return
449         */
450        private String getLocalPDBFilePath(String pdbId) {
451
452                String pdbFile = null ;
453                File f = null ;
454
455                // this are the possible PDB file names...
456                String fpath ;
457                String ppath ;
458
459                if ( pdbDirectorySplit){
460                        // pdb files are split into subdirectories based on their middle position...
461                        String middle = pdbId.substring(1,3).toLowerCase();
462                        fpath = path+lineSplit + middle + lineSplit + pdbId;
463                        ppath = path +lineSplit +  middle + lineSplit + "pdb"+pdbId;
464                } else {
465                        fpath = path+lineSplit + pdbId;
466                        ppath = path +lineSplit + "pdb"+pdbId;
467                }
468
469                String[] paths = new String[]{fpath,ppath};
470
471
472                for ( int p=0;p<paths.length;p++ ){
473                        String testpath = paths[p];
474                        //System.out.println(testpath);
475                        for (int i=0 ; i<extensions.size();i++){
476                                String ex = (String)extensions.get(i) ;
477                                //System.out.println("PDBFileReader testing: "+testpath+ex);
478                                f = new File(testpath+ex) ;
479
480                                if ( f.exists()) {
481
482                                        pdbFile = testpath+ex ;
483
484                                        // we have found the file locally
485
486                                        if ( params.isUpdateRemediatedFiles()){
487                                                long lastModified = f.lastModified();
488
489                                                if (lastModified < lastRemediationDate) {
490                                                        // the file is too old, replace with newer version
491                                                        System.out.println("replacing file " + pdbFile +" with latest remediated file from PDB.");
492                                                        pdbFile = null;
493
494                                                        return null;
495                                                }
496                                        }
497
498                                        return pdbFile;
499                                }
500
501                                if ( pdbFile != null) break;
502                        }
503                }
504                return null;
505        }

511        /**
512         * Returns an input stream for a biological assembly file based on the passed in pdbId and 
513         * the biological assembly id {@link #setBioAssemblyId(int)}. Files are cached in a local directory.
514         * @param pdbId
515         * @return InputStream
516         * @throws IOException
517         * @author Peter Rose
518         * @since 3.2
519         */
520        private InputStream getInputStreamBioAssembly(String pdbId)
521                        throws IOException
522                        {
523                loadedBioAssembly = true;
524                InputStream inputStream = null;
525
526                if ( pdbId.length() < 4)
527                        throw new IOException("the provided ID does not look like a PDB ID : " + pdbId);
528
529                checkPath();
530
531                // make sure path follows unix convention
532                String uPath = FileDownloadUtils.toUnixPath(path);
533
534                // create local subdirectory for biological assembly files if it doesn't exist
535                String dir = uPath + LOCAL_BIO_ASSEMBLY_DIRECTORY;
536
537                File tmp = new File(dir);
538                if ( ! tmp.exists()){
539                        tmp.mkdir();
540                }
541
542                String fpath ;
543
544                if (pdbDirectorySplit){
545                        // pdb files are split into subdirectories based on their middle position...
546                        String middle = pdbId.substring(1,3).toLowerCase();
547                        fpath = dir + "/" + middle + "/";
548                } else {
549                        fpath = dir + "/";
550                }  
551
552                String fileName = fpath + getBiologicalAsssemblyFileName(pdbId.toLowerCase(), bioAssemblyId);
553                File f = new File(fileName) ;
554                
555                // check if bio assembly file exists in local cache
556                if ( f.exists()) {
557                        InputStreamProvider isp = new InputStreamProvider();
558
559                        try {
560                                inputStream = isp.getInputStream(fileName);
561                                return inputStream;
562                        } catch (Exception e){
563                                e.printStackTrace();
564                                // something is wrong with the file!
565                                // it probably should be downloaded again...
566                        }
567                } else if (bioAssemblyFallback) {
568                        if (pdbDirectorySplit){
569                                // pdb files are split into subdirectories based on their middle position...
570                                String middle = pdbId.substring(1,3).toLowerCase();
571                                fpath = uPath + middle + "/";
572                        } else {
573                                fpath = uPath;
574                        }  
575                        fileName = fpath + "pdb" + pdbId + ".ent.gz";
576                        f = new File(fileName);
577
578                        if (f.exists()) {
579
580                                try {
581                                        InputStreamProvider isp = new InputStreamProvider();
582                                        inputStream = isp.getInputStream(fileName);
583                                        System.out.println("Loaded original PDB file as a fallback." + fileName);
584                                        loadedBioAssembly = false;
585                                        return inputStream;
586                                } catch (Exception e) {
587                                        // now try autofetch next
588                                }
589                        }
590                }
591
592                if (autoFetch){//from here we try our online search
593                        if(fetchCurrent && !fetchFileEvenIfObsolete) {
594                                String current = PDBStatus.getCurrent(pdbId);
595
596                                if(current == null) {
597                                        // either an error or there is not current entry
598                                        current = pdbId;
599                                }
600                                inputStream  = downloadAndGetInputStreamBioAssembly(current, BIO_ASSEMBLY_FILES_PATH);
601                                if (inputStream != null) {
602                                        return inputStream;
603                                }
604                        } else if(fetchFileEvenIfObsolete && PDBStatus.getStatus(pdbId) == Status.OBSOLETE) {
605                                String message = "No biological assembly with PDB code " + pdbId + " found!" ;
606                                throw new IOException (message);
607                        } else {
608                                inputStream = downloadAndGetInputStreamBioAssembly(pdbId, BIO_ASSEMBLY_FILES_PATH);
609                                if (inputStream != null) {
610                                        return inputStream;
611                                }
612                        }
613                } 
614
615                // if biological assembly file cannot be found, and bioAssemblyFallBack is true,
616                // get the original PDB file as a fall back (i.e. for NMR structures, where the
617                // PDB file represents the biological assembly).
618                if (bioAssemblyFallback) {      
619                        inputStream = getInputStream(pdbId);            
620                        if (inputStream != null) {
621                                System.out.println("Biological assembly file for PDB ID: " + pdbId+  " is not available. " +
622                                                "Loaded original PDB file as a fallback from local cache.");
623                                return getInputStream(pdbId);
624                        }
625                }
626
627                return null;
628                        }

630        private  File downloadPDB(String pdbId, String pathOnServer){
631
632                if ((path == null) || (path.equals(""))){
633                        // accessing temp. OS directory:         
634                        String property = "java.io.tmpdir";
635
636                        String tempdir = System.getProperty(property);
637
638                        if ( !(tempdir.endsWith(lineSplit) ) )
639                                tempdir = tempdir + lineSplit;
640                        System.err.println("you did not set the path in PDBFileReader, don;t know where to write the downloaded file to");
641                        System.err.println("assuming default location is temp directory: " + tempdir);
642                        path = tempdir;
643                }
644
645
646
647                File realFile = null;
648
649                pdbId = pdbId.toLowerCase();
650                String middle = pdbId.substring(1,3);
651
652                if ( pdbDirectorySplit) {
653                        String dir = path+lineSplit+middle;
654                        File directoryCheck = new File (dir);
655                        if ( ! directoryCheck.exists()){
656                                directoryCheck.mkdir();
657                        }
658
659                        realFile =new File(dir+lineSplit+"pdb"+ pdbId+".ent.gz");
660                } else {
661
662                        realFile = new File(path+lineSplit+"pdb"+pdbId+".ent.gz");
663                }
664
665
666                String serverName = System.getProperty(PDB_FILE_SERVER_PROPERTY);
667
668                if ( serverName == null)
669                        serverName = DEFAULT_PDB_FILE_SERVER;
670
671                String ftp = String.format("ftp://%s%s%s/pdb%s.ent.gz", serverName,pathOnServer,middle, pdbId);
672
673                //System.out.println("Fetching " + ftp);
674
675
676                try {
677                        URL url = new URL(ftp);
678
679                        FileDownloadUtils.downloadGzipCompressedFile(url, realFile);
680
681                } catch (Exception e){
682                        System.err.println("Problem while downloading PDB ID " + pdbId + " from FTP server." );
683                        System.err.println("Failed URL: " + ftp);
684                        e.printStackTrace();
685                        return null;
686                }
687
688
689                return realFile;
690        }

695        /**
696         * Downloads a biological assembly file. If this file cannot be found, it will download the original 
697         * PDB file (i.e. for NMR structures), if bioAssemblyFallback has been set to true;
698         * @param pdbId
699         * @param pathOnServer
700         * @return File of biological assembly
701         * @author Peter Rose
702         * @since 3.2
703         */
704        private  File downloadPDBBiologicalAssembly(String pdbId, String pathOnServer){ 
705                loadedBioAssembly = true;
706
707                checkPath();
708
709                String fileName = getBiologicalAsssemblyFileName(pdbId, bioAssemblyId);
710
711                pdbId = pdbId.toLowerCase();
712                String middle = pdbId.substring(1,3);
713
714                String serverName = System.getProperty(PDB_FILE_SERVER_PROPERTY);
715
716                if ( serverName == null)
717                        serverName = DEFAULT_PDB_FILE_SERVER;
718
719                String ftp = String.format("ftp://%s%s%s/%s", serverName,pathOnServer,middle,fileName);
720
721                System.out.println("Fetching " + ftp);
722
723                URL url = null;
724                try {
725                        url = new URL(ftp);
726                } catch (MalformedURLException e1) {
727                        System.err.println("Problem while downloading Biological Assembly " + pdbId + " from FTP server." );
728                        e1.printStackTrace();
729                        return null;
730                }
731
732                // check if the file exists on the FTP site. If biological assembly file does not exist
733                // and the fallback has been set, get the original PDB file instead (i.e. for NMR structures).
734
735                File f = null;
736                
737                try {
738                        f = downloadFileIfAvailable(url, pdbId,fileName);
739                } catch (IOException ioe){
740                        // ignore here because it prob. means the file does not exist...
741                }
742
743                if ( f == null) {
744
745                        if (bioAssemblyFallback) {
746
747                                System.out.println("Biological unit file for PDB ID: " + pdbId+  " is not available. " +
748                                                "Downloading original PDB file as a fallback.");
749
750                                loadedBioAssembly = false;
751
752                                String fallBackPDBF = getLocalPDBFilePath(pdbId);
753
754                                if ( fallBackPDBF != null)
755                                        return new File(fallBackPDBF);
756
757                                return downloadPDB(pdbId, CURRENT_FILES_PATH);
758                        } 
759                        return null;
760                }
761
762                return f;
763
