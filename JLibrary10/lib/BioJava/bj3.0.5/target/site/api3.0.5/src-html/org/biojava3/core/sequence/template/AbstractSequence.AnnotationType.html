091    /**
092     * A ProxySequenceReader allows abstraction of both the storage of the sequence data and the location
093     * of the sequence data. A variety of use cases are possible. A ProxySequenceReader that knows the offset and of the sequence in
094     * a large fasta file. A ProxySequenceReader that can pull Sequence data from Uniprot, NCBI or a custom database.
095     * If the ProxySequecneReader implements various interfaces then the sequence will set those interfaces so that calls to
096     * various methods will be valid.
097     *
098     * @param proxyLoader
099     * @param compoundSet
100     */
101    public AbstractSequence(SequenceReader<C> proxyLoader, CompoundSet<C> compoundSet) {
102        setCompoundSet(compoundSet);
103        setProxySequenceReader(proxyLoader);
104    }

106    /**
107     * Very important method that allows external mappings of sequence data and features. This method
108     * will gain additional interface inspection that allows external data sources with knowledge
109     * of features for a sequence to be supported. 
110     *  
111     * @param proxyLoader
112     */
113    public void setProxySequenceReader(SequenceReader<C> proxyLoader) {
114        this.sequenceStorage = proxyLoader;
115        if (proxyLoader instanceof FeaturesKeyWordInterface) {
116            this.setFeaturesKeyWord((FeaturesKeyWordInterface) sequenceStorage);
117        }
118        if (proxyLoader instanceof DatabaseReferenceInterface) {
119            this.setDatabaseReferences((DatabaseReferenceInterface) sequenceStorage);
120        }
121        if(getAccession() == null && proxyLoader instanceof UniprotProxySequenceReader){ // we have lots of unsupported operations for this call so quick fix to allow this tow rork
122            this.setAccession(proxyLoader.getAccession());
123        }
124    }

305    /**
306     * Return features at a sequence position by type
307     * @param featureType
308     * @param bioSequencePosition
309     * @return
310     */
311    public List<FeatureInterface<AbstractSequence<C>, C>> getFeatures(String featureType, int bioSequencePosition) {
312        ArrayList<FeatureInterface<AbstractSequence<C>, C>> featureHits =
313                new ArrayList<FeatureInterface<AbstractSequence<C>, C>>();
314        List<FeatureInterface<AbstractSequence<C>, C>> features = getFeaturesByType(featureType);
315        if (features != null) {
316            for (FeatureInterface<AbstractSequence<C>, C> feature : features) {
317                if (bioSequencePosition >= feature.getLocations().getStart().getPosition() && bioSequencePosition <= feature.getLocations().getEnd().getPosition()) {
318                    featureHits.add(feature);
319                }
320            }
321        }
322        return featureHits;
323    }

352    /**
353     * Method to help set the proper details for a feature as it relates to a sequence
354     * where the feature needs to have a location on the sequence
355     * @param bioStart
356     * @param bioEnd
357     * @param feature
358     */
359    public void addFeature(int bioStart, int bioEnd, FeatureInterface<AbstractSequence<C>, C> feature) {
360        SequenceLocation<AbstractSequence<C>, C> sequenceLocation =
361                new SequenceLocation<AbstractSequence<C>, C>(bioStart, bioEnd, this);
362        feature.setLocation(sequenceLocation);
363        addFeature(feature);
364    }
365
366    /**
367     * Add a feature to this sequence. The feature will be added to the collection where the order is start position and if more than
368     * one feature at the same start position then longest is added first. This helps on doing feature layout for displaying features
369     * in SequenceFeaturePanel
370     * @param feature
371     */
372    public void addFeature(FeatureInterface<AbstractSequence<C>, C> feature) {
373        features.add(feature);
374        ArrayList<FeatureInterface<AbstractSequence<C>, C>> featureList = groupedFeatures.get(feature.getType());
375        if (featureList == null) {
376            featureList = new ArrayList<FeatureInterface<AbstractSequence<C>, C>>();
377            groupedFeatures.put(feature.getType(), featureList);
378        }
379        featureList.add(feature);
380        Collections.sort(features, AbstractFeature.LOCATION_LENGTH);
381        Collections.sort(featureList, AbstractFeature.LOCATION_LENGTH);
382    }
383
384    /**
385     * Remove a feature from the sequence
386     * @param feature
387     */
388    public void removeFeature(FeatureInterface<AbstractSequence<C>, C> feature) {
389        features.remove(feature);
390        ArrayList<FeatureInterface<AbstractSequence<C>, C>> featureList = groupedFeatures.get(feature.getType());
391        if (featureList != null) {
392            featureList.remove(feature);
393            if (featureList.isEmpty()) {
394                groupedFeatures.remove(feature.getType());
395            }
396        }
397    }
