044        /**
045         * 
046         * @param group a {@link Group} in structure.
047         * @param isAminoAcid true if it is an amino acid.
048         * @return the {@link StructureGroup} of the group.
049         */
050        public static StructureGroup getStructureGroup(Group group, boolean isAminoAcid) {
051                ResidueNumber resNum = group.getResidueNumber();
052                return new StructureGroup(resNum, group.getPDBName(), isAminoAcid);
053        }
054        
055        /**
056         * 
057         * @param atom a {@link Atom} in structure.
058         * @param isParentAminoAcid true if the containing group is an amino acid.
059         * @return the {@link StructureAtom} of the atom.
060         */
061        public static StructureAtom getStructureAtom(Atom atom, boolean isParentAminoAcid) {
062                StructureGroup strucGroup = getStructureGroup(atom.getGroup(), isParentAminoAcid);
063                return new StructureAtom(strucGroup, atom.getName());
064        }
065        
066        /**
067         * 
068         * @param atom1 the first {@link Atom} in structure.
069         * @param isParentAminoAcid1 true if the first containing group is an amino acid..
070         * @param atom2 the second {@link Atom} in structure.
071         * @param isParentAminoAcid2 true if the second containing group is an amino acid..
072         * @return the {@link StructureAtomLinkage} of the two atoms.
073         */
074        public static StructureAtomLinkage getStructureAtomLinkage(Atom atom1, 
075                boolean isParentAminoAcid1, Atom atom2, boolean isParentAminoAcid2) {
076                StructureAtom strucAtom1 = getStructureAtom(atom1, isParentAminoAcid1);
077                StructureAtom strucAtom2 = getStructureAtom(atom2, isParentAminoAcid2);
078                double distance = getAtomDistance(atom1, atom2);
079                return new StructureAtomLinkage(strucAtom1, strucAtom2, distance);
080        }
081        
082        /**
083         * 
084         * @param atom1 the first {@link Atom} in structure.
085         * @param atom2 the second {@link Atom} in structure.
086         * @return the distance between the two atoms in Angstrom.
087         */
088        public static double getAtomDistance(Atom atom1, Atom atom2) {
089                double distance;
090                try {
091                        distance = Calc.getDistance(atom1, atom2);
092                } catch (StructureException e) {
093                        throw new AssertionError();
094                }
095                
096                return distance;
097        }

099        /**
100         * Find a linkage between two groups within tolerance of bond length,
101         * from potential atoms.
102         * @param group1 the first {@link Group}.
103         * @param group2 the second {@link Group}.
104         * @param potentialNamesOfAtomOnGroup1 potential names of the atom on the first group.
105         *                If null, search all atoms on the first group.
106         * @param potentialNamesOfAtomOnGroup2 potential names of the atom on the second group.
107         *                If null, search all atoms on the second group.
108         * @param ignoreNCLinkage true to ignore all N-C linkages
109         * @param bondLengthTolerance bond length error tolerance.
110         * @return an array of two Atoms that form bond between each other
111         *  if found; null, otherwise.
112         */
113        public static Atom[] findNearestAtomLinkage(final Group group1, final Group group2,
114                        List<String> potentialNamesOfAtomOnGroup1, List<String> potentialNamesOfAtomOnGroup2,
115                        final boolean ignoreNCLinkage, double bondLengthTolerance) {
116                List<Atom[]> linkages = findAtomLinkages(group1, group2, 
117                                potentialNamesOfAtomOnGroup1, potentialNamesOfAtomOnGroup2,
118                                ignoreNCLinkage, bondLengthTolerance);
119                
120                Atom[] ret = null;
121                double minDistance = Double.POSITIVE_INFINITY;
122
123                for (Atom[] linkage : linkages) {
124                        double distance;
125                        try {
126                                distance = Calc.getDistance(linkage[0], linkage[1]);
127                        } catch (StructureException e) {
128                                throw new AssertionError();
129                        }
130                        
131                        if (distance < minDistance) {
132                                minDistance = distance;
133                                ret = linkage;
134                        }
135                }
136                
137                return ret;
138        }

140        /**
141         * Find linkages between two groups within tolerance of bond length,
142         * from potential atoms.
143         * @param group1 the first {@link Group}.
144         * @param group2 the second {@link Group}.
145         * @param ignoreNCLinkage true to ignore all N-C linkages
146         * @param bondLengthTolerance bond length error tolerance.
147         * @return a list, each element of which is an array of two Atoms that form bond 
148         * between each other.
149         */
150        public static List<Atom[]> findAtomLinkages(final Group group1,
151                        final Group group2, final boolean ignoreNCLinkage,
152                        final double bondLengthTolerance) {
153                return findAtomLinkages(group1, group2,
154                                null, null, ignoreNCLinkage, bondLengthTolerance);
155        }
156        
157        /**
158         * Find linkages between two groups within tolerance of bond length,
159         * from potential atoms.
160         * @param group1 the first {@link Group}.
161         * @param group2 the second {@link Group}.
162         * @param potentialNamesOfAtomOnGroup1 potential names of the atom on the first group.
163         *                If null, search all atoms on the first group.
164         * @param potentialNamesOfAtomOnGroup2 potential names of the atom on the second group.
165         *                If null, search all atoms on the second group.
166         * @param ignoreNCLinkage true to ignore all N-C linkages
167         * @param bondLengthTolerance bond length error tolerance.
168         * @return a list, each element of which is an array of two Atoms that form bond 
169         * between each other.
170         */
171        public static List<Atom[]> findAtomLinkages(final Group group1, 
172                        final Group group2,
173                        List<String> potentialNamesOfAtomOnGroup1,
174                        List<String> potentialNamesOfAtomOnGroup2,
175                        final boolean ignoreNCLinkage,
176                        final double bondLengthTolerance) {
177                if (group1==null || group2==null) {
178                        throw new IllegalArgumentException("Null group(s).");
179                }
180                
181                if (bondLengthTolerance<0) {
182                        throw new IllegalArgumentException("bondLengthTolerance cannot be negative.");
183                }
184                
185                List<Atom[]> ret = new ArrayList<Atom[]>();
186                
187                if (potentialNamesOfAtomOnGroup1 == null) {
188                        // if empty name, search for all atoms
189                        potentialNamesOfAtomOnGroup1 = getAtomNames(group1);
190                }
191                
192                if (potentialNamesOfAtomOnGroup2 == null) {
193                        // if empty name, search for all atoms
194                        potentialNamesOfAtomOnGroup2 = getAtomNames(group2);
195                }
196                
197                for (String namesOfAtomOnGroup1 : potentialNamesOfAtomOnGroup1) {
198                        for (String namesOfAtomOnGroup2 : potentialNamesOfAtomOnGroup2) {
199                                Atom[] atoms = findLinkage(group1, group2, namesOfAtomOnGroup1,
200                                                namesOfAtomOnGroup2, bondLengthTolerance);
201                                if (atoms != null) {
202                                        if (ignoreNCLinkage &&
203                                                        ((atoms[0].getName().equals("N") && atoms[1].getName().equals("C"))
204                                                                        || (atoms[0].getName().equals("C") && atoms[1].getName().equals("N")))
205                                                                ) {
206                                                continue;
207                                        }
208                                        
209                                        ret.add(atoms);
210                                }
211                        }
212                }
213                
214                return ret;
215        }

217        /**
218         * Find a linkage between two groups within tolerance of bond length.
219         * @param group1 the first {@link Group}.
220         * @param group2 the second {@link Group}.
221         * @param nameOfAtomOnGroup1 atom name of the first group.
222         * @param nameOfAtomOnGroup2 atom name of the second group.
223         * @param bondLengthTolerance bond length error tolerance.
224         * @return an array of two Atoms that form bond between each other
225         *  if found; null, otherwise.
226         */
227        public static Atom[] findLinkage(final Group group1, final Group group2,
228                        String nameOfAtomOnGroup1, String nameOfAtomOnGroup2,
229                        double bondLengthTolerance) {
230                Atom[] ret = new Atom[2];
231                double distance;
232                
233                try {
234                        ret[0] = group1.getAtom(nameOfAtomOnGroup1);
235                        ret[1] = group2.getAtom(nameOfAtomOnGroup2);
236                        distance = Calc.getDistance(ret[0], ret[1]);
237                } catch (StructureException e) {
238                        return null;
239                }
240                
241                if (ret[0]==null || ret[1]==null) {
242                        return null;
243                }
244                
245                float radiusOfAtom1 = ret[0].getElement().getCovalentRadius();
246                float radiusOfAtom2 = ret[1].getElement().getCovalentRadius();
247                
248                if (Math.abs(distance-radiusOfAtom1 -radiusOfAtom2)
249                                > bondLengthTolerance) {
250                        return null;
251                }
252                
253                return ret;
254        }
255        
256        /**
257         * 
258         * @param group a {@link Group}.
259         * @return all atom names of the group.
260         */
261        public static List<String> getAtomNames(Group group) {
262                List<Atom> atoms = group.getAtoms();
263                if (atoms == null) {
264                        return Collections.emptyList();
265                }
266                
267                int n = atoms.size();
268                List<String> ret = new ArrayList<String>(n);
269                for (int i=0; i<n; i++) {
270                        ret.add(atoms.get(i).getName());
271                }
272                
273                return ret;
274        }
