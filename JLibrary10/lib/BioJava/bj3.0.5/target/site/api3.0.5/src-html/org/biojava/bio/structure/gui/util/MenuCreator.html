024import java.awt.BorderLayout;

032import javax.swing.Box;
033import javax.swing.JButton;
034import javax.swing.JDialog;
035import javax.swing.JEditorPane;
036import javax.swing.JFileChooser;
037import javax.swing.JMenu;
038import javax.swing.JMenuBar;
039import javax.swing.JMenuItem;
040import javax.swing.JScrollPane;
041
042import org.biojava.bio.structure.Structure;
043import org.biojava.bio.structure.io.PDBFileReader;
044
045import org.biojava.bio.structure.align.gui.AlignmentGui;
046import org.biojava.bio.structure.gui.BiojavaJmol;

049/**
050 *  Create the menu for BiojavaJmol
051 * @author Andreas Prlic
052 * @since 1.7
053 */
054public class MenuCreator {
055
056        /** provide a JMenuBar that can be added to a JFrame
057         * 
058         * @return a JMenuBar
059         */
060        public static JMenuBar initMenu(){
061
062                // show a menu
063
064                JMenuBar menu = new JMenuBar();
065
066                JMenu file= new JMenu("File");
067                file.getAccessibleContext().setAccessibleDescription("File Menu");
068
069                JMenuItem openI = new JMenuItem("Open");
070                openI.setMnemonic(KeyEvent.VK_O);
071                openI.addActionListener(new ActionListener(){
072                        public void actionPerformed(ActionEvent e) {
073                                String cmd = e.getActionCommand();                              
074                                if ( cmd.equals("Open")){
075                                        final JFileChooser fc = new JFileChooser();
076
077//                                      In response to a button click:
078                                        int returnVal = fc.showOpenDialog(null);
079                                        if ( returnVal == JFileChooser.APPROVE_OPTION) {
080                                                File file = fc.getSelectedFile();
081                                                
082                                                PDBFileReader reader = new PDBFileReader();
083                                                try {
084                                                        Structure s = reader.getStructure(file);
085                                                        BiojavaJmol jmol = new BiojavaJmol();
086                                                        jmol.setStructure(s);
087                                                        jmol.evalString("select * ; color chain;");
088                                                        jmol.evalString("select *; spacefill off; wireframe off; backbone 0.4;  ");
089
090                                                } catch (Exception ex){
091                                                        ex.printStackTrace();
092                                                }
093                                                
094
095                                        }
096                                }                               
097                        }               
098                });
099                file.add(openI);
100
101                JMenuItem exitI = new JMenuItem("Exit");
102                exitI.setMnemonic(KeyEvent.VK_X);
103                exitI.addActionListener(new ActionListener(){
104
105                        public void actionPerformed(ActionEvent e) {
106                                String cmd = e.getActionCommand();
107
108                                if ( cmd.equals("Exit")){
109                                        System.exit(0);
110                                }                               
111                        }                       
112                });
113
114                file.add(exitI);
115                menu.add(file);
116
117                JMenu align = new JMenu("Align");
118                JMenuItem pairI = new JMenuItem("2 protein structures");
119                pairI.addActionListener(new ActionListener() {
120                        public void actionPerformed(ActionEvent e) {
121                                String cmd = e.getActionCommand();
122
123                                if ( cmd.equals("2 protein structures")){
124                                        MenuCreator.showPairDialog();
125                                }                               
126                        }
127                });
128
129                align.add(pairI);
130
131                menu.add(align);
132
133                JMenu about = new JMenu("About");
134                JMenuItem aboutI = new JMenuItem("PDBview");
135                aboutI.addActionListener(new ActionListener(){
136
137                        public void actionPerformed(ActionEvent e) {
138                                String cmd = e.getActionCommand();
139
140                                if ( cmd.equals("PDBview")){
141                                        MenuCreator.showAboutDialog();
142                                }                               
143                        }                       
144                });
145
146                about.add(aboutI);
147
148                menu.add(Box.createGlue());
149                menu.add(about);
150
151                return menu;
152
153        }
154
155
156        /** provide a display for the pairwise structure alignment
157         * 
158         */
159        private static void showPairDialog(){
160                AlignmentGui gui = AlignmentGui.getInstance();
161                gui.setVisible(true);
162        }
163
164        /** show some info about this gui
165         * 
166         */
167        private static void showAboutDialog(){
168
169                JDialog dialog = new JDialog();
170
171                dialog.setSize(new Dimension(300,300));
172
173                String msg = "This viewer is based on <b>BioJava</b> and <b>Jmol</>. <br>Author: Andreas Prlic <br> ";
174                msg += "Structure Alignment algorithm based on a variation of the PSC++ algorithm by Peter Lackner.";
175
176
177                JEditorPane txt = new JEditorPane("text/html", msg);
178                txt.setEditable(false);
179
180
181                JScrollPane scroll = new JScrollPane(txt);
182
183                Box vBox = Box.createVerticalBox();
184                vBox.add(scroll);
185
186                JButton close = new JButton("Close");
187
188                close.addActionListener(new ActionListener(){
189                        public void actionPerformed(ActionEvent event) {
190                                Object source = event.getSource();
191
192                                JButton but = (JButton)source;
193                                Container parent = but.getParent().getParent().getParent().getParent().getParent().getParent() ;
194
195                                JDialog dia = (JDialog) parent;
196                                dia.dispose();
197                        }
198                });
199
200                Box hBoxb = Box.createHorizontalBox();
201                hBoxb.add(Box.createGlue());
202                hBoxb.add(close,BorderLayout.EAST);
203
204                vBox.add(hBoxb);
205
206                dialog.getContentPane().add(vBox);
207                dialog.setVisible(true);
208
209
210        }
