049/** An interface for the events triggered by a MMcifParser.
050 * The Consumer listens to the events and builds up the protein structure.
051 *
052 * @author Andreas Prlic
053 *  @since 1.7
054 *
055 */
056public interface MMcifConsumer {
057        /** called at start of document
058         *
059         */
060        public void documentStart();
061
062        /** called at end of document
063         *
064         */
065        public void documentEnd();
066
067
068        /** A new AtomSite record has been read. Contains the Atom data
069         *
070         * @param atom
071         */
072        public void newAtomSite(AtomSite atom);
073        public void newEntity(Entity entity);
074        public void newEntityPolySeq(EntityPolySeq epolseq);
075        public void newStructAsym(StructAsym sasym);
076        public void setStruct(Struct struct);
077        public void newDatabasePDBrev(DatabasePDBrev dbrev);
078        public void newDatabasePDBremark(DatabasePDBremark remark);
079        public void newExptl(Exptl exptl);
080        public void newStructRef(StructRef sref);
081        public void newStructRefSeq(StructRefSeq sref);
082        public void newPdbxPolySeqScheme(PdbxPolySeqScheme ppss);
083        public void newPdbxNonPolyScheme(PdbxNonPolyScheme ppss);
084        public void newPdbxEntityNonPoly(PdbxEntityNonPoly pen);
085        public void newStructKeywords(StructKeywords kw);
086        public void newRefine(Refine r);
087        public void newChemComp(ChemComp c);
088        public void newChemCompDescriptor(ChemCompDescriptor ccd);
089        public void newPdbxStructOperList(PdbxStructOperList structOper);
090        public void newPdbxStrucAssembly(PdbxStructAssembly strucAssembly);
091        public void newPdbxStrucAssemblyGen(PdbxStructAssemblyGen strucAssembly);
092        
093        /** AuditAuthor contains the info from the PDB-AUTHOR records.
094         *
095         * @param aa
096         */
097        public void newAuditAuthor(AuditAuthor aa);
098
099        /** This method is called if no particular handler for the provided cif category
100         * has been implemented so far.
101         * @param category The category that is being processed.
102         * @param loopFields the fields of this category.
103         * @param lineData the data that is being provided.
104         */
105        public void newGenericData(String category, List<String> loopFields, List<String> lineData);
106
107        public void setFileParsingParameters(FileParsingParameters params);
108        public FileParsingParameters getFileParsingParameters();
109
110
111}
