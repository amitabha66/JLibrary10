044/**
045 * A utility class for common {@link String} manipulation tasks.

050public class StringManipulationHelper  {

052        /**
053         * we are using Unix endline here, since this is used for testing XML and it
054         * is part of the XML recommendations: <a href
055         * ="http://www.w3.org/TR/REC-xml/#sec-line-ends"
056         * >http://www.w3.org/TR/REC-xml/#sec-line-ends</a>
057         */
058        private static final String UNIX_NEWLINE = "\n";
059
060        private StringManipulationHelper() {
061                // to prevent instantiation
062        }
063
064        
065
066        
067
068        /**
069         * @author andreas
070         * @param stream
071         * @return
072         */
073        public static String convertStreamToString(InputStream stream) {
074                BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
075                StringBuilder sb = new StringBuilder();
076
077                String line = null;
078                try {
079                        while ((line = reader.readLine()) != null) {
080
081                sb.append(line).append(UNIX_NEWLINE);
082                        }
083                } catch (IOException e) {
084                        // e.printStackTrace();
085                } finally {
086                        try {
087                                stream.close();
088                        } catch (IOException e) {
089                                e.printStackTrace();
090                        }
091                }
092
093                return sb.toString();
094        }
095        
096        /**
097         * compares two strings for equality, line by line, ignoring any difference
098         * of end line delimiters contained within the 2 Strings. This method should
099         * be used if and only if two Strings are considered identical when all nodes
100         * are identical including their relative order. Generally useful when
101         * asserting identity of <b>automatically regenerated</b> XML or PDB.
102         * 
103         * @param expected
104         * @param actual
105         */
106        public static boolean equalsToIgnoreEndline(String expected, String actual) {
107                if (expected == null && actual == null) {
108                        return true;
109                }
110                if (expected != null ^ actual != null) {
111                        return false;
112                }
113                Scanner scanner1 = new Scanner(expected);
114                Scanner scanner2 = new Scanner(actual);
115                String line1, line2;
116                while (scanner1.hasNextLine()) {
117                        line1 = scanner1.nextLine();
118                        line2 = scanner2.nextLine();
119                        if (! line1.equals(line2))
120                                return false;
121                }
122                if (scanner2.hasNextLine()) {
123                        return false;
124                }
125
126                return true;
127        }

130        public static boolean equalsToXml(String expected, String actual) {
131                Document expectedDocument=null;
132                Document actualDocument=null;
133                try {
134                        DocumentBuilderFactory documentBuilderFactory = DocumentBuilderFactory.newInstance();
135                        DocumentBuilder documentBuilder = documentBuilderFactory.newDocumentBuilder();
136                        expectedDocument = documentBuilder.parse(new ByteArrayInputStream(expected.getBytes()));
137                        actualDocument = documentBuilder.parse(new ByteArrayInputStream(actual.getBytes()));
138                } catch (ParserConfigurationException e) {
139                        e.printStackTrace();
140                        throw new RuntimeException("Couldn't Parse XML", e);
141                } catch (SAXException e) {
142                        e.printStackTrace();
143                        throw new RuntimeException("Couldn't Parse XML", e);
144                } catch (IOException e) {
145                        e.printStackTrace();
146                        throw new RuntimeException("Couldn't Parse XML", e);
147                }
148                final DocumentType doctype1 = expectedDocument.getDoctype();
149                final DocumentType doctype2 = actualDocument.getDoctype();
150                if (doctype1==null ^ doctype2 == null) {
151                        return false;
152                }else if (doctype1!= null /*&& doctype2 != null*/) {
153                        NamedNodeMap expectedNotations = doctype1.getNotations();
154                        NamedNodeMap actualNotations = doctype2.getNotations();
155                        if (expectedNotations.getLength() == actualNotations.getLength()) {
156                                for (int i = 0; i < expectedNotations.getLength(); i++) {
157                                        Node node= expectedNotations.item(i);
158                                        node.isEqualNode(null);
159                                }
160                        }else{
161                                return false;
162                        }
163                        
164                }
165                
166                throw new UnsupportedOperationException("not yet implemented");
167        }
