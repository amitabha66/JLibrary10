001package org.biojava.bio.structure.align.gui;
002
003import java.awt.BorderLayout;
004import java.awt.Cursor;
005import java.awt.Insets;
006import java.awt.event.ActionEvent;
007import java.awt.event.ActionListener;
008
009import javax.swing.BorderFactory;
010import javax.swing.JButton;
011
012import javax.swing.JPanel;
013import javax.swing.JProgressBar;
014import javax.swing.JScrollPane;
015import javax.swing.JTextArea;
016
017import org.biojava.bio.structure.align.FarmJob;
018import org.biojava.bio.structure.align.events.AlignmentProgressListener;
019
020/** a GUI that allows to watch progress as multiple alignments are being processed.
021 * 
022 * @author Andreas Prlic
023 *
024 */
025public class GUIAlignmentProgressListener extends JPanel implements AlignmentProgressListener, ActionListener {
026
027
028
029
030        /**
031         * 
032         */
033        private static final long serialVersionUID = 1L;
034
035        int alignmentsProcessed;
036
037        JProgressBar progressBar;
038        private JTextArea taskOutput;
039        private JButton stopButton;
040
041        FarmJob farmJob;
042
043        public GUIAlignmentProgressListener(){
044
045                super(new BorderLayout());
046                stopButton = new JButton("Stop");
047                stopButton.setActionCommand("Stop");
048                stopButton.addActionListener(this);
049
050                progressBar = new JProgressBar(0, 100);
051                progressBar.setValue(0);
052                progressBar.setStringPainted(true);
053
054                taskOutput = new JTextArea(5, 20);
055                taskOutput.setMargin(new Insets(5,5,5,5));
056                taskOutput.setEditable(false);
057
058                JPanel panel = new JPanel();
059                panel.add(stopButton);
060                panel.add(progressBar);
061
062                add(panel, BorderLayout.PAGE_START);
063                add(new JScrollPane(taskOutput), BorderLayout.CENTER);
064                setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
065
066        }
067        
068        
069
070         public FarmJob getFarmJob() {
071                return farmJob;
072        }
073
074
075
076        public void setFarmJob(FarmJob parent) {
077                this.farmJob = parent;
078        }
079
080
081
082        /**
083     * Invoked when the user presses the stop button.
084     */
085    public void actionPerformed(ActionEvent evt) {
086        
087        //System.out.println("stopping!");
088        logStatus("terminating");
089        logStatus(" Total alignments processed: " + alignmentsProcessed);
090        stopButton.setEnabled(false);
091        setCursor(Cursor.getPredefinedCursor(Cursor.WAIT_CURSOR));
092        progressBar.setIndeterminate(true);
093        progressBar.setStringPainted(false);
094        System.out.println("terminating jobs");
095        
096        farmJob.terminate();
097       
098        setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
099        progressBar.setIndeterminate(false);
100        
101    }
102
103   
104        public void alignmentEnded() {
105                
106                alignmentsProcessed++;
107                
108                //System.out.println("aligned " + alignmentsProcessed );
109                int v = progressBar.getValue();
110                
111                progressBar.setValue(v+1);
112                progressBar.setString(v+"");
113        synchronized(this){notifyAll();}
114
115        }
116
117        public void alignmentStarted(String name1, String name2) {      
118                logStatus("#" + progressBar.getValue() + " starting alignment of " + name1 + " " + name2);
119        }
120
121        public void downloadingStructures(String name) {                
122                logStatus("Downloading " + name );
123        }
124
125        public void logStatus(String message) {         
126                taskOutput.append(message+"\n");
127        }
128
129        public void requestingAlignmentsFromServer(int nrAlignments) {
130                logStatus("Requesting " + nrAlignments + " alignments to be calculated");
131                progressBar.setMaximum(nrAlignments);
132                progressBar.setValue(0);        
133        synchronized(this){notifyAll();}
134
135        }
136
137        public void sentResultsToServer(int nrAlignments, String serverMessage) {
138                logStatus("sent alignment results back to server. Server responded: >"+serverMessage+"<");
139                progressBar.setValue(0);
140                synchronized(this){notifyAll();}
141        }
142
143
144}
