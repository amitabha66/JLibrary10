001package org.biojava.bio.structure.align.xml;
002
003import java.io.IOException;
004import java.io.PrintWriter;
005import java.io.StringReader;
006import java.io.StringWriter;
007import java.util.SortedSet;
008import java.util.TreeSet;
009
010import javax.xml.parsers.DocumentBuilder;
011import javax.xml.parsers.DocumentBuilderFactory;
012
013
014import org.biojava3.core.util.PrettyXMLWriter;
015
016import org.w3c.dom.Document;
017
018import org.w3c.dom.NamedNodeMap;
019import org.w3c.dom.Node;
020import org.w3c.dom.NodeList;
021import org.xml.sax.InputSource;
022
023public class RepresentativeXMLConverter {
024
025
026        public static final String toXML(SortedSet<String> representatives){
027                StringWriter sw = new StringWriter();
028                PrintWriter writer = new PrintWriter(sw);
029
030                PrettyXMLWriter xml = new PrettyXMLWriter(writer);
031                try {
032                        xml.openTag("representatives");
033
034                        for ( String repr : representatives){
035                                xml.openTag("pdbChain");
036                                xml.attribute("name", repr);                                                            
037                                xml.closeTag("pdbChain");
038                        }
039                        xml.closeTag("representatives");
040                } catch(IOException ex){
041                        ex.printStackTrace();
042                }
043                return sw.toString();
044        }
045
046        public static final SortedSet<String> fromXML(String xml){
047                SortedSet<String> representatives = new TreeSet<String>();
048                try {
049                        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
050                        DocumentBuilder db = factory.newDocumentBuilder();
051                        InputSource inStream = new InputSource();
052                        inStream.setCharacterStream(new StringReader(xml));
053                        Document doc = db.parse(inStream);
054
055                        // normalize text representation
056                        doc.getDocumentElement().normalize();
057
058
059                        //Element rootElement = doc.getDocumentElement();
060
061                        NodeList listOfPairs = doc.getElementsByTagName("pdbChain");
062                        //int numArrays = listOfArrays.getLength();
063
064                        // go over the blocks
065                        for(int i=0; i<listOfPairs.getLength() ; i++)
066                        {
067                                Node pair       = listOfPairs.item(i);
068                                //NodeList valList = pair.getChildNodes();
069                                //int numChildren  = valList.getLength();
070
071                                NamedNodeMap map = pair.getAttributes();
072
073                                String name =  map.getNamedItem("name").getTextContent();                               
074                                representatives.add(name);
075                        }
076
077                } catch (Exception e){
078                        e.printStackTrace();
079                }
080                
081                return representatives;
082        }
083}
