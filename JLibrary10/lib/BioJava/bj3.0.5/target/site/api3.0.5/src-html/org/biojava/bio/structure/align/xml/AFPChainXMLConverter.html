032        /** Write the XML representation to a StringWriter
033         * 
034         * @param afpChain
035         * @param swriter
036         * @throws IOException
037         */
038        public synchronized static void toXML(AFPChain afpChain, StringWriter swriter,Atom[] ca1, Atom[]ca2) throws IOException{
039
040                PrintWriter writer = new PrintWriter(swriter);
041                PrettyXMLWriter xml = new PrettyXMLWriter(writer);
042                
043                xml.openTag("AFPChain");
044                
045                printXMLHeader(xml,afpChain);
046
047
048                // that is the initial alignment...
049                // we don't serialize that at the present.
050                //int[] blockResSize = afpChain.getBlockResSize();
051                //int[][][] blockResList = afpChain.getBlockResList();
052
053
054                // get the alignment blocks
055                int blockNum = afpChain.getBlockNum();
056                //int[] optLen       = afpChain.getOptLen();
057                //int[] blockSize    = afpChain.getBlockSize();
058                for(int bk = 0; bk < blockNum; bk ++) {
059                                                
060                        xml.openTag("block");
061
062                        printXMLBlockHeader(xml,afpChain, bk);
063
064                        if ( ca1 == null || ca2 == null) {
065                                try {
066                                        printXMLEQRKnownPositions(xml,afpChain,bk);
067                                } catch (StructureException ex ){
068                                        throw new IOException(ex.getMessage());
069                                }
070                        }
071                        else 
072                                printXMLEQRInferPositions(xml, afpChain,bk,ca1,ca2);
073
074                        printXMLMatrixShift(xml, afpChain, bk);
075
076                        xml.closeTag("block");
077                }
078
079                xml.closeTag("AFPChain");
080
081                writer.close();
082
083
084        }

088        private static void printXMLEQRKnownPositions(PrettyXMLWriter xml,
089                        AFPChain afpChain, int blockNr) throws IOException, StructureException{
090                int[] optLen       = afpChain.getOptLen();
091
092
093                String[][][] pdbAln = afpChain.getPdbAln();
094                if ( pdbAln == null){
095                        throw new StructureException("Can't convert to XML without known the PDB coordinates. Please provide Ca atoms and call toXML(afpChain,ca1, ca2)");
096                }
097
098                for ( int eqrNr = 0 ; eqrNr < optLen[blockNr] ; eqrNr++ ){
099                        
100                        String pdbResnum1 = pdbAln[blockNr][0][eqrNr];
101                        String pdbResnum2 = pdbAln[blockNr][1][eqrNr];
102                        
103                        //System.out.println(eqrNr + " got resnum: " + pdbResnum1 + " " + pdbResnum2);
104                        
105                        String[] spl1 = pdbResnum1.split(":");
106                        String[] spl2 = pdbResnum2.split(":");
107
108                        String chain1 = spl1[0];
109                        String pdbres1 = spl1[1];
110
111                        String chain2 = spl2[0];
112                        String pdbres2 = spl2[1];
113
114                        xml.openTag("eqr");
115                        xml.attribute("eqrNr",eqrNr+"");
116                        xml.attribute("pdbres1",pdbres1);
117                        xml.attribute("chain1", chain1);
118                        xml.attribute("pdbres2",pdbres2);
119                        xml.attribute("chain2", chain2);
120                        xml.closeTag("eqr");
121                }
122        }

125        public static void printXMLEQRInferPositions(PrettyXMLWriter xml,                       
126                        AFPChain afpChain, int bk, Atom[] ca1, Atom[] ca2)  throws IOException{
127                
128                int[] optLen       = afpChain.getOptLen();
129                
130                if ( optLen == null)
131                        return;
132                                
133                int[][][] optAln   = afpChain.getOptAln();
134                
135                
136                for ( int pos=0;pos< optLen[bk];pos++){
137                        int pos1 = optAln[bk][0][pos];
138                        int pos2 = optAln[bk][1][pos];
139                        xml.openTag("eqr");
140                        xml.attribute("eqrNr",pos+"");
141                        xml.attribute("pdbres1",ca1[pos1].getGroup().getResidueNumber().toString());
142                        xml.attribute("chain1", ca1[pos1].getGroup().getChain().getChainID());
143                        xml.attribute("pdbres2",ca2[pos2].getGroup().getResidueNumber().toString());
144                        xml.attribute("chain2", ca2[pos2].getGroup().getChain().getChainID());
145                        
146                        xml.closeTag("eqr");
147                        //System.out.println("aligned position: " + pos1  + ":" + pos2 + 
148                        //" pdbresnum " + ca1[pos1].getGroup().getResidueNumber().toString() + " " +
149                        //ca1[pos1].getParent().getPDBName()+":" + 
150                        //ca2[pos2].getGroup().getResidueNumber().toString() + " " + ca2[pos2].getParent().getPDBName());
151         
152                }        
153
154        }

157        private static void printXMLBlockHeader(PrettyXMLWriter xml,
158                        AFPChain afpChain,int blockNr) throws IOException{
159
160                int bk = blockNr;
161                int[] blockSize    = afpChain.getBlockSize();
162                //if ( blockSize[bk] == 0) {
163                //      return;
164                //}
165                int[] blockGap     = afpChain.getBlockGap();
166                
167                double[]blockScore = afpChain.getBlockScore();
168                double[] blockRmsd = afpChain.getBlockRmsd();
169
170                xml.attribute("blockNr",bk+"");
171                xml.attribute("blockSize", blockSize[bk]+"");
172                xml.attribute("blockScore", String.format("%5.2f",blockScore[bk]).trim());
173                xml.attribute("blockRmsd", String.format("%5.2f",blockRmsd[bk]).trim());
174                xml.attribute("blockGap", blockGap[bk]+"");
175
176        }

179        private static void printXMLMatrixShift(PrettyXMLWriter xml,
180                        AFPChain afpChain, int blockNr)  throws IOException {
181           
182           Matrix[] ms     = afpChain.getBlockRotationMatrix();
183           if ( ms == null || ms.length == 0)
184          return;
185           
186           Matrix matrix = ms[blockNr]; 
187           if ( matrix == null)
188                   return;
189           xml.openTag("matrix");
190                
191                        
192                for (int x=0;x<3;x++){
193                        for (int y=0;y<3;y++){
194                                String key = "mat"+(x+1)+(y+1);
195                                xml.attribute(key,String.format("%.6f",matrix.get(x,y)));
196                        }
197                }
198                xml.closeTag("matrix");
199
200                Atom[]   shifts = afpChain.getBlockShiftVector();
201                Atom shift = shifts[blockNr];
202                xml.openTag("shift");
203                xml.attribute("x", String.format("%.3f",shift.getX()));
204                xml.attribute("y", String.format("%.3f",shift.getY()));
205                xml.attribute("z", String.format("%.3f",shift.getZ()));
206                xml.closeTag("shift");
207
208        }

217        public static void printXMLHeader(PrettyXMLWriter xml, AFPChain afpChain) throws IOException{
218                xml.attribute("name1", afpChain.getName1());
219                xml.attribute("name2", afpChain.getName2());
220                xml.attribute("method", afpChain.getAlgorithmName());
221                xml.attribute("version" , afpChain.getVersion());
222                xml.attribute("alnLength", afpChain.getAlnLength() + "");
223                xml.attribute("blockNum", afpChain.getBlockNum() + "");
224                xml.attribute("gapLen", afpChain.getGapLen() + "");
225                xml.attribute("optLength", afpChain.getOptLength() + "");
226                xml.attribute("totalLenIni", afpChain.getTotalLenIni() + "");
227
228                xml.attribute("alignScore", String.format("%5.2f", afpChain.getAlignScore() ).trim());
229                xml.attribute("chainRmsd",  String.format("%5.2f", afpChain.getChainRmsd() ).trim());
230                xml.attribute("identity",String.format("%5.4f", afpChain.getIdentity() ).trim());
231                xml.attribute("normAlignScore", String.format("%5.2f",afpChain.getNormAlignScore()).trim());
232                xml.attribute("probability", String.format("%.2e", afpChain.getProbability() ).trim());
233                xml.attribute("similarity", String.format("%5.4f", afpChain.getSimilarity() ).trim());
234                
235                xml.attribute("similarity1", afpChain.getCoverage1() + "");
236                xml.attribute("similarity2", afpChain.getCoverage2() + "");
237                xml.attribute("totalRmsdIni", String.format("%5.2f",afpChain.getTotalRmsdIni() ).trim());
238                xml.attribute("totalRmsdOpt", String.format("%5.2f",afpChain.getTotalRmsdOpt() ).trim());
239                xml.attribute("ca1Length", afpChain.getCa1Length()+"");
240                xml.attribute("ca2Length", afpChain.getCa2Length()+"");
241                xml.attribute("afpNum",afpChain.getAfpSet().size()+"");
242                xml.attribute("alignScoreUpdate",String.format("%5.2f",afpChain.getAlignScoreUpdate()).trim());
243                xml.attribute("time", String.format("%d",afpChain.getCalculationTime()));
244                if ( afpChain.getTMScore() != -1){
245                        xml.attribute("tmScore", String.format("%.2f",afpChain.getTMScore()));
246                }
247                
248                // test if alignment is CP:
249                if ( ! AlignmentTools.isSequentialAlignment(afpChain,false)) {
250                        xml.attribute("cp","true");
251                }
252        }
