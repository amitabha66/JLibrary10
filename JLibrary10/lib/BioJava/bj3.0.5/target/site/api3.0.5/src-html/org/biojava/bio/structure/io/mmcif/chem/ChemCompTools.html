164    public static final boolean isStandardChemComp(ChemComp cc){
165
166        String pid = cc.getMon_nstd_parent_comp_id();
167        String one = cc.getOne_letter_code();
168
169        PolymerType polymerType = cc.getPolymerType();
170
171        // standard residues have no parent
172        if ((pid == null) || (pid.equals("?"))){
173          
174                // and they have a one letter code
175            if ( ( one != null) && ( ! one.equals("?") )){
176
177                // peptides and dpeptides must not have X
178                if ( (polymerType == PolymerType.peptide) ||
179                        ( polymerType == PolymerType.dpeptide)) {
180                    return performPeptideCheck(cc, one);
181
182                }
183                if (polymerType == PolymerType.rna){
184                    return performRNACheck(cc);
185                }
186                if (polymerType == PolymerType.dna) {
187
188                    return performDNACheck(cc);
189
190                }
191
192                System.err.println("what is this?" + cc);
193                return false;
194            }
195        }
196        return false;
197    }

200    private static boolean performRNACheck(ChemComp cc) {
201        if (cc.getId().length() == 1)
202            return true;
203        else
204            return false;
205    }

208    private static boolean performDNACheck(ChemComp cc) {
209        if ( cc.getId().equals(UNKNOWN_NUCLEOTIDE))
210            return false;
211
212        Character c = getDNAOneLetter(cc.getId());
213        if ( c==null){
214            // we did not find it in the list of standard nucleotides
215            return false;
216        }
217        return true;
218    }
219
220
221    private static boolean performPeptideCheck(ChemComp cc, String one) {
222        if (one.equals(UNKNOWN_ONE_LETTER_CODE)) {
223            return false;
224        }
225        Character c =  getAminoOneLetter(cc.getId());
226        if ( c==null){
227            // we did not find it in the list of standard aminos
228            return false;
229        }
230        return true;
231    }

237    public static Character getOneLetterCode(ChemComp cc, ChemicalComponentDictionary dictionary){
238        if ( cc.getResidueType() == ResidueType.nonPolymer )
239                return null;
240
241        if ( cc.isStandard())
242            return cc.getOne_letter_code().charAt(0);
243
244        ChemComp parent = dictionary.getParent(cc);
245        if ( parent == null){
246            //System.err.println("parent is null " + cc);
247            return cc.getOne_letter_code().charAt(0);
248        }
249        PolymerType poly = cc.getPolymerType();
250        if (( poly == PolymerType.peptide) || ( poly == PolymerType.dpeptide)){
251            Character c = getAminoOneLetter(parent.getId());
252            if ( c == null)
253                c = UNKNOWN_ONE_LETTER_CODE;
254            return c;
255        }
256        if ( poly == PolymerType.dna){
257            Character c = getDNAOneLetter(parent.getId());
258            if (c == null)
259                c = UNKNOWN_NUCLEOTIDE;
260            return c;
261
262        }
263        return cc.getMon_nstd_parent_comp_id().charAt(0);
264    }
