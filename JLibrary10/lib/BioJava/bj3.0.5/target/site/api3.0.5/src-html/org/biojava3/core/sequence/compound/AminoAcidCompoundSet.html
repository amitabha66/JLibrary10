037/**
038 * Set of proteinogenic amino acids.  Molecular weights are recorded in daltons (Da) as residues of a chain; monomers
039 * outside of a chain would likely have an additional mass of 18.01524 Da contributed by an associated water molecule.
040 *
041 * Currently we have different symbols to handle inserts so not as clean as it should be
042 *
043 * @author Richard Holland
044 * @author Scooter Willis
045 * @author Mark Chapman
046 */
047public class AminoAcidCompoundSet implements CompoundSet<AminoAcidCompound> {

102    public String getStringForCompound(AminoAcidCompound compound) {
103        return compound.toString();
104    }
105
106    public AminoAcidCompound getCompoundForString(String string) {
107        if (string.length() == 0) {
108            return null;
109        }
110        if (string.length() == 3) {
111                return this.aminoAcidCompoundCache3Letter.get(string.toUpperCase());
112        }
113        if (string.length() > this.getMaxSingleCompoundStringLength()) {
114            throw new IllegalArgumentException("String supplied ("+string+") is too long. Max is "+getMaxSingleCompoundStringLength());
115        }
116        return this.aminoAcidCompoundCache.get(string);
117    }
118
119    public int getMaxSingleCompoundStringLength() {
120        return 1;
121    }
122
123
124    public boolean isCompoundStringLengthEqual() {
125        return true;
126    }

128    private final static AminoAcidCompoundSet aminoAcidCompoundSet = new AminoAcidCompoundSet();
129
130    public static AminoAcidCompoundSet getAminoAcidCompoundSet() {
131        return aminoAcidCompoundSet;
132    }

134    public boolean compoundsEquivalent(AminoAcidCompound compoundOne, AminoAcidCompound compoundTwo) {
135        Set<AminoAcidCompound> equivalents = getEquivalentCompounds(compoundOne);
136        return (equivalents != null) && equivalents.contains(compoundTwo);
137    }
138
139    public Set<AminoAcidCompound> getEquivalentCompounds(AminoAcidCompound compound) {
140        if (equivalentsCache.isEmpty()) {
141            // most compounds are equivalent to themselves alone
142            for (AminoAcidCompound c : aminoAcidCompoundCache.values()) {
143                equivalentsCache.put(c, Collections.singleton(c));
144            }
145            // ambiguous Asparagine or Aspartic acid
146            addAmbiguousEquivalents("N", "D", "B");
147            // ambiguous Glutamine or Glutamic acid
148            addAmbiguousEquivalents("E", "Q", "Z");
149            // ambiguous Leucine or Isoleucine
150            addAmbiguousEquivalents("I", "L", "J");
151            // ambiguous gaps
152            AminoAcidCompound gap1, gap2, gap3;
153            Set<AminoAcidCompound> gaps = new HashSet<AminoAcidCompound>();
154            gaps.add(gap1 = aminoAcidCompoundCache.get("-"));
155            gaps.add(gap2 = aminoAcidCompoundCache.get("."));
156            gaps.add(gap3 = aminoAcidCompoundCache.get("_"));
157            equivalentsCache.put(gap1, gaps);
158            equivalentsCache.put(gap2, gaps);
159            equivalentsCache.put(gap3, gaps);
160            // X is never equivalent, even to itself
161            equivalentsCache.put(aminoAcidCompoundCache.get("X"), new HashSet<AminoAcidCompound>());
162        }
163        return equivalentsCache.get(compound);
164    }
165
166    // helper method to initialize the equivalent sets for 2 amino acid compounds and their ambiguity compound
167    private void addAmbiguousEquivalents(String one, String two, String either) {
168        Set<AminoAcidCompound> equivalents;
169        AminoAcidCompound cOne, cTwo, cEither;
170
171        equivalents = new HashSet<AminoAcidCompound>();
172        equivalents.add(cOne = aminoAcidCompoundCache.get(one));
173        equivalents.add(cTwo = aminoAcidCompoundCache.get(two));
174        equivalents.add(cEither = aminoAcidCompoundCache.get(either));
175        equivalentsCache.put(cEither, equivalents);
176
177        equivalents = new HashSet<AminoAcidCompound>();
178        equivalents.add(cOne);
179        equivalents.add(cEither);
180        equivalentsCache.put(cOne, equivalents);
181
182        equivalents = new HashSet<AminoAcidCompound>();
183        equivalents.add(cTwo);
184        equivalents.add(cEither);
185        equivalentsCache.put(cTwo, equivalents);
186    }
187
188    public boolean hasCompound(AminoAcidCompound compound) {
189        return aminoAcidCompoundCache.containsValue(compound);
190    }
191
192    // TODO throwing an error seems unnecessary, should this return a boolean instead? maybe rename to isValidSequence?
193    public void verifySequence(Sequence<AminoAcidCompound> sequence) throws CompoundNotFoundError {
194        for (AminoAcidCompound compound : sequence) {
195            if (!hasCompound(compound)) {
196                throw new CompoundNotFoundError("Compound (" + compound + ") not found in AminoAcidCompoundSet.");
197            }
198        }
199    }
200
201    public List<AminoAcidCompound> getAllCompounds() {
202        return new ArrayList<AminoAcidCompound>(aminoAcidCompoundCache.values());
203    }

206    public boolean isComplementable() {
207        return false;
208    }
