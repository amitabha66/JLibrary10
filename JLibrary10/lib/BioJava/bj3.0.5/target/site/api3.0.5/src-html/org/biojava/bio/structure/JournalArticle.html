211    @Override
212    public String toString() {
213//        JRNL        AUTH   M.HAMMEL,G.SFYROERA,D.RICKLIN,P.MAGOTTI,
214//        JRNL        AUTH 2 J.D.LAMBRIS,B.V.GEISBRECHT
215//        JRNL        TITL   A STRUCTURAL BASIS FOR COMPLEMENT INHIBITION BY
216//        JRNL        TITL 2 STAPHYLOCOCCUS AUREUS.
217//        JRNL        REF    NAT.IMMUNOL.                  V.   8   430 2007
218//        JRNL        REFN                   ISSN 1529-2908
219//        JRNL        PMID   17351618
220//        JRNL        DOI    10.1038/NI1450
221        String eol = System.getProperty("line.separator");
222
223        StringBuffer jrnlString = new StringBuffer();
224
225        StringBuffer authString = new StringBuffer("JRNL        AUTH   ");
226        StringBuffer titlString = new StringBuffer("JRNL        TITL   ");
227        StringBuffer editString = new StringBuffer("JRNL        EDIT   ");
228        StringBuffer refString = new StringBuffer("JRNL        REF    ");
229        StringBuffer publString = new StringBuffer("JRNL        PUBL   ");
230        StringBuffer refnString = new StringBuffer("JRNL        REFN                   ");
231        StringBuffer pmidString = new StringBuffer("JRNL        PMID   ");
232        StringBuffer doiString = new StringBuffer("JRNL        DOI    ");
233
234        for (Author author : authorList) {
235            authString.append(author).append(",");
236        }
237        jrnlString.append(authString.toString()).append(eol);
238        titlString.append(title);
239        jrnlString.append(titlString.toString()).append(eol);
240        if (!editorList.isEmpty()) {
241            for (Author editor : editorList) {
242                editString.append(editor).append(",");
243            }
244            jrnlString.append(editString.toString()).append(eol);
245        }
246//        if (ref.equals("TO BE PUBLISHED")) {
247            refString.append(ref);
248        jrnlString.append(refString.toString()).append(eol);
249//        }
250        if (!publisher.equals("")) {
251            publString.append(publisher);
252            jrnlString.append(publString.toString()).append(eol);
253        }
254        if (!refn.equals("")) {
255            refnString.append(refn);
256            jrnlString.append(refnString.toString()).append(eol);
257        }
258        if (!pmid.equals("")) {
259            pmidString.append(pmid);
260            jrnlString.append(pmidString.toString()).append(eol);
261        }
262        if (!doi.equals("")) {
263            doiString.append(doi);
264            jrnlString.append(doiString.toString()).append(eol);
265        }
266
267        return jrnlString.toString();
268    }
