111        public Chain getMatchingAtomRes(Chain seqRes, List<Chain> atomList)
112                        throws StructureException {
113                Iterator<Chain> iter = atomList.iterator();
114                while(iter.hasNext()){
115                        Chain atomChain = iter.next();
116                        if ( atomChain.getChainID().equals(seqRes.getChainID())){
117                                return atomChain;
118                        }
119                }
120                throw new StructureException("could not match seqres chainID >" + seqRes.getChainID() + "< to ATOM chains!");
121        }

149        /** map the seqres groups to the atomRes chain.
150         *  updates the atomRes chain object with the mapped data
151         *  seqRes chain shuold not be needed after this and atomRes should be continued to be used.
152         *  
153         * @param atomRes
154         * @param seqRes
155         * @throws StructureException 
156         */
157        public void mapSeqresRecords(Chain atomRes, Chain seqRes) throws StructureException {
158                List<Group> seqResGroups = seqRes.getAtomGroups();
159                List<Group> atmResGroups = atomRes.getAtomGroups();
160                
161                
162                if (DEBUG) {
163                        System.err.println("COMPARING " + atomRes.getChainID() + " (" + atmResGroups.size()+") " + seqRes.getChainID() + " (" +seqResGroups.size() +") ");
164                }
165                
166                
167                boolean simpleMatchSuccessful = trySimpleMatch(seqResGroups, atmResGroups);
168
169                if ( simpleMatchSuccessful) {
170                        // update the new SEQRES list
171                        
172                        atomRes.setSeqResGroups(seqResGroups);
173                        return;
174                }
175
176                //System.err.println("Could not map SEQRES to ATOM records easily, need to align...");
177
178                if ( seqRes.getAtomGroups(GroupType.AMINOACID).size() < 1) {
179
180                        if ( seqRes.getAtomGroups(GroupType.NUCLEOTIDE).size() > 1) {
181                                if (DEBUG){
182                                        System.out.println("chain " + seqRes.getChainID() + " is a nucleotide chain, aligning nucs...");
183                                }
184                                align2NucleotideChains(seqRes,atomRes);
185                                return;
186                        } else {
187
188                                if (DEBUG){
189                                        System.out.println("chain " + seqRes.getChainID() + " does not contain amino acids, ignoring...");
190                                }
191                                return;
192                        }
193                }
194
195
196
197
198
199                if ( atomRes.getAtomGroups(GroupType.AMINOACID).size() < 1) {
200                        if (DEBUG){
201                                System.out.println("chain " + atomRes.getChainID() + " does not contain amino acids, ignoring...");
202                        }
203                        return;
204                }
205                if ( DEBUG )
206                        System.out.println("Alignment for chain "+ atomRes.getChainID() );
207
208                
209                boolean noMatchFound = align(seqResGroups,atomRes.getAtomGroups());
210                if ( ! noMatchFound){
211                        atomRes.setSeqResGroups(seqResGroups);
212                }
213                
214        }

235        /** a simple matching approach that tries to do a 1:1 mapping between SEQRES and ATOM records
236         *  returns true if this simple matching approach worked fine
237         *  
238         * @param seqRes
239         * @param atomList
240         * @return
241         */
242        
243        public boolean trySimpleMatch(List<Group> seqResGroups,List<Group> atmResGroups) {
244                // by default first ATOM position is 1
245                //
246                boolean startAt1 = true;
247
248                for ( int atomResPos = 0 ; atomResPos < atmResGroups.size() ; atomResPos++){
249
250                        //                      if ( DEBUG)
251                        //                              System.err.println(" trying to simple match " + atomResPos +"/" + atmResGroups.size());
252
253                        // let's try to match this case
254                        Group atomResGroup = atmResGroups.get(atomResPos);
255
256                        // let's ignore waters
257                        String threeLetterCode = atomResGroup.getPDBName();
258                        if ( excludeTypes.contains(threeLetterCode)){
259                                continue;
260                        }
261
262                        ResidueNumber atomResNum = atomResGroup.getResidueNumber();
263
264                        int seqResPos = atomResNum.getSeqNum();
265
266
267
268                        if ( seqResPos < 0) {
269                                if ( DEBUG )
270                                        System.err.println("ATOM residue number < 0");
271                                return false;
272                        }
273
274                        if ( seqResPos == 0){
275                                // make sure the first SEQRES is matching.
276                                Group seqResGroup = seqResGroups.get(0);
277                                if (  seqResGroup.getPDBName().equals(atomResGroup.getPDBName())){
278                                        // they match! seems in this case the numbering starts with 0...
279                                        startAt1 = false;
280                                } else {
281                                        if ( DEBUG){
282                                                System.err.println("SEQRES position 1  ("+seqResGroup.getPDBName()+
283                                                                ") does not match ATOM PDB res num 0 (" + atomResGroup.getPDBName()+")");
284
285                                        }
286                                        return false;
287
288                                }
289                        }
290
291
292                        if ( startAt1 )
293                                seqResPos--;
294
295                        // another check that would require the alignment approach
296                        if ( startAt1 && seqResPos >=  seqResGroups.size()  )
297                        {
298
299                                // could be a HETATOM...
300                                if ( atomResGroup instanceof AminoAcid) {
301                                        if ( DEBUG )
302                                                System.err.println(" ATOM residue nr: " + seqResPos + " > seqres! " + seqResGroups.size() + " " + atomResGroup);
303                                        return false;
304                                } else if ( atomResGroup instanceof NucleotideImpl) {
305                                        if ( DEBUG )
306                                                System.err.println(" NUCLEOTIDE residue nr: " + seqResPos + " > seqres! " + seqResGroups.size() + " " + atomResGroup);
307                                        return false;
308                                } else {
309                                        // we won't map HETATOM groups...
310                                        continue;
311                                }
312                        }
313
314
315                        //                      if ( seqResPos < 0){
316                        //
317                        //                              System.err.println("What is going on??? " + atomRes.getChainID() + " " + atomResGroup);
318                        //                      }
319
320                        if ( seqResPos >= seqResGroups.size()){
321                                if ( DEBUG)
322                                        System.err.println("seqres groups don't match atom indeces " + seqResPos);
323                                if ( atomResGroup instanceof AminoAcid )
324                                        return false;
325                                else
326                                        continue;
327                        }
328
329                        Group seqResGroup = seqResGroups.get(seqResPos );
330
331                        if ( ! seqResGroup.getPDBName().trim().equals(atomResGroup.getPDBName().trim())){
332                                // a mismatch! something is wrong in the mapping and we need to do an alignment
333                                if ( DEBUG )
334                                        System.err.println("Mismatch of SEQRES pos " + seqResPos + " and ATOM record: " + atomResGroup + " | " + seqResGroup);
335                                return false;
336                        }
337
338                        // the two groups are identical and we can merge them
339                        // replace the SEQRES group with the ATOM group...
340                        if (DEBUG)
341                                System.err.println("merging " + seqResPos + " " + atomResGroup);
342                        seqResGroups.set(seqResPos, atomResGroup);
343
344                }
345
346                // all atom records could get matched correctly!
347                return true;
348
349        }

351        /** returns the full sequence of the Atom records of a parent
352         * with X instead of HETATMSs. The advantage of this is
353         * that it allows us to also align HETATM groups back to the SEQRES.
354         * @param groups the list of groups in a parent
355         * @param positionIndex a Map to keep track of which group is at which sequence position
356         * @return string representations
357         */
358        public static String getFullAtomSequence(List<Group> groups, Map<Integer, Integer> positionIndex){
359
360
361                StringBuffer sequence = new StringBuffer() ;
362                int seqIndex = 0; // track sequence.length()
363                for ( int i=0 ; i< groups.size(); i++){
364                        Group g = (Group) groups.get(i);
365
366                        if ( g instanceof AminoAcid ){
367                                AminoAcid a = (AminoAcid)g;
368                                char oneLetter =a.getAminoType();
369                                if ( oneLetter == '?')
370                                        oneLetter = 'X';
371
372                                positionIndex.put(seqIndex,i);
373                                sequence.append(oneLetter );
374                                seqIndex++;
375                        } else {
376
377                                // exclude solvents
378                                if (  excludeTypes.contains(g.getPDBName()))
379                                        continue;
380
381                                // exclude metals
382                                if ( g.size() == 1 ) {
383                                        try {
384                                                Atom a = g.getAtom(0);
385                                                if (a.getElement().isMetal())
386                                                        continue;
387
388                                                continue;
389                                        } catch (StructureException e){
390                                                e.printStackTrace();
391                                                continue;
392                                        }
393                                }
394
395                                ChemComp cc = g.getChemComp();
396                                if ( 
397                                                ResidueType.lPeptideLinking.equals(cc.getResidueType()) ||
398                                                PolymerType.PROTEIN_ONLY.contains(cc.getPolymerType())  ||
399                                                PolymerType.POLYNUCLEOTIDE_ONLY.contains(cc.getPolymerType())
400                                                ) {
401                                        //System.out.println(cc.getOne_letter_code());
402                                        String c = cc.getOne_letter_code();
403                                        if ( c.equals("?"))
404                                                c = "X";
405
406                                        positionIndex.put(seqIndex,i);
407                                        sequence.append(c);
408                                        seqIndex++;
409                                } else {
410                                        //System.out.println(cc);
411                                        continue;
412                                }
413
414
415                                //sequence.append("X");
416                        }
417
418                }
419
420                return sequence.toString();
421
422        }

425        public boolean alignNucleotideGroups(List<Group> seqRes, List<Group> atomRes) throws StructureException{
426
427                Map<Integer,Integer> seqresIndexPosition = new HashMap<Integer, Integer>();
428                Map<Integer,Integer> atomIndexPosition   = new HashMap<Integer, Integer>();
429
430                String seq1 = getFullAtomSequence(seqRes, seqresIndexPosition);
431                //
432                String seq2 = getFullAtomSequence(atomRes, atomIndexPosition);
433
434                if ( DEBUG ) {
435
436                        System.out.println("align seq1 ("+ seq1.length()+") " + seq1);
437                        System.out.println("align seq2 ("+ seq2.length()+") " + seq2);
438                }
439
440
441                Sequence s1;
442                Sequence s2;
443
444                try {
445                        s1 = new DNASequence(seq1,AmbiguityDNACompoundSet.getDNACompoundSet());
446                } catch (CompoundNotFoundError e){
447                        ///oops perhaps a RNA sequence?
448                        try {
449                                s1 = new RNASequence(seq1,AmbiguityRNACompoundSet.getRNACompoundSet());
450                        } catch (CompoundNotFoundError ex) {
451                                System.err.println("Could not determine compound set for sequence 1 " + seq1);
452                                return false;
453                        }
454                }
455                try {
456                        s2 = new DNASequence(seq2,AmbiguityDNACompoundSet.getDNACompoundSet());
457                } catch (CompoundNotFoundError e){
458                        ///oops perhaps a RNA sequence?
459                        try {
460                                s2 = new RNASequence(seq2,AmbiguityRNACompoundSet.getRNACompoundSet());
461                        } catch (CompoundNotFoundError ex) {
462                                System.err.println("Could not determine compound set for sequence 2 " + seq2);
463                                return false;
464                        }
465                }
466
467                SubstitutionMatrix<NucleotideCompound> matrix = SubstitutionMatrixHelper.getNuc4_2();
468
469                GapPenalty penalty = new SimpleGapPenalty();
470
471                short gop = 8;
472                short extend = 1;
473                penalty.setOpenPenalty(gop);
474                penalty.setExtensionPenalty(extend);
475
476                try {
477                        PairwiseSequenceAligner smithWaterman =
478                                        Alignments.getPairwiseAligner(s1, s2, PairwiseSequenceAlignerType.LOCAL, penalty, matrix);
479
480                        SequencePair pair = smithWaterman.getPair();
481
482
483
484                        if ( pair == null)
485                                throw new StructureException("could not align objects!");
486
487
488                        if ( DEBUG ) {
489                                System.out.println(pair.toString(60));
490                        }
491
492                        boolean noMatchFound = mapDNAChains(seqRes,atomRes,pair,seqresIndexPosition, atomIndexPosition );
493
494                        return noMatchFound;
495                } catch (Exception e){
496                        System.err.println("Problem while aligning ATOM and SEQRES records for " + atomRes.get(0).getChain().getParent().getPDBCode() + " chain: "  +atomRes.get(0).getChain().getChainID());
497                        //e.printStackTrace();
498                        return true;
499                }
500
501        }

504        /** aligns two chains of groups, where the first parent is representing the
505         * list of amino acids as obtained from the SEQRES records, and the second parent
506         * represents the groups obtained from the ATOM records (and containing the actual ATOM information).
507         * This does the actual alignment and if a group can be mapped to a position in the SEQRES then the corresponding
508         * position is repplaced with the group that contains the atoms.
509         *
510         * @param seqRes
511         * @param atomRes
512         * @return true if no match has bee found
513         * @throws StructureException
514         */
515        public boolean align(List<Group> seqRes, List<Group> atomRes) throws StructureException{
516
517                Map<Integer,Integer> seqresIndexPosition = new HashMap<Integer, Integer>();
518                Map<Integer,Integer> atomIndexPosition   = new HashMap<Integer, Integer>();
519
520                String seq1 = getFullAtomSequence(seqRes, seqresIndexPosition);
521                //
522                String seq2 = getFullAtomSequence(atomRes, atomIndexPosition);
523
524                if ( DEBUG ) {
525
526                        System.out.println("align seq1 ("+ seq1.length()+") " + seq1);
527                        System.out.println("align seq2 ("+ seq2.length()+") " + seq2);
528                }
529                ProteinSequence s1 = new ProteinSequence(seq1);
530                ProteinSequence s2 = new ProteinSequence(seq2);
531
532                SubstitutionMatrix<AminoAcidCompound> matrix = SubstitutionMatrixHelper.getBlosum65();
533
534                GapPenalty penalty = new SimpleGapPenalty();
535
536                short gop = 8;
537                short extend = 1;
538                penalty.setOpenPenalty(gop);
539                penalty.setExtensionPenalty(extend);
540
541                try {
542                        PairwiseSequenceAligner<ProteinSequence, AminoAcidCompound> smithWaterman =
543                                        Alignments.getPairwiseAligner(s1, s2, PairwiseSequenceAlignerType.LOCAL, penalty, matrix);
544
545                        SequencePair<ProteinSequence, AminoAcidCompound> pair = smithWaterman.getPair();
546
547
548
549                        if ( pair == null)
550                                throw new StructureException("could not align objects!");
551
552
553                        if ( DEBUG ) {
554                                System.out.println(pair.toString(60));
555                        }
556
557                        boolean noMatchFound = mapChains(seqRes,atomRes,pair,seqresIndexPosition, atomIndexPosition );
558
559                        return noMatchFound;
560                } catch (Exception e){
561                        System.err.println("Problem while aligning ATOM and SEQRES records for " + atomRes.get(0).getChain().getParent().getPDBCode() + " chain: "  +atomRes.get(0).getChain().getChainID());
562                        //e.printStackTrace();
563                        return true;
564                }
565
566        }

569        private boolean mapChains(List<Group> seqResGroups, List<Group> atomRes,
570                        SequencePair<ProteinSequence, AminoAcidCompound> pair,
571                        Map<Integer,Integer> seqresIndexPosition,
572                        Map<Integer,Integer> atomIndexPosition
573                        ) throws StructureException{
574
575
576
577                // at the present stage the future seqREs are still stored as Atom groups in the seqRes parent...
578
579
580                int aligLength = pair.getLength();
581
582                //              System.out.println("sequence length seqres:");
583                //              System.out.println(seqresIndexPosition.keySet().size());
584                //              System.out.println("alignment length: " + aligLength);
585                // System.out.println(gapSymbol.getName());
586
587                // make sure we actually find an alignment
588                boolean noMatchFound = true;
589
590                Compound gapSymbol =  AminoAcidCompoundSet.getAminoAcidCompoundSet().getCompoundForString("-");
591
592                mainLoop:
593                        for (int i = 1; i <= aligLength ; i++) {
594
595                                Compound s =  pair.getCompoundAt(1, i);
596                                Compound a =  pair.getCompoundAt(2,i);
597
598                                // alignment is using internal index start at 1...
599                                int posSeq = pair.getIndexInQueryAt(i)  -1;
600                                int posAtom = pair.getIndexInTargetAt(i) -1;
601
602                                if (  s.equals(gapSymbol) || a.equals(gapSymbol)){
603                                        continue;                               
604                                }
605
606                                if ( s.equals(a)){
607
608                                        // the atom record can be aligned to the SeqRes record!
609                                        // replace the SeqRes group with the Atom group!
610
611                                        Group s1 = seqResGroups.get(seqresIndexPosition.get(posSeq));
612                                        Group a1 = atomRes.get(atomIndexPosition.get(posAtom));
613
614                                        if ( s1 == null || a1 == null){
615                                                /// can't map this position...
616                                                System.err.println("can't map " + i + ":" + s + " " + posSeq +" " + s1 + " atom: " + posAtom + " " + a1 );
617                                                continue mainLoop;
618                                        }
619
620                                        // need to trim the names to allow matching e.g in
621                                        // pdb1b2m
622                                        String pdbNameS = s1.getPDBName();
623                                        String pdbNameA = a1.getPDBName();
624                                        if ( pdbNameS == null || pdbNameA == null ){
625                                                System.err.println("null value found at " + posSeq + " when trying to align " + s1 + " and " + a1 + " " + posAtom);
626                                                throw new StructureException("null value found at group.getPDBName()");
627                                        }
628                                        if (! pdbNameA.equals(pdbNameS)){
629                                                if ( ! pdbNameA.trim().equals(pdbNameS.trim())) {
630                                                        System.err.println(s1 + " " + posSeq + " does not align with " + a1+ " " + posAtom + " should be: " + s + " : " + a);
631                                                        if ( s1.getType().equals(HetatomImpl.type) && a1.getType().equals(HetatomImpl.type)){
632                                                                System.err.println("they seem to be hetatoms, so ignoring mismatch.");
633                                                        }
634                                                        else {
635                                                                //  System.exit(0);// for debug only
636                                                                //System.out.println(lst1.seqString());
637                                                                //System.out.println(lst2.seqString());
638                                                                System.err.println("could not match residues " + s1 + " " + a1);
639                                                                //throw new StructureException("could not match residues " + s1 + " " + a1);
640                                                        }
641
642                                                }
643                                        }
644
645                                        // do the actual replacing of the SEQRES group with the ATOM group
646                                        //                                      if ( s1.getChain().getChainID().equals("A")) {
647                                        //                                              System.out.println(" setting " + posSeq +" " + a1);
648                                        //                                      }
649                                        seqResGroups.set(seqresIndexPosition.get(posSeq),a1);
650                                        noMatchFound = false;
651                                }
652                        }
653
654
655                // now we merge the two chains into one
656                // the Groups that can be aligned are now pointing to the
657                // groups in the Atom records.
658                if (  noMatchFound) {
659
660                        if ( DEBUG )
661                                System.out.println("no alignment found!");
662                }
663                return noMatchFound;
664
665        }

667        private boolean mapDNAChains(List<Group> seqResGroups, List<Group> atomRes,
668                        SequencePair<DNASequence, NucleotideCompound> pair,
669                        Map<Integer,Integer> seqresIndexPosition,
670                        Map<Integer,Integer> atomIndexPosition
671                        ) throws StructureException{
672
673
674
675                // at the present stage the future seqREs are still stored as Atom groups in the seqRes parent...
676
677
678                int aligLength = pair.getLength();
679
680                //              System.out.println("sequence length seqres:");
681                //              System.out.println(seqresIndexPosition.keySet().size());
682                //              System.out.println("alignment length: " + aligLength);
683                // System.out.println(gapSymbol.getName());
684
685                // make sure we actually find an alignment
686                boolean noMatchFound = true;
687
688                Compound gapSymbol =  DNACompoundSet.getDNACompoundSet().getCompoundForString("-");
689
690                mainLoop:
691                        for (int i = 1; i <= aligLength ; i++) {
692
693                                Compound s =  pair.getCompoundAt(1, i);
694                                Compound a =  pair.getCompoundAt(2,i);
695
696                                // alignment is using internal index start at 1...
697                                int posSeq = pair.getIndexInQueryAt(i)  -1;
698                                int posAtom = pair.getIndexInTargetAt(i) -1;
699
700                                if (  s.equals(gapSymbol) || a.equals(gapSymbol)){
701                                        continue;                               
702                                }
703
704                                if ( s.equals(a)){
705
706                                        // the atom record can be aligned to the SeqRes record!
707                                        // replace the SeqRes group with the Atom group!
708
709                                        Group s1 = seqResGroups.get(seqresIndexPosition.get(posSeq));
710                                        Group a1 = atomRes.get(atomIndexPosition.get(posAtom));
711
712                                        if ( s1 == null || a1 == null){
713                                                /// can't map this position...
714                                                System.err.println("can't map " + i + ":" + s + " " + posSeq +" " + s1 + " atom: " + posAtom + " " + a1 );
715                                                continue mainLoop;
716                                        }
717
718                                        // need to trim the names to allow matching e.g in
719                                        // pdb1b2m
720                                        String pdbNameS = s1.getPDBName();
721                                        String pdbNameA = a1.getPDBName();
722                                        if ( pdbNameS == null || pdbNameA == null ){
723                                                System.err.println("null value found at " + posSeq + " when trying to align " + s1 + " and " + a1 + " " + posAtom);
724                                                throw new StructureException("null value found at group.getPDBName()");
725                                        }
726                                        if (! pdbNameA.equals(pdbNameS)){
727                                                if ( ! pdbNameA.trim().equals(pdbNameS.trim())) {
728                                                        System.err.println(s1 + " " + posSeq + " does not align with " + a1+ " " + posAtom + " should be: " + s + " : " + a);
729                                                        if ( s1.getType().equals(HetatomImpl.type) && a1.getType().equals(HetatomImpl.type)){
730                                                                System.err.println("they seem to be hetatoms, so ignoring mismatch.");
731                                                        }
732                                                        else {
733                                                                //  System.exit(0);// for debug only
734                                                                //System.out.println(lst1.seqString());
735                                                                //System.out.println(lst2.seqString());
736                                                                System.err.println("could not match residues " + s1 + " " + a1);
737                                                                //throw new StructureException("could not match residues " + s1 + " " + a1);
738                                                        }
739
740                                                }
741                                        }
742
743                                        // do the actual replacing of the SEQRES group with the ATOM group
744                                        //                                      if ( s1.getChain().getChainID().equals("A")) {
745                                        //                                              System.out.println(" setting " + posSeq +" " + a1);
746                                        //                                      }
747                                        seqResGroups.set(seqresIndexPosition.get(posSeq),a1);
748                                        noMatchFound = false;
749                                }
750                        }
751
752
753                // now we merge the two chains into one
754                // the Groups that can be aligned are now pointing to the
755                // groups in the Atom records.
756                if (  noMatchFound) {
757
758                        if ( DEBUG )
759                                System.out.println("no alignment found!");
760                }
761                return noMatchFound;
762
763        }
