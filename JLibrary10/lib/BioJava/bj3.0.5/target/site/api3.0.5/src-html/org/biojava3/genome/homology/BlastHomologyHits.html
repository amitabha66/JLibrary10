005package org.biojava3.genome.homology;
006
007import java.io.File;
008import java.util.ArrayList;
009import java.util.LinkedHashMap;
010import org.biojava3.genome.query.BlastXMLQuery;
011
012/**
013 *
014 * @author Scooter Willis <willishf at gmail dot com>
015 */
016public class BlastHomologyHits {
017
018    static public LinkedHashMap<String, ArrayList<String>> getMatches(File xmlBlastHits, double ecutoff) throws Exception {
019        LinkedHashMap<String, ArrayList<String>> homologyHits = new LinkedHashMap<String, ArrayList<String>>();
020        BlastXMLQuery blastXMLQuery = new BlastXMLQuery(xmlBlastHits.getAbsolutePath());
021        LinkedHashMap<String, ArrayList<String>> hits = blastXMLQuery.getHitsQueryDef(ecutoff);
022        for (String accessionid : hits.keySet()) {
023            String[] data = accessionid.split(" "); // deal with notes/comments in blast results
024            String id = data[0];
025            ArrayList<String> uniprotProteinHits = hits.get(accessionid);
026            homologyHits.put(id, uniprotProteinHits);
027
028        }
029        return homologyHits;
030    }
031}
