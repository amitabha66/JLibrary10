077   /** Traverse throught the SCOP hierarchy
078    * 
079    */
080   public void traverseHierarchy()
081   {
082      String pdbId = "4HHB";
083      // download SCOP if required and load into memory
084      ScopDatabase scop = ScopFactory.getSCOP();
085      
086      List<ScopDomain> domains = scop.getDomainsForPDB(pdbId);
087      
088      // show the hierachy for the first domain:
089      
090      ScopNode node = scop.getScopNode(domains.get(0).getSunid());
091      
092      while (node != null){
093         
094         System.out.println("This node: sunid:" + node.getSunid() );
095         System.out.println(scop.getScopDescriptionBySunid(node.getSunid()));
096         node = scop.getScopNode(node.getParentSunid());
097      }
098      
099   }

116   public void alignSuperfamily(){     
117      // download SCOP if required and load into memory
118           ScopDatabase scop = ScopFactory.getSCOP();
119      List<ScopDescription> superfams = scop.getByCategory(ScopCategory.Superfamily);
120
121      System.out.println("Total nr. of superfamilies:" + superfams.size());
122
123      
124      // configure where to load PDB files from and 
125      // what information to load
126      AtomCache cache = new AtomCache();      
127      FileParsingParameters fileparams = new FileParsingParameters() ;
128      fileparams.setAlignSeqRes(false);
129      fileparams.setLoadChemCompInfo(true);
130      fileparams.setParseSecStruc(false);
131      cache.setFileParsingParams(fileparams);
132      
133      // get tge first superfamily
134      ScopDescription superfam1 = superfams.get(0);
135      System.out.println("First superfamily: " + superfam1);
136      
137      ScopNode node = scop.getScopNode(superfam1.getSunID());
138      System.out.println("scopNode for first superfamily:" + node);
139      
140      List<ScopDomain> doms4superfam1 = scop.getScopDomainsBySunid(superfam1.getSunID());
141      ScopDomain dom1 = doms4superfam1.get(0);
142      
143      // align the first domain against all others members of this superfamily
144      for ( int i = 1 ; i < doms4superfam1.size() ; i ++){
145
146         ScopDomain dom2 = doms4superfam1.get(i);
147        
148         try {
149            Structure s1 = cache.getStructureForDomain(dom1);
150            Structure s2 = cache.getStructureForDomain(dom2);
151            
152            Atom[] ca1 = StructureTools.getAtomCAArray(s1);
153            Atom[] ca2 = StructureTools.getAtomCAArray(s2);
154            StructureAlignment ce = StructureAlignmentFactory.getAlgorithm(CeMain.algorithmName);
155            AFPChain afpChain = ce.align(ca1, ca2);
156            
157            //System.out.println(afpChain.toCE(ca1, ca2));
158            
159            //StructureAlignmentDisplay.display(afpChain, ca1, ca2);
160            
161            System.out.println(dom1.getScopId() + " vs. " + dom2.getScopId()+ " :" + afpChain.getProbability());
162            double tmScore = AFPChainScorer.getTMScore(afpChain, ca1, ca2);
163            afpChain.setTMScore(tmScore);
164            System.out.println(AfpChainWriter.toScoresList(afpChain));
165            
166         } catch (Exception e){
167            e.printStackTrace();
168         }
169      }
170
171   }
