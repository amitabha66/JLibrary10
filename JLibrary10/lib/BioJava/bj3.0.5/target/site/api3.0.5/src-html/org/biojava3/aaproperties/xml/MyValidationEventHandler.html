001package org.biojava3.aaproperties.xml;
002
003import javax.xml.bind.ValidationEvent;
004import javax.xml.bind.ValidationEventHandler;
005import javax.xml.bind.ValidationEventLocator;
006
007public class MyValidationEventHandler implements ValidationEventHandler{
008        @Override
009        public boolean handleEvent(ValidationEvent ve) {            
010                if (ve.getSeverity() == ValidationEvent.FATAL_ERROR ||  ve.getSeverity() == ValidationEvent.ERROR){
011                        ValidationEventLocator  locator = ve.getLocator();
012                        //print message from valdation event
013                        System.out.println("Message is " + ve.getMessage());
014                        //output line and column number
015                        System.out.println("Column is " + locator.getColumnNumber() + " at line number " + locator.getLineNumber());
016                        return false;
017                }
018                return true;
019        }
020}
