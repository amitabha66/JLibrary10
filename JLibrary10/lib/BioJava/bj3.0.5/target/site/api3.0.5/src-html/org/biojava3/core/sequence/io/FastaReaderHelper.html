022package org.biojava3.core.sequence.io;
023
024import java.io.File;
025import java.io.FileInputStream;
026import java.io.InputStream;
027import java.util.LinkedHashMap;
028import org.biojava3.core.sequence.DNASequence;
029import org.biojava3.core.sequence.ProteinSequence;
030
031import org.biojava3.core.sequence.compound.AminoAcidCompound;
032import org.biojava3.core.sequence.compound.AminoAcidCompoundSet;
033import org.biojava3.core.sequence.compound.DNACompoundSet;
034import org.biojava3.core.sequence.compound.NucleotideCompound;
035
036/**
037 *
038 * @author Scooter Willis <willishf at gmail dot com>
039 */
040public class FastaReaderHelper {
041
042    /**
043     * Selecting lazySequenceLoad=true will parse the FASTA file and figure out the accessionid and offsets and return sequence objects
044     * that can in the future read the sequence from the disk. This allows the loading of large fasta files where you are only interested
045     * in one sequence based on accession id.
046     * @param file
047     * @param lazySequenceLoad
048     * @return
049     * @throws Exception
050     */
051    public static LinkedHashMap<String, DNASequence> readFastaDNASequence(File file, boolean lazySequenceLoad) throws Exception {
052        if (!lazySequenceLoad) {
053            return readFastaDNASequence(file);
054        }
055
056        FastaReader<DNASequence, NucleotideCompound> fastaProxyReader = new FastaReader<DNASequence, NucleotideCompound>(file, new GenericFastaHeaderParser<DNASequence, NucleotideCompound>(), new FileProxyDNASequenceCreator(file, DNACompoundSet.getDNACompoundSet()));
057        return fastaProxyReader.process();
058
059    }
060
061    /**
062     * Read a fasta file containing amino acids with setup that would handle most
063     * cases.
064     *
065     * @param file
066     * @return
067     * @throws Exception
068     */
069    public static LinkedHashMap<String, ProteinSequence> readFastaProteinSequence(
070            File file) throws Exception {
071        FileInputStream inStream = new FileInputStream(file);
072        LinkedHashMap<String, ProteinSequence> proteinSequences = readFastaProteinSequence(inStream);
073        inStream.close();
074        return proteinSequences;
075    }
076
077    /**
078     * Read a fasta file containing amino acids with setup that would handle most
079     * cases. User is responsible for closing InputStream because you opened it
080     *
081     * @param inStream
082     * @return
083     * @throws Exception
084     */
085    public static LinkedHashMap<String, ProteinSequence> readFastaProteinSequence(
086            InputStream inStream) throws Exception {
087        FastaReader<ProteinSequence, AminoAcidCompound> fastaReader = new FastaReader<ProteinSequence, AminoAcidCompound>(
088                inStream,
089                new GenericFastaHeaderParser<ProteinSequence, AminoAcidCompound>(),
090                new ProteinSequenceCreator(AminoAcidCompoundSet.getAminoAcidCompoundSet()));
091        return fastaReader.process();
092    }
093
094    /**
095     * Read a fasta DNA sequence
096     * @param inStream
097     * @return
098     * @throws Exception
099     */
100    public static LinkedHashMap<String, DNASequence> readFastaDNASequence(
101            InputStream inStream) throws Exception {
102        FastaReader<DNASequence, NucleotideCompound> fastaReader = new FastaReader<DNASequence, NucleotideCompound>(
103                inStream,
104                new GenericFastaHeaderParser<DNASequence, NucleotideCompound>(),
105                new DNASequenceCreator(DNACompoundSet.getDNACompoundSet()));
106        return fastaReader.process();
107    }
108
109    /**
110     *
111     * @param file
112     * @return
113     * @throws Exception
114     */
115    public static LinkedHashMap<String, DNASequence> readFastaDNASequence(
116            File file) throws Exception {
117        FileInputStream inStream = new FileInputStream(file);
118        LinkedHashMap<String, DNASequence> dnaSequences = readFastaDNASequence(inStream);
119        inStream.close();
120        return dnaSequences;
121    }
122
123    public static void main(String args[]) throws Exception {
124
125        LinkedHashMap<String, DNASequence> dnaSequences = FastaReaderHelper.readFastaDNASequence(new File("fasta.fna"));
126        for (DNASequence sequence : dnaSequences.values()) {
127            sequence.getRNASequence().getProteinSequence().getSequenceAsString();
128        }
129    }
130}
