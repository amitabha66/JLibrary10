020package org.biojava.bio.structure.secstruc;
021
022public class SecStrucState{
023        public boolean isBend() {
024                return bend;
025        }
026
027        public void setBend(boolean bend) {
028                this.bend = bend;
029        }
030
031        double phi;
032        double psi;
033        double omega;
034
035        float kappa;
036        
037        HBond accept1;
038        HBond accept2;
039        HBond donor1;
040        HBond donor2;
041
042        boolean[] turn;
043        boolean bend;
044        
045        SecStrucType secStruc;
046        SecStrucType threeState;
047
048        public SecStrucState(){
049                phi = 360;
050                psi = 360 ;
051                omega = 360;
052                accept1 = new HBond();
053                accept2 = new HBond();
054                donor1  = new HBond();
055                donor2  = new HBond();
056                turn = new boolean[3];
057                turn[0] = false;
058                turn[1] = false;
059                turn[2] = false;
060                bend = false;
061                secStruc = SecStrucType.coil;
062                threeState = SecStrucType.coil;
063                kappa = 360;
064        }
065
066        public float getKappa() {
067                return kappa;
068        }
069
070        public void setKappa(float kappa) {
071                this.kappa = kappa;
072        }
073
074        public String toString(){
075                StringBuffer buf = new StringBuffer();
076                buf.append(secStruc.toString()+threeState.toString() + " a1:"+accept1 + " a2:" + accept2 + " d1:" + donor1 + " d2:" + donor2);
077                return buf.toString();
078        }
079
080        public boolean[] getTurn() {
081                return turn;
082        }
083
084        public void setTurn(boolean[] turn) {
085                this.turn = turn;
086        }
087
088        public HBond getAccept1() {
089                return accept1;
090        }
091        public void setAccept1(HBond accept1) {
092                this.accept1 = accept1;
093        }
094        public HBond getAccept2() {
095                return accept2;
096        }
097        public void setAccept2(HBond accept2) {
098                this.accept2 = accept2;
099        }
100        public HBond getDonor1() {
101                return donor1;
102        }
103        public void setDonor1(HBond donor1) {
104                this.donor1 = donor1;
105        }
106        public HBond getDonor2() {
107                return donor2;
108        }
109        public void setDonor2(HBond donor2) {
110                this.donor2 = donor2;
111        }
112        public double getPhi() {
113                return phi;
114        }
115        public void setPhi(double phi) {
116                this.phi = phi;
117        }
118        public double getPsi() {
119                return psi;
120        }
121        public void setPsi(double psi) {
122                this.psi = psi;
123        }
124
125
126        public SecStrucType getSecStruc() {
127                return secStruc;
128        }
129
130        public void setSecStruc(SecStrucType secStruc) {
131                this.secStruc = secStruc;
132        }
133
134        public SecStrucType getThreeState() {
135                return threeState;
136        }
137
138        public void setThreeState(SecStrucType threeState) {
139                this.threeState = threeState;
140        }
141
142
143        public double getOmega() {
144                return omega;
145        }
146
147        public void setOmega(double omega) {
148                this.omega = omega;
149        }
150
151
152
153
154}
