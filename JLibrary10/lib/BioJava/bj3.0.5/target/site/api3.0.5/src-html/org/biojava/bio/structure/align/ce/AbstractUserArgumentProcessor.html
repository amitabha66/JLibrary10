080        public void process(String[] argv){
081
082                printAboutMe();
083
084                List<String> mandatoryArgs = getMandatoryArgs();
085
086                for (int i = 0 ; i < argv.length; i++){
087                        String arg   = argv[i];
088
089                        String value = null;
090                        if ( i < argv.length -1)
091                                value = argv[i+1];
092
093                        // if value starts with - then the arg does not have a value.
094                        if (value != null && value.startsWith("-"))
095                                value = null;
096                        else 
097                                i++;
098
099
100                        String[] tmp = {arg,value};
101
102                        //System.out.println(arg + " " + value);
103
104                        try {
105
106                                CliTools.configureBean(params, tmp);  
107
108                        } catch (ConfigurationException e){
109
110                                e.printStackTrace();
111
112                                if ( mandatoryArgs.contains(arg) ) {
113                                        // there must not be a ConfigurationException with mandatory arguments.
114                                        return;
115                                } else {
116                                        // but there can be with optional ...
117                                }
118                        }           
119                }
120
121                if ( params.getPdbFilePath() != null){
122                        System.setProperty(PDB_DIR,params.getPdbFilePath());
123                }
124
125                if ( params.isShowMenu()){
126                        System.err.println("showing menu...");
127                        try {
128                                GuiWrapper.showAlignmentGUI();
129                        } catch (Exception e){
130                                System.err.println(e.getMessage());
131                                e.printStackTrace();
132                        }
133                }
134
135                if ( params.getShowDBresult() != null){
136                        // user wants to view DB search results:
137
138
139                        System.err.println("showing DB results...");
140                        try {
141                                GuiWrapper.showDBResults(params);
142                        } catch (Exception e){
143                                System.err.println(e.getMessage());
144                                e.printStackTrace();
145                        }
146
147                }
148
149                String pdb1  = params.getPdb1();
150                String file1 = params.getFile1();
151
152
153                if (pdb1 != null || file1 != null){
154                        runPairwise();
155                }
156
157                if ( params.getAlignPairs() != null){
158                        runDBSearch();
159                }
160        }

182        private void runDBSearch(){
183
184
185                String pdbFilePath = params.getPdbFilePath();
186
187                if ( pdbFilePath == null || pdbFilePath.equals("")){
188
189                        System.err.println("You did not specify the -pdbFilePath. Can not find PDB files in file system and will assume a temporary location.");
190                        UserConfiguration c = new UserConfiguration();
191                        pdbFilePath = c.getPdbFilePath();
192                }
193
194
195                AtomCache cache = new AtomCache(pdbFilePath, params.isPdbDirSplit());
196
197                String inputFile = params.getAlignPairs();
198
199                if ( inputFile == null || inputFile.equals("")){
200                        System.err.println("Please specify the mandatory argument -inputFile!");
201                        return;
202                }
203
204                String outputFile = params.getOutFile();
205
206                if ( outputFile == null || outputFile.equals("")){
207                        System.err.println("Please specify the mandatory argument -outFile!");
208                        return;
209                }
210
211                System.out.println("running DB search with parameters:" + params);
212
213
214                try {
215                        File f = new File(inputFile);
216
217                        BufferedReader is = new BufferedReader (new InputStreamReader(new FileInputStream(f)));
218
219                        BufferedWriter out = new BufferedWriter(new FileWriter(outputFile, true));
220
221                        StructureAlignment algorithm =  getAlgorithm();
222
223                        String header = "# algorithm:" + algorithm.getAlgorithmName(); 
224                        out.write(header);
225                        out.write(newline);
226
227                        out.write("#Legend: " + newline );
228                        String legend = getDbSearchLegend();
229                        out.write(legend + newline );
230                        System.out.println(legend);
231                        String line = null;
232                        while ( (line = is.readLine()) != null){
233                                if ( line.startsWith("#"))
234                                        continue;
235
236                                String[] spl = line.split(" ");
237
238                                if ( spl.length != 2) {
239                                        System.err.println("wrongly formattted line. Expected format: 4hhb.A 4hhb.B but found " + line);
240                                        continue;
241                                } 
242
243                                String pdb1 = spl[0];
244                                String pdb2 = spl[1];
245
246
247                                Structure structure1 = cache.getStructure(pdb1);
248                                Structure structure2 = cache.getStructure(pdb2);
249
250                                Atom[] ca1;
251                                Atom[] ca2;
252
253
254                                ca1 = StructureTools.getAtomCAArray(structure1);
255                                ca2 = StructureTools.getAtomCAArray(structure2);
256
257                                Object jparams = getParameters();
258
259                                AFPChain afpChain;
260
261                                afpChain = algorithm.align(ca1, ca2, jparams);
262                                afpChain.setName1(pdb1);
263                                afpChain.setName2(pdb2);
264
265                                String result = getDbSearchResult(afpChain);
266                                out.write(result);
267                                System.out.print(result);
268
269                                checkWriteFile(afpChain,ca1,ca2,true);
270                        }
271
272                        out.close();
273                        is.close();
274                } catch(Exception e){
275                        e.printStackTrace();
276                }
277        }

280        private void runPairwise(){
281
282                String name1 = params.getPdb1();
283                String file1 = params.getFile1();
284
285                if ( name1 == null && file1 == null){
286                        throw new RuntimeException("You did not specify the -pdb1 or -file1 parameter. Can not find query PDB id for alignment.");
287                }
288
289                if ( file1 == null) {
290                        if ( name1.length() < 4) {
291                                throw new RuntimeException("-pdb1 does not look like a PDB ID. Please specify PDB code or PDB.chainId.");
292                        }
293                }
294
295
296
297                String name2 = params.getPdb2();
298                String file2 = params.getFile2();
299                if ( name2 == null && file2 == null ){
300                        throw new RuntimeException("You did not specify the -pdb2 or -file2 parameter. Can not find target PDB id for alignment.");
301                }
302
303                if ( file2 == null ){
304                        if ( name2.length() < 4) {
305                                throw new RuntimeException("-pdb2 does not look like a PDB ID. Please specify PDB code or PDB.chainId.");
306                        }
307                }
308
309                // first load two example structures
310
311                Structure structure1 = null;
312                Structure structure2 = null;
313
314                String path = params.getPdbFilePath();
315
316                if ( file1 == null || file2 == null) {
317                        if ( path == null){
318                                System.err.println("You did not specify the -pdbFilePath parameter. Can not find the PDB files in your file system and assuming a temporary location.");
319                                UserConfiguration c = new UserConfiguration();
320                                path = c.getPdbFilePath();
321                        }
322
323                        AtomCache cache = new AtomCache(path, params.isPdbDirSplit());
324                        cache.setAutoFetch(params.isAutoFetch());   
325                        structure1 = getStructure(cache, name1, file1);
326                        structure2 = getStructure(cache, name2, file2);
327                } else {
328
329                        structure1 = getStructure(null, name1, file1);
330                        structure2 = getStructure(null, name2, file2);                  
331                }      
332
333
334
335                if ( structure1 == null){
336                        System.err.println("structure 1 is null, can't run alignment.");
337                        return;
338                }
339
340                if ( structure2 == null){
341                        System.err.println("structure 2 is null, can't run alignment.");
342                        return;
343                }
344
345                if ( name1 == null) {
346                        name1 = structure1.getName();
347                }
348                if ( name2 == null) {
349                        name2 = structure2.getName();
350                }
351
352                //                   default:      new:
353                // 1buz - 1ali : time: 8.3s eqr 68 rmsd 3.1 score 161 | time 6.4 eqr 58 rmsd 3.0 scre 168
354                // 5pti - 1tap : time: 6.2s eqr 48 rmsd 2.67 score 164 | time 5.2 eqr 49 rmsd 2.9 score 151
355                // 1cdg - 8tim
356                // 1jbe - 1ord
357                // 1nbw.A - 1kid
358                // 1t4y - 1rp5
359
360
361                try {
362
363                        Atom[] ca1;
364                        Atom[] ca2;
365
366                        ca1 = StructureTools.getAtomCAArray(structure1);
367                        ca2 = StructureTools.getAtomCAArray(structure2);
368
369                        StructureAlignment algorithm =  getAlgorithm();
370                        Object jparams = getParameters();
371
372                        AFPChain afpChain;
373
374                        afpChain = algorithm.align(ca1, ca2, jparams);
375                        afpChain.setName1(name1);
376                        afpChain.setName2(name2);
377
378                        if ( params.isShow3d()){
379
380                                if (! GuiWrapper.isGuiModuleInstalled()) {
381                                        System.err.println("The biojava-structure-gui module is not installed. Please install!");
382                                } else {
383
384                                        try {
385
386                                                Object jmol = GuiWrapper.display(afpChain,ca1,ca2);
387
388                                                GuiWrapper.showAlignmentImage(afpChain, ca1,ca2,jmol);
389
390                                        } catch (Exception e){
391
392                                                System.err.println(e.getMessage());
393                                                e.printStackTrace();
394                                        }
395                                        //StructureAlignmentJmol jmol = algorithm.display(afpChain,ca1,ca2,hetatms1, nucs1, hetatms2, nucs2);
396
397                                        //String result = afpChain.toFatcat(ca1, ca2);
398                                        //String rot = afpChain.toRotMat();
399
400                                        //DisplayAFP.showAlignmentImage(afpChain, ca1,ca2,jmol);
401                                }
402                        } 
403
404
405                        checkWriteFile(afpChain,ca1, ca2, false);
406
407
408
409
410                        if ( params.isPrintXML()){
411                                String fatcatXML = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
412                                System.out.println(fatcatXML);
413                        }
414                        if ( params.isPrintFatCat()) {
415                                // default output is to XML on sysout...
416                                System.out.println(afpChain.toFatcat(ca1, ca2));
417                        }
418                        if ( params. isPrintCE()){
419                                System.out.println(afpChain.toCE(ca1, ca2));
420                        }
421
422
423
424                } catch (Exception e) {
425                        e.printStackTrace();
426                        return;
427                }
428        }

430        /** check if the result should be written to the local file system
431         * 
432         * @param params2
433         * @param afpChain
434         * @param ca1
435         * @param ca2
436         */
437        private void checkWriteFile( AFPChain afpChain, Atom[] ca1, Atom[] ca2, boolean dbsearch)
438                        throws Exception
439                        {
440                String output = null;
441                if ( params.isOutputPDB()){
442                        if (! GuiWrapper.isGuiModuleInstalled()) {
443                                System.err.println("The biojava-structure-gui module is not installed. Please install!");
444                                output = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
445                        } else {
446
447                                Structure tmp = AFPAlignmentDisplay.createArtificalStructure(afpChain, ca1, ca2);
448                                output = "TITLE  " + afpChain.getAlgorithmName() + " " + afpChain.getVersion()  + " ";
449                                output += afpChain.getName1() + " vs. " + afpChain.getName2();
450                                output += newline;
451                                output += tmp.toPDB();
452                        }
453                } else  if ( params.getOutFile() != null) {
454                        // output by default is XML
455                        // write the XML to a file...
456                        output = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
457
458                } else if ( params.getSaveOutputDir() != null){
459                        output = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
460                }
461
462                // no output requested.
463                if ( output == null)
464                        return;
465
466                String fileName = null;
467
468                if ( dbsearch ){
469                        if ( params.getSaveOutputDir() != null) {
470                                fileName = params.getSaveOutputDir(); 
471                                fileName += getAutoFileName(afpChain);
472                        } else {
473                                fileName = getAutoFileName(afpChain);
474                        }
475                } else 
476
477                        if ( params.getOutFile() != null) {
478                                fileName = params.getOutFile();
479                        }
480
481                if (fileName == null) {
482                        System.err.println("Can't write outputfile. Either provide a filename using -outFile or set -autoOutputFile to true .");
483                        return;
484                }
485                //System.out.println("writing results to " + fileName);
486
487                FileOutputStream out; // declare a file output object
488                PrintStream p; // declare a print stream object
489
490                try
491                {
492                        // Create a new file output stream
493                        out = new FileOutputStream(fileName);
494
495                        // Connect print stream to the output stream
496                        p = new PrintStream( out );
497
498                        p.println (output);
499
500                        p.close();
501                }
502                catch (Exception e)
503                {
504                        System.err.println ("Error writing to file " + params.getOutFile());
505                }
506
507
508                        }

511        private String getAutoFileName(AFPChain afpChain){
512                String fileName =afpChain.getName1()+"_" + afpChain.getName2()+"_"+afpChain.getAlgorithmName();
513
514                if (params.isOutputPDB() )
515                        fileName += ".pdb";
516                else
517                        fileName += ".xml";
518                return fileName;
519        }

522        private Structure getStructure(AtomCache cache, String name1, String file) 
523        {
524
525                PDBFileReader reader = new PDBFileReader();
526                if ( file != null ){
527                        try {
528                                // check if it is a URL:
529                                try {
530                                        URL url = new URL(file);
531                                        System.out.println(url);
532
533                                        Structure s = reader.getStructure(url);
534
535                                        return fixStructureName(s,file);
536
537                                } catch ( Exception e){
538                                        System.err.println(e.getMessage());
539                                }
540                                File f= new File(file);
541                                System.out.println("file from local " + f.getAbsolutePath());
542                                Structure s= reader.getStructure(f);
543                                return fixStructureName(s, file);
544                        } catch (Exception e){
545                                System.err.println("general exception:" + e.getMessage());
546                                System.err.println("unable to load structure from " + file);
547                                return null;
548                        }
549                }
550                try {
551                        Structure s = cache.getStructure(name1);
552                        return s;
553                } catch ( Exception e){
554                        System.err.println(e.getMessage());
555                        System.err.println("unable to load structure from dir: " + cache.getPath() + "/"+ name1);
556                        return null;
557                }
558
559        }

561        /** apply a number of rules to fix the name of the structure if it did not get set during loading.
562         * 
563         * @param s
564         * @param file
565         * @return
566         */
567        private Structure fixStructureName(Structure s, String file) {
568
569                if ( s.getName() != null && (! s.getName().equals("")))
570                        return s;
571
572                s.setName(s.getPDBCode());
573
574                if ( s.getName() == null || s.getName().equals("")){
575                        File f = new File(file);
576                        s.setName(f.getName());
577                }
578                return s;
579        }
