129        /** returns an identical copy of this structure .
130         * @return an identical Structure object
131         */
132        @SuppressWarnings("deprecation")
133        public Structure clone() {
134
135                Structure n = new StructureImpl();
136                // go through whole substructure and clone ...
137
138                // copy structure data
139                if (isNmr()) n.setNmr(true);
140
141                n.setPDBCode(getPDBCode());
142                n.setName(getName());
143                n.setHeader(getHeader());
144                //TODO: do deep copying of data!
145                n.setPDBHeader(pdbHeader);
146                n.setDBRefs(this.getDBRefs());
147                n.setConnections(getConnections());
148                n.setSites(getSites());
149                
150                // go through each chain and clone chain
151                for (int i=0;i<nrModels();i++){
152                        List<Chain> cloned_model = new ArrayList<Chain>();
153
154                        for (int j=0;j<size(i);j++){
155
156                                Chain current_chain = (Chain) getChain(i,j);
157                                Chain cloned_chain  = (Chain) current_chain.clone();
158
159                                cloned_model.add(cloned_chain);
160                        }
161                        n.addModel(cloned_model);
162
163                }
164
165                for (SSBond ssbond: ssbonds){
166                        n.addSSBond(ssbond.clone());
167                }
168
169                //n.setSeqRes(this.getSeqRes());
170                return n ;
171        }

174        public Group findGroup(String chainId, String pdbResnum, int modelnr)
175        throws StructureException {
176
177
178                // if structure is xray there will be only one "model".
179                if ( modelnr > models.size())
180                        throw new StructureException(" no model nr " + modelnr +
181                                        " in this structure. (contains "+models.size()+")");
182
183
184                Chain c = findChain(chainId,modelnr);
185
186                List<Group> groups = c.getAtomGroups();
187
188                // now iterate over all groups in this chain.
189                // in order to find the amino acid that has this pdbRenum.
190
191                Iterator<Group> giter = groups.iterator();
192                while (giter.hasNext()){
193                        Group g =  giter.next();
194                        String rnum = g.getResidueNumber().toString();
195                        //System.out.println(g + " >" + rnum + "< >" + pdbResnum + "<");
196                        // we only mutate amino acids
197                        // and ignore hetatoms and nucleotides in this case
198                        if (rnum.equals(pdbResnum))
199                                return g;
200                }
201
202                throw new StructureException("could not find group " + pdbResnum +
203                                " in chain " + chainId);
204        }

216        public Chain findChain(String chainId, int modelnr) throws StructureException {
217
218                List<Chain> chains = getChains(modelnr);
219
220                // iterate over all chains.
221                Iterator<Chain> iter = chains.iterator();
222                while (iter.hasNext()){
223                        Chain c = iter.next();
224
225                        if (c.getChainID().equals(chainId)) {
226                                return c;
227                        }
228                }
229                throw new StructureException("could not find chain " + chainId);
230        }

316         *
317         */
318        public void addChain(Chain chain, int modelnr) {
319                // if model has not been initialized, init it!
320                chain.setParent(this);
321                if ( models.size() == 0  ) {
322                        List<Chain> model = new ArrayList<Chain>() ;
323                        model.add(chain);
324                        models.add(model);
325
326                } else {
327                        List<Chain> model = models.get(modelnr);
328                        model.add(chain);
329                }
330
331
332
333        }

384    public void setModel(int position, List<Chain> model){
385        if (model == null)
386                throw new IllegalArgumentException("trying to set model to null!");
387
388        for (Chain c: model)
389                c.setParent(this);
390
391        //System.out.println("model size:" + models.size());
392
393        if (models.size() ==0){
394                models.add(model);
395        } else {
396                models.set(position, model);
397        }

400        /** string representation.
401         *
402         */
403        public String toString(){
404                String newline = System.getProperty("line.separator");
405                StringBuffer str = new StringBuffer();
406                str.append("structure ");
407                str.append(name);
408                str.append(" ");
409                str.append(pdb_id);
410                str.append(" ");
411
412                if ( isNmr() ){
413                        str.append( " models: ");
414                        str.append(nrModels());
415                        str.append(newline) ;
416                }
417
418        str.append(pdbHeader.toString());
419        str.append(newline) ;
420
421        for (int i=0;i<nrModels();i++){
422                        if (isNmr() ) {
423                                str.append(" model[");
424                                str.append(i);
425                                str.append("]:");
426                                str.append(newline);
427                        }
428                        str.append(" chains:");
429                        str.append(newline);
430
431                        for (int j=0;j<size(i);j++){
432
433                                Chain cha = (Chain)getChain(i,j);
434                                List<Group> agr = cha.getAtomGroups("amino");
435                                List<Group> hgr = cha.getAtomGroups("hetatm");
436                                List<Group> ngr = cha.getAtomGroups("nucleotide");
437
438                                str.append("chain " + j + ": >"+cha.getChainID()+"< ");
439                                if ( cha.getHeader() != null){
440                                        Compound comp = cha.getHeader();
441                                        String molName = comp.getMolName();
442                                        if ( molName != null){
443                                                str.append(molName);
444                                        }
445                                }
446
447
448                                str.append(newline);
449                str.append(" length SEQRES: ").append(cha.getSeqResLength());
450                str.append(" length ATOM: ").append(cha.getAtomLength());
451                str.append(" aminos: ").append(agr.size());
452                str.append(" hetatms: ").append(hgr.size());
453                                str.append(" nucleotides: "+ngr.size() + newline);
454                        }
455
456                }
457        str.append("DBRefs: "+ dbrefs.size()+ newline);
458        for (DBRef dbref: dbrefs){
459            str.append(dbref.toPDB()).append(newline);
460        }
461        str.append("Molecules: ").append(newline);
462                Iterator<Compound> iter = compounds.iterator();
463                while (iter.hasNext()){
464                        Compound mol = iter.next();
465            str.append(mol).append(newline);
466                }
467
468
469                return str.toString() ;
470        }

579        public boolean hasChain(String chainId) {
580                int modelnr = 0;
581
582                List<Chain> chains = getChains(modelnr);
583                Iterator<Chain> iter = chains.iterator();
584                while ( iter.hasNext()){
585                        Chain c = iter.next();
586                        // we check here with equals because we might want to distinguish between upper and lower case chains!
587                        if ( c.getChainID().equals(chainId))
588                                return true;
589                }
590                return false;
591        }
