075 * This class implements the actual PDB file parsing. Do not access it directly, but

077 *
078 * <h2>Parsing</h2>
079 *
080 * During the PDBfile parsing several Flags can be set. See the {@link #setFileParsingParameters(FileParsingParameters)} methods.

082 *
083 * <p>
084 * To provide excessive memory usage for large PDB files, there is the ATOM_CA_THRESHOLD.
085 * If more Atoms than this threshold are being parsed in a PDB file, the parser will automatically
086 * switch to a C-alpha only representation.
087 * </p>
088 *
089 * <p>
090 * The result of the parsing of the PDB file is a new {@link Structure} object.
091 * </p>
092 *
093 *
094 * For more documentation on how to work with the Structure API please

104 * </p>

108 public {@link Structure} loadStructure(String pathToPDBFile){
109            // The PDBFileParser is wrapped by the PDBFileReader
110                {@link PDBFileReader} pdbreader = new {@link PDBFileReader}();
111
112                {@link Structure} structure = null;
113                try{
114                        structure = pdbreader.getStructure(pathToPDBFile);
115                        System.out.println(structure);
116                } catch (IOException e) {
117                        e.printStackTrace();
118                }
119                return structure;
120        }

338        /** initiate new resNum, either Hetatom, Nucleotide, or AminoAcid */
339        private Group getNewGroup(String recordName,Character aminoCode1, String aminoCode3) {
340
341                if ( params.isLoadChemCompInfo() ){
342                        Group g =  ChemCompGroupFactory.getGroupFromChemCompDictionary(aminoCode3);
343                        if ( g != null)
344                                return g;
345                }
346
347
348                Group group;
349                if ( recordName.equals("ATOM") ) {
350                        if (aminoCode1 == null)  {
351                                // it is a nucleotide
352                                NucleotideImpl nu = new NucleotideImpl();
353                                group = nu;
354
355                        } else if (aminoCode1 == StructureTools.UNKNOWN_GROUP_LABEL){
356                                group = new HetatomImpl();
357
358                        } else {
359                                AminoAcidImpl aa = new AminoAcidImpl() ;
360                                aa.setAminoType(aminoCode1);
361                                group = aa ;
362                        }
363                }
364                else {
365                        if (aminoCode1 != null ) {
366                                AminoAcidImpl aa = new AminoAcidImpl() ;
367                                aa.setAminoType(aminoCode1);
368                                group = aa ;
369                        } else {
370                                group = new HetatomImpl();
371                        }
372                }
373                //              System.out.println("new resNum type: "+ resNum.getType() );
374                return  group ;
375        }

380        /**
381         Handler for
382         HEADER Record Format
383
384         COLUMNS        DATA TYPE       FIELD           DEFINITION
385         ----------------------------------------------------------------------------------
386         1 -  6        Record name     "HEADER"
387         11 - 50        String(40)      classification  Classifies the molecule(s)
388         51 - 59        Date            depDate         Deposition date.  This is the date
389         the coordinates were received by
390         the PDB
391         63 - 66        IDcode          idCode          This identifier is unique within PDB
392
393         */
394        private void pdb_HEADER_Handler(String line) {
395                //System.out.println(line);
396
397                String classification  = line.substring (10, 50).trim() ;
398                String deposition_date = line.substring (50, 59).trim() ;
399                String pdbCode         = line.substring (62, 66).trim() ;
400
401                pdbId = pdbCode;
402                if (DEBUG) {
403                        System.out.println("Parsing entry " + pdbId);
404                }
405                //*really* old files (you'll need to hunt to find these as they
406                //should have been remediated) have headers like below. Plus the
407                //pdbId at positions 72-76 is present in every line
408
409                //HEADER    PROTEINASE INHIBITOR (TRYPSIN)          05-OCT-84   5PTI      5PTI   3
410                //HEADER    TRANSFERASE (ACYLTRANSFERASE)           02-SEP-92   1LAC      1LAC   2
411                if (line.trim().length() > 66) {
412                        if (pdbId.equals(line.substring (72, 76))){
413                                isLegacyFormat = true;
414                                System.out.println(pdbId + " is a LEGACY entry - this will most likely not parse correctly.");
415                        }
416                }
417                header.put("idCode",pdbCode);
418                structure.setPDBCode(pdbCode);
419                header.put("classification",classification);
420                header.put("depDate",deposition_date);
421
422                pdbHeader.setIdCode(pdbCode);
423                pdbHeader.setClassification(classification);
424
425
426                try {
427                        Date dep = dateFormat.parse(deposition_date);
428                        pdbHeader.setDepDate(dep);
429                        header.put("depDate",deposition_date);
430
431                } catch (ParseException e){
432                        e.printStackTrace();
433                }
434
435        }
