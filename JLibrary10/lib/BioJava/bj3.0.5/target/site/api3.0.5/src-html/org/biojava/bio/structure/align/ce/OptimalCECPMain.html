103        /**
104         * Circularly permutes arr in place.
105         * 
106         * <p>Similar to {@link Collections#rotate(List, int)} but with reversed
107         * direction. Perhaps it would be more efficient to use the Collections version?
108         * @param <T>
109         * @param arr The array to be permuted
110         * @param cp The number of residues to shift leftward, or equivalently, the index of
111         *  the first element after the permutation point.
112         */
113        private static <T> void permuteArray(T[] arr, int cp) {
114                // Allow negative cp points for convenience.
115                if(cp == 0) {
116                        return;
117                }
118                if(cp < 0) {
119                        cp = arr.length+cp;
120                }
121                if(cp < 0 || cp >= arr.length) {
122                        throw new ArrayIndexOutOfBoundsException(
123                                        "Permutation point ("+cp+") must be between -ca2.length and ca2.length-1" );
124                }
125                
126                List<T> temp = new ArrayList<T>(cp);
127
128                // shift residues left
129                for(int i=0;i<cp;i++) {
130                        temp.add(arr[i]);
131                }
132                for(int j=cp;j<arr.length;j++) {
133                        arr[j-cp]=arr[j];
134                }
135                for(int i=0;i<cp;i++) {
136                        arr[arr.length-cp+i] = temp.get(i);
137                }
138        }

178        /**
179         * Aligns ca1 with ca2 permuted by <i>cp</i> residues.
180         * <p><strong>WARNING:</strong> Modifies ca2 during the permutation. Be sure
181         * to make a copy before calling this method.
182         * 
183         * @param ca1
184         * @param ca2
185         * @param param
186         * @param cp
187         * @return
188         * @throws StructureException 
189         */
190        public AFPChain alignPermuted(Atom[] ca1, Atom[] ca2, Object param, int cp) throws StructureException {
191                // initial permutation
192                permuteArray(ca2,cp);
193                                
194                // perform alignment
195                AFPChain afpChain = super.align(ca1, ca2, param);
196                
197                // un-permute alignment
198                permuteAFPChain(afpChain, -cp);
199                
200                if(afpChain.getName2() != null) {
201                        afpChain.setName2(afpChain.getName2()+" CP="+cp);
202                }
203                
204                // Specify the permuted
205                return afpChain;
206        }

208        /**
209         * Permute the second protein of afpChain by the specified number of residues.
210         * @param afpChain Input alignment
211         * @param cp Amount leftwards (or rightward, if negative) to shift the 
212         * @return A new alignment equivalent to afpChain after the permutations
213         */
214        private static void permuteAFPChain(AFPChain afpChain, int cp) {
215                
216                int ca2len = afpChain.getCa2Length();
217                                
218                //fix up cp to be positive
219                if(cp == 0) {
220                        return;
221                }
222                if(cp < 0) {
223                        cp = ca2len+cp;
224                }
225                if(cp < 0 || cp >= ca2len) {
226                        throw new ArrayIndexOutOfBoundsException(
227                                        "Permutation point ("+cp+") must be between -ca2.length and ca2.length-1" );
228                }
229                
230                // Fix up optAln
231                permuteOptAln(afpChain,cp);
232                
233                if(afpChain.getBlockNum() > 1)
234                        afpChain.setSequentialAlignment(false);
235                // fix up matrices
236                // ca1 corresponds to row indices, while ca2 corresponds to column indices.
237                afpChain.setDistanceMatrix(permuteMatrix(afpChain.getDistanceMatrix(),0,-cp));
238                // this is square, so permute both
239                afpChain.setDisTable2(permuteMatrix(afpChain.getDisTable2(),-cp,-cp));
240
241                //TODO fix up other AFP parameters?
242                
243        }

245        /**
246         * Permutes <i>mat</i> by moving the rows of the matrix upwards by <i>cp</i>
247         * rows.
248         * @param mat The original matrix
249         * @param cpRows Number of rows upward to move entries
250         * @param cpCols Number of columns leftward to move entries
251         * @return The permuted matrix
252         */
253        private static Matrix permuteMatrix(Matrix mat, int cpRows, int cpCols) {
254                //fix up cp to be positive
255                if(cpRows == 0 && cpCols == 0) {
256                        return mat.copy();
257                }
258                if(cpRows < 0) {
259                        cpRows = mat.getRowDimension()+cpRows;
260                }
261                if(cpRows < 0 || cpRows >= mat.getRowDimension()) {
262                        throw new ArrayIndexOutOfBoundsException( String.format(
263                                        "Can't permute rows by %d: only %d rows.", 
264                                        cpRows, mat.getRowDimension() )
265                        );
266                }
267
268                if(cpCols < 0) {
269                        cpCols = mat.getColumnDimension()+cpCols;
270                }
271                if(cpCols < 0 || cpCols >= mat.getColumnDimension()) {
272                        throw new ArrayIndexOutOfBoundsException( String.format(
273                                        "Can't permute cols by %d: only %d rows.", 
274                                        cpCols, mat.getColumnDimension() )
275                        );
276                }
277                
278                int[] rows = new int[mat.getRowDimension()];
279                for(int i=0;i<rows.length;i++) {
280                        rows[i] = (i+cpRows)%rows.length;
281                }
282                int[] cols = new int[mat.getColumnDimension()];
283                for(int i=0;i<cols.length;i++) {
284                        cols[i] = (i+cpCols)%cols.length;
285                }
286                
287                Matrix newMat = mat.getMatrix(rows, cols);
288                assert(newMat.getRowDimension() == mat.getRowDimension());
289                assert(newMat.getColumnDimension() == mat.getColumnDimension());
290                assert(newMat.get(0, 0) ==
291                        mat.get(cpRows%mat.getRowDimension(), cpCols%mat.getColumnDimension()));
292                
293                
294                return newMat;
295        }

297        /**
298         * Modifies the {@link AFPChain#setOptAln(int[][][]) optAln} of an AFPChain
299         * by permuting the second protein.
300         * 
301         * Sets residue numbers in the second protein to <i>(i-cp)%len</i>
302         * 
303         * @param afpChain
304         * @param cp Amount leftwards (or rightward, if negative) to shift the
305         */
306        private static void permuteOptAln(AFPChain afpChain, int cp)
307        {
308                int ca2len = afpChain.getCa2Length();
309
310                if( ca2len <= 0) {
311                        throw new IllegalArgumentException("No Ca2Length specified in "+afpChain);
312                }
313
314                // Allow negative cp points for convenience.
315                if(cp == 0) {
316                        return;
317                }
318                if(cp <= -ca2len || cp >= ca2len) {
319                        // could just take cp%ca2len, but probably its a bug if abs(cp)>=ca2len
320                        throw new ArrayIndexOutOfBoundsException( String.format(
321                                        "Permutation point %d must be between %d and %d for %s",
322                                        cp, 1-ca2len,ca2len-1, afpChain.getName2() ) );
323                }
324                if(cp < 0) {
325                        cp = cp + ca2len;
326                }
327
328                // the unprocessed alignment
329                int[][][] optAln = afpChain.getOptAln();
330                int[] optLen = afpChain.getOptLen();
331
332                // the processed alignment
333                List<List<List<Integer>>> blocks = new ArrayList<List<List<Integer>>>(afpChain.getBlockNum()*2);
334                
335                //Update residue indices
336                // newi = (oldi-cp) % N
337                for(int block = 0; block < afpChain.getBlockNum(); block++) {
338                        if(optLen[block]<1)
339                                continue;
340                        
341                        // set up storage for the current block
342                        List<List<Integer>> currBlock = new ArrayList<List<Integer>>(2);
343                        currBlock.add( new ArrayList<Integer>());
344                        currBlock.add( new ArrayList<Integer>());
345                        blocks.add(currBlock);
346
347                        // pos = 0 case
348                        currBlock.get(0).add( optAln[block][0][0] );
349                        currBlock.get(1).add( (optAln[block][1][0]+cp ) % ca2len);
350                        
351                        for(int pos = 1; pos < optLen[block]; pos++) {
352                                //check if we need to start a new block
353                                //this happens when the new alignment crosses the protein terminus
354                                if( optAln[block][1][pos-1]+cp<ca2len && 
355                                                optAln[block][1][pos]+cp >= ca2len) {
356                                        currBlock = new ArrayList<List<Integer>>(2);
357                                        currBlock.add( new ArrayList<Integer>());
358                                        currBlock.add( new ArrayList<Integer>());
359                                        blocks.add(currBlock);
360                                }
361                                currBlock.get(0).add( optAln[block][0][pos] );
362                                currBlock.get(1).add( (optAln[block][1][pos]+cp ) % ca2len);
363                        }
364                }
365                
366                // save permuted blocks to afpChain
367                assignOptAln(afpChain,blocks);
368        }

370        /**
371         * Sometimes it's convenient to store an alignment using java collections,
372         * where <tt>blocks.get(blockNum).get(0).get(pos)</tt> specifies the aligned
373         * residue at position <i>pos</i> of block <i>blockNum</i> of the first
374         * protein.
375         * 
376         * This method takes such a collection and stores it into the afpChain's
377         * {@link AFPChain#setOptAln(int[][][]) optAln}, setting the associated
378         * length variables as well.
379         * 
380         * @param afpChain
381         * @param blocks
382         */
383        private static void assignOptAln(AFPChain afpChain, List<List<List<Integer>>> blocks)
384        {
385                
386                int[][][] optAln = new int[blocks.size()][][];
387                int[] optLen = new int[blocks.size()];
388                int optLength = 0;
389                int numBlocks = blocks.size();
390                
391                for(int block = 0; block < numBlocks; block++) {
392                        // block should be 2xN rectangular
393                        assert(blocks.get(block).size() == 2);
394                        assert( blocks.get(block).get(0).size() == blocks.get(block).get(1).size());
395
396                        optLen[block] = blocks.get(block).get(0).size();
397                        optLength+=optLen[block];
398                        
399                        optAln[block] = new int[][] {
400                                        new int[optLen[block]],
401                                        new int[optLen[block]]
402                        };
403                        for(int pos = 0; pos < optLen[block]; pos++) {
404                                optAln[block][0][pos] = blocks.get(block).get(0).get(pos);
405                                optAln[block][1][pos] = blocks.get(block).get(1).get(pos);
406                        }
407                }
408                
409                
410                afpChain.setBlockNum(numBlocks);
411                afpChain.setOptAln(optAln);
412                afpChain.setOptLen(optLen);
413                afpChain.setOptLength(optLength);
414                
415                // TODO I don't know what these do. Should they be set?
416                //afpChain.setBlockSize(blockSize);
417                //afpChain.setBlockResList(blockResList);
418                //afpChain.setChainLen(chainLen);
419                
420        }

422        /**
423         * Finds the optimal alignment between two proteins allowing for a circular
424         * permutation (CP).
425         * 
426         * The precise algorithm is controlled by the 
427         * {@link OptimalCECPParameters parameters}. If the parameter
428         * {@link OptimalCECPParameters#isTryAllCPs() tryAllCPs} is true, all possible
429         * CP sites are tried and the optimal site is returned. Otherwise, the
430         * {@link OptimalCECPParameters#getCPPoint() cpPoint} parameter is used to
431         * determine the CP point, greatly reducing the computation required.
432         * 
433         * @param ca1 CA atoms of the first protein
434         * @param ca2 CA atoms of the second protein
435         * @param param {@link CeParameters} object
436         * @return The best-scoring alignment
437         * @throws StructureException
438         * 
439         * @see #alignOptimal(Atom[], Atom[], Object, AFPChain[])
440         */
441        @Override
442        public AFPChain align(Atom[] ca1, Atom[] ca2, Object param)
443        throws StructureException
444        {
445                if(params.isTryAllCPs()) {
446                        return alignOptimal(ca1,ca2,param,null);
447                } else {
448                        int cpPoint = params.getCPPoint();
449                        return alignPermuted(ca1, ca2, param, cpPoint);
450                }
451        }

453        /**
454         * Finds the optimal alignment between two proteins allowing for a circular
455         * permutation (CP).
456         * 
457         * This algorithm performs a CE alignment for each possible CP site. This is
458         * quite slow. Use {@link #alignHeuristic(Atom[], Atom[], Object)} for a
459         * faster algorithm.
460         * 
461         * @param ca1 CA atoms of the first protein
462         * @param ca2 CA atoms of the second protein
463         * @param param {@link CeParameters} object
464         * @param alignments If not null, should be an empty array of the same length as
465         *  ca2. This will be filled with the alignments from permuting ca2 by 
466         *  0 to n-1 residues.
467         * @return The best-scoring alignment
468         * @throws StructureException
469         */
470        public AFPChain alignOptimal(Atom[] ca1, Atom[] ca2, Object param, AFPChain[] alignments)
471        throws StructureException
472        {
473                long startTime = System.currentTimeMillis();
474                
475                if(alignments.length != ca2.length) {
476                        throw new IllegalArgumentException("scores param should have same length as ca2");
477                }
478                
479                AFPChain unaligned = super.align(ca1, ca2, param);
480                AFPChain bestAlignment = unaligned;
481                
482                if(debug) {
483                        // print progress bar header
484                        System.out.print("|");
485                        for(int cp=1;cp<ca2.length-1;cp++) {
486                                System.out.print("=");
487                        }
488                        System.out.println("|");
489                        System.out.print(".");
490                }
491
492                if(alignments != null) {
493                        alignments[0] = unaligned;
494                }
495                
496                for(int cp=1;cp<ca2.length;cp++) {
497                        // clone ca2 to prevent side effects from propegating
498                        Atom[] ca2p = StructureTools.cloneCAArray(ca2);
499
500                        //permute one each time. Alters ca2p as a side effect
501                        AFPChain currentAlignment = alignPermuted(ca1,ca2p,param,cp);
502                        
503                        // increment progress bar
504                        if(debug) System.out.print(".");
505                        
506                        // fix up names, since cloning ca2 wipes it
507                        try {
508                                currentAlignment.setName2(ca2[0].getGroup().getChain().getParent().getName()+" CP="+cp);
509                        } catch( Exception e) {
510                                //null pointers, empty arrays, etc.
511                        }
512                        double currentScore = currentAlignment.getAlignScore();
513                        
514                        if(alignments != null) {
515                                alignments[cp] = currentAlignment;
516                        }
517                        
518                        if(currentScore>bestAlignment.getAlignScore()) {
519                                bestAlignment = currentAlignment;
520                        }
521                }
522                if(debug) {
523                        long elapsedTime = System.currentTimeMillis()-startTime;
524                        System.out.println();
525                        System.out.format("%d alignments took %.4f s (%.1f ms avg)\n",
526                                        ca2.length, elapsedTime/1000., (double)elapsedTime/ca2.length);
527                }
528
529                
530                return bestAlignment;
531                
532        }
