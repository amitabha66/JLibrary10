063/** A utility class that provides easy access to Structure objects. If you are running a
064 *  script that is frequently re-using the same PDB structures, the AtomCache keeps an 
065 *  in-memory cache of the files for quicker access. The cache is a soft-cache, this 
066 *  means it won't cause out of memory exceptions, but garbage collects the data if the 
067 *  Java virtual machine needs to free up space. The AtomCache is thread-safe.
068 * 
069 * @author Andreas Prlic
070 * @author Spencer Bliven
071 * @author Peter Rose
072 * @since 3.0
073 */

116        /** Creates an instance of an AtomCache that is pointed to the a particular
117         * path in the file system.
118         * 
119         * @param pdbFilePath a directory in the file system to use as a location to cache files.
120         * @param isSplit a flag to indicate if the directory organisation is "split" as on the PDB ftp servers, or if all files are contained in one directory.
121         */
122        public AtomCache(String pdbFilePath, boolean isSplit){
123
124
125                if ( ! pdbFilePath.endsWith(FILE_SEPARATOR))
126                        pdbFilePath += FILE_SEPARATOR;
127
128                // we are caching the binary files that contain the PDBs gzipped
129                // that is the most memory efficient way of caching...
130                // set the input stream provider to caching mode
131                System.setProperty(InputStreamProvider.CACHE_PROPERTY, "true");
132
133                path = pdbFilePath;
134
135                System.setProperty(AbstractUserArgumentProcessor.PDB_DIR,path);
136                //this.cache = cache;
137                this.isSplit = isSplit;
138
139                autoFetch = true;
140                fetchFileEvenIfObsolete = false;
141                fetchCurrent = false;
142
143                currentlyLoading.clear();
144                params = new FileParsingParameters();
145
146                // we don't need this here
147                params.setAlignSeqRes(false);
148                // no secstruc either 
149                params.setParseSecStruc(false);
150                // 
151
152
153                this.strictSCOP = true;
154
155                scopInstallation = null;
156        }

284        /** Returns the representation of a ScopDomain as a BioJava Structure object
285         * 
286         * @param domain a scop domain
287         * @return a Structure object.
288         * @throws IOException
289         * @throws StructureException
290         */
291
292        public Structure getStructureForDomain(ScopDomain domain) throws IOException, StructureException{
293                if ( scopInstallation == null) {
294                        scopInstallation = ScopFactory.getSCOP();
295                }
296
297                Structure s = null;
298
299                String pdbId = domain.getPdbId();
300
301                try {
302                        s = getStructure(pdbId);
303
304                } catch (StructureException ex){
305                        System.err.println("error getting Structure for " + pdbId);
306
307                        throw new StructureException(ex);
308                }
309
310
311                StringWriter range = new StringWriter();
312                range.append("(");
313                int rangePos = 0;
314                for ( String r : domain.getRanges()) {
315                        rangePos++;
316                        range.append(r);
317                        if ( ( domain.getRanges().size()> 1) && (rangePos < domain.getRanges().size())){
318                                range.append(",");
319                        }
320
321                }
322                range.append(")");
323                //System.out.println("getting range for "+ pdbId + " " + range);
324
325                Structure n = StructureTools.getSubRanges(s, range.toString());
326
327
328                // get ligands of chain
329                // add the ligands of the chain...
330                StructureName structureName = new StructureName(domain.getScopId());
331                
332                // this does not work for multi-chain domains as of yet...
333                // ignore multi-chain domains
334                if (! structureName.getChainId().equals(".")){
335                        Chain newChain  = n.getChainByPDB(structureName.getChainId());
336                        Chain origChain = s.getChainByPDB(structureName.getChainId());
337                        List<Group> ligands = origChain.getAtomLigands();
338
339                        for(Group g: ligands){
340                                if ( ! newChain.getAtomGroups().contains(g)) {
341                                        newChain.addGroup(g);
342                                }
343                        }
344                }
345
346                n.setName(domain.getScopId());
347                n.setPDBCode(domain.getScopId());
348
349
350                StringWriter d = new StringWriter();
351                d.append(domain.getClassificationId());
352
353
354
355                if ( scopInstallation != null) {
356                        int sf = domain.getSuperfamilyId();
357
358                        ScopDescription scopDesc = scopInstallation.getScopDescriptionBySunid(sf);
359
360                        if( scopDesc != null){
361                                d.append( " | " );
362                                d.append(scopDesc.getDescription());
363
364                        }
365                }
366                n.getPDBHeader().setDescription(d.toString());
367                return n;
368        }

425        /** Request a Structure based on a <i>name</i>.
426         * 
427         * <pre>
428                Formal specification for how to specify the <i>name</i>:
429
430                name     := pdbID
431                               | pdbID '.' chainID
432                               | pdbID '.' range
433                               | scopID
434                range         := '('? range (',' range)? ')'?
435                               | chainID
436                               | chainID '_' resNum '-' resNum
437                pdbID         := [0-9][a-zA-Z0-9]{3}
438                chainID       := [a-zA-Z0-9]
439                scopID        := 'd' pdbID [a-z_][0-9_]
440                resNum        := [-+]?[0-9]+[A-Za-z]?
441
442
443                Example structures:
444                1TIM     #whole structure
445                4HHB.C     #single chain
446                4GCR.A_1-83     #one domain, by residue number
447                3AA0.A,B     #two chains treated as one structure
448                d2bq6a1     #scop domain
449                </pre>
450         * 
451         * With the additional set of rules:
452         *  
453         *  <ul>
454         *  <li>If only a PDB code is provided, the whole structure will be return including ligands, but the first model only (for NMR).
455         *      <li>Chain IDs are case sensitive, PDB ids are not. To specify a particular chain write as: 4hhb.A or 4HHB.A </li>
456         *  <li>To specify a SCOP domain write a scopId e.g. d2bq6a1. Some flexibility can be allowed in SCOP domain names, see {@link #setStrictSCOP(boolean)}</li>
457         *  <li>URLs are accepted as well</li>
458         *  </ul>
459         *  
460         * @param name
461         * @return a Structure object, or null if name appears improperly formated (eg too short, etc)
462         * @throws IOException The PDB file cannot be cached due to IO errors
463         * @throws StructureException The name appeared valid but did not correspond to a structure.
464         *      Also thrown by some submethods upon errors, eg for poorly formatted subranges.
465         */
466        public Structure getStructure(String name) throws IOException, StructureException{
467
468                if ( name.length() < 4)
469                        throw new IllegalArgumentException("Can't interpret IDs that are shorter than 4 residues!");
470
471                Structure n = null;
472
473                boolean useChainNr = false;
474                boolean useDomainInfo = false;
475                String range = null;
476                int chainNr = -1;
477
478                try {
479
480                        StructureName structureName = new StructureName(name);
481
482                        String pdbId   = null;
483                        String chainId = null;
484
485                        if ( name.length() == 4){
486
487                                pdbId = name; 
488
489                        } else if ( structureName.isScopName()){
490
491                                // return based on SCOP domain ID
492                                return getStructureFromSCOPDomain(name);
493                        } else if ( structureName.isCathID()) {
494                                return getStructureFromCATHDomain(structureName);
495                        } else if (name.length() == 6){
496                                // name is PDB.CHAINID style (e.g. 4hhb.A)
497
498                                pdbId = name.substring(0,4);
499                                if ( name.substring(4,5).equals(CHAIN_SPLIT_SYMBOL)) {
500                                        chainId = name.substring(5,6);
501                                } else if ( name.substring(4,5).equals(CHAIN_NR_SYMBOL)) {
502
503                                        useChainNr = true;      
504                                        chainNr = Integer.parseInt(name.substring(5,6));
505                                }
506
507
508
509                        }  else if ( name.startsWith("file:/") || name.startsWith("http:/") ) {
510
511                                // this is a URL
512                                try {
513
514                                        URL url = new URL(name);
515
516                                        return getStructureFromURL(url);
517
518                                } catch (Exception e){
519                                        e.printStackTrace();
520                                        return null;
521                                }
522
523
524                        } else if ( structureName.isPDPDomain()){
525
526                                // this is a PDP domain definition
527
528                                try {
529                                        return getPDPStructure(name);
530                                } catch (Exception e){
531                                        e.printStackTrace();
532                                        return null;
533                                }
534                        } else if ( name.startsWith(BIOL_ASSEMBLY_IDENTIFIER)){
535
536                                try {
537
538                                        return getBioAssembly(name);
539
540                                } catch (Exception e){
541                                        e.printStackTrace();
542                                        return null;
543                                }
544                        } else if ( (name.length() > 6) 
545                                        &&  ( ! name.startsWith(PDP_DOMAIN_IDENTIFIER) ) 
546                                        && (name.contains(CHAIN_NR_SYMBOL) || name.contains(UNDERSCORE)) 
547                                        && ( ! (name.startsWith("file:/") || name.startsWith("http:/")))
548
549                                        ) {
550
551                                // this is a name + range 
552
553                                pdbId = name.substring(0,4);
554                                // this ID has domain split information...
555                                useDomainInfo = true;
556                                range = name.substring(5);
557
558                        }
559
560                        //System.out.println("got: >" + name + "< " + pdbId + " " + chainId + " useChainNr:" + useChainNr + " " +chainNr + " useDomainInfo:" + useDomainInfo + " " + range);
561
562                        if (pdbId == null) {
563
564                                return null;
565                        }
566
567                        while ( checkLoading(pdbId) ){
568                                // waiting for loading to be finished...
569
570                                try {
571                                        Thread.sleep(100);
572                                } catch (InterruptedException e){
573                                        System.err.println(e.getMessage());
574                                }
575
576                        }
577
578
579                        //long start  = System.currentTimeMillis();
580
581                        Structure s;
582                        flagLoading(pdbId);
583                        try {
584                                PDBFileReader reader = new PDBFileReader();
585                                reader.setPath(path);
586                                reader.setPdbDirectorySplit(isSplit);
587                                reader.setAutoFetch(autoFetch);
588                                reader.setFetchFileEvenIfObsolete(fetchFileEvenIfObsolete);
589                                reader.setFetchCurrent(fetchCurrent);
590
591                                reader.setFileParsingParameters(params);
592
593                                s = reader.getStructureById(pdbId.toLowerCase());
594
595                        } catch (Exception e){
596                                flagLoadingFinished(pdbId);
597                                throw new StructureException(e.getMessage() + " while parsing " + pdbId,e);
598                        }
599                        flagLoadingFinished(pdbId);
600
601                        //long end  = System.currentTimeMillis();
602                        //System.out.println("time to load " + pdbId + " " + (end-start) + "\t  size :" + StructureTools.getNrAtoms(s) + "\t cached: " + cache.size());
603                        if ( chainId == null && chainNr < 0 && range == null) {                                                              
604                                // we only want the 1st model in this case
605                                n = StructureTools.getReducedStructure(s,-1);
606
607                        }
608                        else {
609
610                                if ( useChainNr) {
611                                        //System.out.println("using ChainNr");
612                                        n = StructureTools.getReducedStructure(s, chainNr);
613                                } else if ( useDomainInfo) {
614                                        //System.out.println("calling getSubRanges");
615                                        n = StructureTools.getSubRanges(s, range);
616                                } else  {
617                                        //System.out.println("reducing Chain Id " + chainId);
618                                        n = StructureTools.getReducedStructure(s, chainId);
619                                }
620                        }
621
622
623                } catch (Exception e){
624                        System.err.println("problem loading:" + name);
625                        e.printStackTrace();
626
627                        throw new StructureException(e.getMessage() + " while parsing " + name,e);
628
629                }
630
631                n.setName(name);
632                return n;
633
634
635        }

665        private Structure getStructureFromSCOPDomain(String name)
666                        throws IOException, StructureException {
667                // looks like a SCOP domain!
668                ScopDomain domain;
669                if( this.strictSCOP) {
670                        domain = getScopDomain(name);
671                } else {
672                        domain = guessScopDomain(name);
673                }
674
675                System.out.println(domain);
676                if ( domain != null){
677                        Structure s = getStructureForDomain(domain);
678                        return s;
679                }
680
681                // Guessing didn't work, so just use the PDBID and Chain from name
682                if( !this.strictSCOP) {
683                        Matcher scopMatch = scopIDregex.matcher(name);
684                        if( scopMatch.matches() ) {
685                                String pdbID = scopMatch.group(1);
686                                String chainID = scopMatch.group(2);
687
688                                // None of the actual SCOP domains match. Guess that '_' means 'whole chain'
689                                if( !chainID.equals("_") ) {
690                                        //Add chain identifier
691                                        pdbID += "."+scopMatch.group(2);
692                                }
693                                // Fetch the structure by pdb id
694                                Structure struct = getStructure(pdbID);
695                                if(struct != null) {
696                                        System.err.println("Trying chain "+pdbID);
697                                }
698
699                                return struct;
700                        }
701                }
702
703                throw new StructureException("Unable to get structure for SCOP domain: "+name);
704        }

707        private Structure getStructureFromCATHDomain(StructureName structureName)
708                        throws IOException, StructureException {
709
710
711
712                CathInstallation cathInstall = new CathInstallation(path);
713
714                CathDomain cathDomain = cathInstall.getDomainByCathId(structureName.getName());
715
716                List<CathSegment> segments = cathDomain.getSegments();
717
718                StringWriter range = new StringWriter();
719
720                int rangePos = 0;
721                String chainId = structureName.getChainId();
722                for ( CathSegment segment : segments) {
723                        rangePos++;
724
725                        range.append(chainId);
726                        range.append("_");
727
728                        range.append(segment.getStart());
729                        range.append("-");
730                        range.append(segment.getStop());
731                        if ( ( segments.size()> 1) && (rangePos < segments.size())){
732                                range.append(",");
733                        }
734                }
735
736                String pdbId = structureName.getPdbId();
737
738                Structure s = null;
739
740                try {
741                        s = getStructure(pdbId);
742

775        private Structure getStructureFromURL(URL url) throws IOException, StructureException {
776                // looks like a URL for a file was provided:
777                System.out.println("fetching structure from URL:" + url);
778
779                String queryS = url.getQuery();
780
781                String chainId = null;
782                if ( queryS != null && (queryS.startsWith("chainId="))) {
783                        chainId = queryS.substring(8);
784
785                        String fullu = url.toString();
786
787                        if (fullu.startsWith("file:") && fullu.endsWith("?"+queryS)) {
788                                // for windowze, drop the query part from the URL again
789                                // otherwise there will be a "file not found error" ...
790
791                                String newu = fullu.substring(0,(fullu.length()-(("?"+queryS).length())));
792                                //System.out.println(newu);
793                                url = new URL(newu);
794                        }
795                }
796
797
798                PDBFileReader reader = new PDBFileReader();
799                reader.setPath(path);
800                reader.setPdbDirectorySplit(isSplit);
801                reader.setAutoFetch(autoFetch);
802                reader.setFetchFileEvenIfObsolete(fetchFileEvenIfObsolete);
803                reader.setFetchCurrent(fetchCurrent);
804
805                reader.setFileParsingParameters(params);
806
807                Structure s = reader.getStructure(url);
808                if ( chainId == null)
809                        return StructureTools.getReducedStructure(s,-1);
810                else 
811                        return StructureTools.getReducedStructure(s,chainId);
812        }

815        public static final Pattern scopIDregex = Pattern.compile("d(....)(.)(.)" );
816        /**
817         * <p>Guess a scop domain. If an exact match is found, return that.
818         * 
819         * <p>Otherwise, return the first scop domain found for the specified protein
820         * such that<ul>
821         *   <li>The chains match, or one of the chains is '_' or '.'.
822         *   <li>The domains match, or one of the domains is '_'.
823         * </ul>
824         *   
825         *   
826         * @param name
827         * @return
828         * @throws IOException
829         * @throws StructureException
830         */
831        private ScopDomain guessScopDomain(String name) throws IOException, StructureException {
832                List<ScopDomain> matches = new LinkedList<ScopDomain>();
833
834                // Try exact match first
835                ScopDomain domain = getScopDomain(name);
836                if ( domain != null){
837                        return domain;
838                }
839
840                // Didn't work. Guess it!
841                System.err.println("Warning, could not find SCOP domain: " + name);
842
843                Matcher scopMatch = scopIDregex.matcher(name);
844                if( scopMatch.matches() ) {
845                        String pdbID = scopMatch.group(1);
846                        String chainID = scopMatch.group(2);
847                        String domainID = scopMatch.group(3);
848
849                        if ( scopInstallation == null) {
850                                scopInstallation = ScopFactory.getSCOP();
851                        }
852
853                        for( ScopDomain potentialSCOP : scopInstallation.getDomainsForPDB(pdbID) ) {
854                                Matcher potMatch = scopIDregex.matcher(potentialSCOP.getScopId());
855                                if(potMatch.matches()) {
856                                        if( chainID.equals(potMatch.group(2)) ||
857                                                        chainID.equals("_") || chainID.equals(".") ||
858                                                        potMatch.group(2).equals("_") || potMatch.group(2).equals(".") ) {
859                                                if( domainID.equals(potMatch.group(3)) || domainID.equals("_") || potMatch.group(3).equals("_") ) {
860                                                        // Match, or near match
861                                                        matches.add(potentialSCOP);
862                                                }
863                                        }
864                                }
865                        }
866                }
867
868                Iterator<ScopDomain> match = matches.iterator();
869                if( match.hasNext() ) {
870                        ScopDomain bestMatch = match.next();
871                        System.err.print("Trying domain "+bestMatch.getScopId()+".");
872                        if( match.hasNext() ) {
873                                System.err.print(" Other possibilities: ");
874                                while(match.hasNext()) {
875                                        System.err.print(match.next().getScopId() + " ");
876                                }
877                        }
878                        System.err.println();
879                        return bestMatch;
880                } else {
881                        return null;
882                }
883        }

944        /** 
945         * Loads the biological assembly for a given PDB ID and bioAssemblyId. 
946         * If a bioAssemblyId > 0 is specified, the corresponding biological assembly file will be loaded. Note, the
947         * number of available biological unit files varies. Many entries don't have a biological assembly specified (i.e. NMR structures),
948         * many entries have only one biological assembly (bioAssemblyId=1), and a few structures have multiple biological assemblies.
949         * Set bioAssemblyFallback to true, to download the original PDB file in cases that a biological assembly file is not available.
950         * 
951         * @param pdbId the PDB ID
952         * @param bioAssemblyId the ID of the biological assembly
953         * @param bioAssemblyFallback if true, try reading original PDB file in case the biological assembly file is not available
954         * @return a structure object
955         * @throws IOException 
956         * @throws StructureException 
957         * @author Peter Rose
958         * @since 3.2
959         */
960        public Structure getBiologicalAssembly(String pdbId, int bioAssemblyId, boolean bioAssemblyFallback) throws StructureException, IOException {
961                Structure s;
962                if (bioAssemblyId < 1) {
963                        throw new StructureException("bioAssemblyID must be greater than zero: " + pdbId + 
964                                        " bioAssemblyId " + bioAssemblyId);
965                }
966                PDBFileReader reader = new PDBFileReader();
967                reader.setPath(path);
968                reader.setPdbDirectorySplit(isSplit);
969                reader.setAutoFetch(autoFetch);
970                reader.setFetchFileEvenIfObsolete(fetchFileEvenIfObsolete);
971                reader.setFetchCurrent(fetchCurrent);
972                reader.setFileParsingParameters(params);
973                reader.setBioAssemblyId(bioAssemblyId);
974                reader.setBioAssemblyFallback(bioAssemblyFallback);
975                s = reader.getStructureById(pdbId.toLowerCase());
976                s.setPDBCode(pdbId);
977                return s;
978        }
