001package org.biojava3.aaproperties.xml;
002
003import javax.xml.bind.annotation.XmlAccessType;
004import javax.xml.bind.annotation.XmlAccessorType;
005import javax.xml.bind.annotation.XmlAttribute;
006import javax.xml.bind.annotation.XmlType;
007
008
009@XmlType(name = "Iostope", propOrder = {"name","neutronsNum","mass"})
010@XmlAccessorType(XmlAccessType.NONE)
011public class Isotope {
012        /**
013         * The isotope name i.e. T (tritium)
014         */
015        @XmlAttribute(name = "name", required = true)
016        private String name; 
017        /**
018         * Number of neutrons 
019         */
020        @XmlAttribute(name = "neutronsnum", required = true)
021        private int neutronsNum;
022        /**
023         * Relative Atomic Mass of the isotope
024         */
025        @XmlAttribute(name = "mass", required = true)
026        private double mass; 
027
028        public Isotope(){}
029
030        public Isotope(String name, int neutronsNum, double mass){
031                if(neutronsNum <= 0){
032                        throw new Error("Neutrons number of Isotopes must be > 0.");
033                }
034                if(mass <= 0){
035                        throw new Error("Mass of Isotopes must be > 0.");
036                }
037                this.setName(name);
038                this.setNeutronsNum(neutronsNum);
039                this.setMass(mass);
040        }
041        
042        public String getName(){
043                return this.name;
044        }
045
046        public void setName(String name) {
047                this.name = name;
048        }
049        
050        public int getNeutronsNum() {
051                return neutronsNum;
052        }
053        
054        public void setNeutronsNum(int neutronsNum) {
055                this.neutronsNum = neutronsNum;
056        }
057        
058        public double getMass() {
059                return mass;
060        }
061        
062        public void setMass(double weight) {
063                this.mass = weight;
064        }
065}
