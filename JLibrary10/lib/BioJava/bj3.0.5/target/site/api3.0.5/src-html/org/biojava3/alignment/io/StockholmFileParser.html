037/**
038 * Stockholm file parser.<br>
039 * for more information about the format refer to 
040 * <ul>
041 * <li><a href="ftp://ftp.sanger.ac.uk/pub/databases/Pfam/current_release/userman.txt">ftp://ftp.sanger.ac.uk/pub/databases/Pfam/current_release/userman.txt</a>.</li>
042 * <li><a href="ftp://ftp.sanger.ac.uk/pub/databases/Rfam/CURRENT/USERMAN">ftp://ftp.sanger.ac.uk/pub/databases/Rfam/CURRENT/USERMAN</a>.</li>
043 * <li><a href="http://sonnhammer.sbc.su.se/Stockholm.html">http://sonnhammer.sbc.su.se/Stockholm.html</a>.</li>
044 * </ul>
045 * <pre>
046 * Pfam DESCRIPTION OF FIELDS
047
048   Compulsory fields:
049   ------------------
050
051   AC   Accession number:           Accession number in form PFxxxxx.version or PBxxxxxx.
052   ID   Identification:             One word name for family.
053   DE   Definition:                 Short description of family.
054   AU   Author:                     Authors of the entry.
055   SE   Source of seed:             The source suggesting the seed members belong to one family.
056   GA   Gathering method:           Search threshold to build the full alignment.
057   TC   Trusted Cutoff:             Lowest sequence score and domain score of match in the full alignment.
058   NC   Noise Cutoff:               Highest sequence score and domain score of match not in full alignment.
059   TP   Type:                       Type of family -- presently Family, Domain, Motif or Repeat.
060   SQ   Sequence:                   Number of sequences in alignment.
061   //                               End of alignment.
062
063   Optional fields:
064   ----------------
065
066   DC   Database Comment:           Comment about database reference.
067   DR   Database Reference:         Reference to external database.
068   RC   Reference Comment:          Comment about literature reference.
069   RN   Reference Number:           Reference Number.
070   RM   Reference Medline:          Eight digit medline UI number.
071   RT   Reference Title:            Reference Title.
072   RA   Reference Author:           Reference Author
073   RL   Reference Location:         Journal location.
074   PI   Previous identifier:        Record of all previous ID lines.
075   KW   Keywords:                   Keywords.
076   CC   Comment:                    Comments.
077   NE   Pfam accession:             Indicates a nested domain.
078   NL   Location:                   Location of nested domains - sequence ID, start and end of insert.
079   WK   Wikipedia Reference:        Reference to wikipedia.
080
081   Obsolete fields:
082   -----------
083   AL   Alignment method of seed:   The method used to align the seed members.
084   AM   Alignment Method:           The order ls and fs hits are aligned to the model to build the full align.
085
086 * </pre>
087 * 
088 * @since 3.0.5
089 * @author Amr AL-Hossary
090 * @author Marko Vaz
091 * 
092 */

257        /**
258         * Parses a Stockholm file and returns a {@link StockholmStructure} object with its content.<br>
259         * This function is meant to be used for single access to specific 
260         * file and it closes the file after doing its assigned job. Any subsequent call 
261         * to {@link #parseNext(int)} will throw an exception or will function with unpredicted behavior.
262         * 
263         * @param filename complete(?) path to the file from where to read the content
264         * @return stockholm file content
265         * @throws IOException when an exception occurred while opening/reading/closing the file+
266         * @throws ParserException if unexpected format is encountered
267         */
268        public StockholmStructure parse(String filename) throws IOException,ParserException{
269                InputStream inStream = new InputStreamProvider().getInputStream(filename);
270                StockholmStructure structure = parse(inStream);
271                inStream.close();
272                return structure;
273        }
274        /**
275         * Parses a Stockholm file and returns a {@link StockholmStructure} object with its content.<br>
276         * This function doesn't close the file after doing its assigned job; to allow for further calls of {@link #parseNext(int)}.
277         * @see #parseNext(int)
278         * 
279         * @param filename file from where to read the content. see {@link InputStreamProvider} for more details.
280         * @param max maximum number of files to read, {@link #INFINITY} for all.
281         * @return a vector of {@link StockholmStructure} containing parsed structures.
282         * @throws IOException when an exception occurred while opening/reading/closing the file.
283         * @throws ParserException if unexpected format is encountered
284         */
285        public List<StockholmStructure> parse(String filename, int max) throws IOException,ParserException{
286                InputStreamProvider isp = new InputStreamProvider();
287                InputStream inStream = isp.getInputStream(filename);
288                return parse(inStream, max);
289        }

305        /**parses an {@link InputStream} and returns at maximum <code>max</code> objects contained in
306         * that file.<br>
307         * This method leaves the stream open for further calls of {@link #parse(InputStream, int)} (same function) or {@link #parseNext(int)}.
308         * 
309         * @see #parseNext(int)
310         * @param inStream the stream to parse
311         * @param max maximum number of structures to try to 
312         * parse, {@link #INFINITY} to try to obtain as much as possible. 
313         * @return a {@link List} of {@link StockholmStructure} objects. If there are no more
314         * structures, an empty list is returned.
315         * @throws IOException in case an I/O Exception occurred.
316         */
317        public List<StockholmStructure> parse(InputStream inStream, int max) throws IOException {
318                if (max < INFINITY) {
319                        throw new IllegalArgumentException("max can't be -ve value "+max);
320                }
321                if (inStream != this.cashedInputStream) {
322                        this.cashedInputStream=inStream;
323                        this.internalScanner=null;
324                }
325                
326                if (internalScanner == null) {
327                        internalScanner= new Scanner(inStream);
328                }
329                ArrayList<StockholmStructure> structures= new ArrayList<StockholmStructure>();
330                while (max != INFINITY && max-- >0) {
331                        StockholmStructure structure = parse(internalScanner);
332                        if(structure != null){
333                                structures.add(structure);
334                        }else {
335                                break;
336                        }
337                }
338                return structures;
339        }

341        /**Tries to parse and return as maximum as <code>max</code> structures in the last used file or input stream.<br>
342         * Please consider calling either {@link #parse(InputStream)}, 
343         * {@link #parse(InputStream, int)}, or {@link #parse(String, int)} before calling this function.
344         * @param max
345         * @return
346         * @throws IOException
347         */
348        public List<StockholmStructure> parseNext(int max) throws IOException {
349                return parse(this.cashedInputStream, max);
350        }

352        /**
353         * Parses a Stockholm file and returns a {@link StockholmStructure} object with its content.
354         * This method returns just after reaching the end of structure delimiter line ("//"), leaving any remaining empty lines unconsumed.
355         * 
356         * @param scanner from where to read the file content
357         * @return Stockholm file content, <code>null</code> if couldn't or no more structures.
358         * @throws IOException 
359         * @throws Exception
360         */
361        StockholmStructure parse(Scanner scanner) throws ParserException, IOException {
362                if (scanner == null) {
363                        if(internalScanner != null){
364                                scanner = internalScanner;
365                        }else {
366                                throw new IllegalArgumentException("No Scanner defined");
367                        }
368                }
369                String line = null;
370                int linesCount = 0;
371                try {
372                        while(scanner.hasNextLine()){
373                                line = scanner.nextLine();
374                                // if the file is empty
375                                //this condition will not happen, just left in case we decided to go for buffereedReader again for performance purpose.
376                                if (linesCount == 0 && line == null) {
377                                        throw new IOException("Could not parse Stockholm file, BufferedReader returns null!");
378                                }
379
380                                // ignore empty lines
381                                if ((/*status==STATUS_INSIDE_FILE &&*/ line == null) || line.trim().length()==0) {
382                                        continue;
383                                }
384
385                                if (line.startsWith("#=G")) {
386//                                      // comment line or metadata
387//                                      line = line.substring(1).trim();
388//                                      line = line.substring(1).trim();
389                                        if (line.startsWith(GENERIC_PER_FILE_ANNOTATION,2)) {
390                                                // #=GF <featurename> <generic per-file annotation, free text>
391                                                int firstSpaceIndex=line.indexOf(' ', 5);
392                                                String featureName=line.substring(5, firstSpaceIndex);
393                                                String value= line.substring(firstSpaceIndex).trim();
394                                                handleFileAnnotation(featureName,value);
395                                        } else if (line.startsWith(GENERIC_PER_CONSENSUS_ANNOTATION,2)) {
396                                                //Being in a consensus means we are no longer in a sequence.
397                                                this.status=STATUS_INSIDE_FILE;
398                                                // #=GC <featurename> <generic per-column annotation, exactly 1 char per column>
399                                                int firstSpaceIndex=line.indexOf(' ', 5);
400                                                String featureName=line.substring(5, firstSpaceIndex);
401                                                String value= line.substring(firstSpaceIndex).trim();
402                                                handleConsensusAnnotation(featureName, value);
403                                        } else if (line.startsWith(GENERIC_PER_SEQUENCE_ANNOTATION,2)) {
404                                                // #=GS <seqname> <featurename> <generic per-sequence annotation, free text>
405                                                int index1=line.indexOf(' ', 5);
406                                                String seqName=line.substring(5, index1);
407                                                while (line.charAt(++index1)<= ' ')//i.e. white space
408                                                        ;//keep advancing
409                                                int index2=line.indexOf(' ', index1);
410                                                String featureName=line.substring(index1, index2);
411                                                String value= line.substring(index2).trim();
412                                                handleSequenceAnnotation(seqName,featureName,value);
413                                        } else if (line.startsWith(GENERIC_PER_RESIDUE_ANNOTATION,2)) {
414                                                // #=GR <seqname> <featurename> <generic per-sequence AND per-column mark-up, exactly 1 character per column>
415                                                int index1=line.indexOf(' ', 5);
416                                                String seqName=line.substring(5, index1);
417                                                while (line.charAt(++index1)== ' ') 
418                                                        ;//keep advancing
419                                                int index2=line.indexOf(' ', index1);
420                                                String featureName=line.substring(index1, index2);
421                                                String value= line.substring(index2).trim();
422                                                handleResidueAnnotation(seqName,featureName,value);
423                                        }
424                                }else if(line.startsWith("# STOCKHOLM")) { // it is the header line
425                                        if (status== STATUS_OUTSIDE_FILE) {
426                                                status= STATUS_INSIDE_FILE;
427                                                String[] header = line.split("\\s+");
428                                                this.stockholmStructure = new StockholmStructure();
429                                                this.stockholmStructure.getFileAnnotation().setFormat(header[1]);
430                                                this.stockholmStructure.getFileAnnotation().setVersion(header[2]);
431                                        } else {
432                                                throw new ParserException("Uexpected Format line: ["+line+"]");
433                                        }
434                                } else if (line.trim().equals("//")) {
435                                        status=STATUS_OUTSIDE_FILE;
436                                        break;//should we just break immediately or jump next empty lines?
437                                } else /*if (!line.startsWith("#")) */{
438                                        if (status == STATUS_IN_SEQUENCE) {
439                                                // This line corresponds to a sequence. Something like:
440                                                // O83071/192-246 MTCRAQLIAVPRASSLAEAIACAQKMRVSRVPVYERS
441                                                handleSequenceLine(line);
442//                                      }else if (status==STATUS_OUTSIDE_FILE) {
443//                                              throw new ParserException("The end of file character was allready reached but there are still sequence lines");
444                                        }else {
445                                                System.err.println("Error: Unknown or unexpected line [" +line+"].\nPlease contact the Biojava team.");
446                                                throw new ParserException("Error: Unknown or unexpected line [" +line+"].");
447                                        }
448                                }
449                                linesCount++;
450                        }
451                } catch (IOException e) {
452                        e.printStackTrace();
453                        throw new IOException("Error parsing Stockholm file");
454                }
455                StockholmStructure structure = this.stockholmStructure;
456                this.stockholmStructure=null;
457                if (structure != null) {
458                        int length = -1;
459                        Map<String, StringBuffer> sequences = structure.getSequences();
460                        for (String sequencename : sequences.keySet()) {
461                                StringBuffer sequence = sequences.get(sequencename);
462                                if (length == -1) {
463                                        length = sequence.length();
464                                } else if (length != sequence.length()) {
465                                        throw new RuntimeException(
466                                                        "Sequences have different lengths");
467                                }
468                        }
469                }
470                return structure;
471        }

473        /**
474         * Handles a line that corresponds to a sequence. <br>
475         * e.g.: COATB_BPIKE/30-81
476         * AEPNAATNYATEAMDSLKTQAIDLISQTWPVVTTVVVAGLVIRLFKKFSSKA<br>
477         * N.B.: This function can't tolerate sequences with intrinsic white space.
478         * @param line
479         *            the line to be parsed
480         * @throws Exception
481         */
482        private void handleSequenceLine(String line) throws ParserException {
483                String[] lineContent = line.split("\\s+");
484                if (lineContent.length != 2) {
485                        throw new ParserException("Could not split sequence line into sequence name and sequence:\n"
486                                                        + line);
487                }
488                stockholmStructure.addSequence(lineContent[0], lineContent[1]);
489        }

491        /**
492         * #=GF &lt;feature&gt; &lt;Generic per-File annotation, free text&gt;
493         * @param featureName 
494         * @param value the line to be parsed
495         */
496        private void handleFileAnnotation(String featureName, String value) {
497                if (featureName.equals(GF_ACCESSION_NUMBER)) {
498                        stockholmStructure.getFileAnnotation().setGFAccessionNumber(value);
499                } else if (featureName.equals(GF_IDENTIFICATION)) {
500                        stockholmStructure.getFileAnnotation().setGFIdentification(value);
501                } else if (featureName.equals(GF_DB_REFERENCE)) { 
502                        stockholmStructure.getFileAnnotation().addDBReference(value);
503                } else if (featureName.equals(GF_DEFINITION)) {
504                        stockholmStructure.getFileAnnotation().setGFDefinition(value);
505                } else if (featureName.equals(GF_AUTHOR)) {
506                        stockholmStructure.getFileAnnotation().setGFAuthors(value);
507                } else if (featureName.equals(GF_ALIGNMENT_METHOD)) {
508                        stockholmStructure.getFileAnnotation().setAlignmentMethod(value);
509                } else if (featureName.equals(GF_BUILD_METHOD)) {
510                        stockholmStructure.getFileAnnotation().addGFBuildMethod(value);
511                } else if (featureName.equals(GF_SEARCH_METHOD)) {
512                        stockholmStructure.getFileAnnotation().setGFSearchMethod(value);
513                } else if (featureName.equals(GF_SOURCE_SEED)) {
514                        stockholmStructure.getFileAnnotation().setGFSourceSeed(value);
515                } else if (featureName.equals(GF_SOURCE_STRUCTURE)) {
516                        stockholmStructure.getFileAnnotation().setGFSourceStructure(value);
517                } else if (featureName.equals(GF_GATHERING_THRESHOLD)) {
518                        stockholmStructure.getFileAnnotation().setGFGatheringThreshs(value);
519                } else if (featureName.equals(GF_TRUSTED_CUTOFF)) {
520                        stockholmStructure.getFileAnnotation().setGFTrustedCutoffs(value);
521                } else if (featureName.equals(GF_NOISE_CUTOFF)) {
522                        stockholmStructure.getFileAnnotation().setGFNoiseCutoffs(value);
523                } else if (featureName.equals(GF_TYPE_FIELD)) {
524                        stockholmStructure.getFileAnnotation().setGFTypeField(value);
525                } else if (featureName.equals(GF_PREVIOUS_IDS)) {
526                        stockholmStructure.getFileAnnotation().setGFPreviousIDs(value);
527                } else if (featureName.equals(GF_SEQUENCE)) {
528                        status=STATUS_IN_SEQUENCE;
529                        stockholmStructure.getFileAnnotation().setGFNumSequences(value);
530                } else if (featureName.equals(GF_DB_COMMENT)) {
531                        stockholmStructure.getFileAnnotation().setGFDBComment(value);
532//              } else if (featureName.equals(GF_DB_REFERENCE)) {
533//                      stockholmStructure.getFileAnnotation().addDBReference(value);
534                } else if (featureName.equals(GF_REFERENCE_COMMENT)) {
535                        stockholmStructure.getFileAnnotation().setGFRefComment(value);
536                } else if (featureName.equals(GF_REFERENCE_NUMBER)) {
