034        /**
035         * Checks that the alignment given by afpChain is sequential. This means
036         * that the residue indices of both proteins increase monotonically as
037         * a function of the alignment position (ie both proteins are sorted).
038         * 
039         * This will return false for circularly permuted alignments or other
040         * non-topological alignments. It will also return false for cases where
041         * the alignment itself is sequential but it is not stored in the afpChain
042         * in a sorted manner.
043         * 
044         * Since algorithms which create non-sequential alignments split the
045         * alignment into multiple blocks, some computational time can be saved
046         * by only checking block boundaries for sequentiality. Setting
047         * <tt>checkWithinBlocks</tt> to <tt>true</tt> makes this function slower,
048         * but detects AFPChains with non-sequential blocks.
049         * 
050         * Note that this method should give the same results as
051         * {@link AFPChain#isSequentialAlignment()}. However, the AFPChain version
052         * relies on the StructureAlignment algorithm correctly setting this
053         * parameter, which is sadly not always the case.
054         * 
055         * @param afpChain An alignment
056         * @param checkWithinBlocks Indicates whether individual blocks should be
057         *      checked for sequentiality
058         * @return True if the alignment is sequential.
059         */
060        public static boolean isSequentialAlignment(AFPChain afpChain, boolean checkWithinBlocks) {
061                int[][][] optAln = afpChain.getOptAln();
062                int[] alnLen = afpChain.getOptLen();
063                int blocks = afpChain.getBlockNum();
064                
065                if(blocks < 1) return true; //trivial case
066                if ( alnLen[0] < 1) return true;
067                
068                // Check that blocks are sequential
069                if(checkWithinBlocks) {
070                        for(int block = 0; block<blocks; block++) {
071                                if(alnLen[block] < 1 ) continue; //skip empty blocks
072                                
073                                int prevRes1 = optAln[block][0][0];
074                                int prevRes2 = optAln[block][1][0];
075                                
076                                for(int pos = 1; pos<alnLen[block]; pos++) {
077                                        int currRes1 = optAln[block][0][pos];
078                                        int currRes2 = optAln[block][1][pos];
079                                        
080                                        if(currRes1 < prevRes1) {
081                                                return false;
082                                        }
083                                        if(currRes2 < prevRes2) {
084                                                return false;
085                                        }
086                                        
087                                        prevRes1 = currRes1;
088                                        prevRes2 = currRes2;
089                                }
090                        }
091                }
092
093                // Check that blocks are sequential
094                int prevRes1 = optAln[0][0][alnLen[0]-1];
095                int prevRes2 = optAln[0][1][alnLen[0]-1];
096                
097                for(int block = 1; block<blocks;block++) {
098                        if(alnLen[block] < 1 ) continue; //skip empty blocks
099
100                        if(optAln[block][0][0]<prevRes1) {
101                                return false;
102                        }
103                        if(optAln[block][1][0]<prevRes2) {
104                                return false;
105                        }
106                        
107                        prevRes1 = optAln[block][0][alnLen[block]-1];
108                        prevRes2 = optAln[block][1][alnLen[block]-1];
109                }
110                
111                return true;
112        }

114        /**
115         * Creates a Map specifying the alignment as a mapping between residue indices
116         * of protein 1 and residue indices of protein 2.
117         * 
118         * <p>For example,<pre>
119         * 1234
120         * 5678</pre>
121         * becomes<pre>
122         * 1->5
123         * 2->6
124         * 3->7
125         * 4->8</pre>
126         * 
127         * @param afpChain An alignment
128         * @return A mapping from aligned residues of protein 1 to their partners in protein 2.
129         * @throws StructureException If afpChain is not one-to-one
130         */
131        public static Map<Integer, Integer> alignmentAsMap(AFPChain afpChain) throws StructureException {
132                Map<Integer,Integer> map = new HashMap<Integer,Integer>();
133                
134                int[][][] optAln = afpChain.getOptAln();
135                int[] optLen = afpChain.getOptLen();
136                for(int block = 0; block < afpChain.getBlockNum(); block++) {
137                        for(int pos = 0; pos < optLen[block]; pos++) {
138                                int res1 = optAln[block][0][pos];
139                                int res2 = optAln[block][1][pos];
140                                if(map.containsKey(res1)) {
141                                        throw new StructureException(String.format("Residue %d aligned to both %d and %d.", res1,map.get(res1),res2));
142                                }
143                                map.put(res1,res2);
144                        }
145                }
146                return map;
147        }

158         *  function is undefined for that input.

164        /**
165         * Applies an alignment k times. Eg if alignmentMap defines function f(x),
166         * this returns a function f^k(x)=f(f(...f(x)...)).
167         * 
168         * To allow for functions with different domains and codomains, the identity
169         * function allows converting back in a reasonable way. For instance, if 
170         * alignmentMap represented an alignment between two proteins with different
171         * numbering schemes, the identity function could calculate the offset
172         * between residue numbers, eg I(x) = x-offset.
173         * 
174         * When an identity function is provided, the returned function calculates
175         * f^k(x) = f(I( f(I( ... f(x) ... )) )).
176         * 
177         * @param <S>
178         * @param <T>
179         * @param alignmentMap The input function, as a map (see {@link AlignmentTools#alignmentAsMap(AFPChain)})
180         * @param identity An identity-like function providing the isomorphism between
181         *  the codomain of alignmentMap (of type <T>) and the domain (type <S>).
182         * @param k The number of times to apply the alignment
183         * @return A new alignment. If the input function is not automorphic
184         *  (one-to-one), then some inputs may map to null, indicating that the
185         *  function is undefined for that input.
186         */
187        public static <S,T> Map<S,T> applyAlignment(Map<S, T> alignmentMap, Map<T,S> identity, int k) {
188                
189                // This implementation simply applies the map k times.
190                // If k were large, it would be more efficient to do this recursively,
191                // (eg f^4 = (f^2)^2) but k will usually be small.
192                
193                if(k<0) throw new IllegalArgumentException("k must be positive");
194                if(k==1) {
195                        return new HashMap<S,T>(alignmentMap); 
196                }
197                // Convert to lists to establish a fixed order
198                List<S> preimage = new ArrayList<S>(alignmentMap.keySet()); // currently unmodified
199                List<S> image = new ArrayList<S>(preimage);
200                
201                for(int n=1;n<k;n++) {
202                        int deltasSq = 0;
203                        int numDeltas = 0;
204                        // apply alignment
205                        for(int i=0;i<image.size();i++) {
206                                S pre = image.get(i);
207                                T intermediate = (pre==null?null: alignmentMap.get(pre));
208                                S post = (intermediate==null?null: identity.get(intermediate));
209                                image.set(i, post);
210                        }
211                }
212                
213                
214
215                Map<S, T> imageMap = new HashMap<S,T>(alignmentMap.size());
216                
217                //TODO handle nulls consistently.
218                // assure that all the residues in the domain are valid keys
219                /*
220                for(int i=0;i<preimage.size();i++) {
221                        S pre = preimage.get(i);
222                        T intermediate = (pre==null?null: alignmentMap.get(pre));
223                        S post = (intermediate==null?null: identity.get(intermediate));
224                        imageMap.put(post, null);
225                }
226                */
227                // now populate with actual values
228                for(int i=0;i<preimage.size();i++) {
229                        S pre = preimage.get(i);
230                        
231                        // image is currently f^k-1(x), so take the final step
232                        S preK1 = image.get(i);
233                        T postK = (preK1==null?null: alignmentMap.get(preK1));
234                        imageMap.put(pre,postK);
235                        
236                }
237                return imageMap;
238        }

258        /**
259         * Tries to detect symmetry in an alignment.
260         * 
261     * <p>Conceptually, an alignment is a function f:A->B between two sets of
262     * integers. The function may have simple topology (meaning that if two
263     * elements of A are close, then their images in B will also be close), or
264     * may have more complex topology (such as a circular permutation). This
265     * function checks <i>alignment</i> against a reference function
266     * <i>identity</i>, which should have simple topology. It then tries to 
267     * determine the symmetry order of <i>alignment</i> relative to
268     * <i>identity</i>, up to a maximum order of <i>maxSymmetry</i>.
269     * 
270         * 
271         * <p><strong>Details</strong><br/>
272         * Considers the offset (in number of residues) which a residue moves
273         * after undergoing <i>n</i> alternating transforms by alignment and
274         * identity. If <i>n</i> corresponds to the intrinsic order of the alignment,
275         * this will be small. This algorithm tries increasing values of <i>n</i>
276         * and looks for abrupt decreases in the root mean squared offset.
277         * If none are found at <i>n</i><=maxSymmetry, the alignment is reported as
278         * non-symmetric.
279         * 
280         * @param alignment The alignment to test for symmetry
281         * @param identity An alignment with simple topology which approximates
282         *  the sequential relationship between the two proteins. Should map in the
283         *  reverse direction from alignment.
284         * @param maxSymmetry Maximum symmetry to consider. High values increase
285         *  the calculation time and can lead to overfitting. 
286         * @param minimumMetricChange Percent decrease in root mean squared offsets
287         *  in order to declare symmetry. 0.4f seems to work well for CeSymm.
288         * @return The order of symmetry of alignment, or 1 if no order <= 
289         *  maxSymmetry is found.
290         *  
291         * @see IdentityMap For a simple identity function
292         */
293        public static int getSymmetryOrder(Map<Integer, Integer> alignment, Map<Integer,Integer> identity,
294                        final int maxSymmetry, final float minimumMetricChange) {
295                List<Integer> preimage = new ArrayList<Integer>(alignment.keySet()); // currently unmodified
296                List<Integer> image = new ArrayList<Integer>(preimage);
297                                
298                int bestSymmetry = 1;
299                double bestMetric = Double.POSITIVE_INFINITY; //lower is better
300                boolean foundSymmetry = false;
301                
302                if(debug) {
303                        System.out.println("Symm\tPos\tDelta");
304                }
305                
306                for(int n=1;n<=maxSymmetry;n++) {
307                        int deltasSq = 0;
308                        int numDeltas = 0;
309                        // apply alignment
310                        for(int i=0;i<image.size();i++) {
311                                Integer pre = image.get(i);
312                                Integer intermediate = (pre==null?null: alignment.get(pre));
313                                Integer post = (intermediate==null?null: identity.get(intermediate));
314                                image.set(i, post);
315                                
316                                if(post != null) {
317                                        int delta = post-preimage.get(i);
318                                        
319                                        deltasSq += delta*delta;
320                                        numDeltas++;
321
322                                        if(debug) {
323                                                System.out.format("%d\t%d\t%d\n",n,preimage.get(i),delta);
324                                        }
325                                }
326                                
327                        }
328                        
329                        // Metrics: RMS compensates for the trend of smaller numDeltas with higher order
330                        // Not normalizing by numDeltas favors smaller orders
331                        
332                        double metric = Math.sqrt((double)deltasSq/numDeltas); // root mean squared distance
333                        //double metric = Math.sqrt((double)deltasSq); // root squared distance
334                        
335                        //System.out.format("%d\t%f\n",n,metric);
336                        
337                        if(!foundSymmetry && metric < bestMetric * minimumMetricChange) {
338                                // n = 1 is never the best symmetry
339                                if(bestMetric < Double.POSITIVE_INFINITY) {
340                                        foundSymmetry = true;
341                                }
342                                bestSymmetry = n;
343                                bestMetric = metric;
344                        }
345                        
346                        // When debugging need to loop over everything. Unneeded in production
347                        if(!debug && foundSymmetry) {
348                                break;
349                        }
350                        
351                }
352                if(foundSymmetry) {
353                        return bestSymmetry;
354                } else {
355                        return 1;
356                }
357        }

360        /**
361         * Guesses the order of symmetry in an alignment
362         * 
363         * <p>Uses {@link #getSymmetryOrder(Map alignment, Map identity, int, float)}
364         * to determine the the symmetry order. For the identity alignment, sorts
365         * the aligned residues of each protein sequentially, then defines the ith
366         * residues of each protein to be equivalent.
367         */
368        public static int getSymmetryOrder(AFPChain afpChain, int maxSymmetry, float minimumMetricChange) throws StructureException {
369                // alignment comes from the afpChain alignment
370                Map<Integer,Integer> alignment = AlignmentTools.alignmentAsMap(afpChain);
371
372                // Now construct identity to map aligned residues in sequential order
373                Map<Integer, Integer> identity = guessSequentialAlignment(alignment, true);
374                
375                
376                return AlignmentTools.getSymmetryOrder(alignment,
377                                identity,
378                                maxSymmetry, minimumMetricChange);
379        }

381        /**
382         * Takes a potentially non-sequential alignment and guesses a sequential
383         * version of it. Residues from each structure are sorted sequentially and
384         * then compared directly.
385         * 
386         * <p>The results of this method are consistent with what one might expect
387         * from an identity function, and are therefore useful with
388         * {@link #getSymmetryOrder(Map, Map identity, int, float)}.
389         * <ul>
390         *  <li>Perfect self-alignments will have the same pre-image and image,
391         *      so will map X->X</li>
392         *  <li>Gaps and alignment errors will cause errors in the resulting map,
393         *      but only locally. Errors do not propagate through the whole 
394         *      alignment.</li>
395         * </ul>
396         * 
397         * <h4>Example:</h4>
398         * A non sequential alignment, represented schematically as
399         * <pre>
400         * 12456789
401         * 78912345</pre>
402         * would result in a map
403         * <pre>
404         * 12456789
405         * 12345789</pre>
406         * @param afpChain The non-sequential input alignment
407         * @param inverseAlignment If false, map from structure1 to structure2. If
408         *  true, generate the inverse of that map.
409         * @return A mapping from sequential residues of one protein to those of the other
410         * @throws IllegalArgumentException if the input alignment is not one-to-one.
411         */
412        public static Map<Integer, Integer> guessSequentialAlignment(
413                        Map<Integer,Integer> alignment, boolean inverseAlignment) {
414                Map<Integer,Integer> identity = new HashMap<Integer,Integer>();
415                
416                SortedSet<Integer> aligned1 = new TreeSet<Integer>();
417                SortedSet<Integer> aligned2 = new TreeSet<Integer>();
418                
419                for(Entry<Integer,Integer> pair : alignment.entrySet()) {
420                        aligned1.add(pair.getKey());
421                        if( !aligned2.add(pair.getValue()) )
422                                throw new IllegalArgumentException("Alignment is not one-to-one for residue "+pair.getValue()+" of the second structure.");
423                }
424                
425                Iterator<Integer> it1 = aligned1.iterator();
426                Iterator<Integer> it2 = aligned2.iterator();
427                while(it1.hasNext()) {
428                        if(inverseAlignment) { // 2->1
429                                identity.put(it2.next(),it1.next());
430                        } else { // 1->2
431                                identity.put(it1.next(),it2.next());
432                        }
433                }
434                return identity;
435        }

437        /**
438         * Retrieves the optimum alignment from an AFPChain and returns it as a
439         * java collection. The result is indexed in the same way as 
440         * {@link AFPChain#getOptAln()}, but has the correct size().
441         * <pre>
442         * List<List<List<Integer>>> aln = getOptAlnAsList(AFPChain afpChain);
443         * aln.get(blockNum).get(structureNum={0,1}).get(pos)</pre>
444         * 
445         * @param afpChain
446         * @return
447         */
448        public static List<List<List<Integer>>> getOptAlnAsList(AFPChain afpChain) {
449                int[][][] optAln = afpChain.getOptAln();
450                int[] optLen = afpChain.getOptLen();
451                List<List<List<Integer>>> blocks = new ArrayList<List<List<Integer>>>(afpChain.getBlockNum());
452                for(int blockNum=0;blockNum<afpChain.getBlockNum();blockNum++) {
453                        //TODO could improve speed an memory by wrapping the arrays with
454                        // an unmodifiable list, similar to Arrays.asList(...) but with the
455                        // correct size parameter.
456                        List<Integer> align1 = new ArrayList<Integer>(optLen[blockNum]);
457                        List<Integer> align2 = new ArrayList<Integer>(optLen[blockNum]);
458                        for(int pos=0;pos<optLen[blockNum];pos++) {
459                                align1.add(optAln[blockNum][0][pos]);
460                                align2.add(optAln[blockNum][1][pos]);
461                        }
462                        List<List<Integer>> block = new ArrayList<List<Integer>>(2);
463                        block.add(align1);
464                        block.add(align2);
465                        blocks.add(block);
466                }
467                
468                return blocks;
469        }

474         * A Map<K,V> can be viewed as a function from K to V. This class represents
475         * the identity function. Getting a value results in the value itself.
476         * 
477         * <p>The class is a bit inconsistent when representing its contents. On

481         * @author Spencer Bliven
482         *
483         * @param <K>
484         * @param <V>
485         */
486        public static class IdentityMap<K> extends AbstractMap<K,K> {
487                public IdentityMap() {}
488
489                /**
490                 * @param key
491                 * @return the key
492                 * @throws ClassCastException if key is not of type K
493                 */
494                @SuppressWarnings("unchecked")
495                @Override
496                public K get(Object key) {
497                        return (K)key;
498                }
499                
500                /**
501                 * Always returns the empty set
502                 */
503                @Override
504                public Set<java.util.Map.Entry<K, K>> entrySet() {
505                        return Collections.emptySet();
506                }
507                
508                @Override
509                public boolean containsKey(Object key) {
510                        return true;
511                }
512        }

514        /**
515         * Fundimentally, an alignment is just a list of aligned residues in each
516         * protein. This method converts two lists of ResidueNumbers into an
517         * AFPChain.
518         * @param ca1 CA atoms of the first protein
519         * @param ca2 CA atoms of the second protein
520         * @param aligned1 A list of aligned residues from the first protein
521         * @param aligned2 A list of aligned residues from the second protein.
522         *  Must be the same length as aligned1.
523         * @return An AFPChain representing the alignment. Many properties may be
524         *  null or another default.
525         * @throws IllegalArgumentException if aligned1 and aligned2 have different
526         *  lengths
527         */
528        public static AFPChain createAFPChain(Atom[] ca1, Atom[] ca2,
529                        ResidueNumber[] aligned1, ResidueNumber[] aligned2 ) {
530                //input validation
531                int alnLen = aligned1.length;
532                if(alnLen != aligned2.length) {
533                        throw new IllegalArgumentException("Alignment lengths are not equal");
534                }
535                
536                AFPChain a = new AFPChain();
537                a.setName1(ca1[0].getGroup().getChain().getParent().getName());
538                a.setName2(ca2[0].getGroup().getChain().getParent().getName());
539                
540                a.setBlockNum(1);
541                a.setCa1Length(ca1.length);
542                a.setCa2Length(ca2.length);
543                
544                a.setOptLength(alnLen);
545                a.setOptLen(new int[] {alnLen});
546                
547
548                Matrix[] ms = new Matrix[a.getBlockNum()];
549                a.setBlockRotationMatrix(ms);
550                Atom[] blockShiftVector = new Atom[a.getBlockNum()];
551                a.setBlockShiftVector(blockShiftVector);
552                // TODO run superimposer (not required for display)
553                // TODO more properties
554
555                
556                String[][][] pdbAln = new String[1][2][alnLen];
557                for(int i=0;i<alnLen;i++) {
558                        pdbAln[0][0][i] = aligned1[i].getChainId()+":"+aligned1[i].toString();
559                        pdbAln[0][1][i] = aligned2[i].getChainId()+":"+aligned2[i].toString();
560                }
561                
562                a.setPdbAln(pdbAln);
563                
564                // convert pdbAln to optAln, and fill in some other basic parameters
565                AFPChainXMLParser.rebuildAFPChain(a, ca1, ca2);
566
567                return a;
568        }

570        /**
571         * After the alignment changes (optAln, optLen, blockNum, at a minimum),
572         * many other properties which depend on the superposition will be invalid.
573         * 
574         * This method re-runs a rigid superposition over the whole alignment
575         * and repopulates the required properties, including RMSD (TotalRMSD) and
576         * TM-Score.
577         * @param afpChain
578         * @param ca1
579         * @param ca2
580         * @throws StructureException
581         * @see {@link CECalculator#calc_rmsd(Atom[], Atom[], int, boolean, boolean)}
582         *  contains much of the same code, but stores results in a CECalculator
583         *  instance rather than an AFPChain 
584         */
585        public static void updateSuperposition(AFPChain afpChain, Atom[] ca1, Atom[] ca2) throws StructureException {
586                
587                // Create arrays of aligned atoms
588                int optLength = afpChain.getOptLength();
589                int alnPos = 0;
590                Atom[] aln1 = new Atom[optLength];
591                Atom[] aln2 = new Atom[optLength];
592                
593                int[] optLen = afpChain.getOptLen();
594                int[][][] optAln = afpChain.getOptAln();
595                int blockNum = afpChain.getBlockNum();
596                for(int blk=0;blk<blockNum;blk++) {
597                        for(int pos=0;pos<optLen[blk];pos++) {
598                                int res1 = optAln[blk][0][pos];
599                                int res2 = optAln[blk][1][pos];
600                                aln1[alnPos] = ca1[res1];
601                                aln2[alnPos] = ca2[res2];
602                                alnPos++;
603                        }
604                }
605                
606                // Superimpose
607                SVDSuperimposer svd = new SVDSuperimposer(aln1, aln2);
608                
609                Matrix matrix = svd.getRotation();
610                Atom shift = svd.getTranslation();
611                
612                // Apply transformation to ca2
613                for(Atom a : ca2) {
614                        Calc.rotate(a.getGroup(), matrix);
615                        Calc.shift(a.getGroup(), shift);
616                }
617                
618                double rmsd = svd.getRMS(ca1, ca2);
619                double tm = svd.getTMScore(aln1, aln2, ca1.length, ca2.length);
620
621                // Store new transformation back to AFPChain
622                //TODO convolve current superposition with previous transformation?
