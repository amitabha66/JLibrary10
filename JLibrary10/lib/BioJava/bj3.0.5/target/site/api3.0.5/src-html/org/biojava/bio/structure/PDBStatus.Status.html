033/**
034 * Methods for getting the status of a PDB file (current, obsolete, etc)
035 * and for accessing different versions of the structure.
036 * 
037 * <p>All methods query the 
038 * <a href="http://www.rcsb.org/pdb/rest/idStatus?structureId=1HHB,3HHB,4HHB">
039 * PDB website.</a>
040 * 
041 * <p>PDB supersessions form a directed acyclic graph, where edges point from an
042 * obsolete ID to the entry that directly superseded it. For example, here are  
043 * edges from one portion of the graph:<br/>
044 * 
045 * 1CAT -> 3CAT<br/>
046 * 3CAT -> 7CAT<br/>
047 * 3CAT -> 8CAT<br/>
048 * 
049 * <p>The methods {@link #getReplaces(String, boolean) getReplaces(pdbId, false)}/ 
050 * {@link #getReplacement(String, boolean, boolean) getReplacement(pdbId, false, true)}
051 * just get the incoming/outgoing edges for a single node. The recursive versions 
052 * ({@link #getReplaces(String, boolean) getReplaces(pdbId, true)}, 
053 * {@link #getReplacement(String, boolean, boolean) getReplacement(pdbId, true, false)})
054 * will do a depth-first search up/down the tree and return a list of all nodes ]
055 * reached.
056 * 
057 * <p>Finally, the getCurrent() method returns a single PDB ID from among the 
058 * results of 
059 * {@link #getReplacement(String, boolean) getReplacement(pdbId, true)}. 
060 * To be consistent with the old REST ordering, this is the PDB ID that occurs 
061 * last alphabetically.
062 * 
063 * <p>Results are cached to reduce server load.
064 * 
065 * @author Spencer Bliven <sbliven@ucsd.edu>
066 * @author Amr AL-Hossary
067 * @since 3.0.2
068 */
069public class PDBStatus {
070        public static final String DEFAULT_PDB_SERVER = "www.rcsb.org";
071        public static final String PDB_SERVER_PROPERTY = "PDB.SERVER";

080        /**
081         * Represents the status of PDB IDs. 'OBSOLETE' and 'CURRENT' are the most
082         * common.
083         * @author Spencer Bliven <sbliven@ucsd.edu>
084         *
085         */
086        public enum Status {
087                OBSOLETE,
088                CURRENT,
089                AUTH,
090                HOLD,
091                HPUB,
092                POLC,
093                PROC,
094                REFI,
095                REPL,
096                WAIT,
097                WDRN,
098                MODEL,
099                UNKNOWN;

102                /**
103                 * 
104                 * @param statusStrUpper
105                 * @return
106                 * @throws IllegalArgumentException If the string is not recognized
107                 */
108                public static Status fromString(String statusStr) {
109                        Status status;
110                        String statusStrUpper = statusStr.toUpperCase();
111                        if(statusStrUpper.equalsIgnoreCase("OBSOLETE"))
112                                status = Status.OBSOLETE;
113                        else if(statusStrUpper.equalsIgnoreCase("CURRENT"))
114                                status = Status.CURRENT;
115                        else if(statusStrUpper.equalsIgnoreCase("AUTH"))
116                                status = Status.AUTH;
117                        else if(statusStrUpper.equalsIgnoreCase("HOLD"))
118                                status = Status.HOLD;
119                        else if(statusStrUpper.equalsIgnoreCase("HPUB"))
120                                status = Status.HPUB;
121                        else if(statusStrUpper.equalsIgnoreCase("POLC"))
122                                status = Status.POLC;
123                        else if(statusStrUpper.equalsIgnoreCase("PROC"))
124                                status = Status.PROC;
125                        else if(statusStrUpper.equalsIgnoreCase("REFI"))
126                                status = Status.REFI;
127                        else if(statusStrUpper.equalsIgnoreCase("REPL"))
128                                status = Status.REPL;
129                        else if(statusStrUpper.equalsIgnoreCase("WAIT"))
130                                status = Status.WAIT;
131                        else if(statusStrUpper.equalsIgnoreCase("WDRN"))
132                                status = Status.WDRN;
133                        else if(statusStrUpper.equalsIgnoreCase("MODEL"))
134                                status = Status.MODEL;
135                        else if(statusStrUpper.equalsIgnoreCase("UNKNOWN"))
136                                status = Status.UNKNOWN;
137                        else {
138                                throw new IllegalArgumentException("Unable to parse status '"+statusStrUpper+"'.");
139                        }
140                        return status;
141                }
142        }

144        /**
145         * Get the status of the PDB in question.
146         * 
147         * @param pdbId
148         * @return The status, or null if an error occurred.
149         */
150        public static Status getStatus(String pdbId) {
151                Status[] statuses = getStatus(new String[] {pdbId});
152                if(statuses != null) {
153                        assert(statuses.length == 1);
154                        return statuses[0];
155                } else {
156                        return null;
157                }
158        }

160        /**
161         * Get the status of the a collection of PDBs in question in a single query.
162         * 
163         * @see #getStatus(String)
164         * @param pdbIds
165         * @return The status array, or null if an error occurred.
166         */
167        public static Status[] getStatus(String[] pdbIds) {
168                Status[] statuses = new Status[pdbIds.length];
169
170                List<Map<String,String>> attrList = getStatusIdRecords(pdbIds);
171                //Expect a single record
172                if(attrList == null || attrList.size() != pdbIds.length) {
173                        System.err.println("Error getting Status for "+pdbIds+" from the PDB website.");
174                        return null;
175                }
176
177
178                for(int pdbNum = 0;pdbNum<pdbIds.length;pdbNum++) {
179                        //Locate first element of attrList with matching structureId.
180                        //attrList is usually short, so don't worry about performance
181                        boolean foundAttr = false;
182                        for( Map<String,String> attrs : attrList) {
183
184                                //Check that the record matches pdbId
185                                String id = attrs.get("structureId");
186                                if(id == null || !id.equalsIgnoreCase(pdbIds[pdbNum])) {
187                                        continue;
188                                }
189
190                                //Check that the status is given
191                                String statusStr = attrs.get("status");
192                                if(statusStr == null ) {
193                                        System.err.println("Error: No status returned for "+pdbIds[pdbNum]);
194                                        statuses[pdbNum] = null;
195                                }
196
197                                Status status = Status.fromString(statusStr);
198                                if(status == null) {
199                                        System.err.println("Error: Unknown status '"+statusStr+"'");
200                                        statuses[pdbNum] = null;
201                                }
202
203                                statuses[pdbNum] = status;
204                                foundAttr = true;
205                        }
206                        if(!foundAttr) {
207                                System.err.println("Error: No result found for "+pdbIds[pdbNum]);
208                                statuses[pdbNum] = null;
209                        }
210                }
211
212                return statuses;
213        }

217         * the first element from
218         * {@link #getReplacement(String,boolean) getReplacement(oldPdbId,true,false)}

224                List<String> replacements =  getReplacement(oldPdbId,true, false);
225                if(replacements != null && replacements.size()>0)
226                        return replacements.get(0);
227                else
228                        return null;

231        /**
232         * Gets the PDB which superseded oldPdbId. For CURRENT IDs, this will
233         * be itself. For obsolete IDs, the behavior depends on the recursion 
234         * parameter. If false, only IDs which directly supersede oldPdbId are
235         * returned. If true, the replacements for obsolete records are recursively 
236         * fetched, yielding a list of all current replacements of oldPdbId.
237         * 
238         * 
239         * 
240         * @param oldPdbId A pdb ID
241         * @param recurse Indicates whether the replacements for obsolete records 
242         *              should be fetched.
243         * @param includeObsolete Indicates whether obsolete records should be
244         *              included in the results.
245         * @return The PDB which replaced oldPdbId. This may be oldPdbId itself, for
246         *              current records. A return value of null indicates that the ID has
247         *              been removed from the PDB or that an error has occurred.
248         */
249        public static List<String> getReplacement(String oldPdbId, boolean recurse, boolean includeObsolete) {
250                List<Map<String,String>> attrList = getStatusIdRecords(new String[] {oldPdbId});
251                //Expect a single record
252                if(attrList == null || attrList.size() != 1) {
253                        System.err.println("Error getting Status for "+oldPdbId+" from the PDB website.");
254                        return null;
255                }
256
257                Map<String,String> attrs = attrList.get(0);
258
259                //Check that the record matches pdbId
260                String id = attrs.get("structureId");
261                if(id == null || !id.equalsIgnoreCase(oldPdbId)) {
262                        System.err.println("Error: Results returned from the query don't match "+oldPdbId);
263                        return null;
264                }
265
266                //Check that the status is given
267                String statusStr = attrs.get("status");
268                if(statusStr == null ) {
269                        System.err.println("Error: No status returned for "+oldPdbId);
270                        return null;
271                }
272
273                Status status = Status.fromString(statusStr);
274                if(status == null ) {
275                        System.err.println("Error: Unknown status '"+statusStr+"'");
276                        return null;
277                }
278
279                // If we're current, just return
280                LinkedList<String> results = new LinkedList<String>();
281                switch(status) {
282                case CURRENT:
283                        results.add(oldPdbId);
284                        return results;
285                case OBSOLETE: {
286                        String replacementStr = attrs.get("replacedBy");
287                        if(replacementStr == null) {
288                                System.err.format("Error: %s is OBSOLETE but lacks a replacedBy attribute.\n",oldPdbId);
289                                return null;
290                        }
291                        replacementStr = replacementStr.toUpperCase();
292                        //include this result
293                        if(includeObsolete) {
294                                results.add(oldPdbId);
295                        }
296                        // Some PDBs are not replaced.
297                        if(replacementStr.equals("NONE")) {
298                                return results; //empty
299                        }
300
301                        String[] replacements = replacementStr.split(" ");
302                        Arrays.sort(replacements, new Comparator<String>() {
303                                public int compare(String o1, String o2) {
304                                        return o2.compareToIgnoreCase(o1);
305                                }
306                        });
307                        for(String replacement : replacements) {
308
309                                // Return the replacement.
310                                if(recurse) {
311                                        List<String> others = PDBStatus.getReplacement(replacement, recurse, includeObsolete);
312                                        mergeReversed(results,others);
313                                }
314                                else {
315                                        if(includeObsolete) {
316                                                mergeReversed(results,Arrays.asList(replacement));
317                                        } else {
318                                                // check status of replacement
319                                                Status replacementStatus = getStatus(replacement);
320                                                switch(replacementStatus) {
321                                                case OBSOLETE:
322                                                        //ignore obsolete
323                                                        break;
324                                                case CURRENT:
325                                                default:
326                                                        // include it
327                                                        mergeReversed(results,Arrays.asList(replacement));
328                                                }
329                                        }
330                                }
331                        }
332
333
334                        return results;
335                }
336                case UNKNOWN:
337                        return null;
338                default: { //TODO handle other cases explicitly. They might have other syntax than "replacedBy"
339                        String replacementStr = attrs.get("replacedBy");
340
341                        if(replacementStr == null) {
342                                // If no "replacedBy" attribute, treat like we're current
343                                // TODO is this correct?
344                                results.add(oldPdbId);
345                                return results;
346                        }
347
348                        replacementStr = replacementStr.toUpperCase();
349                        // Some PDBs are not replaced.
350                        if(replacementStr.equals("NONE")) {
351                                return null;
352                        }
353
354
355                        //include this result, since it's not obsolete
356                        results.add(oldPdbId);
357
358                        String[] replacements = replacementStr.split(" ");
359                        Arrays.sort(replacements, new Comparator<String>() {
360                                public int compare(String o1, String o2) {
361                                        return o2.compareToIgnoreCase(o1);
362                                }
363                        });
364                        for(String replacement : replacements) {
365
366                                // Return the replacement.
367                                if(recurse) {
368                                        List<String> others = PDBStatus.getReplacement(replacement, recurse, includeObsolete);
369                                        mergeReversed(results,others);
370                                }
371                                else {
372                                        mergeReversed(results,Arrays.asList(replacement));
373                                }
374                        }
375
376
377                        return results;
378                }
379                }
380        }

382        /**
383         * Takes two reverse sorted lists of strings and merges the second into the
384         * first. Duplicates are removed.
385         * 
386         * @param merged A reverse sorted list. Modified by this method to contain
387         *              the contents of other.
388         * @param other A reverse sorted list. Not modified.
389         */
390        private static void mergeReversed(List<String> merged,
391                        final List<String> other) {
392
393                if(other.isEmpty())
394                        return;
395
396                if(merged.isEmpty()) {
397                        merged.addAll(other);
398                        return;
399                }
400
401                ListIterator<String> m = merged.listIterator();
402                ListIterator<String> o = other.listIterator();
403
404                String nextM, prevO;
405                prevO = o.next();
406                while(m.hasNext()) {
407                        // peek at m
408                        nextM = m.next();
409                        m.previous();
410
411                        //insert from O until exhausted or occurs after nextM
412                        while(prevO.compareTo(nextM) > 0) {
413                                m.add(prevO);
414                                if(!o.hasNext()) {
415                                        return;
416                                }
417                                prevO = o.next();
418                        }
419                        //remove duplicates
420                        if(prevO.equals(nextM)) {
421                                if(!o.hasNext()) {
422                                        return;
423                                }
424                                prevO = o.next();
425                        }
426
427                        m.next();
428                }
429                m.add(prevO);
430                while(o.hasNext()) {
431                        m.add(o.next());
432                }
433
434        }

437        /**
438         * Get the ID of the protein which was made obsolete by newPdbId.
439         * 
440         * @param newPdbId PDB ID of the newer structure
441         * @param recurse If true, return all ancestors of newPdbId.
442         *              Otherwise, just go one step newer than oldPdbId.
443         * @return A (possibly empty) list of ID(s) of the ancestor(s) of
444         *              newPdbId, or <tt>null</tt> if an error occurred.
445         */
446        public static List<String> getReplaces(String newPdbId, boolean recurse) {
447                List<Map<String,String>> attrList = getStatusIdRecords(new String[] {newPdbId});
448                //Expect a single record
449                if(attrList == null || attrList.size() != 1) {
450                        //TODO Is it possible to have multiple record per ID?
451                        // They seem to be combined into one record with space-delimited 'replaces'
452                        System.err.println("Error getting Status for "+newPdbId+" from the PDB website.");
453                        return null;
454                }
455
456                Map<String,String> attrs = attrList.get(0);
457
458                //Check that the record matches pdbId
459                String id = attrs.get("structureId");
460                if(id == null || !id.equals(newPdbId)) {
461                        System.err.println("Error: Results returned from the query don't match "+newPdbId);
462                        return null;
463                }
464
465
466                String replacedList = attrs.get("replaces"); //space-delimited list
467                if(replacedList == null) {
468                        // no replaces value; assume root
469                        return new ArrayList<String>();
470                }
471                String[] directDescendents = replacedList.split("\\s");
472
473                // Not the root! Return the replaced PDB.
474                if(recurse) {
475                        // Note: Assumes a proper directed acyclic graph of revisions
476                        // Cycles will cause infinite loops.
477                        List<String> allDescendents = new LinkedList<String>();
478                        for(String replaced : directDescendents) {
479                                List<String> roots = PDBStatus.getReplaces(replaced, recurse);
480                                mergeReversed(allDescendents,roots);
481                        }
482                        mergeReversed(allDescendents,Arrays.asList(directDescendents));
483
484                        return allDescendents;
485                } else {
486                        return Arrays.asList(directDescendents);
487                }
488        }

500        /**
501         * Fetches the status of one or more pdbIDs from the server.
502         * 
503         * <p>Returns the results as a list of Attributes.
504         * Each attribute should contain "structureId" and "status" attributes, and
505         * possibly more.
506         * 
507         * <p>Example:</br>
508         * <tt>http://www.rcsb.org/pdb/rest/idStatus?structureID=1HHB,4HHB</tt></br>
509         *<pre>&lt;idStatus&gt;
510         *  &lt;record structureId="1HHB" status="OBSOLETE" replacedBy="4HHB"/&gt;
511         *  &lt;record structureId="4HHB" status="CURRENT" replaces="1HHB"/&gt;
512         *&lt;/idStatus&gt;
513         * </pre>
514         * 
515         * <p>Results are not guaranteed to be returned in the same order as pdbIDs.
516         * Refer to the structureId property to match them.
517         * 
518         * @param pdbIDs
519         * @return A map between attributes and values
520         */
521        private static List<Map<String, String>> getStatusIdRecords(String[] pdbIDs) {
522
523                List<Map<String,String>> result = new ArrayList<Map<String,String>>(pdbIDs.length);
524
525                String serverName = System.getProperty(PDB_SERVER_PROPERTY);
526
527                if ( serverName == null)
528                        serverName = DEFAULT_PDB_SERVER;
529                else 
530                        System.out.format("Got System property %s=%s\n",PDB_SERVER_PROPERTY,serverName);
531
532                // Build REST query URL
533                if(pdbIDs.length < 1) {
534                        throw new IllegalArgumentException("No pdbIDs specified");
535                }
536                String urlStr = String.format("http://%s/pdb/rest/idStatus?structureId=",serverName);
537                for(String pdbId : pdbIDs) {
538                        pdbId = pdbId.toUpperCase();
539                        //check the cache
540                        if (recordsCache.containsKey(pdbId)) {
541                                //System.out.println("Fetching "+pdbId+" from Cache");
542                                result.add( recordsCache.get(pdbId) );
543                        } else {
544                                urlStr += pdbId + ",";
545                        }
546                }
547
548                // check if any ids still need fetching
549                if(urlStr.charAt(urlStr.length()-1) == '=') {
550                        return result;
551                }
552
553                try {
554                        //System.out.println("Fetching " + urlStr);
555
556                        URL url = new URL(urlStr);
557
558                        InputStream uStream = url.openStream();
559
560                        /* // Print file directly
561                        BufferedReader r = new BufferedReader(new InputStreamReader(uStream));
562                        String line = r.readLine();
563                        while(line != null) {
564                                System.out.println(line);
565                                line = r.readLine();
566                        }
567                        r.close();
568
569                        uStream = url.openStream();
570                         */
571
572
573                        InputSource source = new InputSource(uStream);
574                        SAXParserFactory parserFactory = SAXParserFactory.newInstance();
575                        SAXParser parser = parserFactory.newSAXParser();
576                        XMLReader reader = parser.getXMLReader();
577
578                        PDBStatusXMLHandler handler = new PDBStatusXMLHandler();
579
580                        reader.setContentHandler(handler);
581                        reader.parse(source);
582
583                        // Fetch results of SAX parsing
584                        List<Map<String,String>> records = handler.getRecords();
585
586                        //add to cache
587                        for(Map<String,String> record : records) {
588                                String pdbId = record.get("structureId").toUpperCase();
589                                if(pdbId != null) {
590                                        recordsCache.put(pdbId, record);
591                                }
592                        }
593
594                        // return results
595                        result.addAll(handler.getRecords());                    
596                } catch (Exception e){
597                        System.err.println("Problem getting status for " + pdbIDs.toString() + " from PDB server." );
598                        e.printStackTrace();
599                        return null;
600                }
601
602                return result;
603        }

605        /**
606         * Handles idStatus xml by storing attributes for all record elements.
607         * 
608         * @author Spencer Bliven <sbliven@ucsd.edu>
609         *
610         */
611        private static class PDBStatusXMLHandler extends DefaultHandler {
612                private List<Map<String,String>> records;
613
614                public PDBStatusXMLHandler() {
615                        records = new ArrayList<Map<String,String>>();
616                }
617
618                /**
619                 * @param uri
620                 * @param localName
621                 * @param qName
622                 * @param attributes
623                 * @throws SAXException
624                 * @see org.xml.sax.helpers.DefaultHandler#startElement(java.lang.String, java.lang.String, java.lang.String, org.xml.sax.Attributes)
625                 */
626                @Override
627                public void startElement(String uri, String localName, String qName,
628                                Attributes attributes) throws SAXException {
629                        //System.out.format("Starting element: uri='%s' localName='%s' qName='%s'\n", uri, localName, qName);
630                        if(qName.equals("record")) {
631                                //Convert attributes into a Map, as it should have been.
632                                //Important since SAX reuses Attributes objects for different calls
633                                Map<String,String> attrMap = new HashMap<String,String>(attributes.getLength()*2);
634                                for(int i=0;i<attributes.getLength();i++) {
635                                        attrMap.put(attributes.getQName(i), attributes.getValue(i));
636                                }
637                                records.add(attrMap);
638                        }
639                }
640
641
642                /**
643                 * @param e
644                 * @throws SAXException
645                 * @see org.xml.sax.helpers.DefaultHandler#error(org.xml.sax.SAXParseException)
646                 */
647                @Override
648                public void error(SAXParseException e) throws SAXException {
649                        System.err.println(e.getMessage());
650                        super.error(e);
651                }
652
653
654                public List<Map<String, String>> getRecords() {
655                        return records;
656                }
657        }

659        /** Returns a list of current PDB IDs
660         * 
661         * @return a list of PDB IDs, or null if a problem occurred
662         */
663
664        public static SortedSet<String> getCurrentPDBIds(){
665
666                SortedSet<String> allPDBs = new TreeSet<String>();
667                String serverName = System.getProperty(PDB_SERVER_PROPERTY);
668
669                if ( serverName == null)
670                        serverName = DEFAULT_PDB_SERVER;
671                else 
672                        System.out.format("Got System property %s=%s\n",PDB_SERVER_PROPERTY,serverName);
673
674                // Build REST query URL
675
676                String urlStr = String.format("http://%s/pdb/rest/getCurrent",serverName);
677                try {
678                        URL u = new URL(urlStr);
679
680                        InputStream stream = HTTPConnectionTools.getInputStream(u, 60000);
681
682                        if (stream != null) {
683                                BufferedReader reader = new BufferedReader(
684                                                new InputStreamReader(stream));
685
686                                String line = null;
687
688                                while ((line = reader.readLine()) != null) {
689                                        int index = line.lastIndexOf("structureId=");
690                                        if (index > 0) {
691                                                allPDBs.add(line.substring(index + 13, index + 17));
692                                        }
693                                }
694                        }
695                } catch (Exception e){
696                        e.printStackTrace();
697                        return null;
698                }
699                return allPDBs;
700
701        }
