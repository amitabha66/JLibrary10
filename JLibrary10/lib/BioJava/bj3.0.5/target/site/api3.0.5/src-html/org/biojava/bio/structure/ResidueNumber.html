025package org.biojava.bio.structure;
026
027import java.io.Serializable;
028import java.io.StringWriter;
029
030/** Everything that is needed to uniquely describe a residue position
031 * 
032 * @author Andreas Prlic
033 *
034 */
035public class ResidueNumber implements Serializable, Comparable<ResidueNumber>
036{
037
038        private static final long serialVersionUID = 1773011704758536083L;
039        private String chainId;
040        private Character insCode;
041        private Integer seqNum;
042
043        public ResidueNumber() {
044        }
045
046        public ResidueNumber(String chainId, Integer residueNumber, Character insCode) {
047                this.chainId = chainId;
048                this.seqNum = residueNumber;
049                this.insCode = insCode;
050        }
051
052        public String getChainId()
053        {
054                return chainId;
055        }
056        public void setChainId(String chainId)
057        {
058                this.chainId = chainId;
059        }
060        public Character getInsCode()
061        {
062                return insCode;
063        }
064        public void setInsCode(Character insCode)
065        {
066                this.insCode = insCode;
067        }
068        public Integer getSeqNum()
069        {
070                return seqNum;
071        }
072        public void setSeqNum(Integer seqNum)
073        {
074                this.seqNum = seqNum;
075        }
076
077        
078
079        
080
081        @Override
082        public boolean equals(Object obj) {
083                if (this == obj)
084                        return true;
085                if (obj == null)
086                        return false;
087                if (getClass() != obj.getClass())
088                        return false;
089                ResidueNumber other = (ResidueNumber) obj;
090                if (chainId == null) {
091                        if (other.chainId != null)
092                                return false;
093                } else if (!chainId.equals(other.chainId))
094                        return false;
095                if (insCode == null) {
096                        if (other.insCode != null)
097                                return false;
098                } else if (!insCode.equals(other.insCode))
099                        return false;
100                if (seqNum == null) {
101                        if (other.seqNum != null)
102                                return false;
103                } else if (!seqNum.equals(other.seqNum))
104                        return false;
105                
106                return true;
107        }
108        
109        @Override
110        public int hashCode() {
111                final int prime = 31;
112                int result = 1;
113                result = prime * result + ((chainId == null) ? 0 : chainId.hashCode());
114                result = prime * result + ((insCode == null) ? 0 : insCode.hashCode());
115                result = prime * result + ((seqNum == null) ? 0 : seqNum.hashCode());
116                return result;
117        }
118
119        /**
120         * @return The residue number and insertion code as a string, eg "74A"
121         * @see java.lang.Object#toString()
122         */
123        @Override
124        public String toString() {
125
126                StringWriter writer = new StringWriter();
127                //         if ( chainId != null){
128                        //                 writer.append(chainId);
129                        //                 writer.append(":");
130                        //         }
131        writer.append(String.valueOf(seqNum));
132                if (  insCode != null && ( insCode != ' '))
133                        writer.append(insCode);
134
135                return writer.toString();
136        }
137
138        /**
139         * @return The chain, number, and insertion code as a string, eg "B  74A" or "A    1 "
140         */
141        public String toPDB() {
142                String insCodeS ;
143                if ( insCode != null)
144                        insCodeS = insCode+"";
145                else insCodeS = " ";
146                return String.format("%s%4d%-2s", chainId, seqNum, insCodeS);
147        }
148
149        
150        /** Convert a string representation of a residue number to a residue number object.
151         * The string representation can be a integer followed by a character.
152         * 
153         * @param pdb_code
154         * @return a ResidueNumber object
155         */
156        public static ResidueNumber fromString(String pdb_code) {
157                ResidueNumber residueNumber = new ResidueNumber();
158                Integer resNum = null;
159                String icode = null;            
160                
161                try {
162                        resNum = Integer.parseInt(pdb_code);
163                } catch ( NumberFormatException e){
164                        // there is an insertion code..
165                
166                        // Split at any position that's either:
167                        // preceded by a digit and followed by a non-digit, or
168                        // preceded by a non-digit and followed by a digit.
169                        String[] spl = pdb_code.split("(?<=\\d)(?=\\D)|(?<=\\D)(?=\\d)");
170                        if ( spl.length == 2){
171                                resNum = Integer.parseInt(spl[0]);
172                                icode = spl[1];
173                        }
174                        
175                }
176                                
177                residueNumber.setSeqNum(resNum);
178                if ( icode == null)
179                        residueNumber.setInsCode(null);
180                else if ( icode.length() > 0)
181                        residueNumber.setInsCode(icode.charAt(0));
182                return residueNumber;
183        }
184
185
186        public int compareTo(ResidueNumber other) {
187                
188                // chain id
189                if (chainId != null && other.chainId != null) {
190                        if (!chainId.equals(other.chainId)) return chainId.compareTo(other.chainId);
191                }
192                if (chainId != null && other.chainId == null) {
193                        return 1;
194                } else if (chainId == null && other.chainId != null) {
195                        return -1;
196                }
197                
198                // sequence number
199                if (seqNum != null && other.seqNum != null) {
200                        if (!seqNum.equals(other.seqNum)) return seqNum.compareTo(other.seqNum);
201                }
202                if (seqNum != null && other.seqNum == null) {
203                        return 1;
204                } else if (seqNum == null && other.seqNum != null) {
205                        return -1;
206                }
207                
208                // insertion code
209                if (insCode != null && other.insCode != null) {
210                        if (!insCode.equals(other.insCode)) return insCode.compareTo(other.insCode);
211                }
212                if (insCode != null && other.insCode == null) {
213                        return 1;
214                } else if (insCode == null && other.insCode != null) {
215                        return -1;
216                }
217                
218                return 0;
219        }
