304        private void buildAligMap(){
305                apos.clear();
306
307                int gap   = 0;
308                int gpos1 = 0;
309                int gpos2 = 0;
310
311                for (int pos = 0 ; pos < idx1.length ; pos ++){
312
313                        int p1 = idx1[pos];
314                        int p2 = idx2[pos];
315
316                        int end = Math.max(p1,p2);
317
318                        //System.out.println("p1: " + p1 + " p2: " + p2 );
319                        // fill up  gaps...
320                        for (;gap<end;gap++){
321
322                                AlignedPosition m = new AlignedPosition();
323                                if ( gpos1 < p1){
324                                        m.setPos1(gpos1);
325
326                                        gpos1++;
327                                }
328                                if ( gpos2 < p2){
329                                        m.setPos2(gpos2);
330
331                                        gpos2++;
332                                }
333                                m.setEquivalent(AlignedPosition.NOT_ALIGNED);
334
335                                //System.out.println(m + " => " + end);
336                                apos.add(m);                            
337                        }
338
339                        // add this aligned position
340                        AlignedPosition m = new AlignedPosition();
341
342                        m.setPos1(p1);
343                        m.setPos2(p2);
344                        m.setEquivalent(AlignedPosition.EQUIVALENT);
345
346                        //System.out.println(m);
347                        apos.add(m);
348                        gpos1++;
349                        gpos2++;
350                        gap++;
351
352                }
353
354                //System.out.println(apos);
355
356        }

359        private void setAtoms(Structure s, SequenceScalePanel panel){
360                if ( structurePairAligner == null){
361                        System.err.println("StructurePairAligner has not been set");
362                        return;
363                }
364                Atom[] ca1 = structurePairAligner.getAlignmentAtoms(s);
365                Chain c = new ChainImpl();
366                c.setChainID("1");
367                for (Atom atom : ca1) {
368                        
369                        Group g = atom.getGroup();
370                        
371                        Chain parentChain = g.getChain();
372                        
373                        c.addGroup(g);
374                        // hack for Jmol?                       
375                        g.setChain(parentChain);
376                }
377                panel.setChain(c);
378
379        }

463        private int getMaxSequenceLength(){
464                int l1 = panel1.getChain().getAtomGroups("amino").size();
465                int l2 = panel2.getChain().getAtomGroups("amino").size();
466                if ( l1 > l2)
467                        return l1;
468                else return l2;
469        }

473        /** calculate the float that is used for display.
474         * 1 * scale = size of 1 amino acid (in pixel).
475         * maximum @see MAX_SCALE
476         * @param zoomFactor
477         * @return a float that is the display "scale" - an internal value required for paintin.
478         * user should only interact with the zoomfactor ...
479         */
480        private float getScaleForZoom(int zoomFactor){
481
482                if ( zoomFactor > 100)
483                        zoomFactor = 100;
484                if ( zoomFactor < 1)
485                        zoomFactor = 1;
486
487
488                int DEFAULT_X_START        
489                = SequenceScalePanel.DEFAULT_X_START;
490                int DEFAULT_X_RIGHT_BORDER = SequenceScalePanel.DEFAULT_X_RIGHT_BORDER;
491
492                int seqLength = getMaxSequenceLength();
493                // the maximum width depends on the size of the parent Component
494
495                int width=getWidth();
496
497
498                float s = width / (float) ( seqLength + DEFAULT_X_START + DEFAULT_X_RIGHT_BORDER );
499                //logger.info("scale for 100% " + s + " " + seqLength + " " + zoomFactor);
500
501                s = (100) * s / ( zoomFactor * 1.0f) ;
502
503                if ( s > MAX_SCALE)
504                        s = MAX_SCALE;
505
506                //logger.info("but changed to " + s);
507                return s;
508        }
509
510        /** a value of 100 means that the whole sequence should be displayed in the current visible window
511         * a factor of 1 means that one amino acid shoud be drawn as big as possible   
512         *  
513         * @param zoomFactor - a value between 1 and 100
514         *
515         *  
516         */
517        public void calcScale(int zoomFactor){
518
519                float s = getScaleForZoom(zoomFactor);
520                scale = s;
521                //logger.info("calc scale zoom:"+zoomFactor+ " s: " + s);
522                panel1.setScale(s);
523                panel2.setScale(s);
524                panel1.repaint();
525                panel2.repaint();
526
527
528                //return scale;
529
530        }
