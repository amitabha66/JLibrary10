044/**
045 * This class wraps a QBlast search request parameter {@code Map} by adding several convenient parameter addition
046 * methods. Other QBlast URL API parameters should be added using
047 * {@link #setAlignmentOption(BlastAlignmentParameterEnum, String)}
048 * <p/>
049 * Required parameters are {@code PROGRAM} and {@code DATABASE}, other parameters are optional
050 * 
051 * @author Sylvain Foisy, Diploide BioIT
052 * @author Gediminas Rimsa
053 */
054public class NCBIQBlastAlignmentProperties implements RemotePairwiseAlignmentProperties {
055        private static final long serialVersionUID = 7158270364392309841L;
056
057        private Map<BlastAlignmentParameterEnum, String> param = new HashMap<BlastAlignmentParameterEnum, String>();
058
059        /**
060         * This method forwards to {@link #getAlignmentOption(BlastAlignmentParameterEnum)}. Consider using it instead.
061         */
062        @Override
063        public String getAlignmentOption(String key) {
064                return getAlignmentOption(BlastAlignmentParameterEnum.valueOf(key));
065        }
066
067        /**
068         * This method forwards to {@link #setAlignmentOption(BlastAlignmentParameterEnum, String)}. Consider using it
069         * instead.
070         */
071        @Override
072        public void setAlignementOption(String key, String val) {
073                setAlignmentOption(BlastAlignmentParameterEnum.valueOf(key), val);
074        }
075
076        /**
077         * Gets parameters, which are currently set
078         */
079        public Set<String> getAlignmentOptions() {
080                Set<String> result = new HashSet<String>();
081                for (BlastAlignmentParameterEnum parameter : param.keySet()) {
082                        result.add(parameter.name());
083                }
084                return result;
085        }
086
087        /**
088         * Gets the value of specified parameter or {@code null} if it is not set.
089         */
090        public String getAlignmentOption(BlastAlignmentParameterEnum key) {
091                return param.get(key);
092        }
093
094        /**
095         * Sets the value of specified parameter
096         */
097        public void setAlignmentOption(BlastAlignmentParameterEnum key, String val) {
098                param.put(key, val);
099        }
100
101        /**
102         * Removes given parameter
103         */
104        public void removeAlignmentOption(BlastAlignmentParameterEnum key) {
105                param.remove(key);
106        }
107
108        /**
109         * @return {@linkplain BlastProgramEnum} used for blast run
110         */
111        public BlastProgramEnum getBlastProgram() {
112                BlastProgramEnum program = BlastProgramEnum.valueOf(getAlignmentOption(PROGRAM));
113                boolean isMegablast = BlastProgramEnum.blastn == program && getAlignmentOption(MEGABLAST).equals("on");
114                return !isMegablast ? program : BlastProgramEnum.megablast;
115        }
116
117        /**
118         * Sets the program to be used with blastall
119         * 
120         * @param program : one of blastall programs
121         */
122        public void setBlastProgram(BlastProgramEnum program) {
123                if (BlastProgramEnum.megablast != program) {
124                        setAlignmentOption(PROGRAM, program.name());
125                        removeAlignmentOption(MEGABLAST);
126                } else {
127                        setAlignmentOption(PROGRAM, BlastProgramEnum.blastn.name());
128                        setAlignmentOption(MEGABLAST, "on");
129                }
130        }

132        /**
133         * @return name of database used with blastall
134         */
135        public String getBlastDatabase() {
136                return getAlignmentOption(DATABASE);
137        }

158        /**
159         * @return double value of EXPECT parameter used for blast run
160         */
161        public double getBlastExpect() {
162                if (param.containsKey(EXPECT)) {
163                        return Double.parseDouble(getAlignmentOption(EXPECT));
164                }
165                return 10;
166        }
167
168        /**
169         * Sets the EXPECT parameter to be use with blastall
170         * <p>
171         * Example: if you want a EXPECT of 1e-10, pass {@code Double.parseDouble("1e-10")} as a parameter
172         * <p>
173         * Blastall equivalent: -e
174         * 
175         * @param expect: a double value of EXPECT parameter
176         */
177        public void setBlastExpect(double expect) {
178                setAlignmentOption(EXPECT, Double.toString(expect));
179        }
180
181        /**
182         * Returns the value of the WORD_SIZE parameter used for this blast run
183         * 
184         * @return int value of WORD_SIZE used by this search
185         * @throws IllegalArgumentException when program type is not set and program type is not supported
186         */
187        public int getBlastWordSize() {
188                if (param.containsKey(WORD_SIZE)) {
189                        return Integer.parseInt(getAlignmentOption(WORD_SIZE));
190                }
191
192                // return default word size value
193                try {
194                        BlastProgramEnum programType = getBlastProgram();
195                        switch (programType) {
196                        case blastn:
197                                return 11;
198                        case megablast:
199                                return 28;
200                        case blastp:
201                        case blastx:
202                        case tblastn:
203                        case tblastx:
204                                return 3;
205                        default:
206                                throw new UnsupportedOperationException("Blast program " + programType.name() + " is not supported.");
207                        }
208                } catch (IllegalArgumentException e) {
209                        throw new IllegalArgumentException("Blast program " + getBlastProgram() + " is not supported.", e);
210                }
211        }
212
213        /**
214         * Sets the WORD_SIZE parameter to be use with blastall
215         * <p>
216         * <b>WARNING!!</b> At this point, the method does not verify the validity of your choice; for example, word size of
217         * greater than 5 with blastp returns error messages from QBlast. Word size range depends on the algorithm chosen.
218         * <p>
219         * More at http://www.ncbi.nlm.nih.gov/staff/tao/URLAPI/new/node74.html
220         * <p>
221         * Blastall equivalent: -W
222         * 
223         * @param word: an int used to set WORD_SIZE
224         */
225        public void setBlastWordSize(int word) {
226                setAlignmentOption(WORD_SIZE, Integer.toString(word));
227        }
228
229        /**
230         * Returns the value for the GAP_CREATION parameter (first half of GAPCOSTS parameter)
231         * 
232         * @return an integer value for gap creation used by this search, -1 if not set or not a number
233         */
234        public int getBlastGapCreation() {
235                String gapCosts = getAlignmentOption(GAPCOSTS);
236                try {
237                        String gapCreation = gapCosts.split("\\+")[0];
238                        return Integer.parseInt(gapCreation);
239                } catch (Exception e) {
240                        return -1;
241                }
242        }
243
244        /**
245         * Returns the value for the gap extension parameter (second half of GAPCOSTS parameter)
246         * 
247         * @return an integer for the value for gap extension used by this search, -1 if not set or not a number
248         */
249        public int getBlastGapExtension() {
250                String gapCosts = getAlignmentOption(GAPCOSTS);
251                try {
252                        String gapExtension = gapCosts.split("\\+")[1];
253                        return Integer.parseInt(gapExtension);
254                } catch (Exception e) {
255                        return -1;
256                }
257        }
258
259        /**
260         * Returns the actual string for the GAPCOSTS parameter which is used to build the URL
261         * 
262         * @return the string representation of the GAPCOSTS parameter formatted for the URL
263         */
264        public String getBlastGapCosts() {
265                return getAlignmentOption(GAPCOSTS);
266        }
267
268        /**
269         * Sets the GAPCOSTS parameter
270         * 
271         * @param gapCreation integer to use as gap creation value
272         * @param gapExtension integer to use as gap extension value
273         */
274        public void setBlastGapCosts(int gapCreation, int gapExtension) {
275                String gc = Integer.toString(gapCreation);
276                String ge = Integer.toString(gapExtension);
277                setAlignmentOption(GAPCOSTS, gc + "+" + ge);
278        }

286                return getAlignmentOption(MATRIX_NAME);
287        }
288
289        /**
290         * Sets the value for the MATRIX parameter to use for blastall
291         * <p>
292         * Blastall equivalent: -M
293         * 
294         * @param matrix : a String to use as gap creation value
295         * @see BlastMatrixEnum
296         */
297        public void setBlastMatrix(BlastMatrixEnum matrix) {
298                setAlignmentOption(MATRIX_NAME, matrix.name());
299
300                boolean gapCostsSet = getBlastGapCreation() != -1 || getBlastGapExtension() != -1;
301
302                if (!gapCostsSet) {
303                        /*
304                         * Setting default values for -G/-E if no other values have been set is necessary because, since BLOSUM62 is
305                         * default, the expected values are -G 11 -E 1. If your matrix choice is different, the request will fail,
306                         * implicitly expecting GAPCOSTS=11+1
307                         */
308                        switch (matrix) {
309                        case PAM30:
310                                setBlastGapCosts(9, 1);
311                                break;
312                        case PAM70:
313                                setBlastGapCosts(10, 1);
314                                break;
315                        case PAM250:
316                                setBlastGapCosts(14, 2);
317                                break;
318                        case BLOSUM45:
319                                setBlastGapCosts(15, 2);
320                                break;
321                        case BLOSUM50:
322                                setBlastGapCosts(13, 2);
323                                break;
324                        case BLOSUM80:
325                        case BLOSUM90:
326                                setBlastGapCosts(10, 1);
327                                break;
328                        }
329                }
330        }

332        /**
333         * Sets the QUERY_FROM and QUERY_TO parameters to be use by blast. Do not use if you want to use the whole sequence.<br/>
334         * Blastall equivalent: -L
335         * 
336         * @param start QUERY_FROM parameter
337         * @param end QUERY_TO parameter
338         */
339        public void setBlastFromToPosition(int start, int end) {
340                if (start >= end) {
341                        throw new IllegalArgumentException("Start index must be less than end index");
342                }
343                setAlignmentOption(QUERY_FROM, String.valueOf(start));
344                setAlignmentOption(QUERY_TO, String.valueOf(end));
345        }
346
347        /**
348         * @return an integer value for the QUERY_FROM parameter
349         * @see #setBlastFromToPosition(int, int)
350         */
351        public int getBlastFromPosition() {
352                return Integer.parseInt(getAlignmentOption(QUERY_FROM));
353        }

356         * @return QUERY_TO parameter
357         * @see #setBlastFromToPosition(int, int)
358         */
359        public int getBlastToPosition() {
360                return Integer.parseInt(getAlignmentOption(QUERY_TO));
361        }

363        /**
364         * This method is to be used if a request is to use non-default values at submission. Useful for the following
365         * blastall parameters:
366         * <ul>
367         * <li>-r: integer to reward for match. Default = 1</li>
368         * <li>-q: negative integer for penalty to allow mismatch. Default = -3</li>
369         * <li>-y: dropoff for blast extensions in bits, using default if not specified. Default = 20 for blastn, 7 for all
370         * others (except megablast for which it is not applicable).</li>
371         * <li>-X: X dropoff value for gapped alignment, in bits. Default = 30 for blastn/megablast, 15 for all others.</li>
372         * <li>-Z: final X dropoff value for gapped alignement, in bits. Default = 50 for blastn, 25 for all others (except
373         * megablast for which it is not applicable)</li>
374         * <li>-P: equals 0 for multiple hits 1-pass, 1 for single hit 1-pass. Does not apply to blastn ou megablast.</li>
375         * <li>-A: multiple hits window size. Default = 0 (for single hit algorithm)</li>
376         * <li>-I: number of database sequences to save hits for. Default = 500</li>
377         * <li>-Y: effective length of the search space. Default = 0 (0 represents using the whole space)</li>
378         * <li>-z: a real specifying the effective length of the database to use. Default = 0 (0 represents the real size)</li>
379         * <li>-c: an integer representing pseudocount constant for PSI-BLAST. Default = 7</li>
380         * <li>-F: any filtering directive</li>
381         * </ul>
382         * <p>
383         * WARNING!! This method is still very much in flux and might not work as expected...
384         * </p>
385         * <p>
386         * You have to be aware that at no moment is there any error checking on the use of these parameters by this class.
387         * </p>
388         * 
389         * @param advancedOptions : a String with any number of optional parameters with an associated value.
390         */
391        public void setBlastAdvancedOptions(String advancedOptions) {
392                // Escaping white spaces with + char to comply with QBlast specifications
393                setAlignmentOption(OTHER_ADVANCED, advancedOptions.replaceAll(" ", "+"));
394        }
395
396        /**
397         * @return the String with the advanced options
398         */
399        public String getBlastAdvancedOptions() {
400                return getAlignmentOption(OTHER_ADVANCED);
401        }
