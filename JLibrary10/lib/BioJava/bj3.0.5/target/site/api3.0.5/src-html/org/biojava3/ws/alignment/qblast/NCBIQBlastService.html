050/**
051 * Provides a simple way of submitting BLAST request to the QBlast service at NCBI.
052 * <p>
053 * NCBI provides a Blast server through a CGI-BIN interface. This service simply encapsulates an access to it by giving
054 * users access to get/set methods to fix sequence, program and database as well as advanced options.
055 * </p>
056 * <p>
057 * The philosophy behind this service is to disconnect submission of Blast requests from collection of Blast results.
058 * This is done so to allow a user to submit multiple Blast requests while allowing recovery of the reports at a later
059 * time.
060 * </p>
061 * <p>
062 * Presently, only blastall programs are accessible.
063 * </p>
064 * 
065 * @author Sylvain Foisy, Diploide BioIT
066 * @author Gediminas Rimsa
067 */
068public class NCBIQBlastService implements RemotePairwiseAlignmentService {
069        /**
070         * Number of milliseconds by which expected job execution time is incremented if it is not finished yet. Subsequent
071         * calls to {@link #isReady(String, long)} method will return false until at least this much time passes.
072         */
073        public static final long WAIT_INCREMENT = 3000;
074
075        private static final MapToStringTransformer MAP_TO_STRING_TRANSFORMER = new MapToStringTransformer();
076        private static final String SERVICE_URL = "http://www.ncbi.nlm.nih.gov/blast/Blast.cgi";
077        private static final String DEFAULT_EMAIL = "anonymous@biojava.org";
078        private static final String DEFAULT_TOOL = "biojava3";
079
080        private URL serviceUrl;
081        private String email = DEFAULT_EMAIL;
082        private String tool = DEFAULT_TOOL;
083
084        private Map<String, BlastJob> jobs = new HashMap<String, BlastJob>();
085
086        public NCBIQBlastService() {
087                try {
088                        serviceUrl = new URL(SERVICE_URL);
089                } catch (MalformedURLException e) {
090                        throw new RuntimeException("It looks like the URL for NCBI QBlast service is wrong. Cause: " + e.getMessage(), e);
091                }
092        }
093
094        /**
095         * A simple method to check the availability of the QBlast service. Sends {@code Info} command to QBlast
096         * 
097         * @return QBlast info output concatenated to String
098         * @throws Exception if unable to connect to the NCBI QBlast service
099         */
100        public String getRemoteBlastInfo() throws Exception {
101                OutputStreamWriter writer = null;
102                BufferedReader reader = null;
103                try {
104                        URLConnection serviceConnection = setQBlastServiceProperties(serviceUrl.openConnection());
105                        writer = new OutputStreamWriter(serviceConnection.getOutputStream());
106                        writer.write("CMD=Info");
107                        writer.flush();
108                        reader = new BufferedReader(new InputStreamReader(serviceConnection.getInputStream()));
109                        StringBuilder sb = new StringBuilder();
110                        String line;
111                        while ((line = reader.readLine()) != null) {
112                                sb.append(line);
113                                sb.append(System.getProperty("line.separator"));
114                        }
115                        return sb.toString();
116                } catch (IOException e) {
117                        throw new Exception("Impossible to get info from QBlast service at this time. Cause: " + e.getMessage(), e);
118                } finally {
119                        IOUtils.close(reader);
120                        IOUtils.close(writer);
121                }
122        }

141        /**
142         * Sends the Blast request via the Put command of the CGI-BIN interface. Uses all of the parameters specified in
143         * {@code alignmentProperties} (parameters PROGRAM and DATABASE are required).
144         * 
145         * @param query : a {@code String} representing a sequence or Genbank ID
146         * @param alignmentProperties : a {@code RemotePairwiseAlignmentProperties} object representing alignment properties
147         * @return the request id for this sequence, necessary to fetch results after completion
148         * @throws Exception if unable to connect to the NCBI QBlast service or if no sequence or required parameters
149         *             PROGRAM and DATABASE are not set
150         */
151        @Override
152        public String sendAlignmentRequest(String query, RemotePairwiseAlignmentProperties alignmentProperties) throws Exception {
153                Map<String, String> params = new HashMap<String, String>();
154                for (String key : alignmentProperties.getAlignmentOptions()) {
155                        params.put(key, alignmentProperties.getAlignmentOption(key));
156                }
157
158                if (query == null || query.isEmpty()) {
159                        throw new IllegalArgumentException("Impossible to execute QBlast request. The sequence has not been set.");
160                }
161                if (!params.containsKey(PROGRAM.name())) {
162                        throw new IllegalArgumentException("Impossible to execute QBlast request. Parameter PROGRAM has not been set.");
163                }
164                if (!params.containsKey(DATABASE.name())) {
165                        throw new IllegalArgumentException("Impossible to execute QBlast request. Parameter DATABASE has not been set.");
166                }
167
168                params.put(CMD.name(), "Put");
169                params.put(QUERY.name(), query);
170                params.put(TOOL.name(), getTool());
171                params.put(EMAIL.name(), getEmail());
172
173                String putCmd = MAP_TO_STRING_TRANSFORMER.transform(params);
174
175                OutputStreamWriter writer = null;
176                BufferedReader reader = null;
177                try {
178                        URLConnection serviceConnection = setQBlastServiceProperties(serviceUrl.openConnection());
179                        writer = new OutputStreamWriter(serviceConnection.getOutputStream());
180                        writer.write(putCmd);
181                        writer.flush();
182
183                        // Get the response
184                        reader = new BufferedReader(new InputStreamReader(serviceConnection.getInputStream()));
185
186                        // find request id and time of execution
187                        BlastJob job = new BlastJob();
188                        String line;
189                        while ((line = reader.readLine()) != null) {
190                                if (!line.contains("class=\"error\"") && !line.contains("Message ID#")) {
191                                        // if there is no error, capture RID and RTOE
192                                        if (line.contains("RID = ")) {
193                                                String[] arr = line.split("=");
194                                                job.setId(arr[1].trim());
195                                        } else if (line.contains("RTOE = ")) {
196                                                String[] arr = line.split("=");
197                                                job.setStartTimestamp(System.currentTimeMillis());
198                                                job.setExpectedExecutionTime(Long.parseLong(arr[1].trim()) * 1000);
199                                        }
200                                        jobs.put(job.getId(), job);
201                                } else {
202                                        // handle QBlast error message
203
204                                        // Capture everything to the left of this HTML statement...
205                                        String[] tmp = line.split("</p></li></ul>");
206
207                                        // Only the error message is on the right side of this...
208                                        String[] moreTmp = tmp[0].split("<p class=\"error\">");
209                                        throw new Exception("NCBI QBlast refused this request because: " + moreTmp[1].trim());
210                                }
211
212                        }
213                        if (job != null && job.getId() != null) {
214                                return job.getId();
215                        } else {
216                                throw new Exception("Unable to retrieve request ID");
217                        }
218                } catch (IOException e) {
219                        throw new IOException("An error occured submiting sequence to BLAST server. Cause: " + e.getMessage(), e);
220                } finally {
221                        IOUtils.close(reader);
222                        IOUtils.close(writer);
223                }
224        }

226        /**
227         * Wrapper method for {@link #isReady(String, long)}, omitting unnecessary {@code present} property.
228         * 
229         * @see #isReady(String, long)
230         */
231        public boolean isReady(String id) throws Exception {
232                return isReady(id, 0);
233        }
234
235        /**
236         * Checks for completion of request.
237         * <p/>
238         * If expected execution time (RTOE) is available for request, this method will always return false until that time
239         * passes. This is done to prevent sending unnecessary requests to the server.
240         * 
241         * @param id : request id, which was returned by {@code sendAlignmentRequest} method
242         * @param present : is not used, can be any value
243         * @return a boolean value telling if the request has been completed
244         */
245        @Override
246        public boolean isReady(String id, long present) throws Exception {
247                BlastJob job = jobs.get(id);
248                if (job != null) {
249                        long expectedJobFinishTime = job.getStartTimestamp() + job.getExpectedExecutionTime();
250                        if (System.currentTimeMillis() < expectedJobFinishTime) {
251                                return false;
252                        }
253                } else {
254                        // it might be a valid job from another session
255                        job = new BlastJob();
256                        job.setId(id);
257                        job.setStartTimestamp(System.currentTimeMillis());
258                        job.setExpectedExecutionTime(0);
259                }
260
261                OutputStreamWriter writer = null;
262                BufferedReader reader = null;
263                try {
264                        String checkRequest = "CMD=Get&RID=" + job.getId();
265                        URLConnection serviceConnection = setQBlastServiceProperties(serviceUrl.openConnection());
266                        writer = new OutputStreamWriter(serviceConnection.getOutputStream());
267                        writer.write(checkRequest);
268                        writer.flush();
269                        reader = new BufferedReader(new InputStreamReader(serviceConnection.getInputStream()));
270
271                        String line;
272                        while ((line = reader.readLine()) != null) {
273                                if (line.contains("READY")) {
274                                        jobs.put(job.getId(), job);
275                                        return true;
276                                } else if (line.contains("WAITING")) {
277                                        job.setExpectedExecutionTime(job.getExpectedExecutionTime() + WAIT_INCREMENT);
278                                        jobs.put(job.getId(), job);
279                                        return false;
280                                } else if (line.contains("UNKNOWN")) {
281                                        throw new IllegalArgumentException("Unknown request id - no results exist for it. Given id = " + id);
282                                }
283                        }
284                        return false;
285                } catch (IOException ioe) {
286                        throw new Exception("It is not possible to fetch Blast report from NCBI at this time. Cause: " + ioe.getMessage(), ioe);
287                } finally {
288                        IOUtils.close(reader);
289                        IOUtils.close(writer);
290                }
291        }

293        /**
294         * Extracts the actual Blast report for given request id according to options provided in {@code outputProperties}
295         * argument.
296         * <p/>
297         * If the results are not ready yet, sleeps until they are available. If sleeping is not desired, call this method
298         * after {@code isReady} returns true
299         * 
300         * @param id : request id, which was returned by {@code sendAlignmentRequest} method
301         * @param outputProperties : an object specifying output formatting options
302         * @return an {@code InputStream} of results
303         * @throws Exception if it is not possible to recover the results
304         */
305        public InputStream getAlignmentResults(String id, RemotePairwiseAlignmentOutputProperties outputProperties) throws Exception {
306                Map<String, String> params = new HashMap<String, String>();
307                for (String key : outputProperties.getOutputOptions()) {
308                        params.put(key, outputProperties.getOutputOption(key));
309                }
310                OutputStreamWriter writer = null;
311
312                while (!isReady(id)) {
313                        Thread.sleep(WAIT_INCREMENT + 100);
314                }
315
316                params.put(CMD.name(), "Get");
317                params.put(RID.name(), id);
318                params.put(TOOL.name(), getTool());
319                params.put(EMAIL.name(), getEmail());
320                String getCmd = MAP_TO_STRING_TRANSFORMER.transform(params);
321
322                try {
323                        URLConnection serviceConnection = setQBlastServiceProperties(serviceUrl.openConnection());
324                        writer = new OutputStreamWriter(serviceConnection.getOutputStream());
325                        writer.write(getCmd);
326                        writer.flush();
327                        return serviceConnection.getInputStream();
328                } catch (IOException ioe) {
329                        throw new Exception("It is not possible to fetch Blast report from NCBI at this time. Cause: " + ioe.getMessage(), ioe);
330                } finally {
331                        IOUtils.close(writer);
332                }
333        }

335        /**
336         * Sends a delete request for given request id. Optional operation, ignores IOExceptions.<br/>
337         * Can be used after results of given search are no longer needed to be kept on Blast server
338         * 
339         * @param id request id, as returned by {@code sendAlignmentRequest} method
340         */
341        public void sendDeleteRequest(String id) {
342                OutputStreamWriter writer = null;
343                try {
344                        String deleteRequest = "CMD=Delete&RID=" + id;
345                        URLConnection serviceConnection = setQBlastServiceProperties(serviceUrl.openConnection());
346                        writer = new OutputStreamWriter(serviceConnection.getOutputStream());
347                        writer.write(deleteRequest);
348                        writer.flush();
349                } catch (IOException ignore) {
350                        // ignore it this is an optional operation
351                } finally {
352                        IOUtils.close(writer);
353                }
354        }

356        /**
357         * Sets properties for given URLConnection
358         * 
359         * @param conn URLConnection to set properties for
360         * @return given object after setting properties
361         */
362        private URLConnection setQBlastServiceProperties(URLConnection conn) {
363                conn.setDoOutput(true);
364                conn.setUseCaches(false);
365                conn.setRequestProperty("User-Agent", "Biojava/NCBIQBlastService");
366                conn.setRequestProperty("Connection", "Keep-Alive");
367                conn.setRequestProperty("Content-type", "application/x-www-form-urlencoded");
368                conn.setRequestProperty("Content-length", "2000");
369                return conn;
370        }
