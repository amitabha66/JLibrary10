043/**
044 * Implements a data structure for a {@link Sequence} within an alignment.
045 *
046 * @author Mark Chapman
047 * @param <C> each element of the {@link Sequence} is a {@link Compound} of type C
048 */
049public class SimpleAlignedSequence<S extends Sequence<C>, C extends Compound> implements AlignedSequence<S, C> {
050
051    private static final String gap = "-";
052
053    // always stored
054    private AlignedSequence<S, C> prev;
055    private S original;
056    private int length, numBefore, numAfter;
057    private Location location;
058
059    // cached (lazily initialized)
060    private int numGaps = -1;
061    private int[] alignmentFromSequence, sequenceFromAlignment;
062
063    /**
064     * Creates an {@link AlignedSequence} for the given {@link Sequence} in a global alignment.
065     *
066     * @param original the original {@link Sequence} before alignment
067     * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
068     * @throws IllegalArgumentException if given sequence does not fit in alignment
069     */
070    public SimpleAlignedSequence(S original, List<Step> steps) {
071        this(original, steps, 0, 0);
072    }
073
074    /**
075     * Creates an {@link AlignedSequence} for the given {@link Sequence} in a local alignment.
076     *
077     * @param original the original {@link Sequence} before alignment
078     * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
079     * @param numBefore number of {@link Compound}s before a local alignment
080     * @param numAfter number of {@link Compound}s after a local alignment
081     * @throws IllegalArgumentException if given sequence does not fit in alignment
082     */
083    public SimpleAlignedSequence(S original, List<Step> steps, int numBefore, int numAfter) {
084        this.original = original;
085        this.numBefore = numBefore;
086        this.numAfter = numAfter;
087        length = steps.size();
088        setLocation(steps);
089    }
090
091    /**
092     * Creates a new {@link AlignedSequence} for the given {@link AlignedSequence} in a global alignment.
093     *
094     * @param prev the previous {@link AlignedSequence} before this alignment
095     * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
096     * @throws IllegalArgumentException if given sequence does not fit in alignment
097     */
098    public SimpleAlignedSequence(AlignedSequence<S, C> prev, List<Step> steps) {
099        this(prev, steps, 0, 0);
100    }
101
102    /**
103     * Creates a new {@link AlignedSequence} for the given {@link AlignedSequence} in a local alignment.
104     *
105     * @param prev the previous {@link AlignedSequence} before this alignment
106     * @param steps lists whether the sequence aligns a {@link Compound} or gap at each index of the alignment
107     * @param numBefore number of {@link Compound}s before a local alignment
108     * @param numAfter number of {@link Compound}s after a local alignment
109     * @throws IllegalArgumentException if given sequence does not fit in alignment
110     */
111    public SimpleAlignedSequence(AlignedSequence<S, C> prev, List<Step> steps, int numBefore, int numAfter) {
112        this.prev = prev;
113        this.original = prev.getOriginalSequence();
114        this.numBefore = numBefore;
115        this.numAfter = numAfter;
116        if (prev instanceof SimpleAlignedSequence<?, ?>) {
117            SimpleAlignedSequence<?, ?> p = (SimpleAlignedSequence<?, ?>) prev;
118            this.numBefore += p.numBefore;
119            this.numAfter += p.numAfter;
120        }
121        length = steps.size();
122        setLocation(steps);
123    }
124
125    // methods for AlignedSequence
126
127    @Override
128    public void clearCache() {
129        alignmentFromSequence = null;
130        sequenceFromAlignment = null;
131    }
132
133    @Override
134    public int getAlignmentIndexAt(int sequenceIndex) {
135        if (alignmentFromSequence == null) {
136            alignmentFromSequence = new int[original.getLength()];
137            int s = 1, a = 1;
138            for (int i = 0; i < numBefore; i++, s++) {
139                alignmentFromSequence[s - 1] = a;
140            }
141            for (; s <= alignmentFromSequence.length && a <= length; s++, a++) {
142                while (a <= length && isGap(a)) {
143                    a++;
144                }
145                alignmentFromSequence[s - 1] = a;
146            }
147            a--;
148            for (int i = 0; i < numAfter; i++, s++) {
149                alignmentFromSequence[s - 1] = a;
150            }
151        }
152        return alignmentFromSequence[sequenceIndex - 1];
153    }
154
155    @Override
156    public Point getEnd() {
157        return location.getEnd();
158    }
159
160    @Override
161    public Location getLocationInAlignment() {
162        return location;
163    }
164
165    @Override
166    public int getNumGaps() {
167        if (numGaps == -1) {
168            numGaps = 0;
169            C cGap = getCompoundSet().getCompoundForString(gap);
170            boolean inGap = false;
171            for (C compound : getAsList()) {
172                if (compound == null || compound.equalsIgnoreCase(cGap)) {
173                    if (!inGap) {
174                        numGaps++;
175                        inGap = true;
176                    }
177                } else {
178                    inGap = false;
179                }
180            }
181        }
182        return numGaps;
183    }
184
185    @Override
186    public S getOriginalSequence() {
187        return original;
188    }
189
190    @Override
191    public int getOverlapCount() {
192        // TODO handle circular alignments
193        return 1;
194    }
195
196    @Override
197    public int getSequenceIndexAt(int alignmentIndex) {
198        if (sequenceFromAlignment == null) {
199            sequenceFromAlignment = new int[length];
200            int a = 1, s = numBefore + 1;
201            for (int i = 0; i < getStart().getPosition(); i++, a++) {
202                sequenceFromAlignment[a - 1] = s;
203            }
204            for (; a <= length; a++) {
205                if (!isGap(a)) {
206                    s++;
207                }
208                sequenceFromAlignment[a - 1] = s;
209            }
210        }
211        return sequenceFromAlignment[alignmentIndex - 1];
212    }
213
214    @Override
215    public Point getStart() {
216        return location.getStart();
217    }
218
219    @Override
220    public boolean isCircular() {
221        return location.isCircular();
222    }
223
224    @Override
225    public boolean isGap(int alignmentIndex) {
226        if (getStart().getPosition() <= alignmentIndex && alignmentIndex <= getEnd().getPosition()) {
227            if (!location.isComplex()) {
228                return false;
229            }
230            for (Location sublocation : location) {
231                if (sublocation.getStart().getPosition() <= alignmentIndex &&
232                        alignmentIndex <= sublocation.getEnd().getPosition()) {
233                    return false;
234                }
235            }
236        }
237        return true;
238    }

240    // methods for Sequence

242    @Override
243    public int countCompounds(C... compounds) {
244        int count = 0;
245        List<C> search = Arrays.asList(compounds);
246        for (C compound : getAsList()) {
247            if (search.contains(compound)) {
248                count++;
249            }
250        }
251        return count;
252    }
253
254    @Override
255    public AccessionID getAccession() {
256        return original.getAccession();
257    }
258
259    @Override
260    public List<C> getAsList() {
261        List<C> compounds = new ArrayList<C>();
262        for (int i = 1; i <= length; i++) {
263            compounds.add(getCompoundAt(i));
264        }
265        return compounds;
266    }
267
268    @Override
269    public C getCompoundAt(int alignmentIndex) {
270        return alignmentIndex >= 1 && alignmentIndex <= length && isGap(alignmentIndex) ?
271                getCompoundSet().getCompoundForString(gap) :
272                original.getCompoundAt(getSequenceIndexAt(alignmentIndex));
273    }
274
275    @Override
276    public CompoundSet<C> getCompoundSet() {
277        return original.getCompoundSet();
278    }
279
280    @Override
281    public int getIndexOf(C compound) {
282        for (int i = 1; i <= length; i++) {
283            if (compound.equals(getCompoundAt(i))) {
284                return i;
285            }
286        }
287        return -1;
288    }
289
290    @Override
291    public int getLastIndexOf(C compound) {
292        for (int i = length; i >= 1; i--) {
293            if (compound.equals(getCompoundAt(i))) {
294                return i;
295            }
296        }
297        return -1;
298    }

332    // helper method to initialize the location
333    private void setLocation(List<Step> steps) {
334        List<Location> sublocations = new ArrayList<Location>();
335        int start = 0, step = 0, oStep = numBefore+numAfter, oMax = this.original.getLength(), pStep = 0, pMax =
336                (prev == null) ? 0 : prev.getLength();
337        boolean inGap = true;
338
339        // build sublocations: pieces of sequence separated by gaps
340        for (; step < length; step++) {
341            boolean isGapStep = (steps.get(step) == Step.GAP),
342                    isGapPrev = (pStep < pMax && prev.isGap(pStep + 1));
343            if (!isGapStep && !isGapPrev) {
344                oStep++;
345                if (inGap) {
346                    inGap = false;
347                    start = step + 1;
348                }
349            } else if (!inGap) {
350                inGap = true;
351                sublocations.add(new SimpleLocation(start, step, Strand.UNDEFINED));
352            }
353            if (prev != null && !isGapStep) {
354                pStep++;
355            }
356        }
357        if (!inGap) {
358            sublocations.add(new SimpleLocation(start, step, Strand.UNDEFINED));
359        }
360
361        // combine sublocations into 1 Location
362        location = (sublocations.size() == 1) ? sublocations.get(0) : new SimpleLocation(
363                sublocations.get(0).getStart(), sublocations.get(sublocations.size() - 1).getEnd(), Strand.UNDEFINED,
364                false, sublocations);
365        // TODO handle circular alignments
366
367        // check that alignment has correct number of compounds in it to fit original sequence
368        if (step != length || oStep != oMax || pStep != pMax) {
369            throw new IllegalArgumentException("Given sequence does not fit in alignment.");
370        }
371    }
