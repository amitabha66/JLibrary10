001package demo;
002
003import java.util.Arrays;
004
005import org.biojava3.core.sequence.ProteinSequence;
006import org.biojava3.core.sequence.compound.AminoAcidCompound;
007import org.biojava3.core.sequence.compound.AminoAcidCompoundSet;
008import org.biojava3.core.sequence.loader.UniprotProxySequenceReader;
009import org.biojava3.ronn.Jronn;
010
011public class PredictDisorder {
012        public static void main(String[] args){
013
014                String uniprotID = "O30642";
015                try {
016                        ProteinSequence seq = getUniprot(uniprotID);
017                        System.out.println(seq.toString());
018                        AminoAcidCompoundSet compoundSet = AminoAcidCompoundSet.getAminoAcidCompoundSet();
019                        
020                        compoundSet.verifySequence(seq);
021                        
022                        float[] values = Jronn.getDisorderScores(seq);
023                        
024                        System.out.println(Arrays.toString(values));
025                        
026                } catch (Exception e){
027                        e.printStackTrace();
028                }
029        }
030
031        /** Fetch a protein sequence from the UniProt web site
032         * 
033         * @param uniProtID
034         * @return a Protein Sequence
035         * @throws Exception
036         */
037        private static ProteinSequence getUniprot(String uniProtID) throws Exception {
038
039                AminoAcidCompoundSet set = AminoAcidCompoundSet.getAminoAcidCompoundSet();
040                UniprotProxySequenceReader<AminoAcidCompound> uniprotSequence = new UniprotProxySequenceReader<AminoAcidCompound>(uniProtID,set);
041
042                ProteinSequence seq = new ProteinSequence(uniprotSequence);
043
044                return seq;
045        }
046}
