048         * The sequence argument must be a protein sequence consisting of only non-ambiguous characters.
049         * This method will sum the molecular weight of each amino acid in the
050         * sequence. Molecular weights are based on <a href="http://web.expasy.org/findmod/findmod_masses.html">here</a>.
051         * 
052         * @param sequence
053         *      a protein sequence consisting of non-ambiguous characters only
054         * @return the total molecular weight of sequence + weight of water molecule
055         */
056        public static final double getMolecularWeight(String sequence){
057                sequence = Utils.checkSequence(sequence);
058                ProteinSequence pSequence = new ProteinSequence(sequence);
059                IPeptideProperties pp = new PeptidePropertiesImpl();
060                return pp.getMolecularWeight(pSequence);
061        }

152        /**
153         * An adaptor method that returns the molecular weight of sequence. The sequence argument must be a protein sequence consisting of only non-ambiguous characters.
154         * This method will sum the molecular weight of each amino acid in the
155         * sequence. Molecular weights are based on the AminoAcidCompositionTable. 
156         * Those input files must be XML using the defined schema.
157         * 
158         * @param sequence
159         *      a protein sequence consisting of non-ambiguous characters only
160         * @param aminoAcidCompositionTable
161         *      a amino acid composition table obtained by calling IPeptideProperties.obtainAminoAcidCompositionTable
162         * @return the total molecular weight of sequence + weight of water molecule
163         *      thrown if the method IPeptideProperties.setMolecularWeightXML(File, File) is not successfully called before calling this method.
164         */
165        public static double getMolecularWeightBasedOnXML(String sequence, AminoAcidCompositionTable aminoAcidCompositionTable){
166                sequence = Utils.checkSequence(sequence, aminoAcidCompositionTable.getSymbolSet());
167                ProteinSequence pSequence = new ProteinSequence(sequence, aminoAcidCompositionTable.getAminoAcidCompoundSet());
168                IPeptideProperties pp = new PeptidePropertiesImpl();
169                return pp.getMolecularWeightBasedOnXML(pSequence, aminoAcidCompositionTable);
170        }

415        /**
416         * An adaptor method to return the composition of the 20 standard amino acid in the sequence.
417         * The sequence argument must be a protein sequence consisting of only
418         * non-ambiguous characters.
419         * The composition of an amino acid is the total number of its occurrence,
420         * divided by the total length of the sequence.
421         * 
422         * @param sequence
423         *              a protein sequence consisting of non-ambiguous characters only
424         * @return the composition of the 20 standard amino acid in the sequence
425         */
426        public static final Map<String, Double> getAACompositionString(String sequence){
427                Map<AminoAcidCompound, Double> aa2Composition = getAAComposition(sequence);
428                Map<String, Double> aaString2Composition = new HashMap<String, Double>();
429                for(AminoAcidCompound aaCompound:aa2Composition.keySet()){
430                        aaString2Composition.put(aaCompound.getShortName(), aa2Composition.get(aaCompound));
431                }
432                return aaString2Composition;
433        }
