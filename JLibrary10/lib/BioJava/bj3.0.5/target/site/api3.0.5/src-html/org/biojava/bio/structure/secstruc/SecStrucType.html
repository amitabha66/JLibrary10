001package org.biojava.bio.structure.secstruc;
002
003import java.io.Serializable;
004
005public enum SecStrucType implements Serializable{
006
007        coil("Coil",' '),
008        helix4("alpha Helix",'H'),
009        helix3("3-10 Helix",'G'),
010        helix5("pi helix",'I'),
011        turn("Turn",'T'),       
012        bend("Bend",'S'),
013        extended("Extended",'E'),
014        bridge("Bridge",'B');
015        ;
016
017        public final Character type;
018        public final String name;
019
020
021        private SecStrucType(String name,Character stype){
022                this.name = name;
023                this.type = stype;
024        }
025
026        public static SecStrucType fromCharacter(Character stype){
027
028                for ( SecStrucType c : SecStrucType.values()){
029                        if ( c.type.equals(stype)){
030                                return c;
031                        }
032                }
033
034                return null;
035
036        }
037
038        public String toString(){
039                return type.toString();
040        }
041
042        public boolean isHelixType() {
043                if ( type.equals(helix4.type) || type.equals(helix3.type) || type.equals(helix5.type))
044                        return true;
045                return false;
046        }
047        
048
049}
