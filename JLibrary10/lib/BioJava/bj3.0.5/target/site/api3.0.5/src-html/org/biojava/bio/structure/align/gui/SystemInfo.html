085   public SystemInfo()
086   {
087      border = new EtchedBorder();
088      msg = "";
089      try {
090         Properties props = System.getProperties();
091         /*
092          * Unfortunately, enumerating System.getProperties() returns
093          * them in an unsatisfactory order. To make the display
094          * esthetically pleasing, we'll extract the property names
095          * (i.e. the keys) into a vector, then sort the vector, then
096          * use the vector as an enumeration. props.size() is not
097          * trustworthy (bug in MRJ?)
098          */
099         //border.setLabelText ("System Properties");
100         /*
101          * Count the actual size of the System property list.
102          */
103         int size = 0;
104         Enumeration<?> enumo = props.propertyNames();
105         while (enumo.hasMoreElements()) {
106            ++size;
107            enumo.nextElement();
108         }
109         String[] names = new String[size];
110         enumo = props.propertyNames();
111         for (int i = 0; enumo.hasMoreElements(); i++) {
112            names[i] = (String) enumo.nextElement();
113         }
114         if (size < 1) {
115            msg = "No System Properties";
116         }
117         else {
118            quickSort(names, 0, names.length - 1);
119            for (int i = 0; i < size; i++) {
120               addOneSystemProperty(names[i]);
121            }
122         }
123      }
124      catch (SecurityException e) {
125        // border.setLabelText ("Default Applet Properties");
126         StringTokenizer t = new StringTokenizer(defaultProperties, " ");
127         while (t.hasMoreElements()) {
128            addOneSystemProperty(t.nextToken());
129         }
130      }
131      catch (Exception e) {
132         append("Strange Exception getting system properties: " + e);
133      }
134   }

140   /**
141    * Stripped-down QuickSort.
142    * @param   vector      The vector of strings to sort
143    * @param   startIndex  The first element to sort
144    * @param   endIndex    The last element to sort
145    * 
146    * example use: 
147    * <pre>
148    *    JavaInfo.quickSort(vector, 0, vector.length - 1);
149    *  </pre>  
150    */
151   public static void quickSort(
152           String[]        vector,
153           int             startIndex,
154           int             endIndex
155       )
156   {
157       int     i       = startIndex;
158       int     j       = endIndex;
159       String  pivot   = vector[(i + j) / 2];
160       do {
161           while (i < endIndex && pivot.compareTo(vector[i]) > 0) {
162               ++i;
163           }
164           while (j > startIndex && pivot.compareTo(vector[j]) < 0) {
165               --j;
166           }
167           if (i < j) {
168               String  temp    = vector[i];
169               vector[i]       = vector[j];
170               vector[j]       = temp;
171           }
172           if (i <= j) {
173               ++i;
174               --j;
175           }
176       } while (i <= j);
177       if (startIndex < j) {
178           quickSort(vector, startIndex, j);
179       }
180       if (i < endIndex) {
181           quickSort(vector, i, endIndex);
182       }
183   }

186   public void addOneSystemProperty(
187         String          name
188   )
189   {
190      try {
191         String propValue    = System.getProperty(name);
192         /*
193          * On the Macintosh (under MRJ), a bunch of font names
194          * are loaded into the property list. We toss them
195          * to avoid confusion.
196          */
197         if (propValue != null && !name.equals(propValue)) {
198            append("<b>"+name + "</b>:");
199            boolean isReadable = true;
200            for (int i = 0; i < propValue.length(); i++) {
201               char    c   = propValue.charAt(i);
202               if (isControlCharacter(c)) {
203                  isReadable = false;
204                  break;
205               }
206            }
207            if (!isReadable) {
208               for (int i = 0; i < propValue.length(); i++) {
209                  char    c   = propValue.charAt(i);
210                  if (Character.isLetterOrDigit(c)) {
211                     append(" '" + c + "'");
212                  }
213                  else {
214                     append(" 0x");
215                     if (c < 0x10) {
216                        append("0");
217                     }
218                     append(Integer.toHexString(c));
219                  }
220               }
221            }
222            else if (isURLProperty(name)) {
223               StringBuffer    fixed = new StringBuffer();
224               int     start   = 0;
225               int     hit     = 0;
226               while ((hit = propValue.indexOf('%', start)) >= 0) {
227                  fixed.append(propValue.substring(start, hit));
228                  int     value   =
229                     Integer.parseInt(propValue.substring(hit + 1, hit + 3), 16);
230                  fixed.append(((char) value));
231                  start = hit + 3;
232               }
233               append(" \"" + fixed + "\"");
234            }
235            else {
236               append(" " + propValue);
237            }
238            append("<br/>");
239         } /* If the property name was found */
240      } /* Try to fetch the property name */
241      catch (SecurityException e) {
242         append(name + ": Security Exception\n"  );
243      }
244      catch (Exception e) {
245         append(name     + ": " + e + "\n");
246      }
247   }
248   protected boolean isHexProperty(
249         String          thisName
250   )
251   {
252      int index   = hexPropertyNames.indexOf(" " + thisName + " ");
253      return (index >= 0);
254   }
255   protected boolean isURLProperty(
256         String          thisName
257   )
258   {
259      int index   = urlPropertyNames.indexOf(" " + thisName + " ");
260      return (index >= 0);
261   }
262   /**
263    * Replicate the Java 1.1 isISOControl method (for Java 1.0.2 compatibility)
264    */
265   private boolean isControlCharacter(
266         char            c
267   )
268   {
269      return ((c >= '\u0000' && c <= '\u001f')
270            || (c >= '\u007f' && c <= '\u009f')); 
271   }

277   public void showDialog(){
278      JDialog dialog = new JDialog();
279
280      dialog.setSize(new Dimension(500,650));
281
282      String msg = getMessage();
283      JEditorPane txt = new JEditorPane("text/html", msg);
284      txt.setEditable(false);
285
286      JScrollPane scroll = new JScrollPane(txt);
287      scroll.setSize(new Dimension(300,500));
288      vBox= Box.createVerticalBox();
289      vBox.add(scroll);
290
291      txt.addHyperlinkListener(new HyperlinkListener(){
292
293         public void hyperlinkUpdate(HyperlinkEvent e) {
294
295            if (e.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
296               String href = e.getDescription();
297               BrowserOpener.showDocument(href);
298            }
299            if ( e.getEventType() == HyperlinkEvent.EventType.ENTERED) {
300               // change the mouse curor
301               vBox.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
302            }
303            if (e.getEventType() == HyperlinkEvent.EventType.EXITED) { 
304               vBox.setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
305            }
306         }
307      });

312      JButton close = new JButton("Close");
313
314      close.addActionListener(new ActionListener(){
315         public void actionPerformed(ActionEvent event) {
316            Object source = event.getSource();
317
318            JButton but = (JButton)source;
319            Container parent = but.getParent().getParent().getParent().getParent().getParent().getParent() ;
320
321            JDialog dia = (JDialog) parent;
322            dia.dispose();
323         }
324      });
325
326      Box hBoxb = Box.createHorizontalBox();
327      hBoxb.add(Box.createGlue());
328      hBoxb.add(close,BorderLayout.EAST);
329
330      vBox.add(hBoxb);
331
332      dialog.getContentPane().add(vBox);
333      dialog.setVisible(true);
334
