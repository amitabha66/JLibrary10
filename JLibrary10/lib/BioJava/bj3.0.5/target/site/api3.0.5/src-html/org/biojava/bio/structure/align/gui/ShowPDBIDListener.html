025package org.biojava.bio.structure.align.gui;
026
027import java.awt.event.ActionEvent;
028import java.awt.event.ActionListener;
029
030
031import javax.swing.JCheckBox;
032
033import javax.swing.JOptionPane;
034
035import javax.swing.SwingUtilities;
036
037
038import org.biojava.bio.structure.align.util.UserConfiguration;
039import org.biojava.bio.structure.align.webstart.WebStartMain;
040
041
042public class ShowPDBIDListener
043implements ActionListener {
044        public void actionPerformed(ActionEvent e) {
045                String cmd = e.getActionCommand();                  
046                if ( cmd.equals("Show By ID")){
047
048                        JCheckBox useBioAssembly = new JCheckBox("Show Biological Assembly");
049
050                        String msg = "Which ID to display?";
051                        Object[] params = {msg, useBioAssembly};
052
053                        String pdbId = JOptionPane.showInputDialog(null,
054                                        params,
055                                        "Enter PDB ID, PDB.chainId, or SCOP domain ID",
056                                        JOptionPane.QUESTION_MESSAGE);
057
058                        if ( pdbId != null) {
059                                try {
060                                        pdbId = pdbId.trim();
061                                        UserConfiguration config = WebStartMain.getWebStartConfig();
062
063                                        StructureLoaderThread r = new StructureLoaderThread(config,pdbId, useBioAssembly.isSelected());
064                                        
065                                        StructureLoaderThread.showProgressBar();
066                                        
067                                        r.execute();
068                                        
069
070                                } catch (Exception ex){
071                                        ex.printStackTrace();
072                                }
073                        }        
074                }               
075        }
076
077        
078
079
080}
