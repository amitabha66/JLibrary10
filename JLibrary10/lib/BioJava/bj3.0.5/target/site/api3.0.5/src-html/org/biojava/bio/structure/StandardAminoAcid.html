023package org.biojava.bio.structure;
024
025import java.io.InputStream;
026import java.util.HashMap;
027import java.util.Map;
028
029
030import org.biojava.bio.structure.io.PDBFileParser;
031
032
033/** A class that provides a set of standard amino acids.
034 * 
035 * 
036 * 
037 * @author Andreas Prlic
038 * @author Tamas Horvath provided the standard amino acids.
039 *   
040 *
041 */
042public final class StandardAminoAcid {
043    
044    static private Map<String,AminoAcid> aminoAcids;
045    
046    /** can not be instanciated
047     * 
048     */
049    private StandardAminoAcid() {
050        super();
051        
052    }
053    
054    /**
055     * <p>
056     * Initialize the static StandardAminoAcid resource.
057     * </p>
058     *
059     * <p>
060     * This parses the resource
061     * <code>org/biojava/bio/structure/standardaminos.pdb</code>
062     * and builds a basic set of amino acids.
063     *</p>
064     * @author Tamas Horvath provided the standard amino acids 
065     */
066    static {
067        aminoAcids = new HashMap<String,AminoAcid>();
068    
069        try {
070            InputStream fileStream = StandardAminoAcid.class.getClassLoader().getResourceAsStream(
071                    "org/biojava/bio/structure/standardaminos.pdb"
072            );
073            if (fileStream == null) {
074                throw new Exception("Couldn't locate standardaminos.pdb.  This probably means that your biojava.jar file is corrupt or incorrectly built.");
075            }
076            
077            PDBFileParser parser = new PDBFileParser();
078            Structure s = parser.parsePDBFile(fileStream);
079
080       
081            GroupIterator iter = new GroupIterator(s);
082            while (iter.hasNext()){
083                Group g = (Group) iter.next();
084               
085                if ( g instanceof AminoAcid){
086                    AminoAcid aa = (AminoAcid)g;
087       
088                    aminoAcids.put(aa.getPDBName(),aa);
089                    aminoAcids.put(aa.getAminoType().toString(),aa);
090                    
091                }
092            }
093            
094        } catch (Exception t) {
095            throw new RuntimeException( "Unable to initialize standard aminoacids", t);
096        }
097    }
098    
099    /** get a standard amino acid.
100     * 
101     * @param name the 3- or 1-letter representation of the amino acid.
102     * @return the amino acids, or null if the name can not be matched
103     */
104    public static AminoAcid getAminoAcid(String name){
105        
106        return (AminoAcid) aminoAcids.get(name);
107    }
108    
109}
