001package demo;

003import org.biojava.bio.structure.Structure;
004import org.biojava.bio.structure.align.gui.jmol.StructureAlignmentJmol;
005import org.biojava.bio.structure.quaternary.io.BioUnitDataProviderFactory;
006
007import org.biojava3.structure.StructureIO;
008
009public class DemoShowBiolAssembly {
010
011        public static void main(String[] args){
012
013                try{
014                        // see also: http://www.pdb.org/pdb/101/static101.do?p=education_discussion/Looking-at-Structures/bioassembly_tutorial.html
015                        // good examples: 1stp 1gav 1hv4 1hho 7dfr 3fad  1qqp
016
017                        // assembly 0 ... asym Unit
018                        // assembly 1 ... the first bio assembly
019                        // example 1fah has  2 assemblies (two copies of the assembly in asymmetric unit)
020                        
021                        
022                        // Various interesting symmetries: (see Lawson, 2008)
023                        // Circular    - 1TJA
024                        // Dihedral    - 1ei7
025                        // Icosahedral - 1a34
026                        // Helical     - 1cgm
027                        
028                        
029                        // DNA 173D .. 2
030                        
031                        BioUnitDataProviderFactory.setBioUnitDataProvider(BioUnitDataProviderFactory.pdbProviderClassName);
032                        //Structure bioAssembly = StructureIO.getBiologicalAssembly("4A1I",2);  
033                        
034                        Structure bioAssembly = StructureIO.getBiologicalAssembly("1ei7",1);
035                                                                        
036                        StructureAlignmentJmol jmolPanel = new StructureAlignmentJmol();
037                        //jmolPanel.evalString("set autobond=false");
038                        jmolPanel.setStructure(bioAssembly);
039
040                        // send some commands to Jmol
041                        jmolPanel.evalString("select * ; color structure ; spacefill off; wireframe off; backbone off; cartoon on; select ligands ; spacefill 0.4; color cpk;");
042                        
043                        System.out.println("done!");
044
045                } catch (Exception e) {
046                        e.printStackTrace();
047                }       
048        }
049}
