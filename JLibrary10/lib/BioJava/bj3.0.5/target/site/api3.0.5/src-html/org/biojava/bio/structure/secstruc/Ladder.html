001package org.biojava.bio.structure.secstruc;
002
003/** Connects two connected strand regions to form larger sheets
004 * 
005 * @author andreas
006 *
007 */
008public class Ladder {
009        int from ;
010        int to;
011        int lfrom;
012        int lto;
013        BridgeType btype;
014        int connectedTo;
015        int connectedFrom;
016
017        public int getFrom() {
018                return from;
019        }
020        public void setFrom(int from) {
021                this.from = from;
022        }
023        public int getTo() {
024                return to;
025        }
026        public void setTo(int to) {
027                this.to = to;
028        }
029        public int getLfrom() {
030                return lfrom;
031        }
032        public void setLfrom(int lfrom) {
033                this.lfrom = lfrom;
034        }
035        public int getLto() {
036                return lto;
037        }
038        public void setLto(int lto) {
039                this.lto = lto;
040        }
041        public BridgeType getBtype() {
042                return btype;
043        }
044        public void setBtype(BridgeType btype) {
045                this.btype = btype;
046        }
047        public int getConnectedTo() {
048                return connectedTo;
049        }
050        public void setConnectedTo(int connectedTo) {
051                this.connectedTo = connectedTo;
052        }
053        public int getConnectedFrom() {
054                return connectedFrom;
055        }
056        public void setConnectedFrom(int connectedFrom) {
057                this.connectedFrom = connectedFrom;
058        }
059        @Override
060        public String toString() {
061                return "Ladder [from=" + from + ", to=" + to + ", lfrom=" + lfrom
062                                + ", lto=" + lto + ", btype=" + btype + ", connectedTo="
063                                + connectedTo + ", connectedFrom=" + connectedFrom + "]";
064        }
065        
066
067}
