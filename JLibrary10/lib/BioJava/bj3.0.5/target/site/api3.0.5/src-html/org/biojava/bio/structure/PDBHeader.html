059        /** String representation
060         *
061         */
062        public String toString(){
063                StringBuffer buf = new StringBuffer();
064
065                try {
066
067                        @SuppressWarnings("rawtypes")
068                        Class c = Class.forName("org.biojava.bio.structure.PDBHeader");
069                        Method[] methods  = c.getMethods();
070
071                        for (int i = 0; i < methods.length; i++) {
072                                Method m = methods[i];
073
074                                String name = m.getName();
075
076                                if ( name.substring(0,3).equals("get")) {
077                                        if (name.equals("getClass"))
078                                                continue;
079                                        Object o  = m.invoke(this, new Object[]{});
080                                        if ( o != null){
081                                                buf.append(name.substring(3,name.length()));
082                                                buf.append(": " + o + " ");
083                                                //if ( o instanceof Date) {
084                                                //    buf.append(": " + FlatFileInstallation.dateFormat.format(o) + " ");
085                                                //} else  {
086                                                //    buf.append(": " + o + " ");
087                                                //}
088                                        }
089                                }
090                        }
091                } catch (Exception e){
092                        e.printStackTrace();
093                }
094
095                return buf.toString();
096        }

127        private void printResolution(StringBuffer buf){
128
129                if (getResolution() == DEFAULT_RESOLUTION){
130                        return;
131                }
132
133                DecimalFormat d2 = (DecimalFormat)NumberFormat.getInstance(java.util.Locale.UK);
134                d2.setMaximumIntegerDigits(2);
135                d2.setMinimumFractionDigits(2);
136                d2.setMaximumFractionDigits(2);
137
138                buf.append("REMARK   2 RESOLUTION. ");
139                String x = d2.format(resolution);
140                buf.append(x);
141                buf.append(" ANGSTROMS.");
142                fillLine(buf,34+x.length());
143
144                buf.append(newline);
145        }

147        private void printExpdata(StringBuffer buf){
148                String exp = getTechnique();
149                if ( exp == null || exp.length() == 0)
150                        return;
151
152                buf.append("EXPDTA    ");
153                buf.append(exp);
154
155                // fill up the white space to the right column
156                int l =  exp.length()+ 10;
157                fillLine(buf,l);
158
159                buf.append(newline);
160
161        }

163        private void printAuthors(StringBuffer buf){
164                String authors = getAuthors();
165                if ( authors == null)
166                        return;
167                if ( authors.equals("")){
168                        return;
169                }
170
171                printMultiLine(buf, "AUTHOR   ", authors,',');
172
173        }

175        private void printMultiLine(StringBuffer buf, String lineStart, String data, char breakChar){
176                if ( lineStart.length() !=  9)
177                        System.err.println("lineStart != 9, there will be problems :" + lineStart);
178
179                if ( data.length() < 58) {
180                        buf.append(lineStart);
181                        buf.append(" ");
182                        buf.append(data);
183                        buf.append(newline);
184                        return;
185                }
186                String thisLine = "";
187                int count = 1;
188                while (data.length() > 57) {
189                        // find first whitespace from left
190                        // there are 10 chars to the left, so the cutoff position is 56
191                        boolean charFound = false;
192                        for ( int i =57;i>-1;i--){
193                                char c = data.charAt(i);
194                                if (c == breakChar){                
195                                        // found the whitespace
196
197                                        thisLine = data.substring(0,i+1);
198
199                                        // prevent endless loop
200                                        if (i == 0 )
201                                                i++;
202                                        data = data.substring(i);
203                                        charFound = true;
204                                        //System.out.println(thisLine);
205                                        //System.out.println(title);
206                                        break;
207                                }
208                        }
209                        // for emergencies...  prevents an endless loop
210                        if ( ! charFound){
211                                thisLine = data.substring(0,58);
212                                data = data.substring(57);             
213                        }
214                        if ( ( breakChar == ',' ) && ( data.charAt(0)== ',')) {
215                                data =   data.substring(1);
216                        }
217
218                        //TODO: check structures that have more than 10  lines...
219                        // start printing..
220
221                        buf.append(lineStart);
222                        if ( count > 1) {
223                                buf.append(count);
224                                if ( breakChar != ' ' )
225                                        buf.append(" ");
226                        }
227                        else
228                                buf.append(" ");
229                        buf.append(thisLine);
230
231                        // fill up the white space to the right column
232                        int l =  thisLine.length()+ 10;
233                        while (l < 67){
234                                l++;
235                                buf.append(" ");
236                        }
237
238                        buf.append(newline);
239                        count++;
240
241                }
242
243                // last line...
244                if ( data.trim().length() > 0){
245                        buf.append(lineStart);
246                        buf.append(count);
247                        int filledLeft = 10;
248                        if ( breakChar != ' ' ) {
249                                buf.append(" ");
250                                filledLeft++;
251                        }
252                        buf.append(data);
253                        // fill up the white space to the right column
254                        int l =  data.length()+ filledLeft;
255                        fillLine(buf,l);
256                        buf.append(newline);
257                }
258
259        }
260
261        private void fillLine(StringBuffer buf, int currentPos){
262                int l = currentPos;
263                while (l < 67){
264                        l++;
265                        buf.append(" ");
266                }
267        }

269        private void printHeader(StringBuffer buf){
270
271                String classification = getClassification();
272
273                if (
274                                (classification ==null) ||
275                                (classification.length() == 0))
276                        return;
277
278                // we can;t display this line since the classification is not there...
279
280                buf.append("HEADER    ");
281                buf.append(classification);
282                buf.append(" ");
283
284                // fill up the white space to the right column
285                int l =  classification.length() + 10 ;
286                while (l < 49){
287                        l++;
288                        buf.append(" ");
289                }
290
291                Date d = getDepDate();
292                if ( d !=  null){
293                        // provide correct display of Dep date...
294                        buf.append(dateFormat.format(d));
295                } else {
296                        buf.append("         ");
297                }
298                buf.append("   ");
299
300                String id = getIdCode();
301                if ( id != null){
302                        buf.append(getIdCode());
303                        buf.append(" ");
304                }
305                else
306                        buf.append("    ");
307                buf.append(newline);
308
309
310        }

321
322                if ( (title == null) || ( title.trim().length() == 0) )

324

327        }

350        /** Compare two PDBHeader objects
351         *
352         * @param other a PDBHeader object to compare this one to.
353         * @return true if they are equal or false if they are not.
354         */
355        public boolean equals(PDBHeader other){
356                try {
357
358                        @SuppressWarnings("rawtypes")
359                        Class c = Class.forName("org.biojava.bio.structure.PDBHeader");
360                        Method[] methods  = c.getMethods();
361
362                        for (int i = 0; i < methods.length; i++) {
363                                Method m = methods[i];
364                                String name = m.getName();
365
366                                if ( name.substring(0,3).equals("get")) {
367                                        if (name.equals("getClass"))
368                                                continue;
369                                        Object a  = m.invoke(this,  new Object[]{});
370                                        Object b  = m.invoke(other, new Object[]{});
371                                        if ( a == null ){
372                                                if ( b == null ){
373                                                        continue;
374                                                } else {
375                                                        System.out.println(name + " a is null, where other is " + b);
376                                                        return false;
377                                                }
378                                        }
379                                        if ( b == null) {
380                                                System.out.println(name + " other is null, where a is " + a);
381                                                return false;
382                                        }
383                                        if (! (a.equals(b))){
384                                                System.out.println("mismatch with " + name + " >" + a + "< >" + b + "<");
385                                                return false;
386                                        }
387                                }
388                        }
389                } catch (Exception e){
390                        e.printStackTrace();
391                        return false;
392                }
393                return true;
394        }
