040    public static int ALIGFACTOR = 1000; // constant to shift floats to ints
041    
042    Alignable a;

061    private void align() {
062        
063        rowDim = a.getRows()+1;
064        colDim = a.getCols()+1;
065        
066        openPen = Math.round(ALIGFACTOR * a.getGapOpenCol());
067        elgPen  = Math.round(ALIGFACTOR * a.getGapExtCol());
068        
069        GapArray[] gapCol = new GapArray[colDim];
070        GapArray[] gapRow = new GapArray[rowDim];
071       
072        // init the arrays
073        for ( int i = 0 ; i< colDim;i++){
074            gapCol[i] = new GapArray();
075        }
076        for ( int j = 0 ; j< rowDim;j++){
077            gapRow[j] = new GapArray();
078        }
079        currentGap = new GapArray();
080        
081        AligMatEl[][]aligmat = a.getAligMat();
082        //System.out.println("check: aligmat row:" + aligmat.length + " aligmat col:" + aligmat[0].length);
083        //System.out.println("check rowdim" + rowDim + " colDim " + colDim);
084        int lastValue = aligmat[rowDim-1][colDim-1].getValue();
085        
086//      first cell
087        aligmat[0][0].setValue(0);
088        gapCol[0].setValue(0);
089        gapCol[0].setIndex(0);
090        gapRow[0].setValue(0);
091        gapRow[0].setIndex(0);
092        
093        // set row 0 margin
094        for(int j=1;j<colDim;j++) {
095            aligmat[0][j].setValue(0);
096            gapCol[j].setValue(-(openPen+elgPen));
097            gapCol[j].setIndex(0);
098        }
099        
100        
101        for(int rowCounter=1;rowCounter<rowDim;rowCounter++){
102            
103            // set column 0 margin
104            aligmat[rowCounter][0].setValue(0);
105            gapRow[rowCounter].setValue(-(openPen+elgPen));
106            gapRow[rowCounter].setIndex(0);
107            
108            for(int colCounter=1;colCounter<colDim;colCounter++) {
109                
110                currentCell = aligmat[rowCounter][colCounter];
111
112                // no gap 
113                currentCell.setValue( aligmat[rowCounter-1][colCounter-1].getValue() + currentCell.getValue());
114                currentCell.setRow((short)(rowCounter-1));
115                currentCell.setCol((short)(colCounter-1));
116                
117                // scan column j for gap, gap in seqB
118                openVal = aligmat[rowCounter-1][colCounter].getValue() - (openPen+elgPen);
119                elgVal  = gapCol[colCounter].getValue()-elgPen;
120                
121                currentGap = new GapArray();
122                
123                if ( openVal >= elgVal){
124                    currentGap.setValue(openVal);
125                    currentGap.setIndex(rowCounter-1); 
126                } else {
127                    currentGap.setValue(elgVal);
128                    currentGap.setIndex(gapCol[colCounter].index);
129                }
130                gapCol[colCounter] = currentGap;
131                
132                if (currentGap.getValue() > currentCell.getValue()){
133                    if ( currentGap.getIndex() >= rowDim)
134                        System.err.println("col gap at" + rowCounter + " " + colCounter + " to " + currentGap.getIndex());
135                    currentCell.setValue( currentGap.getValue());
136                    currentCell.setRow((short)currentGap.getIndex());
137                    currentCell.setCol((short)colCounter);
138                }
139                
140//              scan row i for gap, gap in row
141                openVal = aligmat[rowCounter][colCounter-1].getValue()-(openPen+elgPen);
142                elgVal  = gapRow[rowCounter].getValue() - elgPen;
143                
144                currentGap = new GapArray();
145                
146                if (openVal >= elgVal){
147                    currentGap.setValue(openVal);
148                    currentGap.setIndex(colCounter-1); 
149                } else {
150                    currentGap.setValue(elgVal);
151                    currentGap.setIndex(gapRow[rowCounter].getIndex());
152                }
153                gapRow[rowCounter] = currentGap;
154                
155                
156                if ( currentGap.getValue() > currentCell.getValue() ) {
157                    if ( currentGap.getIndex() >= colDim)
158                        System.err.println("row gap at" + rowCounter + " " + colCounter + " to " + currentGap.getIndex());
159                    currentCell.setValue(currentGap.getValue());
160                    currentCell.setRow((short)rowCounter);
161                    currentCell.setCol((short)currentGap.getIndex());
162                }
163                
164                //System.out.println("aligmat " + i + " " + j + " " + currentCell.getValue() + 
165                //        " " + currentCell.getTrack().getRow() + " " +
166                //        currentCell.getTrack().getCol());
167                aligmat[rowCounter][colCounter]=currentCell;
168            }
169            
170        }
171        
172        
173        // last cell in alignment matrix
174        // do not penalize end gaps
175        int rowCount = rowDim -1;
176        int colCount = colDim -1;
177        currentCell = aligmat[rowCount][colCount];
178        //System.out.println("dimension: " + rowCount + " " + colCount);
179        //System.out.println("lastValue " + lastValue + " lastCell " + currentCell.getValue());
180       
181        // no gap
182        currentCell.setValue(aligmat[rowCount-1][colCount-1].getValue() + lastValue);
183        currentCell.setRow((short)(rowCount-1));
184        currentCell.setCol((short)(colCount-1));
185        
186        // scan last column j for gap, gap in seqB
187        // do not penalyze gaps
188        for (int k=1;k<=rowCount;k++) {
189            int val = aligmat[rowCount-k][colCount].getValue();
190            if ( val>currentCell.getValue()){
191                currentCell.setValue(val);
192                //System.out.println("setting row to " + (rowCount ) );
193                currentCell.setRow((short)(rowCount-k ));
194                currentCell.setCol((short)(colCount  ));
195            }
196        }
197        
198        // scan row i for gap, gap in SeqA 
199        // do not penalyze gaps 
200        for (int k=1;k<=colCount;k++) {
201            int val = aligmat[rowCount][colCount-k].getValue();
202            if ( val > currentCell.getValue()){
203                currentCell.setValue(val);
204                currentCell.setRow((short) (rowCount ) );
205                currentCell.setCol((short)(colCount-k  ));
206            }
207        }
208        //aligmat[rowCount][colCount]=currentCell;
209        //System.out.print (" lastcell " + i + " " + j + " " + currentCell);
210        //a.setAligMat(aligmat);
211        //System.out.println("score:");
212        //System.out.println(aligmat[rowDim-1][colDim-1].getValue());
213        a.setScore(aligmat[rowDim-1][colDim-1].getValue() / (float)ALIGFACTOR);
214        //writeToFile(aligmat);
215        setPath();
216           
217    }

247    @SuppressWarnings({ "rawtypes", "unchecked" })
248        private void setPath(){
249        
250        int  n;
251        IndexPair[] backId = new IndexPair[a.getRows()+1+a.getCols()+1];
252        //IndexPair[] path = a.getPath();
253        List path = new ArrayList();
254        
255        backId[0] = new IndexPair((short)(a.getRows()),(short)(a.getCols()));
256        
257        // backtrace, get backId indices, the indices in diagonal store in path
258       
259       
260        int pathsize = 0;
261        
262        AligMatEl[][] aligmat = a.getAligMat();
263       
264
265        //AligMatEl el1 =aligmat[0][0];
266        //System.out.println("element at 00:" + el1);
267        n=1;
268        while ( (backId[n-1].getRow()>=1) && 
269                (backId[n-1].getCol()>=1) 
270               )
271                {
272            // get backtrace index
273            int x = backId[n-1].getRow();
274            int y = backId[n-1].getCol();
275            
276            //System.out.println("path pos n: + " + n + " x:" + x + " y:" + y + " ");
277            //if (( x >= a.getRows()-1) || ( y >=a.getCols()-1))
278            //    System.out.println("x:"+ x + " y " + y + " rows:" + a.getRows() + " cols:" + a.getCols());
279            try {
280                
281                AligMatEl el = null ;
282                try {
283                    el =aligmat[x][y];
284                } catch(Exception e){
285                    
286                    
287                    e.printStackTrace();
288                    for (int f=0; f< n;f++){
289                        System.out.println(backId[f]);
290                    }
291                    
292                   
293                  
294                }
295                //if (( x >= a.getRows()-1) || ( y >=a.getCols()-1))
296                //System.out.println("el at x " + x + " y " + y + " " +  el);
297                
298                if ( el == null)
299                    System.out.println("el = null! x:"+ x + " y " + y);
300                //if ( n>=( backId.length-10))
301                    //System.out.println("setPath:"+ el + " n:" + n + " " + backId.length);   
302                backId[n] = el;
303            } catch (Exception e){
304                e.printStackTrace();
305                System.out.println("x " + x);
306                System.out.println("y " + y);
307                System.out.println(backId[n-2]);
308                System.exit(0);
309            }
310            // get diagonal indeces into path
311            if (((backId[n-1].getRow() - backId[n].getRow()) == 1) 
312                        && (( backId[n-1].getCol() - backId[n].getCol()) == 1)) {
313                //System.out.println("adding to path " + backId[n-1] + " n " + n + " pathsize " + pathsize + " backId.length: " + backId.length);
314                path.add(backId[n-1]);
315                //path[pathsize] = backId[n-1];    
316                pathsize++;
317                    
318            }   
319            n++;
320        }
321        
322        // path is reverse order..
323        // switch order
324        //System.out.println("path size after refine:" +path.size() + " " + pathsize);
325        IndexPair[] newpath = new IndexPair[pathsize];
326        for (int i = 0 ; i < pathsize; i++){
327            IndexPair o = (IndexPair)path.get(pathsize-1-i);
328            IndexPair np = new IndexPair((short)(o.getRow()-1),(short)(o.getCol()-1));
329            newpath[i] = np;
330        }
331        
332        a.setPath(newpath);
333        a.setPathSize(pathsize);
334        
335      
336    }
