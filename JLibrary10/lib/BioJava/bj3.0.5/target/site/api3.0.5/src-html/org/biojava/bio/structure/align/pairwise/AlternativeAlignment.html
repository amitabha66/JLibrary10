297         *
298         * @param ca
299         */
300        private void rotateShiftAtoms(Atom[] ca){
301
302                for (int i  = 0 ; i < ca.length; i++){
303                        Atom c = ca[i];
304
305                        Calc.rotate(c,currentRotMatrix);
306                        Calc.shift(c,currentTranMatrix);
307                        //System.out.println("after " + c);
308                        ca[i] = c;
309                }

311        }

332        public static Matrix getDistanceMatrix(Atom[] ca1, Atom[]ca2){
333
334                int r = ca1.length;
335                int c = ca2.length;
336
337                Matrix out = new Matrix(r,c);
338
339                for (int i=0; i<r; i++) {
340                        Atom a1 = ca1[i];
341                        for (int j=0;j<c;j++){
342                                Atom b1 = ca2[j];
343
344                                try {
345                                        double d = Calc.getDistance(a1,b1);
346                                        out.set(i,j,d);
347                                } catch (StructureException e) {
348                                        e.printStackTrace();
349                                        out.set(i,j,999);
350                                }
351                        }
352                }
353                return out;
354        }

356        private Alignable getInitalStrCompAlignment(
357                        Atom[] ca1,
358                        Atom[]ca2,
359                        StrucAligParameters params
360        ){
361
362                int rows = ca1.length;
363                int cols = ca2.length;
364
365                float gapOpen = params.getGapOpen();
366                float gapExtension = params.getGapExtension();
367                float co = params.getCreate_co();
368
369                Alignable al = new StrCompAlignment(rows,cols);
370                al.setGapExtCol(gapExtension);
371                al.setGapExtRow(gapExtension);
372                al.setGapOpenCol(gapOpen);
373                al.setGapOpenRow(gapOpen);
374
375                AligMatEl[][] aligmat = al.getAligMat();
376
377                //System.out.println(rows + " " + cols );
378
379                aligmat[0][cols]    = new AligMatEl();
380                aligmat[rows][0]    = new AligMatEl();
381                aligmat[rows][cols] = new AligMatEl();
382
383                for (int i = 0; i < rows; i++) {
384                        Atom a1 = ca1[i];
385
386                        aligmat[i][0]    = new AligMatEl();
387                        //aligmat[i][cols] = new AligMatEl();
388
389                        for (int j = 0; j < cols; j++) {
390                                //System.out.println("index " + i + " " + j);
391                                aligmat[0][j] = new AligMatEl();
392
393                                Atom b1 = ca2[j];
394                                double d = 999;
395                                try {
396                                        d = Calc.getDistance(a1,b1);
397                                } catch (StructureException e) {
398                                        e.printStackTrace();
399                                }
400
401                                AligMatEl e = new AligMatEl();
402                                if (d > co) {
403                                        e.setValue(0);
404                                } else {
405                                        double s =  2 * co / ( 1 + ( d/co) * (d/co)) - co;
406                                        e.setValue((int) Math.round(Gotoh.ALIGFACTOR * s));
407                                }
408                                aligmat[i+1][j+1] = e;
409                        }
410                }
411
412                return al;
413        }

457        /** Refinement procedure based on superposition and dynamic programming.
458         * Performs an iterative refinement. Several methods apply such a procedure,
459         * e.g. CE or ProSup. Here we additionally test for circular permutation,
460         * which are in the same frame of superposition as the optimal alignment.
461         * This feature may be switched off by setting permsize to -1.
462         *
463         * @param params the parameters
464         * @param ca1 atoms of structure 1
465         * @param ca2 atoms of structure 2
466         * @throws StructureException
467         */
468
469        public void refine(StrucAligParameters params,Atom[]ca1,Atom[]ca2) throws StructureException{
470                // System.out.println("refine Alternative Alignment #"+ getAltAligNumber()+" l1:" + ca1.length + " l2:" + ca2.length);
471                //              for ( int i= 0 ; i < idx1.length;i++){
472                //              System.out.println(idx1[i] + " " + idx2[i]);
473                //              }
474
475                // avoid any operations on the original Atoms ...
476                Atom[] ca3 = new Atom[ca2.length];
477                for ( int i = 0 ; i < ca2.length;i++){
478                        ca3[i] = (Atom) ca2[i].clone();
479
480                }
481                //Atom[] ca3 = (Atom[]) ca2.clone();
482
483                // do the inital superpos...
484
485                super_pos_alig(ca1,ca3,idx1,idx2,false);
486
487                //JmolDisplay jmol1 = new JmolDisplay(ca1,ca2,"after first alig " );
488                //jmol1.selectEquiv(idx1,idx2);
489                //int[] jmoltmp1 = idx1;
490                //int[] jmoltmp2 = idx2;
491
492                int lenalt =idx1.length;
493                int lenneu = aligpath.length;
494                //Matrix rmsalt = currentRotMatrix;
495
496
497
498                int ml = Math.max(ca1.length, ca3.length);
499                idx1 = new int[ml];
500                idx2 = new int[ml];
501
502                //int m1l = ca1.length;
503                //int m2l = ca3.length;
504
505                //Matrix sim = new Matrix(ca1.length,ca3.length,0);
506
507                int maxiter = params.getMaxIter();
508                for (int iter = 0 ; iter< maxiter; iter++){
509
510                        float  subscore = 0.0f;
511
512                        rotateShiftAtoms(ca3);
513
514                        // do the dynamic alignment...
515                        Alignable ali = getInitalStrCompAlignment(ca1,ca3, params);
516
517                        new Gotoh(ali);
518
519                        this.score = ali.getScore();
520                        //System.out.println("score " + score);
521                        IndexPair[] path = ali.getPath();
522
523                        int pathsize = ali.getPathSize();
524
525
526                        // now for a superimposable permutation. First we need to check the size and
527                        // position of the quadrant left out by the optimal path
528                        int firsta = path[0].getRow();
529                        int firstb = path[0].getCol();
530                        int lasta  = path[pathsize-1].getRow();
531                        int lastb  = path[pathsize-1].getCol();
532
533                        int quada_beg, quada_end, quadb_beg,quadb_end;
534
535                        if ( firsta == 0){
536                                quada_beg = lasta +1;
537                                quada_end = ca1.length -1;
538                                quadb_beg = 0;
539                                quadb_end = firstb-1;
540                        } else {
541                                quada_beg = 0;
542                                quada_end = firsta-1;
543                                quadb_beg = lastb+1;
544                                quadb_end = ca3.length -1;
545                        }
546
547                        // if the unaligned quadrant is larger than permsize
548                        int permsize = params.getPermutationSize();
549
550                        if ( permsize > 0 &&
551                                        (quada_end - quada_beg >= permsize) &&
552                                        ( quadb_end - quadb_beg >= permsize)) {
553                                AligMatEl[][] aligmat = ali.getAligMat();
554                                Matrix submat = new Matrix(quada_end-quada_beg, quadb_end - quadb_beg) ;
555                                //then we copy the score values of the quadrant into a new matrix
556
557                                //System.out.println(quada_beg + " " + quada_end + " " +quadb_beg+" " + quadb_end);
558                                //System.out.println(sim.getRowDimension() + " " + sim.getColumnDimension());
559                                Atom[] tmp1 = new Atom[quada_end - quada_beg ];
560                                Atom[] tmp2 = new Atom[quadb_end - quadb_beg ];
561
562                                for ( int s = quada_beg; s < quada_end; s++){
563                                        //System.out.println(s + " " +( quada_end-s));
564                                        tmp1[quada_end - s -1] = ca1[s];
565                                        for ( int t = quadb_beg; t < quadb_end; t++){
566                                                if ( s == quada_beg )
567                                                        tmp2[quadb_end - t -1 ] = ca3[t];
568
569                                                //System.out.println(s+" " +t);
570                                                //double val = sim.get(s,t);
571                                                double val = aligmat[s][t].getValue();
572                                                submat.set(s-quada_beg,t-quadb_beg,val);
573                                        }
574                                }
575                                // and perform a dp alignment again. (Note, that we fix the superposition).
576
577                                //Alignable subali = getAlignableFromSimMatrix(submat,params);
578                                Alignable subali = getInitalStrCompAlignment(tmp1, tmp2, params);
579                                subscore = subali.getScore();
580                                //System.out.println("join score" + score + " " + subscore);
581                                this.score = score+subscore;
582
583                                IndexPair[] subpath = subali.getPath();
584                                int subpathsize = subali.getPathSize();
585                                for ( int p=0;p<subpath.length;p++){
586                                        IndexPair sp = subpath[p];
587                                        sp.setRow((short)(sp.getRow()+quada_beg));
588                                        sp.setCol((short)(sp.getCol()+quadb_beg));
589                                }
590
591                                // now we join the two solutions
592                                if ( subpathsize > permsize){
593                                        IndexPair[] wholepath = new IndexPair[pathsize+subpathsize];
594                                        for ( int t=0; t < pathsize; t++){
595                                                wholepath[t] = path[t];
596                                        }
597                                        for ( int t=0 ; t < subpathsize; t++){
598                                                wholepath[t+pathsize] = subpath[t];
599                                        }
600                                        pathsize += subpathsize;
601                                        path = wholepath;
602                                        ali.setPath(path);
603                                        ali.setPathSize(pathsize);
604                                }
605
606                        }
607
608
609                        int j =0 ;
610                        //System.out.println("pathsize,path " + pathsize + " " + path.length);
611
612                        for (int i=0; i < pathsize; i++){
613                                int x = path[i].getRow();
614                                int y = path[i].getCol();
615                                //System.out.println(x+ " " + y);
616                                double d = Calc.getDistance(ca1[x], ca3[y]);
617                                //double d = dismat.get(x,y);
618                                // now we apply the evaluation distance cutoff
619                                if ( d < params.getEvalCutoff()){
620                                        //System.out.println("j:"+j+ " " + x+" " + y + " " + d );
621
622                                        idx1[j] = x;
623                                        idx2[j] = y;
624                                        j+=1;
625                                }
626                        }
627
628                        lenneu = j;
629                        //System.out.println("lenalt,neu:" + lenalt + " " + lenneu);
630
631
632                        int[] tmpidx1 = new int[j];
633                        int[] tmpidx2 = new int[j];
634                        //String idx1str ="idx1: ";
635                        //String idx2str ="idx2: ";
636                        for (int i = 0 ; i<j;i++){
637                                //idx1str += idx1[i]+ " ";
638                                //idx2str += idx2[i]+ " ";
639                                tmpidx1[i] = idx1[i];
640                                tmpidx2[i] = idx2[i];
641                        }
642                        //System.out.println(idx1str);
643                        //System.out.println(idx2str);
644                        //jmol.selectEquiv(idx1,idx2);
645
646                        //if ( j > 0)
647                        //  System.out.println("do new superimpos " + idx1.length + " " + idx2.length + " " + idx1[0] + " " + idx2[0]);
648                        super_pos_alig(ca1,ca3,tmpidx1,tmpidx2,false);
649
650                        this.aligpath = path;
651
652                        if (lenneu == lenalt)
653                                break;
654
655                }
656
657
658
659                idx1 = (int[]) FragmentJoiner.resizeArray(idx1,lenneu);
660                idx2 = (int[]) FragmentJoiner.resizeArray(idx2,lenneu);
661                //              new ... get rms...
662                // now use the original atoms to get the rotmat relative to the original structure...
663                super_pos_alig(ca1,ca2,idx1,idx2,true);
664                eqr0 = idx1.length;
665                gaps0 = count_gaps(idx1,idx2);
666
667                getPdbRegions(ca1,ca2);
668                //System.out.println("eqr " + eqr0 + " aligpath len:"+aligpath.length+ " gaps:" + gaps0 + " score " + score);
669        }

671        private void getPdbRegions(Atom[] ca1, Atom[] ca2){
672                pdbresnum1 = new String[idx1.length];
673                pdbresnum2 = new String[idx2.length];
674
675                for (int i =0 ; i < idx1.length;i++){
676                        Atom a1 = ca1[idx1[i]];
677                        Atom a2 = ca2[idx2[i]];
678                        Group p1 = a1.getGroup();
679                        Group p2 = a2.getGroup();
680                        Chain c1 = p1.getChain();
681                        Chain c2 = p2.getChain();
682
683                        String cid1 = c1.getChainID();
684                        String cid2 = c2.getChainID();
685
686                        String pdb1 = p1.getResidueNumber().toString();
687                        String pdb2 = p2.getResidueNumber().toString();
688
689
690                        if ( ! cid1.equals(" "))
691                                pdb1 += ":" + cid1;
692
693
694                        if ( ! cid2.equals(" "))
695                                pdb2 += ":" + cid2;
696
697
698                        pdbresnum1[i] = pdb1;
699                        pdbresnum2[i] = pdb2;
700                }
701        }

733         *
734         * @param idx1
735         * @param idx2
736         * @return the number of gaps in this alignment
737         */
738        private int count_gaps(int[] i1, int[] i2){
739
740                int i0 = i1[0];
741                int j0 = i2[0];
742                int gaps = 0;
743                for (int i =1 ; i<i1.length;i++ ){
744                        if ( Math.abs(i1[i]-i0) != 1 ||
745                                        ( Math.abs(i2[i]-j0) != 1)){
746                                gaps +=1;
747                        }
748                        i0 = i1[i];
749                        j0 = i2[i];
750                }
751
752                return gaps;
753        }

763        /** Superimposes two molecules according to residue index list idx1 and idx2.
764         *  Does not change the original coordinates.
765         *  as an internal result the rotation matrix and shift vectors for are set
766         *
767         * @param ca1 Atom set 1
768         * @param ca2 Atom set 2
769         * @param idx1 idx positions in set1
770         * @param idx2 idx positions in set2
771         * @param getRMS a flag if the RMS should be calculated
772         * @throws StructureException
773         */
774
775

777        private void super_pos_alig(Atom[]ca1,Atom[]ca2,int[] idx1, int[] idx2, boolean getRMS) throws StructureException{
778
779                //System.out.println("superpos alig ");
780                Atom[] ca1subset = new Atom[idx1.length];
781                Atom[] ca2subset = new Atom[idx2.length];
782
783                for (int i = 0 ; i < idx1.length;i++){
784                        //System.out.println("idx1 "+ idx1[i] + " " + idx2[i]);
785                        int pos1 =  idx1[i];
786                        int pos2 =  idx2[i];
787
788                        ca1subset[i] =  ca1[pos1];
789                        ca2subset[i] = (Atom) ca2[pos2].clone();

792                SVDSuperimposer svd = new SVDSuperimposer(ca1subset,ca2subset);

795                //currentRotMatrix.print(3,3);

797                        rotateShiftAtoms(ca2subset);
