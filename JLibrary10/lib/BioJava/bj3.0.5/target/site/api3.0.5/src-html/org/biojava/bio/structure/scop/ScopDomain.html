001package org.biojava.bio.structure.scop;
002
003import java.io.Serializable;
004import java.io.StringWriter;
005import java.util.List;
006
007import javax.xml.bind.annotation.XmlAccessType;
008import javax.xml.bind.annotation.XmlAccessorType;
009import javax.xml.bind.annotation.XmlRootElement;
010
011
012/** Container for the information for a domain. Contains a line in the file
013 * dir.cla.scop.txt_1.75
014 * 
015 * e.g d1dlwa_  1dlw    A:      a.1.1.1 14982   cl=46456,cf=46457,sf=46458,fa=46459,dm=46460,sp=46461,px=14982
016 * @author Andreas Prlic
017 *
018 */
019@XmlRootElement(name = "ScopDomain", namespace ="http://source.rcsb.org")
020@XmlAccessorType(XmlAccessType.PUBLIC_MEMBER)
021public class ScopDomain implements Serializable, Cloneable{
022
023        /**
024         * 
025         */
026        private static final long serialVersionUID = 5890476209571654301L;
027        
028        String scopId;
029        String pdbId;
030        List<String> ranges;
031        String classificationId;
032        Integer sunid;
033
034        int classId;
035        int foldId;
036        int superfamilyId;
037        int familyId;
038        int domainId;
039        int speciesId;
040        int px;
041
042
043
044
045        @Override
046        public String toString() {
047                StringWriter buf = new StringWriter();
048                buf.append(scopId);
049                buf.append("\t") ;
050                buf.append(pdbId);
051                buf.append( "\t");
052
053                int rangePos = 0;
054                for (String range: ranges){
055                        rangePos++;
056
057                        buf.append(range);
058
059                        if ( ( ranges.size()> 1 ) && (rangePos < ranges.size()))
060                                buf.append(",");
061                }
062                buf.append("\t") ;
063                buf.append(classificationId);
064                buf.append("\t") ;
065        buf.append(String.valueOf(sunid));
066                buf.append("\t") ;
067                
068                buf.append("cl=");
069        buf.append(String.valueOf(classId));
070                buf.append(",cf=");
071        buf.append(String.valueOf(foldId));
072                buf.append(",sf=");
073                buf.append(String.valueOf(superfamilyId));
074                buf.append(",fa=");
075                buf.append(String.valueOf(familyId));        
076                buf.append(",dm=");
077        buf.append(String.valueOf(domainId));
078                buf.append(",sp=");
079        buf.append(String.valueOf(speciesId));
080                buf.append(",px=");
081        buf.append(String.valueOf(px));
082                
083                
084                return buf.toString();
085        }
086

088                return scopId;
089        }
090        public void setScopId(String scopId) {
091                this.scopId = scopId;
092        }
093        public String getPdbId() {
094                return pdbId;
095        }
096        public void setPdbId(String pdbId) {
097                this.pdbId = pdbId;
098        }
099        public List<String> getRanges() {
100                return ranges;
101        }
102        public void setRanges(List<String> ranges) {
103                this.ranges = ranges;
104        }
105        public String getClassificationId() {
106                return classificationId;
107        }
108        public void setClassificationId(String classificationId) {
109                this.classificationId = classificationId;
110        }
111        public Integer getSunid() {
112                return sunid;
113        }
114        public void setSunid(Integer sunid) {
115                this.sunid = sunid;
116        }
117        public int getClassId() {
118                return classId;
119        }
120        public void setClassId(int classId) {
121                this.classId = classId;
122        }
123        public int getFoldId() {
124                return foldId;
125        }
126        public void setFoldId(int foldId) {
127                this.foldId = foldId;
128        }
129        public int getSuperfamilyId() {
130                return superfamilyId;
131        }
132        public void setSuperfamilyId(int superfamilyId) {
133                this.superfamilyId = superfamilyId;
134        }
135        public int getFamilyId() {
136                return familyId;
137        }
138        public void setFamilyId(int familyId) {
139                this.familyId = familyId;
140        }
141        public int getDomainId() {
142                return domainId;
143        }
144        public void setDomainId(int domainId) {
145                this.domainId = domainId;
146        }
147        public int getSpeciesId() {
148                return speciesId;
149        }
150        public void setSpeciesId(int speciesId) {
151                this.speciesId = speciesId;
152        }
153        public int getPx() {
154                return px;
155        }
156        public void setPx(int px) {
157                this.px = px;
158        }
159
160        @Override
161        protected Object clone() throws CloneNotSupportedException {
162        
163                
164                ScopDomain n = new ScopDomain();
165                n.setClassId(getClassId());
166                n.setClassificationId(getClassificationId());
167                n.setDomainId(getDomainId());
168                n.setFamilyId(getFamilyId());
169                n.setFoldId(getFoldId());
170                n.setPdbId(getPdbId());
171                n.setPx(getPx());
172                n.setRanges(getRanges());
173                n.setScopId(getScopId());
174                n.setSpeciesId(getSpeciesId());
175                n.setSunid(getSunid());
176                n.setSuperfamilyId(getSuperfamilyId());
177                
178                
179                return n;
180                
181                
182        }
