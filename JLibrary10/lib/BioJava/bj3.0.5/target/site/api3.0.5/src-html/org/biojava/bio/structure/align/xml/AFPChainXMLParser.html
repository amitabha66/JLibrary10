039        /** new utility method that checks that the order of the pair in the XML alignment is correct and flips the direction if needed
040         * 
041         * @param xml
042         * @param name1
043         * @param name1
044         * @param ca1
045         * @param ca2
046         * @return
047         */
048         public static AFPChain fromXML(String xml, String name1, String name2, Atom[] ca1, Atom[] ca2) throws StructureException{
049                        AFPChain[] afps = parseMultiXML( xml);
050                        if ( afps.length > 0 ) {
051
052                                AFPChain afpChain = afps[0];
053                                
054                                String n1 = afpChain.getName1();
055                                String n2 = afpChain.getName2();
056                                
057                                if ( n1 == null )
058                                        n1 = "";
059                                if ( n2 == null)
060                                        n2 = "";
061                                
062                                //System.out.println("from AFPCHAIN: " + n1 + " " + n2);
063                                if ( n1.equals(name2) && n2.equals(name1)){
064                                        // flipped order
065                                        //System.out.println("AfpChain in wrong order, flipping...");
066                                        afpChain  = AFPChainFlipper.flipChain(afpChain);
067                                }
068                                rebuildAFPChain(afpChain, ca1, ca2);
069
070                                return afpChain;
071                        }
072                        return null;
073                 
074         }

076   public static AFPChain fromXML(String xml, Atom[] ca1, Atom[] ca2)
077        {
078                AFPChain[] afps = parseMultiXML( xml);
079                if ( afps.length > 0 ) {
080
081                        AFPChain afpChain = afps[0];
082                        rebuildAFPChain(afpChain, ca1, ca2);
083
084                        return afpChain;
085                }
086                return null;
087        }

104        /** Takes an XML representation of the alignment and flips the positions of name1 and name2
105         * 
106         * @param xml String representing the alignment
107         * @return XML representation of the flipped alignment
108         */
109        public static String flipAlignment(String xml) throws IOException,StructureException{
110                AFPChain[] afps = parseMultiXML( xml);
111                if ( afps.length < 1 ) 
112                        return null;
113
114                if ( afps.length == 1) {
115                        AFPChain newChain = AFPChainFlipper.flipChain(afps[0]);
116                        if ( newChain.getAlgorithmName() == null) {
117                           newChain.setAlgorithmName(AFPChain.DEFAULT_ALGORITHM_NAME);
118                        }
119                        return AFPChainXMLConverter.toXML(newChain);
120                }
121                throw new StructureException("not Implemented yet!");
122        }

125        /**  replace the PDB res nums with atom positions:
126         * 
127         * @param afpChain
128         * @param ca1
129         * @param ca2
130         */
131        public static void rebuildAFPChain(AFPChain afpChain, Atom[] ca1, Atom[] ca2){
132
133           if ( afpChain.getAlgorithmName() == null) {
134              afpChain.setAlgorithmName(AFPChain.DEFAULT_ALGORITHM_NAME);             
135           }
136           if ( afpChain.getVersion() == null){
137              afpChain.setVersion("1.0");
138           }
139           
140                int blockNum  = afpChain.getBlockNum();
141                int ca1Length = afpChain.getCa1Length();
142                int ca2Length = afpChain.getCa2Length();
143
144                int minLength = Math.min(ca1Length, ca2Length);
145                int[][][] optAln = new int[blockNum][2][minLength];
146                                
147                int[][][] blockResList = afpChain.getBlockResList();
148                if ( blockResList == null){
149                   blockResList = new int[blockNum][2][minLength];
150                }
151                int[] optLen = afpChain.getOptLen();
152
153                String[][][] pdbAln = afpChain.getPdbAln();
154                int[] verifiedOptLen = null;
155                if ( optLen != null)
156                  verifiedOptLen = afpChain.getOptLen().clone();
157                else {
158                        System.err.println("did not find optimal alignment, building up empty alignment.");
159                        optLen = new int[1];
160                        optLen[0] = 0;
161                }
162                for (int blockNr = 0 ; blockNr < blockNum ; blockNr++){
163
164                        //System.out.println("got block " + blockNr + " size: " + optLen[blockNr]);
165                   int verifiedEQR = -1;
166                        for ( int eqrNr = 0 ; eqrNr < optLen[blockNr] ; eqrNr++ ){
167                                String pdbResnum1 = pdbAln[blockNr][0][eqrNr];
168                                String pdbResnum2 = pdbAln[blockNr][1][eqrNr];
169
170                                //System.out.println(blockNr + " " + eqrNr + " got resnum: " + pdbResnum1 + " " + pdbResnum2);
171                                String[] spl1 = pdbResnum1.split(":");
172                                String[] spl2 = pdbResnum2.split(":");
173
174                                String chain1 = spl1[0];
175                                String pdbres1 = spl1[1];
176
177                                String chain2 = spl2[0];
178                                String pdbres2 = spl2[1];
179
180                                int pos1 = getPositionForPDBresunm(pdbres1,chain1,ca1);
181                                int pos2 = getPositionForPDBresunm(pdbres2,chain2,ca2);
182                                
183                                if ( pos1 == -1 || pos2 == -1 ){
184                                   // this can happen when parsing old files that contained Calcium atoms...
185                                   System.err.println("AFPChainXMLParser: warning: pos1: " +pos1 + " " + pdbResnum1 + " pos2: " + pos2 + " " + pdbResnum2 +  " should never be -1. Probably parsing an.");
186                                   verifiedOptLen[blockNr]-- ;
187                                   continue;
188                                }
189
190                                verifiedEQR++;
191                                //System.out.println(blockNr + " " + eqrNr + " " + pos1 + " " + pos2);
192                                optAln[blockNr][0][verifiedEQR] = pos1;
193                                optAln[blockNr][1][verifiedEQR] = pos2;
194                                blockResList[blockNr][0][verifiedEQR] = pos1;
195                                blockResList[blockNr][1][verifiedEQR] = pos2;
196                        }
197                }
198
199                afpChain.setOptLen(verifiedOptLen);
200                afpChain.setOptAln(optAln);
201                afpChain.setBlockResList(blockResList);
202                // build up alignment image:
203                AFPAlignmentDisplay.getAlign(afpChain, ca1, ca2);
204
205
206        }

208        public static AFPChain[] parseMultiXML(String xml) {
209                List<AFPChain> afpChains = new ArrayList<AFPChain>();
210
211                try
212                {
213                        //Convert string to XML document
214                        DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
215                        DocumentBuilder db = factory.newDocumentBuilder();
216                        InputSource inStream = new InputSource();
217                        inStream.setCharacterStream(new StringReader(xml));
218                        Document doc = db.parse(inStream);
219
220                        // normalize text representation
221                        doc.getDocumentElement().normalize();
222
223
224                        //Element rootElement = doc.getDocumentElement();
225
226                        NodeList listOfAFPChains = doc.getElementsByTagName("AFPChain");
227                        //int numArrays = listOfArrays.getLength();                     
228                        // go over the blocks
229                        for(int afpPos=0; afpPos<listOfAFPChains.getLength() ; afpPos++)
230                        {
231
232                                AFPChain a = new AFPChain();
233                                a.setAlgorithmName(AFPChain.DEFAULT_ALGORITHM_NAME);
234                                a.setVersion("1.0");
235                                Node rootElement       = listOfAFPChains.item(afpPos);
236
237                                a.setName1(getAttribute(rootElement,"name1"));                          
238                                a.setName2(getAttribute(rootElement,"name2"));
239                                String algoname = getAttribute(rootElement,"method");
240                                if ( algoname != null) {
241                                        a.setAlgorithmName(algoname);                                   
242                                }
243                                String version = getAttribute(rootElement,"version");
244                                if ( version != null)
245                                        a.setVersion(version);
246
247                                a.setAlnLength( new Integer(getAttribute(rootElement,"alnLength")).intValue());
248                                a.setBlockNum(          new Integer(getAttribute(rootElement,"blockNum")).intValue());
249                                a.setGapLen(            new Integer(getAttribute(rootElement,"gapLen")).intValue());
250                                a.setOptLength( new Integer(getAttribute(rootElement,"optLength")).intValue());
251                                a.setTotalLenIni(       new Integer(getAttribute(rootElement,"totalLenIni")).intValue());
252                                a.setBlockNum(          new Integer(getAttribute(rootElement,"blockNum")).intValue());
253
254                                if ( a.getAlgorithmName().equals(CeCPMain.algorithmName)){
255                   a.setSequentialAlignment(a.getBlockNum() == 1);
256                }
257                                
258                                a.setAlignScore(new Double(getAttribute(rootElement,"alignScore")).doubleValue());
259                                a.setChainRmsd(new Double(getAttribute(rootElement,"chainRmsd")).doubleValue());
260                                a.setIdentity(new Double(getAttribute(rootElement,"identity")).doubleValue());
261                                a.setNormAlignScore(new Double(getAttribute(rootElement,"normAlignScore")).doubleValue());
262                                a.setProbability(new Double(getAttribute(rootElement,"probability")).doubleValue());
263                                a.setSimilarity(new Double(getAttribute(rootElement,"similarity")).doubleValue());
264                                a.setTotalRmsdIni(new Double(getAttribute(rootElement,"totalRmsdIni")).doubleValue());
265                                a.setTotalRmsdOpt(new Double(getAttribute(rootElement,"totalRmsdOpt")).doubleValue());
266                                a.setAlignScoreUpdate(new Double(getAttribute(rootElement,"alignScoreUpdate")).doubleValue());
267                                int ca1Length = new Integer(getAttribute(rootElement,"ca1Length")).intValue();
268                                a.setCa1Length(ca1Length);
269                                int ca2Length = new Integer(getAttribute(rootElement,"ca2Length")).intValue();                                          
270                                a.setCa2Length(ca2Length);
271
272                                String tmScoreS = getAttribute(rootElement,"tmScore");
273                                if ( tmScoreS != null) {
274                                        Double tmScore = null;
275                                        try {
276                                         tmScore = Double.parseDouble(tmScoreS);
277                                        } catch (Exception e){                                          
278                                        }
279                                        a.setTMScore(tmScore);
280                                }
281                                
282                                String calcTimeS = getAttribute(rootElement,"time");
283                                Long calcTime = -1L;
284                                if ( calcTimeS != null){
285                                
286                                        try {
287                                                calcTime = Long.parseLong(calcTimeS);
288                                                
289                                        } catch (Exception e){
290                                                e.printStackTrace();
291                                        }                                       
292                                }
293                                a.setCalculationTime(calcTime);
294                                
295                                Matrix[] ms = new Matrix[a.getBlockNum()];
296                                a.setBlockRotationMatrix(ms);
297                                Atom[] blockShiftVector = new Atom[a.getBlockNum()];
298                                a.setBlockShiftVector(blockShiftVector);
299
300                                int afpNum = new Integer(getAttribute(rootElement,"afpNum")).intValue();
301                                List<AFP> afpSet = new ArrayList<AFP>();
302                                for (int afp=0;afp<afpNum;afp++){
303                                        afpSet.add( new AFP()); 
304                                }
305
306                                a.setAfpSet(afpSet);
307
308                                int minLength = Math.min(ca1Length, ca2Length);
309                                a.setFocusRes1(new int[minLength]);
310                                a.setFocusRes2(new int[minLength]);
311
312
313                                //NodeList listOfBlocks = doc.getElementsByTagName("block");
314                                NodeList listOfBlocks = rootElement.getChildNodes();
315
316                                //int numArrays = listOfArrays.getLength();
317
318                                // go over the blocks
319                                for(int i=0; i<listOfBlocks.getLength() ; i++)
320                                {
321                                        Node block       = listOfBlocks.item(i);
322
323                                        // we only look at blocks.
324                                        if (! block.getNodeName().equals("block"))
325                                                continue;
326
327                                        processBlock(block, a, minLength);
328
329
330                                }
331
332                                afpChains.add(a);
333                        }
334                }
335                catch (SAXParseException err) 
336                {
337                        System.out.println ("** Parsing error" + ", line " 
338                                        + err.getLineNumber () + ", uri " + err.getSystemId ());
339                        System.out.println(" " + err.getMessage ());
340                }
341                catch (SAXException e)
342                {
343                        Exception x = e.getException ();
344                        ((x == null) ? e : x).printStackTrace ();
345                }
346                catch (Throwable t)
347                {
348                        t.printStackTrace ();
349                }
350
351                return afpChains.toArray(new AFPChain[afpChains.size()]);
352        }

355        private static  void processBlock(Node block, AFPChain a, int minLength){
356                NodeList valList = block.getChildNodes();
357                int numChildren  = valList.getLength();
358
359                NamedNodeMap map = block.getAttributes();
360
361                int blockNum = a.getBlockNum();
362
363                int[]     optLen                        = a.getOptLen();
364                if ( optLen == null ) 
365                        optLen = new int[blockNum];
366
367                String[][][] pdbAln = a.getPdbAln();
368                if ( pdbAln == null)
369                        pdbAln         = new String[blockNum][2][minLength];
370
371                //int[][][] optAln                      = new int[blockNum][2][minLength];
372                int[]     blockGap = a.getBlockGap();
373                if ( blockGap == null )
374                        blockGap = new int[blockNum];
375                int[]     blockSize= a.getBlockSize();
376                if ( blockSize == null)
377                        blockSize = new int[blockNum];
378
379                double[]  blockScore = a.getBlockScore();
380                if ( blockScore == null)
381                        blockScore = new double[blockNum];
382                double[]  blockRmsd = a.getBlockRmsd();
383                if (blockRmsd == null )
384                        blockRmsd = new double[blockNum];
385                Matrix[] ms     = a.getBlockRotationMatrix();
386                Atom[] shifts = a.getBlockShiftVector();
387
388                int blockNr = new Integer( map.getNamedItem("blockNr").getTextContent()).intValue();
389
390                int thisBlockGap = new Integer(map.getNamedItem("blockGap").getTextContent()).intValue();
391                blockGap[blockNr] = thisBlockGap;
392
393                int thisBlockSize = new Integer(map.getNamedItem("blockSize").getTextContent()).intValue();
394                blockSize[blockNr] = thisBlockSize;
395
396                double thisBlockScore = new Double(map.getNamedItem("blockScore").getTextContent()).doubleValue();
397                blockScore[blockNr] = thisBlockScore;
398
399                double thisBlockRmsd = new Double(map.getNamedItem("blockRmsd").getTextContent()).doubleValue();
400                blockRmsd[blockNr] = thisBlockRmsd;
401
402
403                // parse out the equivalent positions from the file
404                int nrEqr = 0;
405                for ( int e =0; e< numChildren ; e++){
406                        Node  eqr = valList.item(e);
407
408                        if(!eqr.hasAttributes()) continue;
409
410
411                        if ( eqr.getNodeName().equals("eqr")) {
412                                nrEqr++;
413                                NamedNodeMap atts = eqr.getAttributes();
414
415                                int eqrNr = new Integer(atts.getNamedItem("eqrNr").getTextContent()).intValue();
416
417                                String pdbres1 = atts.getNamedItem("pdbres1").getTextContent();
418                                String chain1 = atts.getNamedItem("chain1").getTextContent();
419                                String pdbres2 = atts.getNamedItem("pdbres2").getTextContent();
420                                String chain2 = atts.getNamedItem("chain2").getTextContent();
421
422                                //System.out.println(blockNr + " " + eqrNr + " " + chain1+" " + pdbres1 + ":" + chain2 + " " + pdbres2);
423
424                                pdbAln[blockNr][0][eqrNr] = chain1+":"+pdbres1;
425                                pdbAln[blockNr][1][eqrNr] = chain2+":"+pdbres2;
426
427                                //  A WORK AROUND FOR THE PROBLEM THAT WE DON:T HAVE PDBs LOADED AT THIS TIME...
428
429                                /* int pos1 = getPositionForPDBresunm(pdbres1,chain1,ca1);
430                                int pos2 = getPositionForPDBresunm(pdbres2,chain2,ca2);
431                                //System.out.println("settion optAln " + blockNr + " " + eqrNr + " " + pos1);
432                                optAln[blockNr][0][eqrNr] = pos1;
433                                optAln[blockNr][1][eqrNr] = pos2;
434                                 */
435                        } else if ( eqr.getNodeName().equals("matrix")){
436                                // process Matrix
437                                Matrix m = new Matrix(3,3);
438
439                                for (int i =1 ; i <= 3 ; i++){
440                                        for (int j =1 ; j <= 3 ; j++){
441                                                String att = getAttribute(eqr, "mat" +i + j);
442                                                double val = Double.parseDouble(att);
443                                                m.set(i-1,j-1,val);
444
445                                        }       
446                                }
447                                ms[blockNr] = m;
448
449                        } else if ( eqr.getNodeName().equals("shift")){
450                                Atom shift = new AtomImpl();
451                                double x = Double.parseDouble(getAttribute(eqr, "x"));
452                                double y = Double.parseDouble(getAttribute(eqr, "y"));
453                                double z = Double.parseDouble(getAttribute(eqr, "z"));
454                                shift.setX(x);
455                                shift.setY(y);
456                                shift.setZ(z);
457                                shifts[blockNr] = shift;
458
459                        }
460
461                }
462                //System.out.println("setting block " + blockNr + " eqr: " + nrEqr);
463                optLen[blockNr] = nrEqr;
464
465
466
467                a.setOptLen(optLen);
468                //a.setOptAln(optAln);
469                a.setPdbAln(pdbAln);
470                a.setBlockGap(blockGap);
471                a.setBlockSize(blockSize);
472
473                a.setBlockScore(blockScore);
474                a.setBlockRmsd(blockRmsd);
475
476
477
478        }

480        private static String getAttribute(Node node, String attr){
481                if( ! node.hasAttributes()) 
482                        return null;
483
484                NamedNodeMap atts = node.getAttributes();
485
486                if ( atts == null)
487                        return null;
488
489                Node att = atts.getNamedItem(attr);
490                if ( att == null)
491                        return null;
492
493                String value = att.getTextContent();
494
495                return value;
496
497        }

499        /** get the position of PDB residue nr X in the ato marray
500         * 
501         * @param pdbresnum
502         * @param chainId
503         * @param atoms
504         * @return
505         */
506        private static int getPositionForPDBresunm(String pdbresnum, String chainId , Atom[] atoms){
507                ResidueNumber residueNumber =  ResidueNumber.fromString(pdbresnum);
508                residueNumber.setChainId(chainId);
509                
510                for ( int i =0; i< atoms.length ;i++){
511                        Group g = atoms[i].getGroup();
512                        
513                        //System.out.println(g.getResidueNumber() + "< ? >" + residueNumber +"<");
514                        if ( g.getResidueNumber().equals(residueNumber)){
515                                //System.out.println(g + " == " + residueNumber );
516                                Chain c = g.getChain();
517                                if ( c.getChainID().equals(chainId)){
518                                        return i;
519                                }
520                        }
521                }
522                return -1;
523        }
