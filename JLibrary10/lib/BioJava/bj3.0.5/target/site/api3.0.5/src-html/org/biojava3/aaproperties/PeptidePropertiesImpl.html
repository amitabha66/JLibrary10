034        private double getWaterMoleculeWeight(){
035                final double hydrogenMW = 1.0079;
036                final double hydroxideMW = 17.0073;
037                //H     1.0079  OH      17.0073
038                return hydrogenMW + hydroxideMW;

049        @Override
050        public double getMolecularWeight(ProteinSequence sequence) {
051                double value = 0.0;
052                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
053                char[] seq = getSequence(sequence.toString(), true);//ignore case
054                for(char aa:seq){
055                        AminoAcidCompound c = aaSet.getCompoundForString(aa + "");
056                        if(Constraints.aa2MolecularWeight.containsKey(c)){
057                                value += Constraints.aa2MolecularWeight.get(c);
058                        }
059                }
060                if(value == 0)
061                        return value;
062                else
063                        return value + getWaterMoleculeWeight();
064        }

083        public double getMolecularWeightBasedOnXML(ProteinSequence sequence, AminoAcidCompositionTable aminoAcidCompositionTable){
084                double value = 0.0;
085                char[] seq = sequence.toString().toCharArray();
086                for(char aa:seq){
087                        Double weight = aminoAcidCompositionTable.getMolecularWeight(aa);
088                        if(weight != null){
089                                value += weight;
090                        }
091                }
092                if(value == 0.0)
093                        return value;
094                else
095                        return value + getWaterMoleculeWeight();

109        @Override
110        public AminoAcidCompositionTable obtainAminoAcidCompositionTable(File elementMassFile, File aminoAcidCompositionFile) 
111                throws JAXBException, FileNotFoundException{
112                //Parse elementMassFile
113                ElementTable iTable = new ElementTable();
114                // Get a JAXB Context for the object we created above
115                JAXBContext jc = JAXBContext.newInstance(iTable.getClass());
116                Unmarshaller u = jc.createUnmarshaller();
117                u.setEventHandler(new MyValidationEventHandler()); 
118                iTable = (ElementTable)u.unmarshal(new FileInputStream(elementMassFile));
119                iTable.populateMaps();
120                
121                //Parse aminoAcidCompositionFile
122                AminoAcidCompositionTable aTable = new AminoAcidCompositionTable();
123                // Get a JAXB Context for the object we created above
124                JAXBContext jc2 = JAXBContext.newInstance(aTable.getClass());
125                Unmarshaller u2 = jc2.createUnmarshaller();
126                u2.setEventHandler(new MyValidationEventHandler()); 
127                aTable = (AminoAcidCompositionTable)u2.unmarshal(new FileInputStream(aminoAcidCompositionFile));
128                aTable.computeMolecularWeight(iTable);
129                return aTable;
130        }

132        @Override
133        public double getExtinctionCoefficient(ProteinSequence sequence, boolean assumeCysReduced) {
134                //Tyr => Y
135                //Trp => W
136                //Cys => C
137                //E(Prot) = Numb(Tyr)*Ext(Tyr) + Numb(Trp)*Ext(Trp) + Numb(Cystine)*Ext(Cystine)
138                //where (for proteins in water measured at 280 nm): Ext(Tyr) = 1490, Ext(Trp) = 5500, Ext(Cystine) = 125;
139                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
140                Map<AminoAcidCompound, Integer> extinctAA2Count = this.getExtinctAACount(sequence);
141                
142                double eProt;
143                if(!assumeCysReduced){
144                        eProt = extinctAA2Count.get(aaSet.getCompoundForString("Y")) * 
145                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("Y")) + 
146                                extinctAA2Count.get(aaSet.getCompoundForString("W")) * 
147                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("W")) +
148                                extinctAA2Count.get(aaSet.getCompoundForString("C")) * 
149                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("C"));
150                }else
151                        eProt = extinctAA2Count.get(aaSet.getCompoundForString("Y")) * 
152                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("Y")) + 
153                                extinctAA2Count.get(aaSet.getCompoundForString("W")) *
154                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("W"));
155                
156                return eProt;
157        }

159        @Override
160        public double getAbsorbance(ProteinSequence sequence, boolean assumeCysReduced){
161                //Absorb(Prot) = E(Prot) / Molecular_weight
162                double mw = this.getMolecularWeight(sequence);
163                double eProt = this.getExtinctionCoefficient(sequence, assumeCysReduced);
164                return eProt / mw;
165        }

167        private Map<AminoAcidCompound, Integer> getExtinctAACount(ProteinSequence sequence){
168                //Cys => C, Tyr => Y, Trp => W
169                int numW = 0;
170                int smallW = 0;
171                double numC = 0;
172                double smallC = 0; 
173                int numY = 0;
174                int smallY = 0;
175                for(char aa:sequence.getSequenceAsString().toCharArray()){
176                        switch(aa){
177                        case 'W': numW++; break;
178                        case 'w': smallW++; break;
179                        case 'C': numC += 0.5; break;
180                        case 'c': smallC += 0.5; break;
181                        case 'Y': numY++; break;
182                        case 'y': smallY++; break;
183                        }
184                }
185                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
186                Map<AminoAcidCompound, Integer> extinctAA2Count = new HashMap<AminoAcidCompound, Integer>();
187                //Ignore Case is always true
188                extinctAA2Count.put(aaSet.getCompoundForString("W"), numW + smallW);
189                extinctAA2Count.put(aaSet.getCompoundForString("C"), (int) (numC + smallC));
190                extinctAA2Count.put(aaSet.getCompoundForString("Y"), numY + smallY);
191                return extinctAA2Count;
192        }

195        public double getInstabilityIndex(ProteinSequence sequence) {
196                double sum = 0.0;
197                String s = sequence.getSequenceAsString().toUpperCase();
198                for(int i = 0; i < sequence.getLength() - 1; i++){
199                        String dipeptide = s.substring(i, i+2);
200                        if(Constraints.diAA2Instability.containsKey(dipeptide)){
201                                sum += Constraints.diAA2Instability.get(dipeptide);
202                        }
203                }
204                return sum * 10 / (s.length() - Utils.getNumberOfInvalidChar(s, null, true));
205        }

207        @Override
208        public double getApliphaticIndex(ProteinSequence sequence) {
209//              Aliphatic index = X(Ala) + a * X(Val) + b * ( X(Ile) + X(Leu) )  
210//              where X(Ala), X(Val), X(Ile), and X(Leu) are mole percent (100 X mole fraction) 
211//              of alanine, valine, isoleucine, and leucine. 
212//              The coefficients a and b are the relative volume of valine side chain (a = 2.9) 
213//              and of Leu/Ile side chains (b = 3.9) to the side chain of alanine. 
214//              Ala => A, Val => V, Ile => I, Leu => L
215                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
216                Map<AminoAcidCompound, Double> aa2Composition = getAAComposition(sequence);
217                final double a = 2.9;
218                final double b = 3.9;
219                double xAla = aa2Composition.get(aaSet.getCompoundForString("A"));
220                double xVal = aa2Composition.get(aaSet.getCompoundForString("V"));
221                double xIle = aa2Composition.get(aaSet.getCompoundForString("I"));
222                double xLeu = aa2Composition.get(aaSet.getCompoundForString("L"));
223                return (xAla + (a * xVal) + (b * (xIle + xLeu))) * 100;
224        }

251        private double getIsoelectricPointInnovagen(ProteinSequence sequence){
252                double currentPH = 7.0;
253                double changeSize = 7.0;
254                String sequenceString = sequence.toString();
255                char nTerminalChar = sequenceString.charAt(0);
256                char cTerminalChar = sequenceString.charAt(sequenceString.length() - 1);
257                
258                Map<AminoAcidCompound, Integer> chargedAA2Count = this.getChargedAACount(sequence);
259                double margin;
260                final double difference = 0.0001;
261                
262                while(true){
263                        margin = this.getNetChargeInnovagen(chargedAA2Count, currentPH, nTerminalChar, cTerminalChar);
264                        //Within allowed difference
265                        if(margin <= difference && margin >= -difference) break;
266                        changeSize /= 2.0;
267                        if(margin > 0){
268                                currentPH += changeSize;
269                        }else{
270                                currentPH -= changeSize;
271                        }
272                }
273                return currentPH;
274        }

317        private final double PH_MIN = 0.0; /* minimum pH value */
318        private final double PH_MAX = 14.0; /* maximum pH value */
319        private final double MAXLOOP = 2000.0; /* maximum number of iterations */
320        private final double EPSI = 0.0001; /* desired precision */
321        
322        private double exp10(double pka){
323                return Math.pow(10, pka);
324        }

326        private double getIsoelectricPointExpasy(String sequence){
327                //
328                // Compute the amino-acid composition.
329                //
330                int comp[] = new int[26];
331                for(int i = 0; i < sequence.length(); i++){
332                        int index = sequence.charAt(i) - 'A';
333                        if(index < 0 || index >= 26) continue;
334                        comp[index]++;
335                }
336                //
337                // Look up N-terminal and C-terminal residue.
338                //
339                int nTermResidue = -1;
340                int index = 0;
341                while((nTermResidue < 0 || nTermResidue >= 26) && index < 25){
342                        nTermResidue = sequence.charAt(index++) - 'A';
343                }
344                
345                int cTermResidue = -1;
346                index = 1;
347                while((cTermResidue < 0 || cTermResidue >= 26) && index < 25){
348                        cTermResidue = sequence.charAt(sequence.length() - index++) - 'A';
349                }
350
351                double phMin = PH_MIN;
352                double phMax = PH_MAX;
353                
354                double phMid = 0.0;
355                double charge = 1.0;
356                for (int i = 0; i < MAXLOOP && (phMax - phMin) > EPSI; i++){
357                        phMid = phMin + (phMax - phMin) / 2.0;
358        
359                        charge = getNetChargeExpasy(comp, nTermResidue, cTermResidue, phMid);
360        
361                        if (charge > 0.0) phMin = phMid;
362                        else phMax = phMid;
363                }
364                return phMid;
365        }

391        private double getNetChargeExpasy(String sequence, double pHPoint){
392                //
393                // Compute the amino-acid composition.
394                //
395                int comp[] = new int[26];
396                for(int i = 0; i < sequence.length(); i++){
397                        int index = sequence.charAt(i) - 'A';
398                        if(index < 0 || index >= 26) continue;
399                        comp[index]++;
400                }
401                //
402                // Look up N-terminal and C-terminal residue.
403                //
404                int nTermResidue = sequence.charAt(0) - 'A';
405                int cTermResidue = sequence.charAt(sequence.length() - 1) - 'A';
406                return getNetChargeExpasy(comp, nTermResidue, cTermResidue, pHPoint);
407        }

409        private double getNetChargeExpasy(int comp[], int nTermResidue, int cTermResidue, double ph){
410                double cter = 0.0; 
411                if(cTermResidue >= 0 && cTermResidue < 26) cter = exp10(-cPk[cTermResidue][0]) / (exp10(-cPk[cTermResidue][0]) + exp10(-ph));
412                double nter = 0.0; 
413                if(nTermResidue >= 0 && nTermResidue < 26) nter = exp10(-ph) / (exp10(-cPk[nTermResidue][1]) + exp10(-ph));
414
415                double carg = comp['R' - 'A'] * exp10(-ph) / (exp10(-cPk['R' - 'A'][2]) + exp10(-ph)); 
416                double chis = comp['H' - 'A'] * exp10(-ph) / (exp10(-cPk['H' - 'A'][2]) + exp10(-ph));
417                double clys = comp['K' - 'A'] * exp10(-ph) / (exp10(-cPk['K' - 'A'][2]) + exp10(-ph));
418
419                double casp = comp['D' - 'A'] * exp10(-cPk['D' - 'A'][2]) / (exp10(-cPk['D' - 'A'][2]) + exp10(-ph));
420                double cglu = comp['E' - 'A'] * exp10(-cPk['E' - 'A'][2]) / (exp10(-cPk['E' - 'A'][2]) + exp10(-ph));
421
422                double ccys = comp['C' - 'A'] * exp10(-cPk['C' - 'A'][2]) / (exp10(-cPk['C' - 'A'][2]) + exp10(-ph));
423                double ctyr = comp['Y' - 'A'] * exp10(-cPk['Y' - 'A'][2]) / (exp10(-cPk['Y' - 'A'][2]) + exp10(-ph));
424
425                return (carg + clys + chis + nter) - (casp + cglu + ctyr + ccys + cter);
426        }

434        private double getNetChargeInnovagen(Map<AminoAcidCompound, Integer> chargedAA2Count, double ph, char nTerminalChar, char cTerminalChar){
435                //Constraints.aa2PKa is aleady reinitialized in getChargedAACount hence no need to do it again
436                
437                //Lys => K, Arg => R, His => H
438                //Asp => D, Glu => E, Cys => C, Tyr => Y
439                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
440                
441                double nTerminalCharge = 0.0; 
442                AminoAcidCompound nTermCompound = aaSet.getCompoundForString(nTerminalChar + "");
443                if(Constraints.aa2NTerminalPka.containsKey(nTermCompound)){
444                        nTerminalCharge = this.getPosCharge(Constraints.aa2NTerminalPka.get(nTermCompound), ph);
445                }                       
446                
447                double cTerminalCharge = 0.0;
448                AminoAcidCompound cTermCompound = aaSet.getCompoundForString(cTerminalChar + "");
449                if(Constraints.aa2CTerminalPka.containsKey(cTermCompound)){
450                        cTerminalCharge = this.getNegCharge(Constraints.aa2CTerminalPka.get(cTermCompound), ph);
451                }
452                
453                double kCharge = chargedAA2Count.get(aaSet.getCompoundForString("K")) * this.getPosCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("K")), ph);
454                double rCharge = chargedAA2Count.get(aaSet.getCompoundForString("R")) * this.getPosCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("R")), ph);
455                double hCharge = chargedAA2Count.get(aaSet.getCompoundForString("H")) * this.getPosCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("H")), ph);
456                double dCharge = chargedAA2Count.get(aaSet.getCompoundForString("D")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("D")), ph);
457                double eCharge = chargedAA2Count.get(aaSet.getCompoundForString("E")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("E")), ph);
458                double cCharge = chargedAA2Count.get(aaSet.getCompoundForString("C")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("C")), ph);
459                double yCharge = chargedAA2Count.get(aaSet.getCompoundForString("Y")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("Y")), ph);
460//              if((kCharge + rCharge + hCharge) == 0.0 && (dCharge + eCharge + cCharge + yCharge) == 0.0){
461//                      return 0.0;
462//              }
463                return (nTerminalCharge + kCharge + rCharge + hCharge) - (dCharge + eCharge + cCharge + yCharge + cTerminalCharge);
464        }

474        private Map<AminoAcidCompound, Integer> getChargedAACount(ProteinSequence sequence){
475                //Lys => K, Arg => R, His => H
476                //Asp => D, Glu => E, Cys => C, Tyr => Y
477                int numK = 0;
478                int numR = 0;
479                int numH = 0;
480                int numD = 0;
481                int numE = 0;
482                int numC = 0;
483                int numY = 0;
484                char[] seq = this.getSequence(sequence.getSequenceAsString(), true);
485                for(char aa:seq){
486                        switch(aa){
487                        case 'K': numK++; break;
488                        case 'R': numR++; break;
489                        case 'H': numH++; break;
490                        case 'D': numD++; break;
491                        case 'E': numE++; break;
492                        case 'C': numC++; break;
493                        case 'Y': numY++; break;
494                        }
495                }
496                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
497                Map<AminoAcidCompound, Integer> chargedAA2Count = new HashMap<AminoAcidCompound, Integer>();
498                chargedAA2Count.put(aaSet.getCompoundForString("K"), numK);
499                chargedAA2Count.put(aaSet.getCompoundForString("R"), numR);
500                chargedAA2Count.put(aaSet.getCompoundForString("H"), numH);
501                chargedAA2Count.put(aaSet.getCompoundForString("D"), numD);
502                chargedAA2Count.put(aaSet.getCompoundForString("E"), numE);
503                chargedAA2Count.put(aaSet.getCompoundForString("C"), numC);
504                chargedAA2Count.put(aaSet.getCompoundForString("Y"), numY);
505                return chargedAA2Count;
506        }

521        public Map<AminoAcidCompound, Double> getAAComposition(ProteinSequence sequence) {
522                int validLength = 0;
523                Map<AminoAcidCompound, Double> aa2Composition = new HashMap<AminoAcidCompound, Double>();
524                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
525                for(AminoAcidCompound aa:aaSet.getAllCompounds()){
526                        aa2Composition.put(aa, 0.0);
527                }
528                char[] seq = this.getSequence(sequence.toString(), true);
529                for(char aa:seq){
530                        if(PeptideProperties.standardAASet.contains(aa)){
531                                AminoAcidCompound compound = aaSet.getCompoundForString(aa + "");
532                                aa2Composition.put(compound, aa2Composition.get(compound) + 1.0);
533                                validLength++;
534                        }
535                }
536                if(validLength > 0){
537                        for(AminoAcidCompound aa:aaSet.getAllCompounds()){
538                                aa2Composition.put(aa, aa2Composition.get(aa) / validLength);
539                        }
540                }else{
541                        for(AminoAcidCompound aa:aaSet.getAllCompounds()){
542                                aa2Composition.put(aa, 0.0);
543                        }
544                }
545                return aa2Composition;
546        }
