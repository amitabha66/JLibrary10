063/**
064 * 
065 * Pass in a Uniprot ID and this ProxySequenceReader when passed to a ProteinSequence will get the sequence data and other data elements
066 * associated with the ProteinSequence by Uniprot. This is an example of how to map external databases of proteins and features to the BioJava3
067 * ProteinSequence.
068 * Important to call @see setUniprotDirectoryCache to allow caching of XML files so they don't need to be reloaded each time. Does
069 * not manage cache.
070 * @param <C>
071 */
072public class UniprotProxySequenceReader<C extends Compound> implements ProxySequenceReader<C>, FeaturesKeyWordInterface, DatabaseReferenceInterface {
073

076    private static String uniprotDirectoryCache = null;
077    private String sequence;
078    private CompoundSet<C> compoundSet;
079    private List<C> parsedCompounds = new ArrayList<C>();
080    Document uniprotDoc;

082    /**
083     * The uniprot id is used to retrieve the uniprot XML which is then parsed as a DOM object
084     * so we know everything about the protein. If an error occurs throw an exception. We could
085     * have a bad uniprot id or network error
086     * @param accession
087     * @param compoundSet
088     * @throws Exception
089     */
090    public UniprotProxySequenceReader(String accession, CompoundSet<C> compoundSet) throws Exception {
091
092        setCompoundSet(compoundSet);
093        uniprotDoc = this.getUniprotXML(accession);
094        String seq = this.getSequence(uniprotDoc);
095        setContents(seq);
096    }

102    /**
103     * Once the sequence is retrieved set the contents and make sure everything this is valid
104     * @param sequence
105     */
106    public void setContents(String sequence) {
107        // Horrendously inefficient - pretty much the way the old BJ did things.
108        // TODO Should be optimised.
109        this.sequence = sequence;
110        this.parsedCompounds.clear();
111        for (int i = 0; i < sequence.length();) {
112            String compoundStr = null;
113            C compound = null;
114            for (int compoundStrLength = 1; compound == null && compoundStrLength <= compoundSet.getMaxSingleCompoundStringLength(); compoundStrLength++) {
115                compoundStr = sequence.substring(i, i + compoundStrLength);
116                compound = compoundSet.getCompoundForString(compoundStr);
117            }
118            if (compound == null) {
119                throw new CompoundNotFoundError(compoundStr);
120            } else {
121                i += compoundStr.length();
122            }
123            this.parsedCompounds.add(compound);
124        }
125    }

234    /**
235     *
236     * @return
237     */
238    public AccessionID getAccession() {
239        AccessionID accessionID = new AccessionID();
240        if (uniprotDoc == null) {
241            return accessionID;
242        }
243        try {
244            Element uniprotElement = uniprotDoc.getDocumentElement();
245            Element entryElement = XMLHelper.selectSingleElement(uniprotElement, "entry");
246            Element nameElement = XMLHelper.selectSingleElement(entryElement, "name");
247            accessionID = new AccessionID(nameElement.getTextContent(), DataSource.UNIPROT);
248        } catch (Exception e) {
249            e.printStackTrace();
250        }
251        return accessionID;
252    }

263    /**
264     *
265     * @param accession
266     * @return
267     * @throws Exception
268     */
269    private Document getUniprotXML(String accession) throws Exception {
270        int index = accession.lastIndexOf(".");
271        String key = accession;
272        if (index != -1) {
273            key = accession.substring(0, index);
274        }
275        StringBuilder sb = new StringBuilder();
276        File f = null;
277        if (uniprotDirectoryCache != null && uniprotDirectoryCache.length() > 0) {
278            f = new File(uniprotDirectoryCache + File.separatorChar + key + ".xml");
279            if (f.exists()) {
280                FileReader fr = new FileReader(f);
281                int size = (int) f.length();
282                char[] data = new char[size];
283                fr.read(data);
284                fr.close();
285                sb.append(data);
286                index = sb.indexOf("xmlns="); //strip out name space stuff to make it easier on xpath
287                if (index != -1) {
288                    int lastIndex = sb.indexOf(">", index);
289                    sb.replace(index, lastIndex, "");
290                }
291            }
292
293        }
294
295        // http://www.uniprot.org/uniprot/?query=SORBIDRAFT_03g027040&format=xml
296        if (sb.length() == 0) {
297            String uniprotURL = getUniprotbaseURL() + "/uniprot/?query=" + key + "&format=xml";
298            logger.info("Loading " + uniprotURL);
299            URL uniprot = new URL(uniprotURL);
300            URLConnection uniprotConnection = uniprot.openConnection();
301            BufferedReader in = new BufferedReader(
302                    new InputStreamReader(
303                    uniprotConnection.getInputStream()));
304            String inputLine;
305
306            while ((inputLine = in.readLine()) != null) {
307                sb.append(inputLine);
308            }
309            in.close();
310            index = sb.indexOf("xmlns="); //strip out name space stuff to make it easier on xpath
311            if (index != -1) {
312                int lastIndex = sb.indexOf(">", index);
313                sb.replace(index, lastIndex, "");
314            }
315            if (f != null) {
316                FileWriter fw = new FileWriter(f);
317                fw.write(sb.toString());
318                fw.close();
319            }
320        }
321
322        logger.info("Load complete");
323        try {
324            //       System.out.println(sb.toString());
325            Document document = XMLHelper.inputStreamToDocument(new ByteArrayInputStream(sb.toString().getBytes()));
326            return document;
327        } catch (Exception e) {
328            System.out.println("Exception on xml parse of:" + sb.toString());
329        }
330        return null;
331    }

339    private String getSequence(Document uniprotDoc) throws Exception {
340        Element uniprotElement = uniprotDoc.getDocumentElement();
341        Element entryElement = XMLHelper.selectSingleElement(uniprotElement, "entry");
342        Element sequenceElement = XMLHelper.selectSingleElement(entryElement, "sequence");
343
344        String seqdata = sequenceElement.getTextContent();
345
346        return seqdata;
347    }

397    /**
398     * Get the gene name associated with this sequence. 
399     * @return
400     * @throws Exception
401     */
402    public String getGeneName() throws Exception {
403        if (uniprotDoc == null) {
404            return "";
405        }
406        Element uniprotElement = uniprotDoc.getDocumentElement();
407        Element entryElement = XMLHelper.selectSingleElement(uniprotElement, "entry");
408        Element geneElement = XMLHelper.selectSingleElement(entryElement, "gene");
409        if (geneElement == null) {
410            return "";
411        }
412        Element nameElement = XMLHelper.selectSingleElement(geneElement, "name");
413        if (nameElement == null) {
414            return "";
415        }
416        return nameElement.getTextContent();
417    }

441    /**
442     * Pull uniprot key words which is a mixed bag of words associated with this sequence
443     * @return
444     * @throws Exception
445     */
446    public ArrayList<String> getKeyWords() throws Exception {
447        ArrayList<String> keyWordsList = new ArrayList<String>();
448        if (uniprotDoc == null) {
449            return keyWordsList;
450        }
451        Element uniprotElement = uniprotDoc.getDocumentElement();
452        Element entryElement = XMLHelper.selectSingleElement(uniprotElement, "entry");
453        ArrayList<Element> keyWordElementList = XMLHelper.selectElements(entryElement, "keyword");
454        for (Element element : keyWordElementList) {
455            keyWordsList.add(element.getTextContent());
456        }
457
458        return keyWordsList;
459    }

461    /**
462     * The Uniprot mappings to other database identifiers for this sequence
463     * @return
464     * @throws Exception
465     */
466    public LinkedHashMap<String, ArrayList<DBReferenceInfo>> getDatabaseReferences() throws Exception {
467        LinkedHashMap<String, ArrayList<DBReferenceInfo>> databaseReferencesHashMap = new LinkedHashMap<String, ArrayList<DBReferenceInfo>>();
468        if (uniprotDoc == null) {
469            return databaseReferencesHashMap;
470        }
471
472        Element uniprotElement = uniprotDoc.getDocumentElement();
473        Element entryElement = XMLHelper.selectSingleElement(uniprotElement, "entry");
474        ArrayList<Element> dbreferenceElementList = XMLHelper.selectElements(entryElement, "dbReference");
475        for (Element element : dbreferenceElementList) {
476            String type = element.getAttribute("type");
477            String id = element.getAttribute("id");
478            ArrayList<DBReferenceInfo> idlist = databaseReferencesHashMap.get(type);
479            if (idlist == null) {
480                idlist = new ArrayList<DBReferenceInfo>();
481                databaseReferencesHashMap.put(type, idlist);
482            }
483            DBReferenceInfo dbreferenceInfo = new DBReferenceInfo(type, id);
484            ArrayList<Element> propertyElementList = XMLHelper.selectElements(element, "property");
485            for (Element propertyElement : propertyElementList) {
486                String propertyType = propertyElement.getAttribute("type");
487                String propertyValue = propertyElement.getAttribute("value");
488                dbreferenceInfo.addProperty(propertyType, propertyValue);
489            }
490
491            idlist.add(dbreferenceInfo);
492        }
493
494
495        return databaseReferencesHashMap;
496    }
