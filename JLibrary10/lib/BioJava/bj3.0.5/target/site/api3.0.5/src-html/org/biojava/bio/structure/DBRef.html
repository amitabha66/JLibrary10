121    /** Append the PDB representation of this DBRef to the provided StringBuffer
122     *
123     * @param buf the StringBuffer to write to.
124     */
125    public void toPDB(StringBuffer buf){
126         Formatter formatter = new Formatter(new StringBuilder(),Locale.UK);
127//        DBREF  3ETA A  990  1295  UNP    P06213   INSR_HUMAN    1017   1322
128//        DBREF  3EH2 A    2   767  UNP    P53992   SC24C_HUMAN    329   1094
129//        DBREF 3EH2 A    2   767     UNP   P53992  SC24C_HUMAN   329   1094
130//        DBREF  3ETA A  990  1295  UNP    P06213   INSR_HUMAN    1017   1322
131        formatter.format("DBREF  %4s %1s %4d%1s %4d%1s %-6s %-8s %-12s%6d%1c%6d%1c",
132                idCode, chainId,seqbegin,insertBegin,seqEnd,insertEnd,
133                database,dbAccession,dbIdCode,
134                dbSeqBegin,idbnsBegin,dbSeqEnd,idbnsEnd
135                );
136
137        buf.append(formatter.toString().trim());
138        formatter.close();
139
140    }
141    /** String representation of a DBRef.
142     * @return a String
143     */
144    public String toString(){
145        StringBuffer buf = new StringBuffer();
146
147        try {
148
149            @SuppressWarnings("rawtypes")
150                        Class c = Class.forName("org.biojava.bio.structure.DBRef");
151            Method[] methods  = c.getMethods();
152
153            for (int i = 0; i < methods.length; i++) {
154                Method m = methods[i];
155
156                String name = m.getName();
157
158                if ( name.substring(0,3).equals("get")) {
159                    if (name.equals("getClass"))
160                            continue;
161                    Object o  = m.invoke(this, new Object[]{});
162                    if ( o != null){
163                        buf.append(name.substring(3,name.length()));
164                        buf.append(": " + o + " ");
165                    }
166                }
167            }
168        } catch (Exception e){
169            e.printStackTrace();
170        }
171
172        return buf.toString();
173    }
