340        private void registerBridge(int start, int end, BridgeType btype){
341                if ( start > end)
342                        return;
343
344                boolean found = false;
345                for (Ladder ladder : ladders){
346                        if (shouldExtendLadder(ladder, start,end,btype)) {
347                                // extend laddder
348                                found = true;
349                                ladder.to++;
350                                if ( btype.equals(BridgeType.parallel)) {
351                                        ladder.lto++;
352                                } else {
353                                        ladder.lfrom--;
354                                }
355                                break;
356
357                        }
358                }
359                if ( ! found){
360                        // create new ladder!
361                        Ladder l = new Ladder();
362                        l.setFrom(start);
363                        l.setTo(end);
364                        l.setBtype(btype);
365                        l.setLfrom(start);
366                        l.setLto(end);
367                        ladders.add(l);
368                }
369
370        }

372        private boolean shouldExtendLadder(Ladder ladder, int start, int end,
373                        BridgeType btype) {
374
375                return ( (btype.equals(ladder.getBtype()) ) &&
376                                ( start  == ladder.getTo() +1) &&
377                                (
378                                                (( end == ladder.getLto() + 1) &&
379                                                                ( btype.equals(BridgeType.parallel) )) ||
380                                                                (( end == ladder.getLfrom() - 1 ) &&
381                                                                                (btype.equals(BridgeType.parallel) ))                                                                           
382                                                ) );
383
384
385
386        }

388        private void testBridge(int i) {
389
390                int currpos = i + 3;
391                for ( int foundNrBridges = 0 ; foundNrBridges < 2 && ( currpos < groups.length-1); currpos++){
392
393
394
395                        BridgeType btype = null;
396
397                        if ( (isBonded(i+1, currpos) && isBonded(currpos, i-1) ) ||
398                                        ( isBonded(currpos +1 , i) && isBonded(i, currpos-1)) ) {
399                                // parallel
400                                btype = BridgeType.parallel;
401                        }
402                        else if ( (isBonded(i, currpos) && isBonded(currpos, i)) ||
403                                        (isBonded(i+1, currpos -1 ) && (isBonded(currpos + 1 , i - 1)))) {
404                                // antiparallel
405                                btype = BridgeType.antiparallel;
406                        }
407                        if (btype != null){
408                                foundNrBridges++;
409                                registerBridge(i, currpos, btype);
410                        }
411                }
412
413        }

416                if ( groups.length < 5)
417                        return;
418

421                        /* Create vectors ( Ca i to Ca i-2 ) ; ( Ca i to CA i +2 ) */

425                        SecStrucGroup ip2 = groups[i+2];

457        private void calculateDihedralAngles() throws StructureException {
458
459                // dihedral angles
460                // phi: C-N-CA-C
461                // psi: N-CA-C-N
462                // Chi1: N-CA-CB-CG, N-CA-CB-OG(SER),N-CA-CB-OG1(Thr),
463                // N-CA-CB-CG1(ILE/VAL), N-CA-CB-SG(CYS)
464                // Omega: CA-C-N-CA
465
466                for (int i=0 ; i < groups.length-1 ;  i++){
467
468                        SecStrucGroup a = groups[i];
469
470                        SecStrucGroup b = groups[i+1];
471
472
473                        Atom a_N   = a.getN();
474                        Atom a_CA  = a.getCA();
475                        Atom a_C  = a.getC();
476
477                        Atom b_N  = b.getN();
478                        Atom b_CA = b.getCA();
479                        Atom b_C  = b.getC();
480
481                        double phi = Calc.torsionAngle(a_C,b_N,b_CA,b_C);
482
483                        double psi = Calc.torsionAngle(a_N,a_CA,a_C,b_N);
484
485                        double omega = Calc.torsionAngle(a_CA,a_C,b_N,b_CA);
486
487                        SecStrucState state1 = (SecStrucState) a.getProperty("secstruc");
488                        SecStrucState state2 = (SecStrucState) b.getProperty("secstruc");
489                        state2.setPhi(phi);
490                        state1.setPsi(psi);
491                        state1.setOmega(omega);
492
493                }
494
495        }

549                                p3 -= i;

576        private static SecStrucGroup[] initGroupArray(Structure s) {
577                List<SecStrucGroup> groupList = new ArrayList<SecStrucGroup>();
578                //GroupIterator iter = new GroupIterator(s);
579                for ( Chain c : s.getChains()){
580
581                        for (Group g : c.getAtomGroups()){
582                                //System.out.println(g);
583                                //                       we can also calc secstruc if hetatom is a modified amino acid.
584                                if ( g.hasAminoAtoms()) {
585
586                                        SecStrucGroup sg = new SecStrucGroup();
587                                        sg.setResidueNumber(g.getResidueNumber());
588                                        sg.setPDBFlag(true);
589                                        try {
590                                                sg.setPDBName(g.getPDBName());
591                                        } catch (PDBParseException e){
592                                                e.printStackTrace();
593                                        }
594                                        sg.setParent(g.getChain());
595
596                                        try {
597
598                                                sg.setN((Atom)   g.getAtomByPDBname(" N  ").clone());
599                                                sg.setCA((Atom)  g.getAtomByPDBname(" CA ").clone());
600                                                sg.setC((Atom)   g.getAtomByPDBname(" C  ").clone());
601                                                sg.setO((Atom)   g.getAtomByPDBname(" O  ").clone());
602                                                sg.setOriginal(g);
603                                                // create H in calc_H
604
605
606                                        } catch (StructureException e){
607                                                e.printStackTrace();
608                                                // one of these atoms is not found!
609                                                continue;
610                                        }
611
612                                        SecStrucState state = new SecStrucState();
613                                        Map<String,Object> m = sg.getProperties();
614                                        if ( m == null) {
615                                                m = new HashMap<String, Object>();
616                                                sg.setProperties(m);
617                                        }
618
619                                        m.put("secstruc",state);
620
621
622                                        groupList.add(sg);
623                                } else {
624                                        //System.out.println("not an amino group");
625                                }
626                        }
627                }
628
629                return (SecStrucGroup[]) groupList.toArray(new SecStrucGroup[groupList.size()]);
630        }

670        /** calculate the HBonds between different groups ...
671         * see Creighton page 147 f
672         *
673         */
674        private void calculateHBonds()
675                        throws StructureException
676                        {
677                //System.out.println("groups length: " + groups.length);
678
679                if ( groups.length < 5)
680                        return;
681
682                // skip the first residue , unable to calc H for it ...
683                for (int i=1 ; i < groups.length ;  i++){
684
685                        SecStrucGroup one = groups[i];
686
687                        if ( ! one.hasAtom("H")) {
688                                System.out.println(" no H at " + i);
689                                continue;
690                        }
691
692                        for ( int j = i+1 ; j < groups.length ; j++){
693
694                                SecStrucGroup two = groups[j];
695
696                                // check if distance is  too large.
697                                // if too big - for sure no HBonds ...
698                                double dist = Calc.getDistance(one.getCA(),two.getCA());
699
700                                // speed up...
701                                if ( dist >= CA_MIN_DIST  )
702                                        continue;
703                                //System.out.println("calc " + i + " " + j + " "+  dist);
704
705                                checkAddHBond(i,j);
706
707                                // "backwards" hbonds are not allowed

751        /** calculate HBond energy of two groups in cal/mol ...
752         * see Creighton page 147 f
753         *
754         * Jeffrey, George A., An introduction to hydrogen bonding, Oxford University Press, 1997.
755         * categorizes hbonds with donor-acceptor distances of
756         * 2.2-2.5 &aring; as "strong, mostly covalent",
757         * 2.5-3.2 &aring; as "moderate, mostly electrostatic",
758         * 3.2-4.0 &aring; as "weak, electrostatic".
759         *  Energies are given as 40-14, 15-4, and <4 kcal/mol respectively.
760         *
761         */
762
763
764        public  double calculateHBondEnergy(SecStrucGroup one, SecStrucGroup two )
765                        throws StructureException{
766
767
768                //System.out.println("calcHBondEnergy" + one + "|" + two);
769
770                Atom N = one.getN();
771                Atom H = one.getH();
772
773                Atom O = two.getO();
774                Atom C = two.getC();
775
776                double dno = Calc.getDistance(O,N);
777                double dhc = Calc.getDistance(C,H);
778                double dho = Calc.getDistance(O,H);
779                double dnc = Calc.getDistance(C,N);
780
781                if  ( debug ){
782
783                        System.out.println("     cccc: " + one.getResidueNumber().toString() + " " + one.getPDBName() + " " +two.getResidueNumber().toString() + " " + two.getPDBName() +
784                                        String.format(" O ("+O.getPDBserial()+")..N ("+ N.getPDBserial()+"):%4.1f  |  ho:%4.1f - hc:%4.1f + nc:%4.1f - no:%4.1f " , dno,dho,dhc,dnc,dno));
785
786                }
787                //System.out.println( cn > ch && oh < 3.0f);
788
789                double contact = MINDIST ;
790
791                //               there seems to be a contact!
792                if ( (dno < contact) || (dhc < contact) || (dnc < contact) || (dno < contact)) {
793                        //System.out.println("!!! contact " + one + " " + two);
794                        return HBONDLOWENERGY ;
795                }
796
797                double e1 = Q / dho  - Q / dhc ;
798                double e2 = Q / dnc  - Q / dno ;
799
800                double energy = e1 + e2;
801
802                if ( debug )
803                        System.out.println(String.format("      N (%d) O(%d): %4.1f : %4.2f ",N.getPDBserial(),O.getPDBserial(), (float)dno , energy));
804
805                // bond too weak
806                //if ( energy > HBONDHIGHENERGY)
807                //      return energy;
808
809                // test to avoid bond too strong
810                if ( energy > HBONDLOWENERGY)
811                        return energy;
812
813
814                return HBONDLOWENERGY ;
815
816        }

818        /**
819         * calculate distance between two atoms.
820         *
821         * @param a  an Atom object
822         * @param b  an Atom object
823         * @return a double
824         * @throws StructureException ...
825         */
826        public static BigDecimal getPreciseDistance(Atom a, Atom b)
827                        throws StructureException
828                        {
829                double x = a.getX() - b.getX();
830                double y = a.getY() - b.getY();
831                double z = a.getZ() - b.getZ();
832
833                double s  = x * x  + y * y + z * z;
834                BigSqrt sqrt = new BigSqrt();
835                BigDecimal d = new BigDecimal(s);
836                BigDecimal dist = sqrt.sqrt(d);
837
838                return dist ;
839                        }
