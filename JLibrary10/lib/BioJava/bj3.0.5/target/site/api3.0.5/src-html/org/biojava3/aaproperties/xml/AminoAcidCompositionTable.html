001package org.biojava3.aaproperties.xml;
002
003import java.util.HashMap;
004import java.util.List;
005import java.util.Map;
006import java.util.Set;
007
008import javax.xml.bind.annotation.XmlAccessType;
009import javax.xml.bind.annotation.XmlAccessorType;
010import javax.xml.bind.annotation.XmlElement;
011import javax.xml.bind.annotation.XmlRootElement;
012
013
014@XmlRootElement(name="compoundtable", namespace="http://biojava.org")
015@XmlAccessorType(XmlAccessType.NONE)
016public class AminoAcidCompositionTable {
017        
018        /**
019         * Contains the list of amino acid composition 
020         */
021        @XmlElement(name = "compound", required = true)
022        private List<AminoAcidComposition> aminoacid;
023        
024        /**
025         * Defines the amino acid compound set unique to this table
026         */
027        private ModifiedAminoAcidCompoundSet modifiedAminoAcidCompoundSet;
028        
029        /**
030         * Stores the mapping of amino acid symbol to its molecular weight
031         */
032        private Map<Character, Double> aaSymbol2MolecularWeight;
033        
034        public AminoAcidCompositionTable(){}
035        
036        public AminoAcidCompositionTable(List<AminoAcidComposition> aaList){
037                this.setAminoacid(aaList);
038        }
039
040        public ModifiedAminoAcidCompoundSet getAminoAcidCompoundSet(){
041                return this.modifiedAminoAcidCompoundSet;
042        }
043        
044        public List<AminoAcidComposition> getAminoacid() {
045                return aminoacid;
046        }
047
048        public void setAminoacid(List<AminoAcidComposition> aminoacid) {
049                this.aminoacid = aminoacid;
050        }
051        
052        public Set<Character> getSymbolSet(){
053                return this.aaSymbol2MolecularWeight.keySet();
054        }
055        
056        private void generatesAminoAcidCompoundSet(){
057                this.modifiedAminoAcidCompoundSet = new ModifiedAminoAcidCompoundSet(this.aminoacid, this.aaSymbol2MolecularWeight);
058        }
059        
060        /**
061         * Computes and store the molecular weight of each amino acid by its symbol in aaSymbol2MolecularWeight.
062         * 
063         * @param eTable
064         *              Stores the mass of elements and isotopes
065         */
066        public void computeMolecularWeight(ElementTable eTable){
067                this.aaSymbol2MolecularWeight = new HashMap<Character, Double>();
068                for(AminoAcidComposition a:aminoacid){
069                        //Check to ensure that the symbol is of single character
070                        if(a.getSymbol().length() != 1){
071                                throw new Error(a.getSymbol() + " is not allowed. Symbols must be single character.\r\nPlease check AminoAcidComposition XML file");
072                        }
073                        //Check to ensure that the symbols are not repeated
074                        char c = a.getSymbol().charAt(0);
075                        if(this.aaSymbol2MolecularWeight.keySet().contains(c)){
076                                throw new Error("Symbol " + c + " is repeated.\r\n" +
077                                                "Please check AminoAcidComposition XML file to ensure there are no repeated symbols. Note that this is case-insensitive.\r\n" +
078                                                "This means that having 'A' and 'a' would be repeating.");
079                        }
080                        double total = 0.0;
081                        if(a.getElementList() != null){
082                                for(Name2Count element:a.getElementList()){
083                                        element.getName();
084                                        if(eTable.getElement(element.getName()) == null){
085                                                throw new Error("Element " + element.getName() + " could not be found. " +
086                                                                "\r\nPlease ensure that its name is correct in AminoAcidComposition.xml and is defined in ElementMass.xml.");
087                                        }
088                                        eTable.getElement(element.getName()).getMass();
089                                        total += eTable.getElement(element.getName()).getMass() * element.getCount();
090                                }
091                        }
092                        if(a.getIsotopeList() != null){
093                                for(Name2Count isotope:a.getIsotopeList()){
094                                        isotope.getName();
095                                        if(eTable.getIsotope(isotope.getName()) == null){
096                                                throw new Error("Isotope " + isotope.getName() + " could not be found. " +
097                                                        "\r\nPlease ensure that its name is correct in AminoAcidComposition.xml and is defined in ElementMass.xml.");
098                                        }
099                                        eTable.getIsotope(isotope.getName()).getMass();
100                                        total += eTable.getIsotope(isotope.getName()).getMass() * isotope.getCount();
101                                }
102                        }
103                        c = a.getSymbol().charAt(0);
104                        this.aaSymbol2MolecularWeight.put(c, total);
105                }
106                generatesAminoAcidCompoundSet();
107        }
108        
109        /**
110         * @param aaSymbol
111         *      Standard symbol of Amino Acid
112         * @return the molecular weight given its symbol
113         * @throws NullPointerException 
114         *      thrown if AminoAcidCompositionTable.computeMolecularWeight(ElementTable) is not called before this method
115         */
116        public double getMolecularWeight(Character aaSymbol) throws NullPointerException{
117                if(this.aaSymbol2MolecularWeight == null){
118                        throw new NullPointerException("Please call AminoAcidCompositionTable.computeMolecularWeight(ElementTable) before this method");
119                }
120                Double d = this.aaSymbol2MolecularWeight.get(aaSymbol);
121                if(d == null)
122                        return 0;
123                else
124                        return d; 
125        }
126}
