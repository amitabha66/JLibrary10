038/**
039 * This class serves as a instance registry by maintaining 
040 * a pool of ProteinModification instances.
041 * 
042 * A list of common protein modifications were preloaded
043 * from an XML file.
044 * 
045 * @author Jianjiong Gao
046 * @since 3.0
047 */
048public class ProteinModificationRegistry {
049        private static Set<ProteinModification> registry = null;
050        private static Map<String, ProteinModification> byId = null;
051        private static Map<String, Set<ProteinModification>> byResidId = null;
052        private static Map<String, Set<ProteinModification>> byPsimodId = null;
053        private static Map<String, Set<ProteinModification>> byPdbccId = null;
054        private static Map<String, Set<ProteinModification>> byKeyword = null;
055        private static Map<Component, Set<ProteinModification>> byComponent = null;
056        private static Map<ModificationCategory, Set<ProteinModification>> byCategory = null;
057        private static Map<ModificationOccurrenceType, Set<ProteinModification>> byOccurrenceType = null;
058        
059        private static String DIR_XML_PTM_LIST = "ptm_list.xml";
060        
061        
062        
063        /**
064         * register common protein modifications from XML file.
065         */
066        private static void registerCommonProteinModifications(InputStream inStream) {
067                try {
068                        
069                        ProteinModificationXmlReader.registerProteinModificationFromXml(inStream);
070                } catch (Exception e) {
071                        e.printStackTrace();
072                }
073        }
074        
075        /**
076         * Initialization the static variables and register common modifications.
077         */
078        public static void init() {
079                lazyInit();
080                
081        }
082        
083        /** Initialization the static variables and register common modifications.
084         * Allows external user to provide alternative ptm_list.xml file instead of the one contained in this jar file.
085         * 
086         * @param inStream InputStream to a XML file containing the list of PTMs (as in ptm_list.xml)
087         */
088         
089        public static void init(InputStream inStream) {
090                lazyInit(inStream);
091        }
092        
093        
094        
095        /**
096         * Lazy Initialization the static variables and register common modifications.
097         * just opens the stream to ptm_list.xml and delegates to lazyInit(InputStream) for parsing. 
098         */
099        private static synchronized void lazyInit() {
100                if (registry==null) {   
101                        InputStream isXml = ProteinModification.class.getResourceAsStream(DIR_XML_PTM_LIST);
102                        lazyInit(isXml);
103                }
104        }
105        
106        
107        /**
108         * Lazy Initialization the static variables and register common modifications. 
109         */
110        private static synchronized void lazyInit(InputStream inStream) {
111                if (registry==null) {   
112                                                
113                        registry =      new HashSet<ProteinModification>();
114                        byId = new HashMap<String, ProteinModification>();
115                        byResidId = new HashMap<String, Set<ProteinModification>>();
116                        byPsimodId = new HashMap<String, Set<ProteinModification>>();
117                        byPdbccId = new HashMap<String, Set<ProteinModification>>();
118                        byKeyword = new HashMap<String, Set<ProteinModification>>();
119                        byComponent = new HashMap<Component, Set<ProteinModification>>();
120                        byCategory = new EnumMap<ModificationCategory, Set<ProteinModification>>(
121                                        ModificationCategory.class);
122                        for (ModificationCategory cat:ModificationCategory.values()) {
123                                byCategory.put(cat, new HashSet<ProteinModification>());
124                        }
125                        byOccurrenceType = new EnumMap<ModificationOccurrenceType, Set<ProteinModification>>(
126                                        ModificationOccurrenceType.class);
127                        for (ModificationOccurrenceType occ:ModificationOccurrenceType.values()) {
128                                byOccurrenceType.put(occ, new HashSet<ProteinModification>());
129                        }
130                        registerCommonProteinModifications(inStream);                                   
131                }
132        }
133
134        /**
135         * Register a new ProteinModification.
136         */
137        public static void register(final ProteinModification modification) {
138                if (modification==null) throw new IllegalArgumentException("modification == null!");
139                
140                lazyInit();
141
142                String id = modification.getId();
143                if (byId.containsKey(id)) {
144                        throw new IllegalArgumentException(id+" has already been registered.");
145                }
146                
147                registry.add(modification);
148                byId.put(id, modification);
149                
150                ModificationCategory cat = modification.getCategory();
151                byCategory.get(cat).add(modification);
152                
153                ModificationOccurrenceType occType = modification.getOccurrenceType();
154                byOccurrenceType.get(occType).add(modification);
155
156                
157                ModificationCondition condition = modification.getCondition();
158                List<Component> comps = condition.getComponents();
159                for (Component comp:comps) {
160                        Set<ProteinModification> mods = byComponent.get(comp);
161                        if (mods==null) {
162                                mods = new HashSet<ProteinModification>();
163                                byComponent.put(comp, mods);
164                        }
165                        mods.add(modification);
166                }
167                
168                String pdbccId = modification.getPdbccId();
169                if (pdbccId!=null) {
170                        Set<ProteinModification> mods = byPdbccId.get(pdbccId);
171                        if (mods==null) {
172                                mods = new HashSet<ProteinModification>();
173                                byPdbccId.put(pdbccId, mods);
174                        }
175                        mods.add(modification);
176                }
177                
178                String residId = modification.getResidId();
179                if (residId!=null) {
180                        Set<ProteinModification> mods = byResidId.get(residId);
181                        if (mods==null) {
182                                mods = new HashSet<ProteinModification>();
183                                byResidId.put(residId, mods);
184                        }
185                        mods.add(modification);
186                }
187                
188                String psimodId = modification.getPsimodId();
189                if (psimodId!=null) {                   
190                        Set<ProteinModification> mods = byPsimodId.get(psimodId);
191                        if (mods==null) {
192                                mods = new HashSet<ProteinModification>();
193                                byPsimodId.put(psimodId, mods);
194                        }
195                        mods.add(modification);
196                }
197                
198                for (String keyword : modification.getKeywords()) {
199                        Set<ProteinModification> mods = byKeyword.get(keyword);
200                        if (mods==null) {
201                                mods = new HashSet<ProteinModification>();
202                                byKeyword.put(keyword, mods);
203                        }
204                        mods.add(modification);
205                }
206        }

208        /**
209         * Remove a modification from registry.
210         * @param mod
211         */
212        public static void unregister(ProteinModification modification) {
213                if (modification==null) throw new IllegalArgumentException("modification == null!");
214                
215                registry.remove(modification);
216                
217                byId.remove(modification.getId());
218                
219                Set<ProteinModification> mods;
220                
221                mods = byResidId.get(modification.getResidId());
222                if (mods!=null) mods.remove(modification);
223                
224                mods = byPsimodId.get(modification.getPsimodId());
225                if (mods!=null) mods.remove(modification);
226                
227                mods = byPdbccId.get(modification.getPdbccId());
228                if (mods!=null) mods.remove(modification);
229                
230                for (String keyword : modification.getKeywords()) {
231                        mods = byKeyword.get(keyword);
232                        if (mods!=null) mods.remove(modification);
233                }
234                
235                ModificationCondition condition = modification.getCondition();
236                List<Component> comps = condition.getComponents();
237                for (Component comp : comps) {
238                        mods = byComponent.get(comp);
239                        if (mods!=null) mods.remove(modification);
240                }
241                
242                byCategory.get(modification.getCategory()).remove(modification);
243                byOccurrenceType.get(modification.getOccurrenceType()).remove(modification);
244        }

295        /**
296         * Get ProteinModifications that involves one or more components.
297         * @param comp1 a {@link Component}.
298         * @param comps other {@link Component}s.
299         * @return a set of ProteinModifications that involves all the components.
300         */
301        public static Set<ProteinModification> getByComponent(final Component comp1,
302                        final Component... comps) {
303                lazyInit();
304                Set<ProteinModification> mods = byComponent.get(comp1);
305                if (mods==null) {
306                        return Collections.emptySet();
307                }
308                
309                if (comps.length==0) {
310                        return Collections.unmodifiableSet(mods);
311                } else {
312                        Set<ProteinModification> ret = new HashSet<ProteinModification>(mods);
313                        for (Component comp:comps) {
314                                mods = byComponent.get(comp);
315                                if (mods==null) {
316                                        return Collections.emptySet();
317                                } else {
318                                        ret.retainAll(mods);
319                                }
320                        }
321                        
322                        return ret;
323                }
324        }
