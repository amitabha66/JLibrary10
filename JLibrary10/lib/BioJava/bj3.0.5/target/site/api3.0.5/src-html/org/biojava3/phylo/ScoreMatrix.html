031public class ScoreMatrix
032{
033  String name;
034
035  /**
036   * reference to integer score matrix
037   */
038  int[][] matrix;
039
040  /**
041   * 0 for Protein Score matrix. 1 for dna score matrix
042   */
043  int type;
044
045  ScoreMatrix(String name, int[][] matrix, int type)
046  {
047    this.matrix = matrix;
048    this.type = type;
049  }
050
051  public boolean isDNA()
052  {
053    return type == 1;
054  }
055
056  public boolean isProtein()
057  {
058    return type == 0;
059  }
060
061  public int[][] getMatrix()
062  {
063    return matrix;
064  }
065
066  /**
067   *
068   * @param A1
069   * @param A2
070   * @return score for substituting first char in A1 with first char in A2
071   */
072  public int getPairwiseScore(String A1, String A2)
073  {
074    return getPairwiseScore(A1.charAt(0), A2.charAt(0));
075  }
076
077  
078
079
080  public int getPairwiseScore(char c, char d)
081  {
082    int pog = 0;
083
084    try
085    {
086      int a = (type == 0) ? ResidueProperties.aaIndex[c]
087              : ResidueProperties.nucleotideIndex[c];
088      int b = (type == 0) ? ResidueProperties.aaIndex[d]
089              : ResidueProperties.nucleotideIndex[d];
090
091      pog = matrix[a][b];
092    } catch (Exception e)
093    {
094      // System.out.println("Unknown residue in " + A1 + " " + A2);
095    }
096
097    return pog;
098  }
099
100}
