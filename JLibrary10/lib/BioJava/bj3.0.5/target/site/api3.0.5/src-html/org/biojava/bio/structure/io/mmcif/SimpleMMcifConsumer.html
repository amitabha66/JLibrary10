181        /** initiate new group, either Hetatom, Nucleotide, or AminoAcid */
182        private Group getNewGroup(String recordName,Character aminoCode1, long seq_id,String groupCode3) {
183
184                if ( params.isLoadChemCompInfo() ){
185                        Group g = ChemCompGroupFactory.getGroupFromChemCompDictionary(groupCode3);
186                        if ( g != null) {
187                                if ( g instanceof AminoAcidImpl) {
188                                        AminoAcidImpl aa = (AminoAcidImpl) g;
189                                        aa.setId(seq_id);
190                                } else if ( g instanceof NucleotideImpl) {
191                                        NucleotideImpl nuc =  (NucleotideImpl) g;
192                                        nuc.setId(seq_id);
193                                } else if ( g instanceof HetatomImpl) {
194                                        HetatomImpl het = (HetatomImpl)g;
195                                        het.setId(seq_id);
196                                }
197                                return g;
198                        }
199
200                }
201
202
203                Group group;
204                if ( recordName.equals("ATOM") ) {
205                        if (aminoCode1 == null)  {
206                                // it is a nucleotide
207                                NucleotideImpl nu = new NucleotideImpl();
208                                group = nu;
209                                nu.setId(seq_id);
210
211                        } else if (aminoCode1 == StructureTools.UNKNOWN_GROUP_LABEL){
212                                HetatomImpl h = new HetatomImpl();
213                                h.setId(seq_id);
214                                group = h;
215
216                        } else {
217                                AminoAcidImpl aa = new AminoAcidImpl() ;
218                                aa.setAminoType(aminoCode1);
219                                aa.setId(seq_id);
220                                group = aa ;
221                        }
222                }
223                else {
224                        HetatomImpl h = new HetatomImpl();
225                        h.setId(seq_id);
226                        group = h;
227                }               
228                //System.out.println("new group type: "+ group.getType() );
229                return  group ;
230        }
231        /** test if the chain is already known (is in current_model
232         * ArrayList) and if yes, returns the chain
233         * if no -> returns null
234         */
235        private Chain isKnownChain(String chainID, List<Chain> chains){
236
237                for (int i = 0; i< chains.size();i++){
238                        Chain testchain =  chains.get(i);
239                        //System.out.println("comparing chainID >"+chainID+"< against testchain " + i+" >" +testchain.getName()+"<");
240                        if (chainID.equals(testchain.getChainID())) {
241                                //System.out.println("chain "+ chainID+" already known ...");
242                                return testchain;
243                        }
244                }
245
246                return null;
247        }

250        /** during mmcif parsing the full atom name string gets truncated, fix this...
251         *
252         * @param name
253         * @return
254         */
255        private String fixFullAtomName(String name){
256
257                if (name.equals("N")){
258                        return " N  ";
259                }
260                if (name.equals("CA")){
261                        return " CA ";
262                }
263                if (name.equals("C")){
264                        return " C  ";
265                }
266                if (name.equals("O")){
267                        return " O  ";
268                }
269                if (name.equals("CB")){
270                        return " CB ";
271                }
272                if (name.equals("CG"))
273                        return " CG ";
274
275                if (name.length() == 2)
276                        return " " + name + " ";
277
278                if (name.length() == 1)
279                        return " " + name + "  ";
280
281                if (name.length() == 3)
282                        return " " + name ;
283
284                return name;
285        }

287        public void newAtomSite(AtomSite atom) {
288
289                // Warning: getLabel_asym_id is not the "chain id" in the PDB file
290                // it is the internally used chain id.
291                // later on we will fix this...
292
293                // later one needs to map the asym id to the pdb_strand_id
294
295                //TODO: add support for MAX_ATOMS
296
297                boolean startOfNewChain = false;
298
299                //String chain_id      = atom.getAuth_asym_id();
300                String chain_id      = atom.getLabel_asym_id();         
301                String fullname      = fixFullAtomName(atom.getLabel_atom_id());                
302                String recordName    = atom.getGroup_PDB();
303                String residueNumberS = atom.getAuth_seq_id();
304                Integer residueNrInt = Integer.parseInt(residueNumberS);
305                // the 3-letter name of the group:
306                String groupCode3    = atom.getLabel_comp_id();
307                if ( groupCode3.length() == 1){
308                        groupCode3 = "  " + groupCode3;
309                }
310                if ( groupCode3.length() == 2){
311                        groupCode3 = " " + groupCode3;
312                }
313                Character aminoCode1 = null;
314                if ( recordName.equals("ATOM") )
315                        aminoCode1 = StructureTools.get1LetterCode(groupCode3);
316                else {
317                        aminoCode1 = StructureTools.get1LetterCode(groupCode3);
318                        // for nucleotides this will be null..
319                        if (aminoCode1 != null &&  aminoCode1.equals(StructureTools.UNKNOWN_GROUP_LABEL)) 
320                                aminoCode1 = null;
321                }
322                String insCodeS = atom.getPdbx_PDB_ins_code();
323                Character insCode = null;
324                if (!  insCodeS.equals("?")) {
325                        insCode = insCodeS.charAt(0);
326                }
327                // we store the internal seq id in the Atom._id field
328                // this is not a PDB file field but we need this to internally assign the insertion codes later
329                // from the pdbx_poly_seq entries..
330
331                long seq_id = -1;
332                try {
333                        seq_id = Long.parseLong(atom.getLabel_seq_id());
334                } catch (NumberFormatException e){
335
336                }
337
338                String nmrModel = atom.getPdbx_PDB_model_num();
339
340                if ( current_nmr_model == null) {
341                        current_nmr_model = nmrModel;
342                }
343
344                if (! current_nmr_model.equals(nmrModel)){
345                        current_nmr_model = nmrModel;
346
347                        // add previous data
348                        if ( current_chain != null ) {
349                                current_chain.addGroup(current_group);
350                        }
351
352                        // we came to the beginning of a new NMR model
353                        structure.setNmr(true);
354                        structure.addModel(current_model);
355                        current_model = new ArrayList<Chain>();
356                        current_chain = null;
357                        current_group = null;
358                }
359
360
361                if (current_chain == null) {
362                        current_chain = new ChainImpl();
363                        current_chain.setChainID(chain_id);
364                        current_model.add(current_chain);
365                        startOfNewChain = true;
366                }
367
368                //System.out.println("BEFORE: " + chain_id + " " + current_chain.getName());
369                if ( ! chain_id.equals(current_chain.getChainID()) ) {
370
371                        startOfNewChain = true;
372
373                        // end up old chain...
374                        current_chain.addGroup(current_group);
375
376                        // see if old chain is known ...
377                        Chain testchain ;
378                        testchain = isKnownChain(current_chain.getChainID(),current_model);
379
380                        //System.out.println("trying to re-using known chain " + current_chain.getName() + " " + chain_id);             
381                        if ( testchain != null && testchain.getChainID().equals(chain_id)){
382                                //System.out.println("re-using known chain " + current_chain.getName() + " " + chain_id);                               
383
384                        } else {
385
386                                testchain = isKnownChain(chain_id,current_model);
387                        }
388
389                        if ( testchain == null) {
390                                //System.out.println("unknown chain. creating new chain.");
391
392                                current_chain = new ChainImpl();
393                                current_chain.setChainID(chain_id);
394
395                        }   else {
396                                current_chain = testchain;
397                        }
398
399                        if ( ! current_model.contains(current_chain))
400                                current_model.add(current_chain);
401
402                } 
403
404
405                ResidueNumber residueNumber = new ResidueNumber(chain_id,residueNrInt, insCode);
406
407                if (current_group == null) {
408
409                        current_group = getNewGroup(recordName,aminoCode1,seq_id, groupCode3);
410
411                        //current_group.setPDBCode(residueNumber);
412
413                        current_group.setResidueNumber(residueNumber);
414                        try { 
415                                current_group.setPDBName(groupCode3);
416                        } catch (PDBParseException e){
417                                System.err.println(e.getMessage());
418                        }
419                }
420
421                if ( startOfNewChain){
422                        current_group = getNewGroup(recordName,aminoCode1,seq_id, groupCode3);
423
424                        // current_group.setPDBCode(residueNumber);
425                        current_group.setResidueNumber(residueNumber);
426                        try {
427                                current_group.setPDBName(groupCode3);
428                        } catch (PDBParseException e){
429                                e.printStackTrace();
430                        }
431                }
432
433                Group altGroup = null;
434                String altLocS = atom.getLabel_alt_id();
435                Character altLoc = ' ';
436                if ( altLocS.length()>0) {
437                        altLoc = altLocS.charAt(0);
438                        if ( altLoc.equals('.') )
439                                altLoc = ' ';
440
441                }
442
443                // check if residue number is the same ...
444                // insertion code is part of residue number
445                if ( ! residueNumber.equals(current_group.getResidueNumber())) {
446                        //System.out.println("end of residue: "+current_group.getPDBCode()+" "+residueNrInt);
447                        current_chain.addGroup(current_group);
448
449                        current_group = getNewGroup(recordName,aminoCode1,seq_id,groupCode3);
450                        //current_group.setPDBCode(pdbCode);
451                        try {
452                                current_group.setPDBName(groupCode3);
453                        } catch (PDBParseException e){
454                                e.printStackTrace();
455                        }
456                        current_group.setResidueNumber(residueNumber);
457                        //                        System.out.println("Made new group:  " + groupCode3 + " " + resNum + " " + iCode);
458
459                } else {
460                        // same residueNumber, but altLocs...
461
462                        // test altLoc
463                        if ( ! altLoc.equals(' ') && ( ! altLoc.equals('.'))) {                                                                                         
464                                altGroup = getCorrectAltLocGroup( altLoc,recordName,aminoCode1,groupCode3, seq_id);
465                                //System.out.println("found altLoc! " + altLoc + " " + current_group + " " + altGroup);
466                        }
467                }
468
469                if ( params.isHeaderOnly())
470                        return;
471
472                atomCount++;
473                //System.out.println("fixing atom name for  >" + atom.getLabel_atom_id() + "< >" + fullname + "<");
474
475                if ( params.isParseCAOnly() ){
476                        // yes , user wants to get CA only
477                        // only parse CA atoms...
478                        if (! fullname.equals(" CA ")){
479                                //System.out.println("ignoring " + line);
480                                atomCount--;
481                                return;
482                        }
483                }
484
485
486
487
488                //see if chain_id is one of the previous chains ...
489
490                Atom a = convertAtom(atom);
491
492                //see if chain_id is one of the previous chains ...
493                if ( altGroup != null) {
494                        altGroup.addAtom(a);
495                        altGroup = null;
496                }
497                else {
498                        current_group.addAtom(a);
499                }
500
501                //System.out.println(">" + atom.getLabel_atom_id()+"< " + a.getGroup().getPDBName() + " " + a.getGroup().getChemComp()  );
502
503                //System.out.println(current_group);
504
505        }

507        /** convert a MMCif AtomSite object to a BioJava Atom object
508         *
509         * @param atom the mmmcif AtomSite record
510         * @return an Atom
511         */
512        private Atom convertAtom(AtomSite atom){
513
514
515                Atom a = new AtomImpl();
516
517                a.setPDBserial(Integer.parseInt(atom.getId()));
518                a.setName(atom.getLabel_atom_id());
519
520                a.setFullName(fixFullAtomName(atom.getLabel_atom_id()));
521
522
523                double x = Double.parseDouble (atom.getCartn_x());
524                double y = Double.parseDouble (atom.getCartn_y());
525                double z = Double.parseDouble (atom.getCartn_z());
526                a.setX(x);
527                a.setY(y);
528                a.setZ(z);
529
530                double occupancy = Double.parseDouble(atom.getOccupancy());
531                a.setOccupancy(occupancy);
532
533                double temp = Double.parseDouble(atom.getB_iso_or_equiv());
534                a.setTempFactor(temp);
535
536                String alt = atom.getLabel_alt_id();
537                if (( alt != null ) && ( alt.length() > 0) && (! alt.equals("."))){
538                        a.setAltLoc(new Character(alt.charAt(0)));
539                } else {
540                        a.setAltLoc(new Character(' '));
541                }
542
543                Element element = Element.R;
544                try {
545                        element = Element.valueOfIgnoreCase(atom.getType_symbol());
546                }  catch (IllegalArgumentException e){}
547                a.setElement(element);
548
549                return a;
550
551        }

554        private Group getCorrectAltLocGroup( Character altLoc,
555                        String recordName, Character aminoCode1, String groupCode3, long seq_id) {
556
557                // see if we know this altLoc already;
558                List<Atom> atoms = current_group.getAtoms();
559                if ( atoms.size() > 0) {
560                        Atom a1 = atoms.get(0);
561                        // we are just adding atoms to the current group
562                        // probably there is a second group following later...
563                        if (a1.getAltLoc().equals(altLoc)) {
564
565                                return current_group;
566                        }
567                }
568
569                List<Group> altLocs = current_group.getAltLocs();
570                for ( Group altLocG : altLocs ){
571                        atoms = altLocG.getAtoms();
572                        if ( atoms.size() > 0) {
573                                for ( Atom a1 : atoms) {
574                                        if (a1.getAltLoc().equals( altLoc)) {
575
576                                                return altLocG;
577                                        }
578                                }
579                        }
580                }
581
582                // no matching altLoc group found.
583                // build it up.
584
585                if ( groupCode3.equals(current_group.getPDBName())) {
586                        if ( current_group.getAtoms().size() == 0)
587                                return current_group;
588                        //System.out.println("cloning current group");
589                        Group altLocG = (Group) current_group.clone();
590                        current_group.addAltLoc(altLocG);
591                        return altLocG; 
592                }
593
594
595                Group altLocG = getNewGroup(recordName,aminoCode1,seq_id,groupCode3);
596
597                try {
598                        altLocG.setPDBName(groupCode3);
599                } catch (PDBParseException e) {
600                        e.printStackTrace();
601                }
602                altLocG.setResidueNumber(current_group.getResidueNumber());
603                current_group.addAltLoc(altLocG);
604                return altLocG;
605        }

631        public void documentEnd() {
632
633                // a problem occurred earlier so current_chain = null ...
634                // most likely the buffered reader did not provide data ...
635                if ( current_chain != null ) {
636
637                        current_chain.addGroup(current_group);
638                        if (isKnownChain(current_chain.getChainID(),current_model) == null) {
639                                current_model.add(current_chain);
640                        }
641                } else {
642                        if ( DEBUG){
643                                System.err.println("current chain is null at end of document.");
644                        }
645                }
646
647                structure.addModel(current_model);
648
649                // Goal is to reproduce the PDB files exactly:
650                // What has to be done is to use the auth_mon_id for the assignment. For this
651
652                // map entities to Chains and Compound objects...
653
654
655                for (StructAsym asym : structAsyms) {
656                        if ( DEBUG )
657                                System.out.println("entity " + asym.getEntity_id() + " matches asym id:" + asym.getId() );
658
659                        Chain s = getEntityChain(asym.getEntity_id());
660                        Chain seqres = (Chain)s.clone();
661                        seqres.setChainID(asym.getId());
662
663
664                        seqResChains.add(seqres);
665                        if ( DEBUG )
666                                System.out.println(" seqres: " + asym.getId() + " " + seqres + "<") ;
667
668                }
669
670
671                if ( params.isAlignSeqRes() ){
672
673                        SeqRes2AtomAligner aligner = new SeqRes2AtomAligner();                  
674                        aligner.align(structure,seqResChains);
675                }
676
677
678                //TODO: add support for these:
679
680                //structure.setConnections(connects);
681                //structure.setCompounds(compounds);
682                //linkChains2Compound(structure);
683
684                // mismatching Author assigned chain IDS and PDB internal chain ids:
685                // fix the chain IDS in the current model:
686
687                Set<String> asymIds = asymStrandId.keySet();
688
689                for (int i =0; i< structure.nrModels() ; i++){
690                        List<Chain>model = structure.getModel(i);
691
692                        List<Chain> pdbChains = new ArrayList<Chain>();
693
694                        for (Chain chain : model) {
695                                for (String asym : asymIds) {

698                                                if (DEBUG)
699                                                        System.out.println("renaming " + asym  + " to : " + asymStrandId.get(asym));

701                                                chain.setChainID(asymStrandId.get(asym));

703                                                Chain known =  isKnownChain(chain.getChainID(), pdbChains);
704                                                if ( known == null ){
705                                                        pdbChains.add(chain);
706                                                } else {
707                                                        // and now we join the 2 chains together again, because in cif files the data can be split up...
708                                                        for ( Group g : chain.getAtomGroups()){
709                                                                known.addGroup(g);
710                                                        }
711                                                }
712
713                                                break;
714                                        }
715                                }
716                        }
717                        structure.setModel(i,pdbChains);
718                }
719
720
721                // set the oligomeric state info in the header...
722
723                PDBHeader header = structure.getPDBHeader();
724                header.setNrBioAssemblies(strucAssemblies.size());
725
726                // the more detailed mapping of chains to rotation operations happens in StructureIO...
727                // TODO clean this up and move it here...
728                //header.setBioUnitTranformationMap(tranformationMap);
729                Map<Integer,List<ModelTransformationMatrix>> transformationMap = new HashMap<Integer, List<ModelTransformationMatrix>>();
730                int total = strucAssemblies.size();
731
732                for ( int defaultBioAssembly = 1 ; defaultBioAssembly <= total; defaultBioAssembly++){
733                        //List<ModelTransformationMatrix>tmp = getBioUnitTransformationList(pdbId, i +1);
734
735                        PdbxStructAssembly psa = strucAssemblies.get(defaultBioAssembly-1);
736                        List<PdbxStructAssemblyGen> psags = new ArrayList<PdbxStructAssemblyGen>(1);
737
738                        for ( PdbxStructAssemblyGen psag: strucAssemblyGens ) {
739                                if ( psag.getAssembly_id().equals(defaultBioAssembly+"")) {
740                                        psags.add(psag);
741                                }
742                        }
743
744                        //System.out.println("psags: " + psags.size());
745                        BiologicalAssemblyBuilder builder = new BiologicalAssemblyBuilder();
746
747                        // these are the transformations that need to be applied to our model
748                        List<ModelTransformationMatrix> transformations = builder.getBioUnitTransformationList(psa, psags, structOpers);
749
750                        transformationMap.put(defaultBioAssembly,transformations);
751                        //System.out.println("mmcif header: " + (defaultBioAssembly+1) + " " + transformations.size() +" " +  transformations);
752
753                }
754                structure.getPDBHeader().setBioUnitTranformationMap(transformationMap);

1152        /** finds the residue in the internal representation and fixes the residue number and insertion code
1153         *
1154         * @param ppss
1155         */
1156        private void replaceGroupSeqPos(PdbxPolySeqScheme ppss){
1157
1158                if (ppss.getAuth_seq_num().equals("?"))
1159                        return;
1160                // at this stage we are still using the internal asym ids...
1161                List<Chain> matchinChains = getChainsFromAllModels(ppss.getAsym_id());
1162
1163                long sid = Long.parseLong(ppss.getSeq_id());
1164                for (Chain c: matchinChains){
1165                        Group target = null;
1166                        for (Group g: c.getAtomGroups()){
1167
1168                                if ( g instanceof AminoAcidImpl){
1169                                        AminoAcidImpl aa = (AminoAcidImpl)g;
1170                                        if (aa.getId() == sid ) {
1171                                                // found it:
1172                                                target = g;
1173                                                break;
1174                                        }
1175                                }
1176                                else if ( g instanceof NucleotideImpl) {
1177                                        NucleotideImpl n = (NucleotideImpl)g;
1178                                        if ( n.getId() == sid) {
1179                                                target = g;
1180                                                break;
1181                                        }
1182                                } else if ( g instanceof HetatomImpl){
1183                                        HetatomImpl h = (HetatomImpl)g;
1184                                        if ( h.getId() == sid){
1185                                                target =h;
1186                                                break;
1187                                        }
1188                                }
1189                        }
1190                        if (target == null){
1191                                logger.info("could not find group at seq. position " + 
1192                                                ppss.getSeq_id() + " in internal chain " + c.getChainID() + ". " + ppss);
1193                                continue;
1194                        }
1195
1196                        if (! target.getPDBName().equals(ppss.getMon_id())){
1197                                logger.info("could not match PdbxPolySeqScheme to chain:" + ppss);
1198                                continue;
1199                        }
1200
1201                        // fix the residue number to the one used in the PDB files...
1202                        Integer pdbResNum = Integer.parseInt(ppss.getAuth_seq_num());
1203                        // check the insertion code...
1204                        String insCodeS = ppss.getPdb_ins_code();
1205                        Character insCode = null;
1206                        if ( ( insCodeS != null) && (! insCodeS.equals(".")) && insCodeS.length()>0){
1207                                //pdbResNum += insCode
1208                                insCode = insCodeS.charAt(0);
1209                        }
1210
1211                        ResidueNumber residueNumber = new ResidueNumber(null, pdbResNum,insCode);
1212                        target.setResidueNumber(residueNumber);
1213                }
1214        }
