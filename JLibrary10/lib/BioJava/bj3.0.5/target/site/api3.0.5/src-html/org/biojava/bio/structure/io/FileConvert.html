080        /** align a string to the right
081         * length is the total length the new string should take, inlcuding spaces on the left
082         * incredible that this tool is missing in java !!!
083         */
084        private static String alignRight(String input, int length){
085
086
087                int n = input.length();
088                if ( n >= length)
089                        return input;
090
091                String spaces = "                           " ;
092                int diff = length - n ;
093                StringBuffer s = new StringBuffer();
094
095                s.append(spaces.substring(0,diff));
096                s.append(input);
097
098                return s.toString();
099        }
100
101        private static String alignLeft(String input, int length){
102                if (input.length() >= length) {
103                        return input;
104                }
105
106                String spaces = "                           " ;
107                input += spaces.substring(0, length - input.length() );
108                return input;
109
110        }

129        /** prints the connections in PDB style
130         *
131         * Thanks to Tamas Horvath for this one
132         */
133        private String printPDBConnections(){
134
135
136                StringBuffer str = new StringBuffer();
137
138                List<Map<String, Integer>> cons = structure.getConnections();
139                for (int cnr = 0; cnr<cons.size();cnr++){
140                        Map<String,Integer> con =  cons.get(cnr);
141                        Integer as = (Integer) con.get("atomserial");
142
143                        String atomserial =  "";
144
145                        String bond1 = "";
146                        String bond2 = "";
147                        String bond3 = "";
148                        String bond4 = "";
149                        String hyd1  = "";
150                        String hyd2  = "";
151                        String salt1 = "";
152                        String hyd3  = "";
153                        String hyd4  = "";
154                        String salt2 = "";
155
156
157
158                        if (con.containsKey("bond1"))  bond1 = con.get("bond1").toString();
159                        if (con.containsKey("bond2"))  bond2 = con.get("bond2").toString();
160                        if (con.containsKey("bond3"))  bond3 = con.get("bond3").toString();
161                        if (con.containsKey("bond4"))  bond4 = con.get("bond4").toString();
162                        if (con.containsKey("hyd1"))   hyd1  = con.get("hyd1").toString();
163                        if (con.containsKey("hyd2"))   hyd2  = con.get("hyd2").toString();
164                        if (con.containsKey("salt1"))  salt1 = con.get("salt1").toString();
165                        if (con.containsKey("hyd3"))   hyd3  = con.get("hyd3").toString();
166                        if (con.containsKey("hyd4"))   hyd4  = con.get("hyd4").toString();
167                        if (con.containsKey("salt2"))  salt2 = con.get("salt2").toString();
168
169                        atomserial = alignRight(""+as,5) ;
170                        bond1      = alignRight(bond1,5) ;
171                        bond2      = alignRight(bond2,5) ;
172                        bond3      = alignRight(bond3,5) ;
173                        bond4      = alignRight(bond4,5) ;
174                        hyd1       = alignRight(hyd1,5)  ;
175                        hyd2       = alignRight(hyd2,5)  ;
176                        salt1      = alignRight(salt1,5) ;
177                        hyd3       = alignRight(hyd3,5)  ;
178                        hyd4       = alignRight(hyd4,5)  ;
179                        salt2      = alignRight(salt2,5) ;
180
181                        String connectLine = "CONECT" + atomserial + bond1 + bond2 + bond3 +
182                        bond4 + hyd1 + hyd2 + salt1 + hyd3 + hyd4 + salt2;
183
184            str.append(connectLine).append(newline);
185                }
186                return str.toString();
187        }

189        /** Convert a structure into a PDB file.
190         * @return a String representing a PDB file.
191         */
192        public String toPDB() {
193
194
195                StringBuffer str = new StringBuffer();
196                //int i = 0 ;
197
198
199
200                // TODO: print all the PDB header informaton in PDB style
201                // some objects (PDBHeader, Compound) are still missing
202                //
203
204                PDBHeader header = structure.getPDBHeader();
205                header.toPDB(str);
206
207
208                //REMARK 800
209                if (!structure.getSites().isEmpty()) {
210            str.append("REMARK 800                                                                      ").append(newline);
211            str.append("REMARK 800 SITE                                                                 ").append(newline);
212                        for (Site site : structure.getSites()) {
213                                site.remark800toPDB(str);
214                        }
215                }
216                //DBREF
217                for (DBRef dbref : structure.getDBRefs()){
218                        dbref.toPDB(str);
219                        str.append(newline);
220                }
221                //SSBOND
222                for (SSBond ssbond : structure.getSSBonds()){
223                        ssbond.toPDB(str);
224                        str.append(newline);
225                }
226                //SITE
227                for (Site site : structure.getSites()) {
228                        try {
229                                site.toPDB(str);
230                        } catch (Exception e){
231                                e.printStackTrace();
232                        }
233                }
234
235                //
236                // print the atom records
237                //
238
239                // do for all models
240                int nrModels = structure.nrModels() ;
241                if ( structure.isNmr()) {
242                        str.append("EXPDTA    NMR, "+ nrModels+" STRUCTURES"+newline) ;
243                }
244                for (int m = 0 ; m < nrModels ; m++) {
245                        List<Chain> model = structure.getModel(m);
246                        // todo support NMR structures ...
247                        if ( structure.isNmr()) {
248                                str.append("MODEL      " + (m+1)+ newline);
249                        }
250                        // do for all chains
251                        int nrChains = model.size();
252                        for ( int c =0; c<nrChains;c++) {
253                                Chain  chain   = model.get(c);
254                                //String chainID = chain.getChainID();
255                                //if ( chainID.equals(DEFAULTCHAIN) ) chainID = " ";
256                                // do for all groups
257                                int nrGroups = chain.getAtomLength();
258                                for ( int h=0; h<nrGroups;h++){
259
260                                        Group g= chain.getAtomGroup(h);
261
262                                        
263                                        toPDB(g,str);
264                                        
265                                        
266                                }
267                        }
268
269                        if ( structure.isNmr()) {
270                str.append("ENDMDL").append(newline);
271                        }
272
273
274
275                }
276
277                if ( doPrintConnections() )
278                        str.append(printPDBConnections());
279
280                return str.toString() ;
281        }

329        /**
330         * print ATOM record in the following syntax
331        <pre>
332    ATOM      1  N   ASP A  15     110.964  24.941  59.191  1.00 83.44           N
333*
334COLUMNS        DATA TYPE       FIELD         DEFINITION
335---------------------------------------------------------------------------------
3361 -  6        Record name     "ATOM  "
3377 - 11        Integer         serial        Atom serial number.
33813 - 16        Atom            name          Atom name.
33917             Character       altLoc        Alternate location indicator.
34018 - 20        Residue name    resName       Residue name.
34122             Character       chainID       Chain identifier.
34223 - 26        Integer         resSeq        Residue sequence number.
34327             AChar           iCode         Code for insertion of residues.
34431 - 38        Real(8.3)       x             Orthogonal coordinates for X in
345Angstroms.
34639 - 46        Real(8.3)       y             Orthogonal coordinates for Y in
347Angstroms.
34847 - 54        Real(8.3)       z             Orthogonal coordinates for Z in
349Angstroms.
35055 - 60        Real(6.2)       occupancy     Occupancy.
35161 - 66        Real(6.2)       tempFactor    Temperature factor.
35273 - 76        LString(4)      segID         Segment identifier, left-justified.
35377 - 78        LString(2)      element       Element symbol, right-justified.
35479 - 80        LString(2)      charge        Charge on the atom.
355</pre>
356*/
357        public static void toPDB(Atom a, StringBuffer str) {
358
359                Group g = a.getGroup();
360                Chain c = g.getChain();
361                String chainID = c.getChainID();
362
363                String type = g.getType() ;
364
365                String record = "" ;
366                if ( type.equals("hetatm") ) {
367                        record = "HETATM";
368                } else {
369                        record = "ATOM  ";
370                }
371
372
373                // format output ...
374                //int groupsize  = g.size();
375                String resName = g.getPDBName(); 
376                String pdbcode = g.getResidueNumber().toString();
377                //String line    = "" ;
378
379
380                int    seri       = a.getPDBserial()        ;
381                String serial     = alignRight(""+seri,5)   ;
382                String fullname   = a.getFullName()         ;
383
384                // System.out.println(" fullname: " + fullname + " : " + a.getAltLoc() + " : " + pdbcode);
385
386                Character  altLoc = a.getAltLoc()           ;
387                String resseq = "" ;
388                if ( hasInsertionCode(pdbcode) )
389                        resseq     = alignRight(""+pdbcode,5);
390                else
391                        resseq     = alignRight(""+pdbcode,4)+" ";
392                String x          = alignRight(""+d3.format(a.getX()),8);
393                String y          = alignRight(""+d3.format(a.getY()),8);
394                String z          = alignRight(""+d3.format(a.getZ()),8);
395                String occupancy  = alignRight(""+d2.format(a.getOccupancy()),6) ;
396                String tempfactor = alignRight(""+d2.format(a.getTempFactor()),6);
397
398                //System.out.println("fullname,zise:" + fullname + " " + fullname.length());
399
400                String leftResName = alignLeft(resName,3);
401
402                StringBuffer s = new StringBuffer();
403                s.append(record);
404                s.append(serial);
405                s.append(" ");
406                s.append(fullname);
407                s.append(altLoc);
408                s.append(leftResName);
409                s.append(" ");
410                s.append(chainID);
411                s.append(resseq);
412                s.append("   ");
413                s.append(x);
414                s.append(y);
415                s.append(z);
416                s.append(occupancy);
417                s.append(tempfactor);
418
419                Element e = a.getElement();
420                
421                String eString = e.toString().toUpperCase();
422                
423                if ( e.equals(Element.R)) {
424                        eString = "X";
425                }
426                str.append(String.format("%-76s%2s", s.toString(),eString));
427                str.append(newline);
428
429        }

431        /** test if pdbserial has an insertion code */
432        private static boolean hasInsertionCode(String pdbserial) {
433                try {
434                        Integer.parseInt(pdbserial) ;
435                } catch (NumberFormatException e) {
436                        return true ;
437                }
438                return false ;
439        }

442        /** convert a protein Structure to a DAS Structure XML response .
443         * @param xw  a XMLWriter object
444         * @throws IOException ...
445         *
446         */
447        @SuppressWarnings("deprecation")
448        public void toDASStructure(XMLWriter xw)
449        throws IOException
450        {
451
452                /*xmlns="http://www.sanger.ac.uk/xml/das/2004/06/17/dasalignment.xsd" xmlns:align="http://www.sanger.ac.uk/xml/das/2004/06/17/alignment.xsd" xmlns:xsd="http://www.w3.org/2001/XMLSchema-instance" xsd:schemaLocation="http://www.sanger.ac.uk/xml/das/2004/06/17/dasalignment.xsd http://www.sanger.ac.uk/xml/das//2004/06/17/dasalignment.xsd"*/
453
454                if ( structure == null){
455                        System.err.println("can not convert structure null");
456                        return;
457                }
458
459                Map<String,Object> header = structure.getHeader();
460
461                xw.openTag("object");
462                xw.attribute("dbAccessionId",structure.getPDBCode());
463                xw.attribute("intObjectId"  ,structure.getPDBCode());
464                // missing modification date
465                String modificationDate = (String)header.get("modDate") ;
466                xw.attribute("objectVersion",modificationDate);
467                xw.attribute("type","protein structure");
468                xw.attribute("dbSource","PDB");
469                xw.attribute("dbVersion","20070116");
470                xw.attribute("dbCoordSys","PDBresnum,Protein Structure");
471
472                // do we need object details ???
473                xw.closeTag("object");
474
475
476                // do for all models
477                for (int modelnr = 0;modelnr<structure.nrModels();modelnr++){
478
479                        // do for all chains:
480                        for (int chainnr = 0;chainnr<structure.size(modelnr);chainnr++){
481                                Chain chain = (Chain)structure.getChain(modelnr,chainnr);
482                                xw.openTag("chain");
483                                xw.attribute("id",chain.getChainID());
484                                xw.attribute("SwissprotId",chain.getSwissprotId() );
485                                if (structure.isNmr()){
486                                        xw.attribute("model",Integer.toString(modelnr+1));
487                                }
488
489                                //do for all groups:
490                                for (int groupnr =0;
491                                groupnr<chain.getAtomLength()
492                                ;groupnr++){
493                                        Group gr = chain.getAtomGroup(groupnr);
494                                        xw.openTag("group");
495                                        xw.attribute("name",gr.getPDBName());
496                                        xw.attribute("type",gr.getType());
497                                        xw.attribute("groupID",gr.getResidueNumber().toString());
498
499
500                                        // do for all atoms:
501                                        //Atom[] atoms  = gr.getAtoms();
502                                        List<Atom> atoms =  gr.getAtoms();
503                                        for (int atomnr=0;atomnr<atoms.size();atomnr++){
504                                                Atom atom = (Atom)atoms.get(atomnr);
505                                                xw.openTag("atom");
506                                                xw.attribute("atomID",Integer.toString(atom.getPDBserial()));
507                                                xw.attribute("atomName",atom.getFullName());
508                                                xw.attribute("x",Double.toString(atom.getX()));
509                                                xw.attribute("y",Double.toString(atom.getY()));
510                                                xw.attribute("z",Double.toString(atom.getZ()));
511                                                xw.closeTag("atom");
512                                        }
513                                        xw.closeTag("group") ;
514                                }
515
516                                xw.closeTag("chain");
517                        }
518                }
519
520
521                if ( doPrintConnections() ) {
522                        // do connectivity for all chains:
523
524                        List<Map<String,Integer>> cons = structure.getConnections();
525                        for (int cnr = 0; cnr<cons.size();cnr++){
526
527
528                                /*
529                 the HashMap for a single CONECT line contains the following fields:
530                 <ul>
531                 <li>atomserial (mandatory) : Atom serial number
532                 <li>bond1 .. bond4 (optional): Serial number of bonded atom
533                 <li>hydrogen1 .. hydrogen4 (optional):Serial number of hydrogen bonded atom
534                 <li>salt1 .. salt2 (optional): Serial number of salt bridged atom
535                 </ul>
536                                 */
537
538                                Map<String, Integer> con = (Map<String, Integer>)cons.get(cnr);
539                                Integer as = (Integer)con.get("atomserial");
540                                int atomserial = as.intValue();
541
542
543                                List<Integer> atomids = new ArrayList<Integer>() ;
544
545                                // test salt and hydrogen first //
546                                if (con.containsKey("salt1")) atomids.add(con.get("salt1"));
547                                if (con.containsKey("salt2")) atomids.add(con.get("salt2"));
548
549                                if (atomids.size()!=0){
550                                        addConnection(xw,"salt",atomserial,atomids);
551                                        atomids = new ArrayList<Integer>() ;
552                                }
553                                if (con.containsKey("hydrogen1")) atomids.add(con.get("hydrogen1"));
554                                if (con.containsKey("hydrogen2")) atomids.add(con.get("hydrogen2"));
555                                if (con.containsKey("hydrogen3")) atomids.add(con.get("hydrogen3"));
556                                if (con.containsKey("hydrogen4")) atomids.add(con.get("hydrogen4"));
557                                if (atomids.size()!=0){
558                                        addConnection(xw,"hydrogen",atomserial,atomids);
559                                        atomids = new ArrayList<Integer>() ;
560                                }
561
562                                if (con.containsKey("bond1")) atomids.add(con.get("bond1"));
563                                if (con.containsKey("bond2")) atomids.add(con.get("bond2"));
564                                if (con.containsKey("bond3")) atomids.add(con.get("bond3"));
565                                if (con.containsKey("bond4")) atomids.add(con.get("bond4"));
566
567                                if (atomids.size()!=0){
568                                        addConnection(xw,"bond",atomserial,atomids);
569                                }
570                        }
571                }
572        }

574        private void addConnection(XMLWriter xw,String connType, int atomserial, List<Integer> atomids){
575                try{
576                        xw.openTag("connect");
577                        xw.attribute("atomSerial",Integer.toString(atomserial));
578                        xw.attribute("type",connType);
579                        for (int i=0;i<atomids.size();i++){
580                                Integer atomid = atomids.get(i);
581                                if ( atomid == null)
582                                        continue;
583                                int aid = atomid.intValue();
584                                xw.openTag("atomID");
585                                xw.attribute("atomID",Integer.toString(aid));
586                                xw.closeTag("atomID");
587                        }
588                        xw.closeTag("connect");
589                } catch( Exception e) {
590                        e.printStackTrace();
591                }
592        }
