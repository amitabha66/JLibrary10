001package org.biojava3.core.sequence.io;
002
003import java.util.Collection;
004
005import junit.framework.TestCase;
006
007import org.biojava3.core.sequence.ProteinSequence;
008import org.biojava3.core.sequence.compound.AminoAcidCompoundSet;
009import org.biojava3.core.sequence.io.CasePreservingProteinSequenceCreator;
010
011public class CasePreservingProteinSequenceCreatorTest extends TestCase {
012
013        public void testConstructor() {
014                CasePreservingProteinSequenceCreator creator = new CasePreservingProteinSequenceCreator(AminoAcidCompoundSet.getAminoAcidCompoundSet());
015                
016                String seq = "aCDEfgHI-Jkl";
017                ProteinSequence prot = (ProteinSequence) creator.getSequence(seq, 0);
018                Collection<Object> uppercase = prot.getUserCollection();
019                
020                //test some assumptions. Hopefully work on non-english locals too?
021                assertFalse(Character.isUpperCase('-'));
022                assertFalse(Character.isUpperCase('.'));
023                
024                assertEquals("Lengths differ",seq.length(),uppercase.size());
025                
026                int i=0;
027                for(Object obj : uppercase) {
028                        assertTrue("Not a Boolean",obj instanceof Boolean);
029                        Boolean bool = (Boolean)obj;
030                        assertEquals("Doesn't match case of "+seq.charAt(i),(Boolean)Character.isUpperCase(seq.charAt(i)),bool);
031                        i++;
032                }
033        }
034}
