152   private void testAlignment(String name1, String name2, Atom[] ca1, Atom[] ca2, boolean doRigid) throws StructureException,IOException{
153
154
155      Atom[] ca3 = StructureTools.cloneCAArray(ca2);
156
157
158      AFPChain afpChain = doAlign(name1, name2, ca1,ca2,doRigid);
159
160
161
162      String fatcat = afpChain.toFatcat(ca1, ca2);
163      String xml          = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
164
165
166      //System.out.println(xml);
167      AFPChain newChain = AFPChainXMLParser.fromXML (xml, ca1, ca3);
168
169      // test blockNum and optLen arrays 
170      int blockNum = afpChain.getBlockNum();
171      int[] optLen = afpChain.getOptLen();
172
173      assertTrue("The nr of aligned blocks is not the same! " + blockNum + " " + newChain.getBlockNum() , blockNum == newChain.getBlockNum());
174
175
176
177      for ( int i =0 ; i < blockNum ; i++){
178         int newLenI = newChain.getOptLen()[i];
179         assertTrue("The values in the optLen field don't match! pos:" + i + " orig:" + optLen[i] + " new:" +  newLenI,optLen[i] == newLenI);
180      }
181
182      // test the internal optAlign data structure:
183
184      int[][][] optAln1 = afpChain.getOptAln();
185      int[][][] optAln2 = newChain.getOptAln();
186
187      for(int i = 0; i < blockNum; i ++)  {          
188         for(int j = 0; j < optLen[i]; j ++) {
189            int p1 = optAln1[i][0][j];
190            int p2 = optAln1[i][1][j];
191
192            int n1 = optAln2[i][0][j];
193            int n2 = optAln2[i][1][j];
194
195            assertTrue(p1 == n1);
196            assertTrue(p2 == n2);
197
198         }
199      }
200
201      String fatcat2 = newChain.toFatcat(ca1, ca3);
202      //System.out.println("*** RESULT2 "+result2);
203
204      assertEquals(fatcat,fatcat2);
205
206      String xml2 = AFPChainXMLConverter.toXML(newChain, ca1, ca3);
207      assertEquals(xml, xml2);
208
209   }

212   public AFPChain doAlign(String name1, String name2, Atom[] ca1, Atom[] ca2 , boolean doRigid) throws StructureException,IOException{
213      FatCatParameters params = new FatCatParameters();
214
215      FatCat fatCat = new FatCat();
216      AFPChain afpChain;
217      if ( doRigid)
218         afpChain = fatCat.alignRigid(ca1,ca2,params);
219      else
220         afpChain = fatCat.alignFlexible(ca1,ca2,params);
221
222      afpChain.setName1(name1);
223      afpChain.setName2(name2);
224      return afpChain;
225
226   }
227
228   public String[] align (String name1, String name2, Atom[] ca1, Atom[] ca2 , boolean doRigid) throws StructureException,IOException{
229
230      AFPChain afpChain = doAlign(name1, name2, ca1, ca2, doRigid);
231      // flexible original results:
232      String fatcat = afpChain.toFatcat(ca1,ca2);
233      //System.out.println(result1);
234
235
236      String xml = AFPChainXMLConverter.toXML(afpChain,ca1,ca2);
237
238      String[] result = new String[2];
239      result[0] = fatcat;
240      result[1] = xml;
241      return result;
242   }

310         assertTrue(xmlnew1.equals(result1[1]));
