030   public void printFirstMismatch(String s1, String s2){
031      String[] spl1 = s1.split(newline);
032      String[] spl2 = s2.split(newline);
033
034      for (int i = 0 ; i < spl1.length ; i++){
035
036         String line1 = spl1[i];
037
038         if ( i >= spl2.length){
039            System.err.println("s2 does not contain line " + (i+1));
040            return;
041         }
042         String line2 = spl2[i];
043
044         if ( line1.equals(line2)){
045            continue;
046         }
047
048         System.err.println("mismatch in line: " + (i+1));
049
050         for ( int j = 0 ; j < line1.length();j++){
051            char c1 = line1.charAt(j);
052
053            if ( j >= line2.length()){
054               System.err.println("s2 is shorter than s1. length s1:" + line1.length() + " length2:" + line2.length() );
055               return;
056            }
057
058            char c2 = line2.charAt(j);
059            if ( c1 != c2){
060
061               System.err.println("line1: " + line1.substring(0,j+1));
062               System.err.println("line2: " + line2.substring(0,j+1));
063
064               System.err.println("mismatch at position " + (j+1) + " c1: "+ c1 + " " + c2);
065             
066               return;
067            }
068         }
069
070
071      }
072
073   }
