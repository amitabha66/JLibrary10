047        /**
048         * Test of process method, of class FastaReader.
049         */
050        @Test
051        public void testProcess() throws Exception {
052                System.out.println("process");
053                InputStream inStream = this.getClass().getResourceAsStream("/PF00104_small.fasta");
054                assertNotNull(inStream);
055
056
057                FastaReader<ProteinSequence,AminoAcidCompound> fastaReader = new FastaReader<ProteinSequence,AminoAcidCompound>(inStream, new GenericFastaHeaderParser<ProteinSequence,AminoAcidCompound>(), new ProteinSequenceCreator(AminoAcidCompoundSet.getAminoAcidCompoundSet()));
058                LinkedHashMap<String,ProteinSequence> proteinSequences = fastaReader.process();
059                inStream.close();
060
061                //Should have 282 sequences
062                //System.out.println("Expecting 283 got " + proteinSequences.size());
063                assertEquals(proteinSequences.size() ,  283 );
064
065                int seqNum = 0;
066                for(String id:proteinSequences.keySet()) {
067                        ProteinSequence proteinSequence = proteinSequences.get(id);
068                        switch(seqNum) {
069                        case 0:
070                                assertEquals(proteinSequence.getAccession().getID(),"A2D504_ATEGE/1-46");
071                                assertEquals(proteinSequence.getSequenceAsString(),"-----------------FK-N----LP-LED----------------Q----ITL--IQY-----------SWM----------------------CL-SSFA------LSWRSYK---HTNSQFLYFAPDLVF-----------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------");
072                                break;
073                        case 281:
074                                //System.out.println(proteinSequence.getAccession());
075                                //System.out.println(proteinSequence.getSequenceAsString());
076                                assertEquals(proteinSequence.getAccession().getID(),"Q9PU76_CRONI/141-323");
077                                assertEquals(proteinSequence.getSequenceAsString(),"VETVTELTEFAKSI-PGFS-N----LD-LND----------------Q----VTL--LKY-----------GVY----------------------EA-IFAM------LASVMNK---DGMPVAYGNGFITRE------------------------------------------------------------------------------------------------------------------------------------------------------------FLKSLRKPFCDIMEPKFDFA-MKF-NSL-E-LDDSDI--------------------SLFVA-AIIC-CGDRPG-------------------------------------------LVNV--GHIEKMQESIVHVLKL-H-----LQN---------NH---PD----------------------------DI------F--------LFP-KLLQKMAD-LRQLV-----------------TEH-AQLV--QIIKK---TESDAHLHPLL-------QEI---");
078                                break;
079                        case 282:
080                                assertEquals(proteinSequence.getAccession().getID(),"Q98SJ1_CHICK/15-61");
081                                assertEquals(proteinSequence.getSequenceAsString(),"---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------Q-----------------NW------Q--------RFY-QLTKLLDS-MHDVV-----------------ENL-LSFC--FQTFLDKSM--SIEFPEML-------AEI---");
082                                break;
083                        }
084                        seqNum++;
085                }
086                assertEquals(seqNum,283);
087        }
