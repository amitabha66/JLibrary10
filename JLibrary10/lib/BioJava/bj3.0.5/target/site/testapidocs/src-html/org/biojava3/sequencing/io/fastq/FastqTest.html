021package org.biojava3.sequencing.io.fastq;
022
023import java.lang.reflect.Field;
024import java.lang.reflect.Modifier;
025
026import junit.framework.TestCase;
027
028/**
029 * Unit test for Fastq.
030 */
031public final class FastqTest
032    extends TestCase
033{
034
035    public void testImmutable()
036    {
037        Class<Fastq> cls = Fastq.class;
038        assertTrue(Modifier.isPublic(cls.getModifiers()));
039        assertTrue(Modifier.isFinal(cls.getModifiers()));
040        Field[] fields = cls.getDeclaredFields();
041        for (Field field : fields)
042        {
043            assertTrue(Modifier.isPrivate(field.getModifiers()));
044            assertTrue(Modifier.isFinal(field.getModifiers()) ||
045                    (Modifier.isVolatile(field.getModifiers()) && Modifier.isTransient(field.getModifiers())));
046        }
047    }
048
049    public void testConstructor()
050    {
051        Fastq fastq = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
052        assertNotNull(fastq);
053
054        try
055        {
056            new Fastq(null, "sequence", "quality_", FastqVariant.FASTQ_SANGER);
057            fail("ctr(null description) expected IllegalArgumentException");
058        }
059        catch (IllegalArgumentException e)
060        {
061            // expected
062        }
063        try
064        {
065            new Fastq("description", null, "quality_", FastqVariant.FASTQ_SANGER);
066            fail("ctr(null sequence) expected IllegalArgumentException");
067        }
068        catch (IllegalArgumentException e)
069        {
070            // expected
071        }
072        try
073        {
074            new Fastq("description", "sequence", null, FastqVariant.FASTQ_SANGER);
075            fail("ctr(null quality) expected IllegalArgumentException");
076        }
077        catch (IllegalArgumentException e)
078        {
079            // expected
080        }
081        try
082        {
083            new Fastq("description", "sequence", "quality_", null);
084            fail("ctr(null variant) expected IllegalArgumentException");
085        }
086        catch (IllegalArgumentException e)
087        {
088            // expected
089        }
090    }
091
092    public void testDescription()
093    {
094        Fastq fastq = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
095        assertTrue(fastq.getDescription() != null);
096        assertEquals("description", fastq.getDescription());
097    }
098
099    public void testSequence()
100    {
101        Fastq fastq = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
102        assertTrue(fastq.getSequence() != null);
103        assertEquals("sequence", fastq.getSequence());
104    }
105
106    public void testQuality()
107    {
108        Fastq fastq = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
109        assertTrue(fastq.getQuality() != null);
110        assertEquals("quality_", fastq.getQuality());
111    }
112
113    public void testVariant()
114    {
115        Fastq fastq = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
116        assertTrue(fastq.getVariant() != null);
117        assertEquals(FastqVariant.FASTQ_SANGER, fastq.getVariant());
118    }
119
120    public void testEquals()
121    {
122        Fastq fastq0 = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
123        Fastq fastq1 = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
124
125        assertFalse(fastq0.equals(null));
126        assertFalse(fastq1.equals(null));
127        assertFalse(fastq0.equals(new Object()));
128        assertFalse(fastq1.equals(new Object()));
129        assertTrue(fastq0.equals(fastq0));
130        assertTrue(fastq1.equals(fastq1));
131        assertFalse(fastq0 == fastq1);
132        assertFalse(fastq0.equals(fastq1));
133        assertFalse(fastq1.equals(fastq0));
134    }
135
136    public void testHashCode()
137    {
138        Fastq fastq0 = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
139        Fastq fastq1 = new Fastq("description", "sequence", "quality_", FastqVariant.FASTQ_SANGER);
140
141        assertEquals(fastq0.hashCode(), fastq0.hashCode());
142        assertEquals(fastq1.hashCode(), fastq1.hashCode());
143        if (fastq0.equals(fastq1))
144        {
145            assertEquals(fastq0.hashCode(), fastq1.hashCode());
146            assertEquals(fastq1.hashCode(), fastq0.hashCode());
147        }
148        if (fastq1.equals(fastq0))
149        {
150            assertEquals(fastq0.hashCode(), fastq1.hashCode());
151            assertEquals(fastq1.hashCode(), fastq0.hashCode());
152        }
153    }
