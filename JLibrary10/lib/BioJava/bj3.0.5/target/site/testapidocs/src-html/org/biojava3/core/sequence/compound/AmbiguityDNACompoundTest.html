001package org.biojava3.core.sequence.compound;
002
003import static org.junit.Assert.assertEquals;
004
005import java.util.ArrayList;
006import java.util.List;
007
008import org.junit.Test;
009
010public class AmbiguityDNACompoundTest {
011
012  private AmbiguityDNACompoundSet set = AmbiguityDNACompoundSet.getDNACompoundSet();
013
014  @Test
015  public void testAmbiguity() {
016    NucleotideCompound actual = set.getAmbiguity(getCompounds("M","V"));
017    assertEquals("Checking M & G = V", getCompounds("V")[0], actual);
018  }
019
020  @Test
021  public void testBasicAmbiguity() {
022    NucleotideCompound actual = set.getAmbiguity(getCompounds("A","C"));
023    assertEquals("Checking A & C = M", getCompounds("M")[0], actual);
024  }
025
026  private NucleotideCompound[] getCompounds(String... compoundStrings) {
027    List<NucleotideCompound> c = new ArrayList<NucleotideCompound>();
028    for(String s: compoundStrings) {
029      c.add(set.getCompoundForString(s));
030    }
031    return c.toArray(new NucleotideCompound[0]);
032  }
033}
