001package org.biojava.bio.structure.util;
002
003import junit.framework.TestCase;
004
005import org.biojava3.core.util.StringManipulationHelper;
006
007public class StringManipulationTestsHelper extends TestCase {
008
009        /**
010         * Asserts that two strings are equal, line by line, ignoring any difference
011         * of end line delimiters contained within the 2 Strings. This method should
012         * be used if and only if two XMLs are considered identical when all nodes
013         * are identical including their relative order. Generally useful when
014         * asserting identity of <b>automatically regenerated</b> XML.
015         * 
016         * @param expected
017         * @param actual
018         */
019        public static void assertEqualsIgnoreEndline(String expected, String actual) {
020                assertTrue(StringManipulationHelper.equalsToIgnoreEndline(expected, actual));
021        }
022
023        /**
024         * Asserts that two XML-representing strings are equal, by recursively
025         * comparing each node's set of properties & children nodes. This method
026         * should be used when two XMLs are considered identical when all nodes are
027         * identical regardless to their order
028         * 
029         * @param expectedXml
030         * @param actualXml
031         */
032        public static void assertEqualsXml(String expectedXml, String actualXml) {
033                assertTrue(StringManipulationHelper.equalsToXml(expectedXml, actualXml));
034        }
035        
036        /**
037         * @param s
038         * @param pdb
039         */
040        public static void compareString(String s, String pdb) {
041                for (int i = 0; i < s.length(); i++) {
042                        System.out.println("@" + i + "\t>" + s.charAt(i) + ":"
043                                        + pdb.charAt(i) + "<\t" + Integer.toHexString(s.charAt(i))
044                                        + ":" + Integer.toHexString(pdb.charAt(i)));
045                        if (Character.toUpperCase(s.charAt(i)) != Character.toUpperCase(pdb.charAt(i))) {
046                                break;
047                        }
048                }
049        }
050        
051        public void testNothing(){
052                assertTrue(true);
053        }
054}
