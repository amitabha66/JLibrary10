027        @Test
028        public void shortExample3() throws Exception{
029                String[] sequences = new String[3];
030                sequences[0] = "QIKDLLVSSSTDLDTTLVLVNAIYFKGMWKTAFNAEDTREMPFHVTKQESKPVQMMCMNNSFNVATLPAE";
031                sequences[1] = "KMKILELPFASGDLSMLVLLPDEVSDLERIEKTINFEKLTEWTNPNTMEKRRVKVYLPQMKIEEKYNLTS";
032                sequences[2] = "VLMALGMTDLFIPSANLTGISSAESLKISQAVHGAFMELSEDGIEMAGSTGVIEDIKHSPESEQFRADHP";
033                 
034                File aminoAcidCompositionFile = new File("./src/main/resources/AminoAcidComposition.xml");
035                AminoAcidCompositionTable table = PeptideProperties.obtainAminoAcidCompositionTable(aminoAcidCompositionFile);
036                 
037                for(String sequence:sequences){
038                    System.out.println("Molecular Weight: " + PeptideProperties.getMolecularWeightBasedOnXML(sequence, table));
039                }
040        }
041        
042        @Test
043        public void shortExample4(){
044                String sequence = "QIKDLLVSSSTDLDTTLVLVNAIYFKGMWKTAFNAEDTREMPFHVTKQESKPVQMMCMNNSFNVATLPAE";
045                
046                //Enrichment of a specific amino acid type
047                System.out.println("Composition of A: " + PeptideProperties.getEnrichment(sequence, "A"));
048                
049                //Enrichment of a list of amino acid types
050                Map<String, Double> composition = PeptideProperties.getAACompositionString(sequence);
051                for(String aa:composition.keySet()){
052                        System.out.println("Composition of " + aa + ": " + composition.get(aa));
053                }
054        }
055

057        @Test
058        public void shortExample5(){
059                String sequence = "QIKDLLVSSSTDLDTTLVLVNAIYFKGMWKTAFNAEDTRECMPFHVTKQESKPVQMMCMNNSFNVATLPAE";
060                
061                //Absorbance
062                System.out.println("Absorbance (Cys Reduced): " + PeptideProperties.getAbsorbance(sequence, true));
063                System.out.println("Absorbance (Cys Not Reduced): " + PeptideProperties.getAbsorbance(sequence, false));
064                
065                //Extinction Coefficient
066                System.out.println("Extinction Coefficient (Cys Reduced): " + PeptideProperties.getExtinctionCoefficient(sequence, true));
067                System.out.println("Extinction Coefficient (Cys Not Reduced): " + PeptideProperties.getExtinctionCoefficient(sequence, false));
068                
069                //Instability Index
070                System.out.println("Instability Index: " + PeptideProperties.getInstabilityIndex(sequence));
071                
072                //Apliphatic Index
073                System.out.println("Apliphatic Index: " + PeptideProperties.getApliphaticIndex(sequence));
074                
075                //Average Hydropathy Value
076                System.out.println("Average Hydropathy Value: " + PeptideProperties.getAvgHydropathy(sequence));
077                
078                //Isoelectric Point
079                System.out.println("Isoelectric Point: " + PeptideProperties.getIsoelectricPoint(sequence));
080                
081                //Net Charge
082                System.out.println("Net Charge at pH 7: " + PeptideProperties.getNetCharge(sequence));
083        }
