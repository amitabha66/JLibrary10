128        public void ensureClaInstalled(){
129                if ( installedCla.get())
130                        return;
131
132                if ( ! claFileAvailable()){
133                        try {
134                                downloadClaFile();
135                        } catch (Exception e){
136                                e.printStackTrace();
137                                installedCla.set(false);
138                                return;
139                        }                       
140                } 
141
142                try {
143                        parseClassification();
144
145                } catch (Exception e){
146                        e.printStackTrace();
147                        installedCla.set(false);
148                        return;
149                }
150                installedCla.set(true);
151
152        }

250         public List<ScopNode> getTree(ScopDomain domain){
251                 ScopNode node = getScopNode(domain.getSunid());
252
253
254                 List<ScopNode> tree = new ArrayList<ScopNode>();
255                 while (node != null){
256
257                         //System.out.println("This node: sunid:" + node.getSunid() );
258                         //System.out.println(getScopDescriptionBySunid(node.getSunid()));
259                         node = getScopNode(node.getParentSunid());
260                         if ( node != null)
261                                 tree.add(node);
262                 }
263                 Collections.reverse(tree);
264                 return tree;
265         }

267         /* (non-Javadoc)
268          * @see org.biojava.bio.structure.scop.ScopDatabase#filterByDomainName(java.lang.String)
269          */
270         public List<ScopDomain> filterByDomainName(String query) {
271
272                 List<ScopDomain > domains = new ArrayList<ScopDomain>();
273                 if (query.length() <5){
274                         return domains;
275                 }
276
277                 String pdbId = query.substring(1,5);
278
279                 List<ScopDomain> doms = getDomainsForPDB(pdbId);
280
281
282                 if ( doms == null)
283                         return domains;
284
285                 query = query.toLowerCase();
286                 for ( ScopDomain d: doms){
287                         if ( d.getScopId().toLowerCase().contains(query)){
288                                 domains.add(d);
289                         }
290                 }
291
292                 return domains;
293         }

337         public ScopDomain getDomainByScopID(String scopId) {
338                 ensureClaInstalled();
339
340                 if ( scopId.length() < 6) {
341                         throw new IllegalArgumentException("Does not look like a scop ID! " + scopId);
342                 }
343                 String pdbId = scopId.substring(1,5);
344                 List<ScopDomain> doms = getDomainsForPDB(pdbId);
345                 if ( doms == null)
346                         return null;
347                 for ( ScopDomain d : doms){
348                         if ( d.getScopId().equalsIgnoreCase(scopId)) 
349                                 return d;
350                 }
351
352                 return null;
353         }

390         private void parseHierarchy(BufferedReader buffer) throws IOException {
391                 String line = null;
392
393                 int counter =0;
394                 while ((line = buffer.readLine ()) != null) {
395                         if ( line.startsWith("#"))
396                                 continue;
397
398                         String[] spl  = line.split("\t");
399
400                         if ( spl.length != 3 ) {
401                                 System.err.println("parseHierarchy: Can't parse line " + line +" (length: " + spl.length+")");
402                                 continue;
403                         }
404                         counter++;
405                         int sunid       = Integer.parseInt(spl[0]);
406                         int parentSunid = -1;
407
408                         if ( sunid != 0)
409                                 parentSunid = Integer.parseInt(spl[1]);
410
411                         String children = spl[2];
412                         String[] childIds = children.split(",");
413
414                         List<Integer> chis = new ArrayList<Integer>();
415
416                         for ( String id : childIds){
417                                 if ( id.equals("-"))
418                                         continue;
419                                 chis.add(Integer.parseInt(id));
420                         }
421
422                         ScopNode node = new ScopNode();
423
424                         node.setSunid(sunid);
425                         node.setParentSunid(parentSunid);
426                         node.setChildren(chis);
427
428                         scopTree.put(sunid, node);
429                 }
430                 System.out.println("parsed " + counter + " scop sunid nodes.");
431         }

445         private void parseDescriptions(BufferedReader buffer) throws IOException {
446                 String line = null;
447
448                 int counter = 0;
449                 while ((line = buffer.readLine ()) != null) {
450                         if ( line.startsWith("#"))
451                                 continue;
452
453                         String[] spl  = line.split("\t");
454
455                         if ( spl.length != 5 ) {
456                                 System.err.println("parseDescriptions: Can't parse line " + line +" (length: " + spl.length+")");
457                                 continue;
458                         }
459                         counter++;
460
461                         //46464  dm  a.1.1.2 -   Hemoglobin I
462                         int sunID = Integer.parseInt(spl[0]);
463                         ScopCategory category =  ScopCategory.fromString(spl[1]);
464                         String classificationId = spl[2];
465                         String name = spl[3];
466                         String desc = spl[4];
467
468                         ScopDescription c = new ScopDescription();
469                         c.setSunID(sunID);
470                         c.setCategory(category);
471                         c.setClassificationId(classificationId);
472                         c.setName(name);
473                         c.setDescription(desc);
474
475                         sunidMap.put(new Integer(sunID), c);
476
477                 }
478                 System.out.println("parsed " + counter + " scop sunid descriptions.");
479         }

483         private void parseClassification(BufferedReader buffer) throws IOException {
484                 String line = null;
485
486                 int counter = 0;
487                 while ((line = buffer.readLine ()) != null) {
488                         if ( line.startsWith("#"))
489                                 continue;
490
491                         String[] spl  = line.split("\t");
492
493                         if ( spl.length != 6){
494                                 System.err.println("Can't parse line " + line);
495                                 continue;
496
497                         }
498                         counter++;
499
500                         String scopId = spl[0];
501                         String pdbId = spl[1];
502                         String range = spl[2];
503                         String classificationId = spl[3];
504                         Integer sunid = Integer.parseInt(spl[4]);
505                         String tree = spl[5];
506
507                         String[] rangeSpl = range.split(",");
508
509
510                         ScopDomain d = new ScopDomain();
511                         d.setScopId(scopId);
512                         d.setPdbId(pdbId);
513
514                         d.setRanges(Arrays.asList(rangeSpl));
515
516                         d.setClassificationId(classificationId);
517                         d.setSunid(sunid);
518
519                         String[] treeSplit = tree.split(",");
520
521                         if (  treeSplit.length != 7 ) {
522                                 System.err.println("can't process: " + tree );
523                         }
524
525                         int classId =Integer.parseInt(treeSplit[0].substring(3));
526                         int foldId = Integer.parseInt(treeSplit[1].substring(3));
527                         int familyId = Integer.parseInt(treeSplit[2].substring(3));
528                         int superfamilyId = Integer.parseInt(treeSplit[3].substring(3));                       
529                         int domainId = Integer.parseInt(treeSplit[4].substring(3));
530                         int speciesId = Integer.parseInt(treeSplit[5].substring(3));
531                         int px = Integer.parseInt(treeSplit[6].substring(3));
532
533                         d.setClassId(classId);
534                         d.setFoldId(foldId);
535                         d.setSuperfamilyId(superfamilyId);
536                         d.setFamilyId(familyId);
537                         d.setDomainId(domainId);
538                         d.setSpeciesId(speciesId);
539                         d.setPx(px);
540
541                         List<ScopDomain> domainList;
542                         if ( domainMap.containsKey(pdbId)){
543                                 domainList = domainMap.get(pdbId);
544                         } else {
545                                 domainList = new ArrayList<ScopDomain>();
546                                 domainMap.put(pdbId,domainList);
547                         }
548
549                         domainList.add(d);
550                         if ( sunid == 47763)
551                                 System.out.println("FOUND DOMAIN!!!! " + sunid + " " + d);
552                 }
553                 System.out.println("parsed "+ counter + " scop sunid domains.");
554
555         }

590         private void downloadFileFromRemote(URL remoteURL, File localFile) throws FileNotFoundException, IOException{
591                 System.out.println("downloading " + remoteURL + " to: " + localFile);
592                 FileOutputStream out = new FileOutputStream(localFile);
593
594                 InputStream in = remoteURL.openStream();
595                 byte[] buf = new byte[4 * 1024]; // 4K buffer
596                 int bytesRead;
597                 while ((bytesRead = in.read(buf)) != -1) {
598                         out.write(buf, 0, bytesRead);
599                 }
600                 in.close();
601                 out.close();
602
603
604         }

669         /* (non-Javadoc)
670          * @see org.biojava.bio.structure.scop.ScopDatabase#getScopDomainsBySunid(java.lang.Integer)
671          */
672         public List<ScopDomain> getScopDomainsBySunid(Integer sunid)
673         {
674
675                 ensureClaInstalled();
676
677                 List<ScopDomain> domains = new ArrayList<ScopDomain>();
678
679                 for (String pdbId: domainMap.keySet()){
680                         for (ScopDomain d : domainMap.get(pdbId)){
681                                 if ( d.getPx() == sunid) {
682                                         domains.add(d);
683                                         continue;
684                                 } else if ( d.getSpeciesId() == sunid ){
685                                         domains.add(d);
686                                         continue;
687                                 }else if ( d.getDomainId() == sunid ){
688                                         domains.add(d);
689                                         continue;
690                                 }else if ( d.getFamilyId() == sunid ){
691                                         domains.add(d);
692                                         continue;
693                                 }else if ( d.getSuperfamilyId() == sunid ){
694                                         domains.add(d);
695                                         continue;
696                                 }else if ( d.getFoldId() == sunid ){
697                                         domains.add(d);
698                                         continue;
699                                 }else if ( d.getClassId() == sunid ){
700                                         domains.add(d);
701                                         continue;
702                                 }
703                         }
704                 }
705                 return domains;
706
707         }
