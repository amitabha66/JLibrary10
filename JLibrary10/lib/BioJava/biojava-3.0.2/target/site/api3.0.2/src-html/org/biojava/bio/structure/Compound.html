108        @SuppressWarnings("unchecked")
109        public String toString(){
110                StringBuffer buf = new StringBuffer();
111                buf.append("Compound: " + molId + " " +molName + " ");
112                /* disabled for the moment
113
114         buf.append(" chains: " );
115        Iterator<Chain> iter = chainList.iterator();
116        while (iter.hasNext()){
117                Chain c = iter.next();
118                buf.append (c.getName() + " ");
119        }
120
121                 */
122                try {
123                        @SuppressWarnings("rawtypes")
124                        Class c = Class.forName("org.biojava.bio.structure.Compound");
125                        Method[] methods  = c.getMethods();
126
127                        for (int i = 0; i < methods.length; i++) {
128                                Method m = methods[i];
129
130                                String name = m.getName();
131                                if ( name.substring(0,3).equals("get")) {
132                                        if (name.equals("getMolId"))
133                                                continue;
134                                        if ( name.equals("getMolName"))
135                                                continue;
136
137                                        Object o  = m.invoke(this, new Object[]{});
138                                        if ( o instanceof String){
139                                                if ( o != null)
140                                                        buf.append(name.substring(3, name.length())+": "+ o + " ");
141                                        }
142                                        if ( o instanceof List){
143                                                if ( o != null)
144                                                        buf.append(name.substring(3,name.length())+": ");
145
146                                                List<Object>lst = (List<Object>)o;
147                                                for (Object obj : lst){
148                                                        if ( obj instanceof Chain){
149                                                                continue;
150                                                        }
151                                                        buf.append(obj + " ");
152                                                }
153
154                                        }
155                                }
156
157                        }
158
159                } catch (Exception e){
160                        e.printStackTrace();
161                }
162
163
164                //if ( organismScientific != null)
165                //      buf.append(" organism scientific: " + organismScientific);
166
167
168                return buf.toString();
169        }

197        public void showCompound() {
198                System.out.println("COMPOUND INFO:");
199                if (this.molId != null) {
200                        System.out.println("Mol ID: " + this.molId);
201                }
202                if (this.chainId != null) {
203                        System.out.println("Chain: " + this.chainId);
204                        //this.refChainId = chainId
205                }
206                if (this.molName != null) {
207                        System.out.println("Mol Name: " + this.molName);
208                }
209                if (this.title != null) {
210                        System.out.println("Title: " + this.title);
211                }
212                if (this.synonyms != null) {
213                        for (String x : this.synonyms) {
214                                System.out.println("Synomym: " + x);
215                        }
216                }
217                if (this.ecNums != null) {
218                        for (String x : this.ecNums) {
219                                System.out.println("EC: " + x);
220                        }
221                }
222                if (this.fragment != null) {
223                        System.out.println("Fragment? " + this.fragment);
224                }
225                if (this.engineered != null) {
226                        System.out.println("Engineered? " + this.engineered);
227                }
228                if (this.mutation != null) {
229                        System.out.println("Mutation? " + this.mutation);
230                }
231                if (this.biologicalUnit != null) {
232                        System.out.println("Biological Unit: " + this.biologicalUnit);
233                }
234                if (this.details != null) {
235                        System.out.println("Details: " + this.details);
236                }
237                if (this.numRes != null) {
238                        System.out.println("No. Residues: " + this.numRes);
239                }
240                //System.out.println( "\n"
241
242        }
