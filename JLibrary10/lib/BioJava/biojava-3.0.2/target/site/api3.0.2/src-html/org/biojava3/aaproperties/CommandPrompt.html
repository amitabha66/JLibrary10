036        private static AminoAcidCompositionTable checkForValidityAndObtainAATable(String inputLocation, int propertyListSize, String aminoAcidCompositionLocation,
037                        String elementMassLocation) throws Exception{
038                if(inputLocation == null) {
039                        showHelp();
040                        throw new Error("Please do provide location of input file.");
041                }
042                if(propertyListSize == 0){
043                        showHelp();
044                        throw new Error("Please at least specify a property to compute.");
045                }
046                AminoAcidCompositionTable aaTable = null;
047                if(aminoAcidCompositionLocation != null && elementMassLocation == null){
048                        aaTable = PeptideProperties.obtainAminoAcidCompositionTable(new File(aminoAcidCompositionLocation));
049                }else if(aminoAcidCompositionLocation != null && elementMassLocation != null){
050                        aaTable = PeptideProperties.obtainAminoAcidCompositionTable(new File(aminoAcidCompositionLocation, elementMassLocation));
051                }else if(aminoAcidCompositionLocation == null && elementMassLocation != null){
052                        throw new Error("You have define the location of Element Mass XML file. Please also define the location of Amino Acid Composition XML file");
053                }
054                return aaTable;
055        }
056        
057        private static void readInputAndGenerateOutput(String outputLocation, List<Character> propertyList, List<Character> specificList,
058                        String delimiter, String inputLocation, AminoAcidCompositionTable aaTable, int decimalPlace) throws Exception{
059                PrintStream output;
060                if(outputLocation != null)
061                        output = new PrintStream(new File(outputLocation));
062                else
063                        output = System.out;
064                printHeader(output, propertyList, specificList, delimiter);
065                LinkedHashMap<String, ProteinSequence> a = readInputFile(inputLocation, aaTable);
066                //Need for the last sequence
067                for(Entry<String, ProteinSequence> entry:a.entrySet()){
068                        compute(output, entry.getValue().getOriginalHeader(), entry.getValue().getSequenceAsString().trim(), delimiter, aaTable, propertyList, specificList,
069                                        decimalPlace);
070                }
071                output.close();
072        }

074        public static void run(String[] args) throws Exception{
075                /*
076                 * Parse input arguments
077                 */
078                List<Character> propertyList = new ArrayList<Character>();
079                List<Character> specificList = new ArrayList<Character>();
080                String inputLocation = null;
081                String outputLocation = null;
082                String aminoAcidCompositionLocation = null;
083                String elementMassLocation = null;
084                String delimiter = ",";
085                int decimalPlace = 4;
086                
087                for(int i = 0; i < args.length; i++){
088                        if(args[i].charAt(0) != '-' || args[i].length() != 2){
089                                showHelp();
090                                throw new Error("Unknown option: " + args[i]);
091                        }else{
092                                switch(args[i].charAt(1)){
093                                //Required
094                                case 'i': inputLocation = args[++i]; break;
095                                //Optional
096                                case 'o': outputLocation = args[++i]; break;
097                                case 'f':
098                                        i++;
099                                        if(args[i].equalsIgnoreCase("csv")) delimiter = ",";
100                                        else if(args[i].equalsIgnoreCase("tsv")) delimiter = "\t";
101                                        else throw new Error("Invalid value for -f: " + args[i] + ". Please choose either csv or tsv only.");
102                                        break;
103                                case 'x': aminoAcidCompositionLocation = args[++i]; break;
104                                case 'y': elementMassLocation = args[++i]; break;
105                                case 'd': decimalPlace = Integer.parseInt(args[++i]); break;
106                                //Properties
107                                case 'a':
108                                        propertyList.add('1');
109                                        propertyList.add('2');
110                                        propertyList.add('3');
111                                        propertyList.add('4');
112                                        propertyList.add('5');
113                                        propertyList.add('6');
114                                        propertyList.add('7');
115                                        propertyList.add('8');
116                                        propertyList.add('9');
117                                        break;
118                                case '1': propertyList.add('1'); break;
119                                case '2': propertyList.add('2'); break;
120                                case '3': propertyList.add('3'); break;
121                                case '4': propertyList.add('4'); break;
122                                case '5': propertyList.add('5'); break;
123                                case '6': propertyList.add('6'); break;
124                                case '7': propertyList.add('7'); break;
125                                case '8': propertyList.add('8'); break;
126                                case '9': propertyList.add('9'); break;
127                                case '0': 
128                                        propertyList.add('0'); 
129                                        i++;
130                                        if(args[i].length() != 1) throw new Error("Invalid value: " + args[i] + ". Amino Acid Symbol should be of single character");
131                                        specificList.add(args[i].toUpperCase().charAt(0)); 
132                                        break;
133                                default: 
134                                        showHelp();
135                                        throw new Error("Unknown option: " + args[i]);
136                                }
137                        }
138                }
139
140                /*
141                 * Check for validity of input arguments
142                 */
143                AminoAcidCompositionTable aaTable = checkForValidityAndObtainAATable(inputLocation, propertyList.size(), aminoAcidCompositionLocation, 
144                                elementMassLocation);
145                
146                /*
147                 * Read input file and generate output
148                 */
149                readInputAndGenerateOutput(outputLocation, propertyList, specificList, delimiter, inputLocation, aaTable, decimalPlace);
150        }

152        private static LinkedHashMap<String, ProteinSequence> readInputFile(String inputLocation, AminoAcidCompositionTable aaTable) throws Exception{
153                FileInputStream inStream = new FileInputStream(inputLocation);
154                FastaReader<ProteinSequence, AminoAcidCompound> fastaReader;
155                if(aaTable == null){
156                        CompoundSet<AminoAcidCompound>    set = CaseFreeAminoAcidCompoundSet.getAminoAcidCompoundSet();
157                        fastaReader = new FastaReader<ProteinSequence, AminoAcidCompound>(
158                                        inStream, new GenericFastaHeaderParser<ProteinSequence, AminoAcidCompound>(), 
159                                        new ProteinSequenceCreator(set));
160                }else{
161                        fastaReader = new FastaReader<ProteinSequence, AminoAcidCompound>(
162                                        inStream, new GenericFastaHeaderParser<ProteinSequence, AminoAcidCompound>(), 
163                                        new ProteinSequenceCreator(aaTable.getAminoAcidCompoundSet()));
164                }
165                return fastaReader.process();
166        }

168        public enum PropertyName{MolecularWeight, Absorbance_True, Absorbance_False, ExtinctionCoefficient_True, ExtinctionCoefficient_False, 
169                InstabilityIndex, ApliphaticIndex, AverageHydropathyValue, IsoelectricPoint, NetCharge_pH_7, A, R, 
170                N, D, C, E, Q, G, H, I, L,
171                K, M, F, P, S, T, W, Y, V};

173        private static void printHeader(PrintStream output, List<Character> propertyList, List<Character> specificList, String delimiter) throws IOException{
174                int specificCount = 0;
175                /* 
176                 * 1 Molecular weight
177                 * 2 Absorbance (assumed Cys reduced and assume Cys to form cystines)
178                 * 3 Extinction coefficient (assumed Cys reduced and assume Cys to form cystines)
179                 * 4 Instability index
180                 * 5 Apliphatic index
181                 * 6 Average hydropathy value
182                 * 7 Isoelectric point
183                 * 8 Net charge at pH 7
184                 * 9 Composition of the 20 standard amino acid
185                 * 0 Composition of the specific amino acid
186                 */
187                List<String> sList = new ArrayList<String>();
188                sList.add("SequenceName");
189                for(Character c:propertyList){
190                        switch(c){
191                        case '1': sList.add(PropertyName.MolecularWeight.toString()); break;
192                        case '2': sList.add(PropertyName.Absorbance_True.toString()); sList.add(PropertyName.Absorbance_False.toString()); break;
193                        case '3': sList.add(PropertyName.ExtinctionCoefficient_True.toString()); sList.add(PropertyName.ExtinctionCoefficient_False.toString()); break;
194                        case '4': sList.add(PropertyName.InstabilityIndex.toString()); break;
195                        case '5': sList.add(PropertyName.ApliphaticIndex.toString()); break;
196                        case '6': sList.add(PropertyName.AverageHydropathyValue.toString()); break;
197                        case '7': sList.add(PropertyName.IsoelectricPoint.toString()); break;
198                        case '8': sList.add(PropertyName.NetCharge_pH_7.toString()); break;
199                        case '9': 
200                                sList.add(PropertyName.A.toString()); sList.add(PropertyName.R.toString()); 
201                                sList.add(PropertyName.N.toString()); sList.add(PropertyName.D.toString()); 
202                                sList.add(PropertyName.C.toString()); sList.add(PropertyName.E.toString());
203                                sList.add(PropertyName.Q.toString()); sList.add(PropertyName.G.toString());
204                                sList.add(PropertyName.H.toString()); sList.add(PropertyName.I.toString());
205                                sList.add(PropertyName.L.toString()); sList.add(PropertyName.K.toString());
206                                sList.add(PropertyName.M.toString()); sList.add(PropertyName.F.toString());
207                                sList.add(PropertyName.P.toString()); sList.add(PropertyName.S.toString());
208                                sList.add(PropertyName.T.toString()); sList.add(PropertyName.W.toString());
209                                sList.add(PropertyName.Y.toString()); sList.add(PropertyName.V.toString());
210                                break;
211                        case '0': sList.add("" + specificList.get(specificCount++)); break;
212                        }
213                }
214                for(int i = 0; i < sList.size(); i++){
215                        if(i != 0) output.print(delimiter);
216                        output.print(sList.get(i));
217                }
218                output.println();
219                output.flush();
220        }

222        private static void compute(PrintStream output, String header, String sequence, String delimiter, 
223                        AminoAcidCompositionTable aaTable, List<Character> propertyList, List<Character> specificList, int decimalPlace){
224                /* 
225                 * 1 Molecular weight
226                 * 2 Absorbance (assumed Cys reduced and assume Cys to form cystines)
227                 * 3 Extinction coefficient
228                 * 4 Instability index
229                 * 5 Apliphatic index
230                 * 6 Average hydropathy value
231                 * 7 Isoelectric point
232                 * 8 Net charge at pH 7
233                 * 9 Composition of the 20 standard amino acid
234                 * 0 Composition of the specific amino acid
235                 */
236                ProteinSequence pSequence;
237                CompoundSet<AminoAcidCompound> aaSet;
238                if(aaTable != null){
239                        sequence = Utils.checkSequence(sequence, aaTable.getSymbolSet());
240                        pSequence = new ProteinSequence(sequence, aaTable.getAminoAcidCompoundSet());
241                        aaSet = aaTable.getAminoAcidCompoundSet(); 
242                }else{
243                        sequence = Utils.checkSequence(sequence);
244                        pSequence = new ProteinSequence(sequence);
245                        aaSet = AminoAcidCompoundSet.getAminoAcidCompoundSet();
246                }
247                IPeptideProperties pp = new PeptidePropertiesImpl();
248                
249                int specificCount = 0;
250                List<Double> dList = new ArrayList<Double>();
251                for(Character c:propertyList){
252                        switch(c){
253                        case '1': 
254                                if(aaTable == null) 
255                                        dList.add(pp.getMolecularWeight(pSequence));
256                                else 
257                                        dList.add(pp.getMolecularWeight(pSequence));
258                                break;
259                        case '2': 
260                                dList.add(pp.getAbsorbance(pSequence, true)); 
261                                dList.add(pp.getAbsorbance(pSequence, false)); 
262                                break;
263                        case '3': 
264                                dList.add(pp.getExtinctionCoefficient(pSequence, true));
265                                dList.add(pp.getExtinctionCoefficient(pSequence, false)); 
266                                break;
267                        case '4': dList.add(pp.getInstabilityIndex(pSequence)); break;
268                        case '5': dList.add(pp.getApliphaticIndex(pSequence)); break;
269                        case '6': dList.add(pp.getAvgHydropathy(pSequence)); break;
270                        case '7': dList.add(pp.getIsoelectricPoint(pSequence)); break;
271                        case '8': dList.add(pp.getNetCharge(pSequence)); break;
272                        case '9': 
273                                Map<AminoAcidCompound, Double> aaCompound2Double = pp.getAAComposition(pSequence);
274                                //(A, R, N, D, C, E, Q, G, H, I, L, K, M, F, P, S, T, W, Y, V)
275                                dList.add(aaCompound2Double.get(Constraints.A));
276                                dList.add(aaCompound2Double.get(Constraints.R));
277                                dList.add(aaCompound2Double.get(Constraints.N));
278                                dList.add(aaCompound2Double.get(Constraints.D));
279                                dList.add(aaCompound2Double.get(Constraints.C));
280                                dList.add(aaCompound2Double.get(Constraints.E));
281                                dList.add(aaCompound2Double.get(Constraints.Q));
282                                dList.add(aaCompound2Double.get(Constraints.G));
283                                dList.add(aaCompound2Double.get(Constraints.H));
284                                dList.add(aaCompound2Double.get(Constraints.I));
285                                dList.add(aaCompound2Double.get(Constraints.L));
286                                dList.add(aaCompound2Double.get(Constraints.K));
287                                dList.add(aaCompound2Double.get(Constraints.M));
288                                dList.add(aaCompound2Double.get(Constraints.F));
289                                dList.add(aaCompound2Double.get(Constraints.P));
290                                dList.add(aaCompound2Double.get(Constraints.S));
291                                dList.add(aaCompound2Double.get(Constraints.T));
292                                dList.add(aaCompound2Double.get(Constraints.W));
293                                dList.add(aaCompound2Double.get(Constraints.Y));
294                                dList.add(aaCompound2Double.get(Constraints.V));
295                                break;
296                        case '0': dList.add(pp.getEnrichment(pSequence, aaSet.getCompoundForString("" + specificList.get(specificCount++)))); break;
297                        }
298                }
299                output.print(header.replace(delimiter, "_"));
300                for(int i = 0; i < dList.size(); i++){
301                        output.print(delimiter + Utils.roundToDecimals(dList.get(i), decimalPlace));
302                }
303                output.println();
304                output.flush();
305        }
