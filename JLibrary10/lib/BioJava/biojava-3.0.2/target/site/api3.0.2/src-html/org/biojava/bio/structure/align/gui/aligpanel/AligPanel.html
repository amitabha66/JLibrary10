170public void paintComponent(Graphics g){
171
172      super.paintComponent(g);
173
174      Graphics2D g2D = (Graphics2D) g;
175
176
177      g2D.setRenderingHint(RenderingHints.KEY_TEXT_ANTIALIASING,
178            RenderingHints.VALUE_TEXT_ANTIALIAS_ON);
179
180      g2D.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
181            RenderingHints.VALUE_ANTIALIAS_ON);
182
183
184      // only draw within the ranges of the Clip
185      //Rectangle drawHere = g2D.getClipBounds();        
186
187
188      //int startpos = coordManager.getSeqPos(0,drawHere.x);       
189      //int endpos   = coordManager.getSeqPos(0,drawHere.x+drawHere.width-2);
190
191
192      char[] seq1 = afpChain.getAlnseq1();
193      char[] seq2 = afpChain.getAlnseq2();
194      char[] symb = afpChain.getAlnsymb();
195
196      int startpos = 0;
197      int endpos = afpChain.getAlnLength();             
198
199      String summary = afpChain.toString();
200      g2D.drawString(summary, 20, coordManager.getSummaryPos());
201
202      Color significantCol = Color.red;
203      if ( afpChain.isSignificantResult())
204         significantCol = Color.green;
205
206      g2D.setPaint(significantCol);
207      // draw a darker backgroun
208      Rectangle sig = new Rectangle(10,10,10,10);                               
209      g2D.fill(sig);
210      boolean isFATCAT = false;
211      if ( afpChain.getAlgorithmName().startsWith("jFatCat")){
212         isFATCAT = true;
213      }
214      for ( int i = startpos ; ((i <= endpos) && ( i < afpChain.getAlnLength())) ;i++){
215
216         // TODO:
217         // color amino acids by hydrophobicity
218         char c1 = seq1[i];
219         char c2 = seq2[i];
220         boolean isGapped = false;
221         g2D.setFont(seqFont);
222
223         List<Integer> alignedPos = null;
224         if ( colorByAlignmentBlock){
225            alignedPos = DisplayAFP.getEQRAlignmentPos(afpChain);
226         }
227         if ( isFATCAT){
228            char s = symb[i];
229            if ( s != ' ') {
230               isGapped = false;
231               g2D.setFont(eqFont);
232            }
233            else
234               isGapped = true;
235         } else {
236            if ( c1 !=  '-' && c2 != '-' ){
237               // no gap
238               g2D.setFont(eqFont);
239            } else {
240               isGapped = true;
241
242            }
243         }
244
245         Point p1 = coordManager.getPanelPos(0,i);
246         int xpos1 = p1.x;
247         int ypos1 = p1.y;
248         Point p2 = coordManager.getPanelPos(1, i);
249         int xpos2 = p2.x;
250         int ypos2 = p2.y;
251         int blockNum = afpChain.getBlockNum();
252         if (! isGapped) {
253            Color bg = Color.white;
254            Color bg2 = Color.white;
255            Color end1 = ColorUtils.rotateHue(ColorUtils.orange,  (1.0f  / 24.0f) * blockNum  );
256            Color end2 = ColorUtils.rotateHue(ColorUtils.cyan,    (1.0f  / 24.0f) * (blockNum  +1)) ;
257            
258            if ( colorByAlignmentBlock) {
259
260               if (! alignedPos.contains(i)){
261
262                  // unaligned!
263                  bg = Color.white;
264                  bg2 = Color.white;
265               } else  {
266                  
267                  int colorPos = 0;
268                  if (isFATCAT ) {
269                     int block = 0;
270                     char s = symb[i];
271                     try {
272                        block = Integer.parseInt(s+"") - 1;
273                        bg  = ColorUtils.getIntermediate(ColorUtils.orange, end1, blockNum, block);
274                        bg2   = ColorUtils.getIntermediate(ColorUtils.cyan, end2, blockNum, block);
275                        //bg = ColorUtils.rotateHue(ColorUtils.orange,  (1.0f  / 24.0f) * block  );
276                        //bg2 = ColorUtils.rotateHue(ColorUtils.cyan,  (1.0f  / 16.0f) * block );
277                     } catch (Exception e){}
278                     
279                     if ( colorPos > ColorUtils.colorWheel.length){
280                        colorPos = ColorUtils.colorWheel.length % colorPos ;
281                     }
282                  } else {
283                     colorPos = AFPAlignmentDisplay.getBlockNrForAlignPos(afpChain, i);
284                     bg  = ColorUtils.getIntermediate(ColorUtils.orange, end1, blockNum, colorPos);
285                     bg2   = ColorUtils.getIntermediate(ColorUtils.cyan, end2, blockNum, colorPos);
286                     //bg = ColorUtils.rotateHue(ColorUtils.orange,  (1.0f  / 24.0f) * colorPos );
287                     //bg2 = ColorUtils.rotateHue(ColorUtils.cyan,  (1.0f  / 16.0f) * colorPos);
288                  }
289                  
290                   
291                 
292               }
293            } else {
294
295               bg = Color.LIGHT_GRAY;
296               bg2 = Color.LIGHT_GRAY;
297            }
298            
299            // draw a darker background
300            g2D.setPaint(bg);
301            Rectangle rec = new Rectangle(p1.x-1,p1.y-11, (p2.x-p1.x)+12, (p2.y-p1.y)+1);
302            g2D.fill(rec);
303            g2D.setPaint(bg2);
304            Rectangle rec2 = new Rectangle(p1.x-1,p1.y+4, (p2.x-p1.x)+12, (p2.y-p1.y)-3);
305            g2D.fill(rec2);
306
307            //g2D.setPaint(Color.black);
308            //g2D.draw(rec);
309         }
310         if ( colorBySimilarity){
311            if ( c1 == c2){
312               Color bg = COLOR_EQUAL;
313               g2D.setPaint(bg);
314               Rectangle rec = new Rectangle(p1.x-1,p1.y-11, (p2.x-p1.x)+12, (p2.y-p1.y)+12);                           
315               g2D.fill(rec);
316            } else if (AFPAlignmentDisplay.aaScore(c1, c2) > 0) {
317               Color bg = COLOR_SIMILAR;
318               g2D.setPaint(bg);
319               Rectangle rec = new Rectangle(p1.x-1,p1.y-11, (p2.x-p1.x)+12, (p2.y-p1.y)+12);                           
320               g2D.fill(rec);
321            }
322         }
323
324         //if ( selectionStart != null && selectionEnd != null){
325         //     if ( i >= selectionStart.getPos1() && i <= selectionEnd.getPos1()) {
326
327         if ( isSelected(i)){
328            // draw selection
329            Color bg = Color.YELLOW;
330            g2D.setPaint(bg);
331            // draw a darker backgroun
332            Rectangle rec = new Rectangle(p1.x-1,p1.y-11, (p2.x-p1.x)+12, (p2.y-p1.y)+12);                              
333            g2D.fill(rec);
334            //  }
335         }
336
337         // draw the AA sequence
338         g2D.setColor(Color.black);
339         g2D.drawString(c1+"",xpos1,ypos1);
340         g2D.drawString(c2+"" ,xpos2,ypos2);
341
342
343
344
345         //System.out.println(seq1[i] + " " + xpos1 + " " + ypos1 + " " + seq2[i] + xpos2 + " " + ypos2);
346      }
347
348      int nrLines = (afpChain.getAlnLength() -1) / AFPChainCoordManager.DEFAULT_LINE_LENGTH;
349
350
351      for ( int i = 0 ; i <= nrLines ; i++){
352
353         try {
354            // draw legend at i
355            Point p1 = coordManager.getLegendPosition(i,0);
356            Point p2 = coordManager.getLegendPosition(i,1);
357
358
359            int aligPos = i * AFPChainCoordManager.DEFAULT_LINE_LENGTH ;
360            Atom a1 = DisplayAFP.getAtomForAligPos(afpChain, 0,aligPos, ca1,false);
361            Atom a2 = DisplayAFP.getAtomForAligPos(afpChain, 1,aligPos, ca2,false);
362            String label1 = JmolTools.getPdbInfo(a1,false);                             
363            String label2 = JmolTools.getPdbInfo(a2,false);
364            g2D.drawString(label1, p1.x,p1.y);
365            g2D.drawString(label2, p2.x,p2.y);
366
367            Point p3 = coordManager.getEndLegendPosition(i,0);
368            Point p4 = coordManager.getEndLegendPosition(i,1);
369
370            aligPos = i * AFPChainCoordManager.DEFAULT_LINE_LENGTH + AFPChainCoordManager.DEFAULT_LINE_LENGTH -1 ;
371            if ( aligPos > afpChain.getAlnLength())
372               aligPos = afpChain.getAlnLength() - 1;
373            Atom a3 = DisplayAFP.getAtomForAligPos(afpChain, 0,aligPos, ca1,true);
374            Atom a4 = DisplayAFP.getAtomForAligPos(afpChain, 1,aligPos, ca2,true);
375
376            String label3 = JmolTools.getPdbInfo(a3,false);
377            String label4 = JmolTools.getPdbInfo(a4,false);
378
379            g2D.drawString(label3, p3.x,p3.y);
380            g2D.drawString(label4, p4.x,p4.y);
381
382
383         } catch (StructureException e){
384            e.printStackTrace();
385         }
386      }

414   private void updateJmolDisplay() {
415
416      if ( jmol == null)
417         return;
418
419      int size = afpChain.getAlnLength();
420
421      StringBuffer cmd = new StringBuffer("select ");
422
423      int nrSelected = 0;
424      try {
425
426         for (int i = 0 ; i< size ; i++){
427            if ( selection.get(i)){
428
429               Atom a1 = DisplayAFP.getAtomForAligPos(afpChain, 0,i, ca1, false);
430               Atom a2 = DisplayAFP.getAtomForAligPos(afpChain, 1,i, ca2, false);
431
432               String select1 = "";
433
434               if ( a1 != null ) 
435                  select1 = JmolTools.getPdbInfo(a1);
436               String select2 = "" ;
437               if ( a2 != null)
438                  select2 = JmolTools.getPdbInfo(a2);
439
440               // nothing to display
441               if ( select1.equals("") && select2.equals(""))
442                  continue;
443
444               if ( nrSelected > 0)
445                  cmd.append(", ");
446
447               cmd.append(select1);
448               cmd.append(", ");
449               cmd.append(select2);
450               nrSelected++;
451            }
452         }
453
454
455      } catch (StructureException e){
456         e.printStackTrace();
457      }
458      if ( nrSelected == 0)
459         cmd.append(" none;");
460      else
461         cmd.append("; set display selected;");
462
463      jmol.evalString(cmd.toString());
464
465
466   }

555   public void actionPerformed(ActionEvent e) {
556      String cmd = e.getActionCommand();
557      // print is handled by superclass
558      if ( cmd.equals(MenuCreator.PRINT)) {
559         super.actionPerformed(e);
560      } else if (cmd.equals(MenuCreator.TEXT_ONLY)){
561         String result = AfpChainWriter.toWebSiteDisplay(afpChain, ca1, ca2);
562         DisplayAFP.showAlignmentImage(afpChain, result);
563      } else if ( cmd.equals(MenuCreator.PAIRS_ONLY)) {
564         String result = AfpChainWriter.toAlignedPairs(afpChain, ca1, ca2) ;
565         DisplayAFP.showAlignmentImage(afpChain, result);
566      } else if (cmd.equals(MenuCreator.FATCAT_TEXT)){
567         String result = afpChain.toFatcat(ca1, ca2);
568         result += AFPChain.newline;
569         result += afpChain.toRotMat();
570         DisplayAFP.showAlignmentImage(afpChain, result);
571      } else if ( cmd.equals(MenuCreator.SELECT_EQR)){
572         selectEQR();
573      } else if ( cmd.equals(MenuCreator.SIMILARITY_COLOR)){
574         colorBySimilarity(true);
575      } else if ( cmd.equals(MenuCreator.EQR_COLOR)){
576         colorBySimilarity(false);
577      } else if ( cmd.equals(MenuCreator.FATCAT_BLOCK)){
578         colorByAlignmentBlock();
579      }
580      else {
581         System.err.println("Unknown command:" + cmd);
582      }
583
584   }

602   private void selectEQR() {
603
604      selection.clear();
605
606      List<Integer> pos1 = DisplayAFP.getEQRAlignmentPos(afpChain);
607
608      for (int pos : pos1){
609         selection.flip(pos);
610      }
611      mouseMoLi.triggerSelectionLocked(true);
612      updateJmolDisplay();
613      this.repaint();
