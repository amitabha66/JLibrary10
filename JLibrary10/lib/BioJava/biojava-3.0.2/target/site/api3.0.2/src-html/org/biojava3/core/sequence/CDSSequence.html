023package org.biojava3.core.sequence;
024
025
026import org.biojava3.core.sequence.compound.NucleotideCompound;
027import org.biojava3.core.sequence.template.CompoundSet;
028
029/**
030 * Represents a exon or coding sequence in a gene. It has a parent {@link TranscriptSequence}
031 * where a TranscriptSequence is the child of a GeneSequence
032 * Not important for protein construction but the phase is used if outputting the gene
033 * to a gff3 file. {@link http://www.sequenceontology.org/gff3.shtml}
034 * @author Scooter Willis
035 */
036public class CDSSequence extends DNASequence {
037
038    //private static final Logger log = Logger.getLogger(CDSSequence.class.getName());
039    Integer phase = 0; // 0, 1, 2 
040    TranscriptSequence parentTranscriptSequence;
041
042    /**
043     *
044     * @param parentSequence
045     * @param bioBegin
046     * @param bioEnd
047     * @param phase
048     */
049    public CDSSequence(TranscriptSequence parentSequence, int bioBegin, int bioEnd, int phase) {
050        parentTranscriptSequence = parentSequence;
051        this.setParentSequence(parentTranscriptSequence);
052        setBioBegin(bioBegin);
053        setBioEnd(bioEnd);
054        this.phase = phase;
055
056    }
057
058        @Override
059    public int getLength() {
060        return Math.abs(this.getBioEnd() - this.getBioBegin()) + 1;
061    }
062
063    /**
064     *
065     * @return get the phase
066     */
067    public Integer getPhase() {
068        return phase;
069    }
070
071    /**
072     *
073     * @return get the strand
074     */
075    public Strand getStrand() {
076        return parentTranscriptSequence.getStrand();
077    }
078
079    /**
080     * A CDS sequence if negative stranded needs to be reverse complement
081     * to represent the actual coding sequence. When getting a ProteinSequence
082     * from a TranscriptSequence this method is callled for each CDSSequence
083     * {@link http://www.sequenceontology.org/gff3.shtml}
084     * {@link http://biowiki.org/~yam/bioe131/GFF.ppt}
085     * @return coding sequence
086     */
087    public String getCodingSequence() {
088        String sequence = this.getSequenceAsString(getBioBegin(), getBioEnd(), getStrand());
089        if (getStrand() == Strand.NEGATIVE) {
090            //need to take complement of sequence because it is negative and we are returning a coding sequence
091            StringBuilder b = new StringBuilder(getLength());
092            CompoundSet<NucleotideCompound> compoundSet = this.getCompoundSet();
093            for (int i = 0; i < sequence.length(); i++) {
094                String nucleotide = sequence.charAt(i) + "";
095                NucleotideCompound nucleotideCompound = compoundSet.getCompoundForString(nucleotide);
096                b.append(nucleotideCompound.getComplement().getShortName());
097            }
098            sequence = b.toString();
099        }
100        //  sequence = sequence.substring(phase);
101        return sequence;
102    }
103}
