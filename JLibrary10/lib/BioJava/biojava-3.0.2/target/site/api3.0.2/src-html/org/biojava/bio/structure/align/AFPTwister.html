053   /**
054    * calculate the total rmsd of the blocks
055    * output a merged pdb file for both proteins                                                                                                                                       
056     protein 1, in chain A                                                                                                                                                            
057    protein 2 is twisted according to the twists detected, in chain B
058    * @return  twisted Groups
059    */     
060   public static Group[] twistPDB(AFPChain afpChain,Atom[] ca1, Atom[] ca2) throws StructureException {
061      //--------------------------------------------------------                                                                                                                         
062
063      if ( afpChain.isShortAlign())
064         return new Group[0];
065
066      List<AFP> afpSet = afpChain.getAfpSet();
067
068      int blockNum = afpChain.getBlockNum();
069
070      int     i,  b2, e2;
071
072
073      //superimposing according to the initial AFP-chaining
074      Atom[] origCA      = StructureTools.cloneCAArray(ca2);
075      Atom[] iniTwistPdb = StructureTools.cloneCAArray(ca2);
076
077      int[] blockResSize = afpChain.getBlockResSize();
078
079      int[][][] blockResList = afpChain.getBlockResList();
080
081      int[] afpChainList    = afpChain.getAfpChainList();
082      int[] block2Afp       = afpChain.getBlock2Afp();
083      int[] blockSize       = afpChain.getBlockSize();
084      int[] focusAfpList    = afpChain.getFocusAfpList();
085
086      if ( focusAfpList == null){
087         focusAfpList = new int[afpChain.getMinLen()];
088         afpChain.setFocusAfpList(focusAfpList);
089      }
090
091      int  focusAfpn = 0;
092      e2 = 0;
093      b2 = 0;
094
095
096      if ( debug)
097         System.out.println("blockNUm at twister:" + blockNum);
098
099      for(int bk = 0; bk < blockNum; bk ++)       {
100
101         // THIS IS TRANSFORMING THE ORIGINAL ca2 COORDINATES, NO CLONING...
102         // copies the atoms over to iniTwistPdb later on in modifyCod
103
104         transformOrigPDB(blockResSize[bk], blockResList[bk][0], blockResList[bk][1],  ca1, ca2, null,-1);
105
106         //transform pro2 according to comparison of pro1 and pro2 at give residues                                                                                         
107         if(bk > 0)      { 
108            b2 = e2; 
109         }
110
111         if ( debug ){
112            System.out.println("b2 is " + b2 + " before  modifyCon");
113         }
114
115
116         if(bk < (blockNum - 1))   {
117
118            //bend at the middle of two consecutive AFPs
119            int afpPos = afpChainList[block2Afp[bk] + blockSize[bk] - 1];
120            AFP a1 = afpSet.get(afpPos);
121            e2 = a1.getP2() ;
122
123            int afpPos2 = afpChainList[block2Afp[bk + 1]];
124            AFP a2 = afpSet.get(afpPos2);
125            e2 = (a2.getP2() - e2)/ 2 + e2;
126            if (debug)
127               System.err.println("e2 : " + e2);
128         } else{ 
129            // last one is until the end...
130            e2 = ca2.length; 
131         }
132
133         // this copies the coordinates over into iniTwistPdb
134         cloneAtomRange(iniTwistPdb, ca2, b2, e2);
135
136         //bound[bk] = e2;
137         for(i = 0; i < blockSize[bk]; i ++)     {                       
138            focusAfpList[focusAfpn] = afpChainList[block2Afp[bk] + i];
139            focusAfpn++;
140         }
141      }
142
143
144      int focusResn = afp2Res(afpChain, focusAfpn,  focusAfpList, 0);
145
146      int totalLenIni = focusResn;
147      afpChain.setTotalLenIni(totalLenIni);
148
149      if(debug)    
150         System.out.println(String.format("calrmsdini for %d residues", focusResn));
151
152      double totalRmsdIni = calCaRmsd(ca1,iniTwistPdb, focusResn, afpChain.getFocusRes1() , afpChain.getFocusRes2() );
153
154
155      afpChain.setTotalRmsdIni(totalRmsdIni);
156      if ( debug ) {
157         System.out.println("got iniRMSD: " + totalRmsdIni);
158         if ( totalRmsdIni == 5.76611141613097) {
159            System.out.println(afpChain.getAlnseq1());
160            System.out.println(afpChain.getAlnsymb());
161            System.out.println(afpChain.getAlnseq2());
162            // System.exit(0);
163         }
164      }
165      afpChain.setFocusAfpList(focusAfpList);
166      afpChain.setBlock2Afp(block2Afp);
167      afpChain.setAfpChainList(afpChainList);
168
169      return twistOptimized(afpChain,ca1,origCA); 
170   }

172   /**       superimposing according to the optimized alignment
173    * 
174    * @param afpChain
175    * @param ca1
176    * @param ca2
177    * @return Group array twisted.
178    * @throws StructureException
179    */
180   public static Group[] twistOptimized(AFPChain afpChain,Atom[] ca1, Atom[] ca2) throws StructureException {
181
182      Atom[] optTwistPdb = new Atom[ca2.length];
183
184      int gPos = -1;
185      for (Atom a : ca2){ 
186         gPos++;
187         optTwistPdb[gPos] = a;
188      }
189
190      int blockNum = afpChain.getBlockNum();
191
192      int b2 =  0;
193      int e2 =  0;
194      int focusResn = 0;
195      int[] focusRes1 = afpChain.getFocusRes1();
196      int[] focusRes2 = afpChain.getFocusRes2();
197
198      if(focusRes1 == null) {
199          focusRes1 = new int[afpChain.getCa1Length()];
200          afpChain.setFocusRes1(focusRes1);
201      }
202      if(focusRes2 == null) {
203          focusRes2 = new int[afpChain.getCa2Length()];
204          afpChain.setFocusRes2(focusRes2);
205      }
206      
207      int[] optLen      = afpChain.getOptLen();
208      int[][][] optAln  = afpChain.getOptAln();
209
210      for(int bk = 0; bk < blockNum; bk ++)       {
211         //System.out.println("transforming "+ optLen[bk] + " " +  bk);
212         //  THIS IS TRANSFORMING THE ORIGINAL ca2 COORDINATES, NO CLONING...
213         // copies the atoms over to iniTwistPdb later on in modifyCod
214         transformOrigPDB(optLen[bk], optAln[bk][0], optAln[bk][1], ca1, ca2,afpChain,bk);
215
216         //transform pro2 according to comparison of pro1 and pro2 at give residues                                                                                         
217         if(bk > 0)      { b2 = e2; }
218         if(bk < blockNum - 1)   { //bend at the middle of two consecutive blocks                                                                                           
219            e2 = optAln[bk][1][optLen[bk] - 1];
220            e2 = (optAln[bk + 1][1][0] - e2)/ 2 + e2;
221         }
222         else    { e2 = ca2.length; }
223         //System.out.println("modifyCod: " + b2 + " " + e2);
224         cloneAtomRange(optTwistPdb, ca2, b2, e2);
225         //bound[bk] = e2;
226         for(int i = 0; i < optLen[bk]; i ++)        {
227            // System.out.println("bk " + bk + " i " + i + " optLen:" + optLen[bk] + " " + optAln[bk][0][i]);
228            focusRes1[focusResn] = optAln[bk][0][i];
229            focusRes2[focusResn] = optAln[bk][1][i];
230            focusResn ++;
231         }
232      }
233      int totalLenOpt = focusResn;
234      if(debug)       System.out.println(String.format("calrmsdopt for %d residues\n", focusResn));
235      double totalRmsdOpt = calCaRmsd(ca1, optTwistPdb, focusResn, focusRes1, focusRes2);
236      if ( debug ) System.out.println("got opt RMSD: " + totalRmsdOpt);
237      int optLength = afpChain.getOptLength();
238
239      if(totalLenOpt != optLength)    {
240         System.err.println(String.format("Warning: final alignment length is different %d %d\n", totalLenOpt, optLength));
241      }
242      if(debug) System.out.println(String.format("final alignment length %d, rmsd %.3f\n", focusResn, totalRmsdOpt));
243
244      afpChain.setTotalLenOpt(totalLenOpt);
245      afpChain.setTotalRmsdOpt(totalRmsdOpt);
246
247      Group[] retGroups = StructureTools.cloneGroups(optTwistPdb);
248
249      return retGroups;
250
251   }

254   /**
255    * transform the coordinates in the ca2 according to the superimposing of the given position pairs.
256    * No Cloning, transforms input atoms.
257    */                                                                                                                                             
258   // orig name: transPdb
259   private static void transformOrigPDB(int n, int[] res1, int[] res2, Atom[] ca1, Atom[] ca2, AFPChain afpChain, int blockNr)
260   throws StructureException
261   {
262      if ( debug ){
263         System.err.println("transforming original coordinates " + n + " len1: " + ca1.length + " res1:" + res1.length + " len2: "+ ca2.length + " res2: " + res2.length);
264      }
265
266      Atom[] cod1 = getAtoms(ca1, res1, n,false); 
267      Atom[] cod2 = getAtoms(ca2, res2, n,false);
268
269      //double  *cod1 = pro1->Cod4Res(n, res1);
270      //double  *cod2 = pro2->Cod4Res(n, res2);
271
272      Matrix r;
273      Atom t;
274
275      SVDSuperimposer svd = new SVDSuperimposer(cod1, cod2);
276
277      r = svd.getRotation();
278      t = svd.getTranslation();
279
280      if ( debug)
281         System.out.println("transPdb: transforming orig coordinates with matrix: " + r);
282      
283      if ( afpChain != null ){
284         Matrix[] ms = afpChain.getBlockRotationMatrix();
285         if ( ms == null)
286            ms = new Matrix[afpChain.getBlockNum()];
287
288         ms[blockNr]= r;
289
290         Atom[] shifts = afpChain.getBlockShiftVector();
291         if ( shifts == null)
292            shifts = new Atom[afpChain.getBlockNum()];
293         shifts[blockNr] = t;
294
295         afpChain.setBlockRotationMatrix(ms);
296         afpChain.setBlockShiftVector(shifts);
297      }
298
299      for (Atom a : ca2){
300         //Calc.rotate(a, r);
301         //Calc.shift(a, t);
302
303         Calc.rotate(a.getGroup(),r);
304         Calc.shift(a.getGroup(),t);
305
306      }
307
308
309   }

311   // like Cod4Res
312   // most likely the clone flag is not needed
313   private static Atom[] getAtoms(Atom[] ca, int[] positions, int length, boolean clone){
314
315      List<Atom> atoms = new ArrayList<Atom>();
316      for ( int i = 0 ; i < length ; i++){
317         int p = positions[i];
318         Atom a;
319         if ( clone ){
320            a = (Atom)ca[p].clone();
321            a.setGroup((Group)ca[p].getGroup().clone());
322         }
323         else {
324            a = ca[p];
325         }
326         atoms.add(a);
327      }
328      return (Atom[]) atoms.toArray(new Atom[atoms.size()]);
329   }

333   /** Clones and copies the Atoms from p2 into p1 range is between r1 and r2 
334    * 
335    * @param p1 
336    * @param p2
337    * @param r1
338    * @param r2
339    */
340   // orig name: modifyCod
341   private static void cloneAtomRange(Atom[] p1, Atom[] p2, int r1, int r2)
342   throws StructureException
343   {
344
345      if ( debug )
346         System.err.println("modifyCod from: " + r1 + " to: " + r2);
347
348      // special clone method, can;t use StructureTools.cloneCAArray, since we access the data
349      // slightly differently here.
350      List<Chain> model = new ArrayList<Chain>();
351      for(int i = r1; i < r2; i ++) {
352
353         Group g = p2[i].getGroup();
354         Group newG = (Group)g.clone();
355
356         p1[i] = newG.getAtom(StructureTools.caAtomName);
357         Chain parentC = g.getChain();
358
359         Chain newChain = null;
360
361         for( Chain c: model){
362            if (c.getChainID().equals(parentC.getChainID())){
363               newChain = c;
364               break;
365            }
366         }
367
368         if ( newChain == null){
369            newChain = new ChainImpl();
370            newChain.setChainID(parentC.getChainID());
371            model.add(newChain);
372         }
373
374         newChain.addGroup(newG);
375
376
377      } //modify caCod                                                                                                                                                                                               
378
379   }

383   /**
384    * Return the rmsd of the CAs between the input pro and this protein, at given positions.
385    * quite similar to transPdb but while that one transforms the whole ca2, this one only works on the res1 and res2 positions.
386    *  
387    *  Modifies the coordinates in the second set of Atoms (pro).
388    *                                                                                                                               
389    * @return rmsd of CAs
390    */
391   private static double calCaRmsd(Atom[] ca1, Atom[] pro, int resn, int[] res1, int[] res2) throws StructureException
392   { 
393
394
395
396      Atom[] cod1 = getAtoms(ca1,  res1,resn,false);
397      Atom[] cod2 = getAtoms(pro,  res2,resn,false);
398
399      Matrix r;
400      Atom t;
401
402      if ( cod1.length == 0  || cod2.length == 0){
403         System.out.println("length of atoms  == 0!");
404         return 99;
405      }
406      SVDSuperimposer svd = new SVDSuperimposer(cod1, cod2);
407
408      r = svd.getRotation();
409      t = svd.getTranslation();
410            
411      for (Atom a : cod2){
412         //Calc.rotate(a, r);
413         //Calc.shift(a, t);
414
415         Calc.rotate(a.getGroup(), r);
416         Calc.shift(a.getGroup(),  t);
417      }
418
419//      if ( showAlignmentSteps){
420//         StructureAlignmentJmol jmol = new StructureAlignmentJmol();
421//
422//         jmol.setAtoms(cod1);
423//         jmol.evalString("select * ; wireframe off; spacefill off;  backbone on; color chain; select ligand;color cpk; wireframe 40;spacefill 120;  ");

429//         jmol2.setTitle("calCaRmsd - pdb2");
430//      }
431      return SVDSuperimposer.getRMS(cod1, cod2);

434   /**
435    * Set the list of equivalent residues in the two proteins given a list of AFPs    
436    * 
437    * WARNING: changes the values for FocusRes1, focusRes2 and FocusResn in afpChain!
438    *                                                                                                                                                 
439    * @param afpChain the AFPChain to store resuts
440    * @param afpn nr of afp
441    * @param afpPositions
442    * @param listStart
443    * @return nr of eq residues
444    */
445
446   public static int afp2Res(AFPChain afpChain, int afpn, int[] afpPositions , int listStart   )
447   {
448      int[] res1 = afpChain.getFocusRes1(); 
449      int[] res2 = afpChain.getFocusRes2();
450      int minLen = afpChain.getMinLen();
451      
452      int n = 0;
453      
454      List<AFP> afpSet =afpChain.getAfpSet();
455           
456      for(int i = listStart; i < listStart+afpn; i ++)      {
457         //System.out.println("i:"+i+" " + list.length + " " + afpn);
458         int a = afpPositions[i];
459         //System.out.println(String.format("afp %d %d\n", i, a));
460         for(int j = 0; j < afpSet.get(a).getFragLen(); j ++)     {
461            if(n >= minLen) {              
462               throw new RuntimeException("Error: too many residues in AFPChainer.afp2Res!");
463            }
464            res1[n] = afpSet.get(a).getP1() + j;
465            res2[n] = afpSet.get(a).getP2() + j;
466            n ++;
467         }
468      }
469      
470      afpChain.setFocusRes1(res1);
471      afpChain.setFocusRes2(res2);
472      afpChain.setFocusResn(n);
473      
474      if ( n == 0 ){
475         System.err.println("warning: n=0!!! + " + afpn + " " + listStart + " " + afpPositions.length);
476      }
477      return n;
478   }
