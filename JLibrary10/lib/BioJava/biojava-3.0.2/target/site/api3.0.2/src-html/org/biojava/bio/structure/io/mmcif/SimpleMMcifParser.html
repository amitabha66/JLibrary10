136        public void parse(BufferedReader buf)
137        throws IOException {
138
139                triggerDocumentStart();
140
141
142                // init container objects...
143                struct = new Struct();
144                String line = null;
145
146                boolean inLoop = false;
147
148                List<String> loopFields = new ArrayList<String>();
149                List<String> lineData   = new ArrayList<String>();
150
151                String category = null;
152
153
154                // the first line is a data_PDBCODE line, test if this looks like a mmcif file
155                line = buf.readLine();
156                if (!line.startsWith("data_")){
157                        System.err.println("this does not look like a valid MMcif file! The first line should be data_1XYZ, but is " + line);
158                        triggerDocumentEnd();
159                        return;
160                }
161
162                while ( (line = buf.readLine ()) != null ){
163                        //System.out.println(inLoop + " " + line);
164
165
166                        if ( inLoop){
167
168                                if (line.startsWith(LOOP_END)){
169                                        // reset all data
170                                        inLoop = false;
171                                        lineData.clear();
172                                        category=null;
173                                        loopFields.clear();
174                                        continue;
175
176
177                                }
178
179                                if ( line.startsWith(FIELD_LINE)){
180                                        // found another field.
181                                        String txt = line.trim();
182                                        //System.out.println("line: " + txt);
183                                        if ( txt.indexOf('.') > -1){
184
185                                                String[] spl = txt.split("\\.");
186                                                //System.out.println(spl.length);
187                                                category = spl[0];
188                                                String attribute = spl[1];
189                                                loopFields.add(attribute);
190                                                if ( spl.length > 2){
191                                                        System.err.println("found nested attribute, not supported, yet!");
192                                                }
193                                        } else {
194                                                category = txt;
195                                        }
196
197
198                                } else {
199
200                                        // in loop and we found a data line
201                                        lineData = processLine(line, buf, loopFields.size());
202                                        if ( lineData.size() != loopFields.size()){
203                                                System.err.println("did not find enough data fields...");
204
205                                        }
206
207                                        endLineChecks(category, loopFields,lineData);
208
209                                        lineData.clear();
210
211                                }
212
213                        } else {
214                                // not in loop
215
216                                if ( line.startsWith(LOOP_START)){
217                                        loopFields.clear();
218                                        inLoop = true;
219                                        category=null;
220                                        lineData.clear();
221                                        continue;
222                                } else if (line.startsWith(LOOP_END)){
223                                        inLoop = false;
224                                        if ( category != null)
225                                                endLineChecks(category, loopFields, lineData);
226                                        category = null;
227                                        loopFields.clear();
228                                        lineData.clear();
229                                } else {
230                                        // a boring normal line
231                                        //System.out.println("boring data line: " + line + " " + inLoop + " " );
232                                        List<String> data = processLine(line, buf, 2);
233                                        //System.out.println("got a single line " + data);
234                                        if ( data.size() < 1){
235                                                // this can happen if empty lines at end of file
236                                                lineData.clear();
237                                                continue;
238                                        }
239                                        String key = data.get(0);
240                                        int pos = key.indexOf(".");
241                                        if ( pos < 0 ) {
242                                                // looks like a chem_comp file
243                                                // line should start with data, otherwise something is wrong!
244                                                if (! line.startsWith("data_")){
245                                                        System.err.println("this does not look like a valid MMcif file! The first line should be data_1XYZ, but is " + line);
246                                                        triggerDocumentEnd();
247                                                        return;
248                                                }
249                                                // ignore the first line...
250                                                category=null;
251                                                lineData.clear();
252                                                continue;
253                                        }
254                                        category = key.substring(0,pos);
255                                        String value = data.get(1);
256                                        loopFields.add(key.substring(pos+1,key.length()));
257                                        lineData.add(value);
258
259
260                                }
261                        }
262                }
263
264                if (struct != null){
265                        triggerStructData(struct);
266                }
267
268                triggerDocumentEnd();
269
270        }

272        private List<String> processSingleLine(String line){
273                //System.out.println("SS processSingleLine " + line);
274                List<String> data = new ArrayList<String>();
275
276                if ( line.trim().length() == 1){
277                        if ( line.startsWith(STRING_LIMIT))
278                                return data;
279                }
280                boolean inString = false;
281                boolean inS1     = false;
282                boolean inS2     = false;
283                String word      = "";
284
285                //System.out.println(line);
286                for (int i=0; i< line.length(); i++ ){
287                        //System.out.println(word);
288                        Character c = line.charAt(i);
289
290                        Character nextC = null;
291                        if (i < line.length() - 1)
292                                nextC = line.charAt(i+1);
293                        if  (c == ' ') {
294
295                                if ( ! inString){
296                                        if ( ! word.equals(""))
297                                                data.add(word.trim());
298                                        word = "";
299                                } else {
300                                        // we are in a string, add the space
301                                        word += c;
302                                }
303
304                        } else if (c == s1 )  {
305
306                                if ( inString){
307
308                                        boolean wordEnd = false;
309                                        if (! inS2) {
310                                                if (nextC != null){
311                                                        //System.out.println("nextC: >"+nextC+"<");
312                                                        if ( Character.isWhitespace(nextC)){
313                                                                i++;
314                                                                wordEnd = true;
315                                                        }
316                                                }
317                                        }
318
319
320                                        if ( wordEnd ) {
321
322                                                // at end of string
323                                                if ( ! word.equals(""))
324                                                        data.add(word.trim());
325                                                word     = "";
326                                                inString = false;
327                                                inS1     = false;
328                                        } else {
329                                                word += c;
330                                        }
331
332                                } else {
333                                        // the beginning of a new string
334                                        inString = true;
335                                        inS1     = true;
336                                }
337                        } else if ( c == s2 ){
338                                if ( inString){
339
340                                        boolean wordEnd = false;
341                                        if (! inS1) {
342                                                if (nextC != null){
343                                                        //System.out.println("nextC: >"+nextC+"<");
344                                                        if ( Character.isWhitespace(nextC)){
345                                                                i++;
346                                                                wordEnd = true;
347                                                        }
348                                                }
349                                        }
350
351                                        if ( wordEnd ) {
352
353                                                // at end of string
354                                                if ( ! word.equals(""))
355                                                        data.add(word.trim());
356                                                word     = "";
357                                                inString = false;
358                                                inS2     = false;
359                                        } else {
360                                                word += c;
361                                        }
362                                } else {
363                                        // the beginning of a new string
364                                        inString = true;
365                                        inS2     = true;
366                                }
367                        } else {
368                                word += c;
369                        }
370
371                }
372                if ( ! word.trim().equals(""))
373                        data.add(word);
374
375                
376                return data;
377
378        }

380        /** get the content of a cif entry
381         *
382         * @param line
383         * @param buf
384         * @return
385         */
386        private List<String> processLine(String line,
387                        BufferedReader buf,
388                        int fieldLength)
389                        throws IOException{
390
391                //System.out.println("XX processLine " + fieldLength + " " + line);
392                // go through the line and process each character
393                List<String> lineData = new ArrayList<String>();
394
395                boolean inString = false;
396
397                String bigWord = null;
398
399                while ( true ){
400
401                        if ( line.startsWith(STRING_LIMIT)){
402                                if (! inString){
403
404                                        inString = true;
405                                        if ( line.length() > 1)
406                                                bigWord = line.substring(1);
407                                        else
408                                                bigWord = "";
409
410
411                                } else {
412                                        // the end of a word
413                                        lineData.add(bigWord);
414                                        bigWord = null;
415                                        inString = false;
416
417                                }
418                        } else {
419                                if ( inString )
420                                        //TODO: make bigWord a stringbuffer...
421                                        bigWord += (line);
422                                else {
423
424                                        List<String> dat = processSingleLine(line);
425
426                                        for (String d : dat){
427                                                lineData.add(d);
428                                        }
429                                }
430                        }
431
432                        //System.out.println("in process line : " + lineData.size() + " " + fieldLength);
433
434                        if ( lineData.size() > fieldLength){
435
436                                System.err.println("wrong data length ("+lineData.size()+
437                                                ") should be ("+fieldLength+") at line " + line + " got lineData: " + lineData);
438                                return lineData;
439                        }
440
441                        if ( lineData.size() == fieldLength)
442                                return lineData;
443
444
445                        line = buf.readLine();
446                        if ( line == null)
447                                break;
448                }
449                return lineData;
450
451        }

658        @SuppressWarnings({ "rawtypes", "unchecked" })
659        private Object buildObject(String className, List<String> loopFields, List<String> lineData) {
660                Object o = null;
661                try {
662                        // build up the Entity object from the line data...
663                        Class c = Class.forName(className);
664
665                        o = c.newInstance();
666
667
668                        int pos = -1 ;
669                        for (String key: loopFields){
670                                pos++;
671
672                                String val = lineData.get(pos);
673                                //System.out.println(key + " " + val);
674                                String u = key.substring(0,1).toUpperCase();
675
676                                try {
677                                        Method m = c.getMethod("set" + u + key.substring(1,key.length()) , String.class);
678                                        m.invoke(o,val);
679                                }
680                                catch( NoSuchMethodException nex){
681
682                                        if (key.indexOf("[") > -1) {
683                                                setArray(c,o,key,val);
684
685                                        } else {
686                                                System.err.println("Trying to set field " + key + " in "+ c.getName() +", but not found! (value:" + val + ")");
687                                                //System.err.println(lineData);
688                                        }
689                                }
690                        }
691                } catch (InstantiationException eix){
692                        eix.printStackTrace();
693                } catch (InvocationTargetException etx){
694                        etx.printStackTrace();
695                } catch (IllegalAccessException eax){
696                        eax.printStackTrace();
697                } catch (ClassNotFoundException ex){
698                        ex.printStackTrace();
699                }
700                return o;
701        }
