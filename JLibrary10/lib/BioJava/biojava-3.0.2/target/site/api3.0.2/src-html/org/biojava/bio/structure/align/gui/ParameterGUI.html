036   @SuppressWarnings("rawtypes")
037public ParameterGUI(StructureAlignment alignment){
038
039      ConfigStrucAligParams params = alignment.getParameters();
040
041      if ( params == null)
042         return;
043      this.params = params;
044
045      String method = alignment.getAlgorithmName();
046      this.setTitle("Parameters for " + method);
047
048
049      List<String> names = params.getUserConfigParameterNames();
050      List<String> keys  = params.getUserConfigParameters();
051      List<Class> types  = params.getUserConfigTypes();
052
053      List<String> helps = params.getUserConfigHelp();
054      
055      // quick check for bugs in params
056      assert(names.size() == keys.size());
057      assert(names.size() == types.size());
058      assert(names.size() == helps.size());
059      
060      textFields = new ArrayList<Component>();
061      Box vBox = Box.createVerticalBox();
062
063      for (int i = 0 ; i < keys.size(); i++){
064         Class type = types.get(i);
065
066         Box hBox = Box.createHorizontalBox();
067         String name = names.get(i);
068         JLabel label = new JLabel(name);
069         String help = helps.get(i);
070         label.setToolTipText(help);
071         String key = keys.get(i);
072         Object value = getValue(key);
073
074         String data = value.toString();
075         Component field;
076         if ( key.equals(CeParameters.SCORING_STRATEGY) ){
077            String[] values = new String[]{"CA only","Sidechain orientation","Angle between sidechains", "CA distance+Angle between sidechains"};
078            JComboBox jcbox = new JComboBox(values);
079            Integer val = (Integer)value;
080            jcbox.setSelectedIndex(val);
081            field = jcbox;
082         } else if ( type == Boolean.class){
083
084            String[] values = new String[]{"true","false"};
085            JComboBox jcbox = new JComboBox(values);
086            if ( data.equalsIgnoreCase("false"))
087               jcbox.setSelectedIndex(1);
088            else 
089               jcbox.setSelectedIndex(0);
090            
091            field = jcbox;
092            
093            //field.setToolTipText(help);
094
095         } else {
096            JTextField tfield = new JTextField(10);
097
098            if ( type == String[].class) {
099               String stuff = "";
100               for ( String da : (String[]) value){
101                  stuff += da + " ";
102               }
103               data = stuff;
104
105
106            }
107            tfield.setText(data);
108            tfield.setToolTipText(help);
109            field = tfield;
110         }
111
112         hBox.add(label);
113         hBox.add(Box.createGlue());
114         hBox.add(field);
115
116         vBox.add(hBox);
117
118         textFields.add(field);
119
120      }
121
122
123      JButton abort = new JButton("Cancel");
124      abort.addActionListener(new ActionListener(){
125         public void actionPerformed(ActionEvent event) {
126            destroy();
127            dispose();           }
128      });
129
130      JButton defaultB = new JButton("Default");
131      defaultB.addActionListener(new ActionListener(){
132         public void actionPerformed(ActionEvent event) {
133            setDefault();
134         }
135      });
136
137      JButton close = new JButton("Apply");
138
139      close.addActionListener(new ActionListener(){
140         public void actionPerformed(ActionEvent event) {
141
142            storeParameters();
143
144            destroy();
145            dispose();           }
146      });
147
148      Box hBox = Box.createHorizontalBox();
149      hBox.add(abort);
150      hBox.add(Box.createGlue());
151      hBox.add(defaultB);
152      hBox.add(Box.createGlue());
153      hBox.add(close);
154
155      vBox.add(hBox);
156      this.getContentPane().add(vBox);
157      this.pack();
158      this.setVisible(true);
159
160
161   }

163   @SuppressWarnings({  "rawtypes" })
164   protected void setDefault() {
165      params.reset();
166
167      List<String> keys  = params.getUserConfigParameters();
168      List<Class> types  = params.getUserConfigTypes();
169      //List<String> names = params.getUserConfigParameterNames();
170      for (int i = 0 ; i < keys.size(); i++){
171
172         Class type = types.get(i);
173         Object data = getValue(keys.get(i));
174         String name = keys.get(i);
175         if ( name.equals(CeParameters.SCORING_STRATEGY)){
176            JComboBox field = (JComboBox)  textFields.get(i);
177            field.setSelectedIndex((Integer)data);
178            field.updateUI();
179         }  else if ( type == Boolean.class){
180            JComboBox field = (JComboBox)  textFields.get(i);
181            if ( data.toString().equalsIgnoreCase("false"))
182               field.setSelectedIndex(1);
183            else 
184               field.setSelectedIndex(0);   
185            field.updateUI();
186
187         } else {
188            JTextField field = (JTextField)textFields.get(i);
189            if ( type.isArray()){
190               String stuff = "";
191               for ( String da : (String[]) data){
192                  stuff += da + " ";
193               }
194               
195               field.setText(stuff);
196            } else {
197               
198               field.setText(data.toString()); 
199            }
200            field.updateUI();
201         }
202
203
204      }
205      this.repaint();
206
207   }

212      params = null;
213   }

215   @SuppressWarnings("rawtypes")
216   protected void storeParameters() {
217      //List<String> names = params.getUserConfigParameterNames();
218      List<String> keys = params.getUserConfigParameters();
219      List<Class> types = params.getUserConfigTypes();
220
221      for (int i = 0 ; i < keys.size(); i++){
222         Class type = types.get(i);
223         String key  = keys.get(i);
224        // String name = keys.get(i);
225         String value = null;
226         System.out.println(key);
227         if ( key.equals(CeParameters.SCORING_STRATEGY)){
228            JComboBox field = (JComboBox)  textFields.get(i);
229            Integer sel = field.getSelectedIndex();
230            value = sel.toString();
231         }
232         else if ( type == Boolean.class){
233            JComboBox field = (JComboBox)  textFields.get(i);
234            int sel = field.getSelectedIndex();
235            Boolean flag = true;
236            if ( sel == 1 )
237               flag = false;
238            value = flag.toString(); 
239         } else {
240            JTextField field = (JTextField)textFields.get(i);
241            value = field.getText();
242         }
243
244         setValue(key, type, value);
245      }
246
247      System.out.println("new parameters: " + params.toString());
248
249   }

251   @SuppressWarnings({ "unchecked", "rawtypes" })
252   private void setValue(String name, Class type, String value) {
253      try {
254         String methodName = "set" + name;
255
256         Class paramC = params.getClass();
257
258         Method m =paramC.getMethod(methodName,type);
259
260
261         Object data = null;
262
263         if ( type == Integer.class){
264            data = Integer.parseInt(value);
265         } else if ( type == Double.class){
266            data = Double.parseDouble(value);
267         } else if ( type == Float.class) {
268            data = Float.parseFloat(value);
269         } else if ( type == Boolean.class) {
270            data = Boolean.parseBoolean(value);
271         } else if ( type == Short.class) {
272            data = Short.parseShort(value);
273         } else if ( type == String[].class) {         
274            data = value.split(" ");
275         }
276
277         if (data == null){
278            System.err.println("Could not set value " + value + " for field " + name);
279            return;
280         }
281         m.invoke(params, data);
282
283
284      } catch (Exception e){
285         e.printStackTrace();
286
287      }
288
289   }
290
291   @SuppressWarnings({ "unchecked", "rawtypes" })
292   private Object  getValue(String name){
293
294      try {
295         String methodName = "get" + name;
296
297         Class paramC = params.getClass();
298
299         Method m =paramC.getMethod(methodName,(Class[])null);
300
301         Object value = m.invoke(params);
302
303         return value;
304      } catch (Exception e){
305         e.printStackTrace();
306         return null;
307      }
