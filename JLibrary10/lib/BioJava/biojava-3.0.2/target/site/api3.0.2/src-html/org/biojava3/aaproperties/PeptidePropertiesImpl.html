031        /**
032         * @return the molecular weight of water
033         */
034        private double getWaterMoleculeWeight(){
035                final double hydrogenMW = 1.0079;
036                final double hydroxideMW = 17.0073;
037                //H     1.0079  OH      17.0073
038                return hydrogenMW + hydroxideMW;
039        }

041        private char[] getSequence(String sequence, boolean ignoreCase){
042                if(ignoreCase){
043                        return sequence.toUpperCase().toCharArray();
044                }else{
045                        return sequence.toCharArray();
046                }
047        }

049        @Override
050        public double getMolecularWeight(ProteinSequence sequence) {
051                double value = 0.0;
052                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
053                char[] seq = getSequence(sequence.toString(), true);//ignore case
054                for(char aa:seq){
055                        AminoAcidCompound c = aaSet.getCompoundForString(aa + "");
056                        if(Constraints.aa2MolecularWeight.containsKey(c)){
057                                value += Constraints.aa2MolecularWeight.get(c);
058                        }
059                }
060                if(value == 0)
061                        return value;
062                else
063                        return value + getWaterMoleculeWeight();
064        }

066        @Override
067        public double getMolecularWeight(ProteinSequence sequence, File aminoAcidCompositionFile) throws JAXBException, FileNotFoundException {
068                File elementMassFile = new File("./src/main/resources/ElementMass.xml");
069                if(elementMassFile.exists() == false){

072                }
073                return getMolecularWeightBasedOnXML(sequence, obtainAminoAcidCompositionTable(elementMassFile, aminoAcidCompositionFile));
074        }

082        @Override
083        public double getMolecularWeightBasedOnXML(ProteinSequence sequence, AminoAcidCompositionTable aminoAcidCompositionTable){
084                double value = 0.0;
085                char[] seq = sequence.toString().toCharArray();
086                for(char aa:seq){
087                        Double weight = aminoAcidCompositionTable.getMolecularWeight(aa);
088                        if(weight != null){
089                                value += weight;
090                        }
091                }
092                if(value == 0.0)
093                        return value;
094                else
095                        return value + getWaterMoleculeWeight();
096        }

109        @Override
110        public AminoAcidCompositionTable obtainAminoAcidCompositionTable(File elementMassFile, File aminoAcidCompositionFile) 
111                throws JAXBException, FileNotFoundException{
112                //Parse elementMassFile
113                ElementTable iTable = new ElementTable();
114                // Get a JAXB Context for the object we created above
115                JAXBContext jc = JAXBContext.newInstance(iTable.getClass());
116                Unmarshaller u = jc.createUnmarshaller();
117                u.setEventHandler(new MyValidationEventHandler()); 
118                iTable = (ElementTable)u.unmarshal(new FileInputStream(elementMassFile));
119                iTable.populateMaps();
120                
121                //Parse aminoAcidCompositionFile
122                AminoAcidCompositionTable aTable = new AminoAcidCompositionTable();
123                // Get a JAXB Context for the object we created above
124                JAXBContext jc2 = JAXBContext.newInstance(aTable.getClass());
125                Unmarshaller u2 = jc2.createUnmarshaller();
126                u2.setEventHandler(new MyValidationEventHandler()); 
127                aTable = (AminoAcidCompositionTable)u2.unmarshal(new FileInputStream(aminoAcidCompositionFile));
128                aTable.computeMolecularWeight(iTable);
129                return aTable;
130        }

132        @Override
133        public double getExtinctionCoefficient(ProteinSequence sequence, boolean assumeCysReduced) {
134                //Tyr => Y
135                //Trp => W
136                //Cys => C
137                //E(Prot) = Numb(Tyr)*Ext(Tyr) + Numb(Trp)*Ext(Trp) + Numb(Cystine)*Ext(Cystine)
138                //where (for proteins in water measured at 280 nm): Ext(Tyr) = 1490, Ext(Trp) = 5500, Ext(Cystine) = 125;
139                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
140                Map<AminoAcidCompound, Integer> extinctAA2Count = this.getExtinctAACount(sequence);
141                
142                double eProt;
143                if(assumeCysReduced == false){
144                        eProt = extinctAA2Count.get(aaSet.getCompoundForString("Y")) * 
145                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("Y")) + 
146                                extinctAA2Count.get(aaSet.getCompoundForString("W")) * 
147                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("W")) +
148                                extinctAA2Count.get(aaSet.getCompoundForString("C")) * 
149                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("C"));
150                }else
151                        eProt = extinctAA2Count.get(aaSet.getCompoundForString("Y")) * 
152                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("Y")) + 
153                                extinctAA2Count.get(aaSet.getCompoundForString("W")) *
154                                Constraints.aa2ExtinctionCoefficient.get(aaSet.getCompoundForString("W"));
155                
156                return eProt;
157        }

159        @Override
160        public double getAbsorbance(ProteinSequence sequence, boolean assumeCysReduced){
161                //Absorb(Prot) = E(Prot) / Molecular_weight
162                double mw = this.getMolecularWeight(sequence);
163                double eProt = this.getExtinctionCoefficient(sequence, assumeCysReduced);
164                return eProt / mw;
165        }

167        private Map<AminoAcidCompound, Integer> getExtinctAACount(ProteinSequence sequence){
168                //Cys => C, Tyr => Y, Trp => W
169                int numW = 0;
170                int smallW = 0;
171                double numC = 0;
172                double smallC = 0; 
173                int numY = 0;
174                int smallY = 0;
175                for(char aa:sequence.getSequenceAsString().toCharArray()){
176                        switch(aa){
177                        case 'W': numW++; break;
178                        case 'w': smallW++; break;
179                        case 'C': numC += 0.5; break;
180                        case 'c': smallC += 0.5; break;
181                        case 'Y': numY++; break;
182                        case 'y': smallY++; break;
183                        }
184                }
185                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
186                Map<AminoAcidCompound, Integer> extinctAA2Count = new HashMap<AminoAcidCompound, Integer>();
187                //Ignore Case is always true
188                extinctAA2Count.put(aaSet.getCompoundForString("W"), numW + smallW);
189                extinctAA2Count.put(aaSet.getCompoundForString("C"), (int) (numC + smallC));
190                extinctAA2Count.put(aaSet.getCompoundForString("Y"), numY + smallY);
191                return extinctAA2Count;
192        }

194        @Override
195        public double getInstabilityIndex(ProteinSequence sequence) {
196                double sum = 0.0;
197                String s = sequence.getSequenceAsString().toUpperCase();
198                for(int i = 0; i < sequence.getLength() - 1; i++){
199                        String dipeptide = s.substring(i, i+2);
200                        if(Constraints.diAA2Instability.containsKey(dipeptide)){
201                                sum += Constraints.diAA2Instability.get(dipeptide);
202                        }
203                }
204                return sum * 10 / (s.length() - Utils.getNumberOfInvalidChar(s, null, true));
205        }

207        @Override
208        public double getApliphaticIndex(ProteinSequence sequence) {
209//              Aliphatic index = X(Ala) + a * X(Val) + b * ( X(Ile) + X(Leu) )  
210//              where X(Ala), X(Val), X(Ile), and X(Leu) are mole percent (100 X mole fraction) 
211//              of alanine, valine, isoleucine, and leucine. 
212//              The coefficients a and b are the relative volume of valine side chain (a = 2.9) 
213//              and of Leu/Ile side chains (b = 3.9) to the side chain of alanine. 
214//              Ala => A, Val => V, Ile => I, Leu => L
215                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
216                Map<AminoAcidCompound, Double> aa2Composition = getAAComposition(sequence);
217                final double a = 2.9;
218                final double b = 3.9;
219                double xAla = aa2Composition.get(aaSet.getCompoundForString("A"));
220                double xVal = aa2Composition.get(aaSet.getCompoundForString("V"));
221                double xIle = aa2Composition.get(aaSet.getCompoundForString("I"));
222                double xLeu = aa2Composition.get(aaSet.getCompoundForString("L"));
223                return (xAla + (a * xVal) + (b * (xIle + xLeu))) * 100;
224        }

226        @Override
227        public double getAvgHydropathy(ProteinSequence sequence) {
228                int validLength = 0;
229                double total = 0.0;
230                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
231                char[] seq = this.getSequence(sequence.toString(), true);
232                for(char aa:seq){
233                        AminoAcidCompound c = aaSet.getCompoundForString(aa + "");
234                        if(Constraints.aa2Hydrophathicity.containsKey(c)){
235                                total += Constraints.aa2Hydrophathicity.get(c);
236                                validLength++;
237                        }
238                }
239                return total / validLength;
240        }

242        @Override
243        public double getIsoelectricPoint(ProteinSequence sequence) {
244                double currentPH = 7.0;
245                double changeSize = 7.0;
246                Map<AminoAcidCompound, Integer> chargedAA2Count = this.getChargedAACount(sequence);
247                double margin;
248                final double difference = 0.0000001;
249                while(true){
250                        margin = this.getNetCharge(chargedAA2Count, currentPH);
251                        //Within allowed difference
252                        if(margin <= difference && margin >= -difference) break;
253                        changeSize /= 2.0;
254                        if(margin > 0){
255                                currentPH += changeSize;
256                        }else{
257                                currentPH -= changeSize;
258                        }
259                }
260                return currentPH;
261        }

263        @Override
264        public double getNetCharge(ProteinSequence sequence) {
265                Map<AminoAcidCompound, Integer> chargedAA2Count = this.getChargedAACount(sequence);
266                return getNetCharge(chargedAA2Count, 7.0);
267        }

269        private double getNetCharge(Map<AminoAcidCompound, Integer> chargedAA2Count, double ph){
270                //Lys => K, Arg => R, His => H
271                //Asp => D, Glu => E, Cys => C, Tyr => Y
272                //(NH2-)        9.69    (-COOH) 2.34
273                final double pkaOfNH2 = 9.69;
274                final double pkaOfCOOH = 2.34;
275                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
276                double nTerminalCharge = this.getPosCharge(pkaOfNH2, ph);
277                double kCharge = chargedAA2Count.get(aaSet.getCompoundForString("K")) * this.getPosCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("K")), ph);
278                double rCharge = chargedAA2Count.get(aaSet.getCompoundForString("R")) * this.getPosCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("R")), ph);
279                double hCharge = chargedAA2Count.get(aaSet.getCompoundForString("H")) * this.getPosCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("H")), ph);
280                double dCharge = chargedAA2Count.get(aaSet.getCompoundForString("D")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("D")), ph);
281                double eCharge = chargedAA2Count.get(aaSet.getCompoundForString("E")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("E")), ph);
282                double cCharge = chargedAA2Count.get(aaSet.getCompoundForString("C")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("C")), ph);
283                double yCharge = chargedAA2Count.get(aaSet.getCompoundForString("Y")) * this.getNegCharge(Constraints.aa2PKa.get(aaSet.getCompoundForString("Y")), ph);
284                double cTerminalCharge = this.getNegCharge(pkaOfCOOH, ph);
285                if((kCharge + rCharge + hCharge) == 0.0 && (dCharge + eCharge + cCharge + yCharge) == 0.0){
286                        return 0.0;
287                }
288                return (nTerminalCharge + kCharge + rCharge + hCharge) - (dCharge + eCharge + cCharge + yCharge + cTerminalCharge);
289        }
290        
291        private double getPosCharge(double pka, double ph){
292                return Math.pow(10, pka) / (Math.pow(10, pka) + Math.pow(10, ph));
293        }
294        
295        private double getNegCharge(double pka, double ph){
296                return Math.pow(10, ph) / (Math.pow(10, pka) + Math.pow(10, ph));
297        }

299        private Map<AminoAcidCompound, Integer> getChargedAACount(ProteinSequence sequence){
300                //Lys => K, Arg => R, His => H
301                //Asp => D, Glu => E, Cys => C, Tyr => Y
302                int numK = 0;
303                int numR = 0;
304                int numH = 0;
305                int numD = 0;
306                int numE = 0;
307                int numC = 0;
308                int numY = 0;
309                char[] seq = this.getSequence(sequence.getSequenceAsString(), true);
310                for(char aa:seq){
311                        switch(aa){
312                        case 'K': numK++; break;
313                        case 'R': numR++; break;
314                        case 'H': numH++; break;
315                        case 'D': numD++; break;
316                        case 'E': numE++; break;
317                        case 'C': numC++; break;
318                        case 'Y': numY++; break;
319                        }
320                }
321                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
322                Map<AminoAcidCompound, Integer> chargedAA2Count = new HashMap<AminoAcidCompound, Integer>();
323                chargedAA2Count.put(aaSet.getCompoundForString("K"), numK);
324                chargedAA2Count.put(aaSet.getCompoundForString("R"), numR);
325                chargedAA2Count.put(aaSet.getCompoundForString("H"), numH);
326                chargedAA2Count.put(aaSet.getCompoundForString("D"), numD);
327                chargedAA2Count.put(aaSet.getCompoundForString("E"), numE);
328                chargedAA2Count.put(aaSet.getCompoundForString("C"), numC);
329                chargedAA2Count.put(aaSet.getCompoundForString("Y"), numY);
330                return chargedAA2Count;
331        }

334        public double getEnrichment(ProteinSequence sequence, AminoAcidCompound aminoAcidCode) {
335                double counter = 0.0;
336                char[] seq = this.getSequence(sequence.getSequenceAsString(), true);
337                for(char aa:seq){
338                        if(aminoAcidCode.getShortName().equals(aa + "")){
339                                counter++;
340                        }
341                }
342                return counter/sequence.getLength();
343        }

345        @Override
346        public Map<AminoAcidCompound, Double> getAAComposition(ProteinSequence sequence) {
347                int validLength = 0;
348                Map<AminoAcidCompound, Double> aa2Composition = new HashMap<AminoAcidCompound, Double>();
349                AminoAcidCompoundSet aaSet = new AminoAcidCompoundSet();
350                for(AminoAcidCompound aa:aaSet.getAllCompounds()){
351                        aa2Composition.put(aa, 0.0);
352                }
353                char[] seq = this.getSequence(sequence.toString(), true);
354                for(char aa:seq){
355                        if(PeptideProperties.standardAASet.contains(aa)){
356                                AminoAcidCompound compound = aaSet.getCompoundForString(aa + "");
357                                aa2Composition.put(compound, aa2Composition.get(compound) + 1.0);
358                                validLength++;
359                        }
360                }
361                if(validLength > 0){
362                        for(AminoAcidCompound aa:aaSet.getAllCompounds()){
363                                aa2Composition.put(aa, aa2Composition.get(aa) / validLength);
364                        }
365                }else{
366                        for(AminoAcidCompound aa:aaSet.getAllCompounds()){
367                                aa2Composition.put(aa, 0.0);
368                        }
369                }
370                return aa2Composition;
371        }
