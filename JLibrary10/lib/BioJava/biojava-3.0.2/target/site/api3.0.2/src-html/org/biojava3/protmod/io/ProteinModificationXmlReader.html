061 * @author Jianjiong Gao
062 * @since 3.0
063 */
064public final class ProteinModificationXmlReader {
065        /**
066         * This is a utility class and thus cannot be instantialized.
067         */
068        private ProteinModificationXmlReader() {}
069        
070        /**
071         * Read protein modifications from XML file and register them.
072         * @param isXml {@link InputStream} of the XML file.
073         * @throws IOException if failed to read the XML file.
074         * @throws ParserConfigurationException if parse errors occur.
075         * @throws SAXException the {@link DocumentBuilder} cannot be created.
076         */
077        public static void registerProteinModificationFromXml(InputStream isXml)
078                        throws IOException, ParserConfigurationException, SAXException {
079                if (isXml==null) {
080                        throw new IllegalArgumentException("Null argument.");
081                }
082                
083                DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
084                DocumentBuilder builder = factory.newDocumentBuilder();
085                Document doc = builder.parse(isXml);
086                
087                NodeList modNodes = doc.getElementsByTagName("Entry");          
088                int modSize = modNodes.getLength();
089                List<Node> nodes;
090                for (int iMod=0; iMod<modSize; iMod++) {
091                        Node modNode = modNodes.item(iMod);
092                        Map<String,List<Node>> infoNodes = getChildNodes(modNode);
093                        
094                        // ID
095                        nodes = infoNodes.get("Id");
096                        if (nodes==null || nodes.size()!=1) {
097                                throw new RuntimeException("Each modification must have exact " +
098                                                "one <Id> field.");
099                        }
100                        String id = nodes.get(0).getTextContent();
101                        
102                        // modification category
103                        nodes = infoNodes.get("Category");
104                        if (nodes==null || nodes.size()!=1) {
105                                throw new RuntimeException("Each modification must have exact " +
106                                                "one <Category> field. See Modification "+id+".");
107                        }
108                        ModificationCategory cat = ModificationCategory.getByLabel(
109                                        nodes.get(0).getTextContent());
110                        if (cat==null) {
111                                throw new RuntimeException(nodes.get(0).getTextContent()+
112                                        " is not defined as an modification category." +
113                                        " See Modification "+id+".");
114                        }
115                        
116                        // occurrence type
117                        nodes = infoNodes.get("Occurrence");
118                        if (nodes==null || nodes.size()!=1) {
119                                throw new RuntimeException("Each modification must have exact " +
120                                                "one <Occurrence> field. See Modification "+id+".");
121                        }
122                        ModificationOccurrenceType occType = ModificationOccurrenceType
123                                .getByLabel(nodes.get(0).getTextContent());
124                        if (occType==null) {
125                                throw new RuntimeException(nodes.get(0).getTextContent()+
126                                        " is not defined as an modification occurence type." +
127                                        " See Modification "+id+".");
128                        }
129                        
130                        // condition
131                        ModificationCondition condition = null;
132                        {
133                                nodes = infoNodes.get("Condition");
134                                if (nodes==null || nodes.size()!=1) {
135                                        throw new RuntimeException("Each modification must have exact " +
136                                                        "one <Condition> field. See Modification "+id+".");
137                                }
138                                
139                                Node compsNode = nodes.get(0);
140                                
141                                // keep track of the labels of component indices
142                                Map<String,Integer> mapLabelComp = new HashMap<String,Integer>();
143
144                                Map<String,List<Node>> compInfoNodes = getChildNodes(compsNode);
145                                
146                                // components
147                                List<Node> compNodes = compInfoNodes.get("Component");
148                                int sizeComp = compNodes.size();
149                                List<Component> comps = new ArrayList<Component>(sizeComp);
150                                for (int iComp=0; iComp<sizeComp; iComp++) {
151                                        Node compNode = compNodes.get(iComp);
152                                        // comp label
153                                        NamedNodeMap compNodeAttrs = compNode.getAttributes();
154                                        Node labelNode = compNodeAttrs.getNamedItem("component");
155                                        if (labelNode==null) {
156                                                throw new RuntimeException("Each component must have a label." +
157                                                                " See Modification "+id+".");
158                                        }
159                                        String label = labelNode.getTextContent();
160                                        
161                                        if (mapLabelComp.containsKey(label)) {
162                                                throw new RuntimeException("Each component must have a unique label." +
163                                                                " See Modification "+id+".");
164                                        }
165                                        
166                                        // comp PDBCC ID
167                                        Set<String> compIds = new HashSet<String>();
168                                        List<Node> compIdNodes = getChildNodes(compNode).get("Id");
169                                        if (compIdNodes!=null) {
170                                                for (Node compIdNode : compIdNodes) {
171                                                        NamedNodeMap compIdNodeAttr = compIdNode.getAttributes();
172                                                        Node compIdSource = compIdNodeAttr.getNamedItem("source");
173                                                        if (compIdSource!=null && compIdSource.getTextContent().equals("PDBCC")) {
174                                                                String strComps = compIdNode.getTextContent();
175                                                                if (strComps.isEmpty()) {
176                                                                        throw new RuntimeException("Empty component." +
177                                                                                        " See Modification "+id+".");
178                                                                }
179                                                                compIds.addAll(Arrays.asList(strComps.split(",")));
180                                                        }
181                                                }
182                                        }
183                                        
184                                        if (compIds.isEmpty()) {
185                                                throw new RuntimeException("Each component must have a PDBCC ID." +
186                                                                " See Modification "+id+".");
187                                        }
188                                        
189                                        // terminal
190                                        boolean nTerminal = false;
191                                        boolean cTerminal = false;
192                                        List<Node> compTermNode = getChildNodes(compNode).get("Terminal");
193                                        if (compTermNode!=null) {
194                                                if (compTermNode.size()!=1) {
195                                                        throw new RuntimeException("Only one <Terminal> condition is allowed for " +
196                                                                        "each component. See Modification "+id+".");
197                                                }
198                                                String nc = compTermNode.get(0).getTextContent();
199                                                if (nc.equals("N")) {
200                                                        nTerminal = true;
201                                                } else if (nc.equals("C")) {
202                                                        cTerminal = true;
203                                                } else {
204                                                        throw new RuntimeException("Only N or C is allowed for <Terminal>." +
205                                                                        " See Modification "+id+".");
206                                                }
207                                        }
208
209                                        // register
210                                        Component comp = Component.of(compIds, nTerminal, cTerminal);
211                                        comps.add(comp);                                                
212                                        mapLabelComp.put(label, comps.size()-1);
213                                }
214                                
215                                // bonds
216                                List<Node> bondNodes = compInfoNodes.get("Bond");
217                                List<ModificationLinkage> linkages = null;
218                                if (bondNodes!=null) {
219                                        int sizeBonds = bondNodes.size();
220                                        linkages = new ArrayList<ModificationLinkage>(sizeBonds);
221                                        for (int iBond=0; iBond<sizeBonds; iBond++) {
222                                                Node bondNode = bondNodes.get(iBond);
223                                                Map<String,List<Node>> bondChildNodes = getChildNodes(bondNode);
224                                                if (bondChildNodes==null) {
225                                                        throw new RuntimeException("Each bond must contain two atoms" +
226                                                                        " See Modification "+id+".");
227                                                }
228                                                
229                                                List<Node> atomNodes = bondChildNodes.get("Atom");
230                                                if (atomNodes==null || atomNodes.size()!=2) {
231                                                        throw new RuntimeException("Each bond must contain two atoms" +
232                                                                        " See Modification "+id+".");
233                                                }
234                                                
235                                                // atom 1
236                                                NamedNodeMap atomNodeAttrs = atomNodes.get(0).getAttributes();
237                                                Node compNode = atomNodeAttrs.getNamedItem("component");
238                                                if (compNode==null) {
239                                                        throw new RuntimeException("Each atom must on a component." +
240                                                                        " See Modification "+id+".");
241                                                }
242                                                String labelComp1 = compNode.getTextContent();
243                                                int iComp1 = mapLabelComp.get(labelComp1);
244                                                
245                                                Node labelNode = atomNodeAttrs.getNamedItem("atom");
246                                                String labelAtom1 = labelNode==null?null:labelNode.getTextContent();
247                                                
248                                                String atom1 = atomNodes.get(0).getTextContent();
249                                                if (atom1.isEmpty()) {
250                                                        throw new RuntimeException("Each atom must have a name. Please use wildcard * if unknown." +
251                                                                        " See Modification "+id+".");
252                                                }
253                                                List<String> potentialAtoms1 = Arrays.asList(atom1.split(","));
254                                                
255                                                // atom 2
256                                                atomNodeAttrs = atomNodes.get(1).getAttributes();
257                                                compNode = atomNodeAttrs.getNamedItem("component");
258                                                if (compNode==null) {
259                                                        throw new RuntimeException("Each atom must on a component." +
260                                                                        " See Modification "+id+".");
261                                                }
262                                                String labelComp2 = compNode.getTextContent();
263                                                int iComp2 = mapLabelComp.get(labelComp2);
264
265                                                labelNode = atomNodeAttrs.getNamedItem("atom");
266                                                String labelAtom2 = labelNode==null?null:labelNode.getTextContent();
267                                                
268                                                String atom2 = atomNodes.get(1).getTextContent();
269                                                if (atom2.isEmpty()) {
270                                                        throw new RuntimeException("Each atom must have a name. Please use wildcard * if unknown." +
271                                                                        " See Modification "+id+".");
272                                                }
273                                                List<String> potentialAtoms2 = Arrays.asList(atom2.split(","));
274                                                
275                                                // add linkage
276                                                ModificationLinkage linkage = new ModificationLinkage(comps,
277                                                                iComp1, potentialAtoms1, labelAtom1,
278                                                                iComp2, potentialAtoms2, labelAtom2);
279                                                linkages.add(linkage);
280                                        }
281                                }
282                                
283                                condition = new ModificationConditionImpl(comps, linkages);
284                        } // end of condition   
285                        
286                        ProteinModificationImpl.Builder modBuilder = 
287                                new ProteinModificationImpl.Builder(id, cat, occType, condition);
288                        
289                        // description
290                        nodes = infoNodes.get("Description");
291                        if (nodes!=null && !nodes.isEmpty()) {
292                                modBuilder.setDescription(nodes.get(0).getTextContent());
293                        }
294                        
295                        // cross references
296                        nodes = infoNodes.get("CrossReference");
297                        if (nodes!=null) {
298                                for (Node node:nodes) {
299                                        Map<String,List<Node>> xrefInfoNodes = getChildNodes(node);
300                                        
301                                        // source
302                                        List<Node> xrefNode = xrefInfoNodes.get("Source");
303                                        if (xrefNode==null || xrefNode.size()!=1) {
304                                                throw new RuntimeException("Error in XML file: " +
305                                                        "a cross reference must contain exactly one <Source> field." +
306                                                        " See Modification "+id+".");
307                                        }
308                                        String xrefDb = xrefNode.get(0).getTextContent();
309                                        
310                                        // id
311                                        xrefNode = xrefInfoNodes.get("Id");
312                                        if (xrefNode==null || xrefNode.size()!=1) {
313                                                throw new RuntimeException("Error in XML file: " +
314                                                        "a cross reference must contain exactly one <Id> field." +
315                                                        " See Modification "+id+".");
316                                        }
317                                        String xrefId = xrefNode.get(0).getTextContent();
318                                        
319                                        // name
320                                        String xrefName = null;
321                                        xrefNode = xrefInfoNodes.get("Name");
322                                        if (xrefNode!=null && !xrefNode.isEmpty()) {
323                                                xrefName = xrefNode.get(0).getTextContent();
324                                        }
325                                        
326                                        if (xrefDb.equals("PDBCC")) {
327                                                modBuilder.setPdbccId(xrefId).setPdbccName(xrefName);
328                                        } else if (xrefDb.equals("RESID")) {
329                                                modBuilder.setResidId(xrefId).setResidName(xrefName);
330                                        } else if (xrefDb.equals("PSI-MOD")) {
331                                                modBuilder.setPsimodId(xrefId).setPsimodName(xrefName);
332                                        }
333                                }
334                        } // end of cross references
335                        
336                        // formula
337                        nodes = infoNodes.get("Formula");
338                        if (nodes!=null && !nodes.isEmpty()) {
339                                modBuilder.setFormula(nodes.get(0).getTextContent());
340                        }
341                        
342                        // keywords
343                        nodes = infoNodes.get("Keyword");
344                        if (nodes!=null && !nodes.isEmpty()) {
345                                for (Node node : nodes) {
346                                        modBuilder.addKeyword(node.getTextContent());
347                                }
348                        }
349                        
350                        ProteinModificationRegistry.register(modBuilder.build());
351                }
352        }
353        
354        /**
355         * Utility method to group child nodes by their names.
356         * @param parent parent node.
357         * @return Map from name to child nodes.
358         */
359        private static Map<String,List<Node>> getChildNodes(Node parent) {
360                if (parent==null)
361                        return Collections.emptyMap();
362                
363                Map<String,List<Node>> children = new HashMap<String,List<Node>>();
364                
365                NodeList nodes = parent.getChildNodes();
366                int nNodes = nodes.getLength();
367                for (int i=0; i<nNodes; i++) {
368                        Node node = nodes.item(i);
369                        if (node.getNodeType()!=Node.ELEMENT_NODE)
370                                continue;
371                        
372                        String name = node.getNodeName();
373                        List<Node> namesakes = children.get(name);
374                        if (namesakes==null) {
375                                namesakes = new ArrayList<Node>();
376                                children.put(name, namesakes);
377                        }
378                        namesakes.add(node);
379                }
380                
381                return children;
382        }
