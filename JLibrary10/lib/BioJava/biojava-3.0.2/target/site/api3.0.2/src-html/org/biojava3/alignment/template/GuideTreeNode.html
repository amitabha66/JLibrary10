024package org.biojava3.alignment.template;
025
026import java.util.concurrent.Future;
027import javax.swing.tree.TreeNode;
028
029import org.biojava3.core.sequence.template.Compound;
030import org.biojava3.core.sequence.template.Sequence;
031
032/**
033 * Defines a data structure for the node in a guide tree used during progressive multiple sequence alignment.
034 * 
035 * @author Mark Chapman
036 * @param <S> each {@link Sequence} in the tree is of type S
037 * @param <C> each element of a {@link Sequence} is a {@link Compound} of type C
038 */
039public interface GuideTreeNode<S extends Sequence<C>, C extends Compound> extends TreeNode {
040
041    /**
042     * Returns the first child node of this node.  For leaf nodes (sequences), this will be null.
043     *
044     * @return the first child node of this node
045     */
046    GuideTreeNode<S, C> getChild1();
047
048    /**
049     * Returns the second child node of this node.  For leaf nodes (sequences), this will be null.
050     *
051     * @return the second child node of this node
052     */
053    GuideTreeNode<S, C> getChild2();
054
055    /**
056     * Returns the difference in height of this node and it's parent node.  A likely meaning of this distance is half
057     * the percent difference between this node and it's sibling node.
058     *
059     * @return the difference in height of this node to it's parent node
060     */
061    double getDistanceToParent();
062
063    /**
064     * Returns the name of this node.  For leaf nodes (sequences), this will likely be the accession ID.
065     *
066     * @return the name of this node
067     */
068    String getName();
069
070    /**
071     * Returns the profile stored at this node.  If the node is a leaf, the profile is that of a single sequence.  If
072     * not, this returns null until {@link #setProfile(Profile)} has been called.
073     *
074     * @return the profile stored at this node
075     */
076    Profile<S, C> getProfile();
077
078    /**
079     * Returns the profile future stored at this node, but does not force the calculation, yet.  This allows alignment
080     * tasks for the entire tree to be queued in a post-order traversal before concurrent execution.
081     *
082     * @return the profile future stored at this node
083     */
084    Future<ProfilePair<S, C>> getProfileFuture();
085
086    /**
087     * Stores the given profile.
088     *
089     * @param profile new profile stored at this node
090     */
091    void setProfile(Profile<S, C> profile);
092
093    /**
094     * Stores the given profile future.  This allows concurrent execution of alignment tasks.
095     *
096     * @param profileFuture new profile to be calculated and then stored at this node
097     */
098    void setProfileFuture(Future<ProfilePair<S, C>> profileFuture);
099
100}
