025package demo;
026
027
028import org.biojava.bio.structure.Atom;
029import org.biojava.bio.structure.Structure;
030import org.biojava.bio.structure.StructureTools;
031import org.biojava.bio.structure.align.StructureAlignment;
032import org.biojava.bio.structure.align.StructureAlignmentFactory;
033
034import org.biojava.bio.structure.align.fatcat.FatCatFlexible;
035import org.biojava.bio.structure.align.gui.DisplayAFP;
036import org.biojava.bio.structure.align.gui.StructureAlignmentDisplay;
037import org.biojava.bio.structure.align.gui.jmol.StructureAlignmentJmol;
038import org.biojava.bio.structure.align.model.AFPChain;
039import org.biojava.bio.structure.align.model.AfpChainWriter;
040import org.biojava.bio.structure.align.util.AtomCache;
041
042
043public class CookBook
044{
045
046
047   public static void main(String[] args){
048
049      String name1="1HNG.B";
050      String name2="1A64.A";
051      
052
053      try {
054
055         // for this example we are going to use the jFatCat-rigid algorithm
056         StructureAlignment algorithm = StructureAlignmentFactory.getAlgorithm(FatCatFlexible.algorithmName);
057
058         // the cache takes care of loading the structures
059         // Downloads files to a temp dir by default
060         AtomCache cache = new AtomCache();
061
062
063         //////////////////////////////
064         // no need to change anything below this line
065         // ////////////////////////////
066
067         // load the structures 
068         Structure structure1 = cache.getStructure(name1);
069         Structure structure2 = cache.getStructure(name2);
070
071         // we are only using the CA atoms in the structure for the alignment
072         Atom[] ca1 = StructureTools.getAtomCAArray(structure1);
073         Atom[] ca2 = StructureTools.getAtomCAArray(structure2);
074
075         // do the actual alignment
076         AFPChain afpChain = algorithm.align(ca1,ca2);
077
078         // just name the two molecules 
079         afpChain.setName1(name1);
080         afpChain.setName2(name2);
081
082         // print and display results:
083
084
085         // flexible original results:
086         System.out.println(afpChain.toFatcat(ca1,ca2));
087
088         // show the alignment in 3D in jmol
089         StructureAlignmentJmol jmol= StructureAlignmentDisplay.display(afpChain, ca1, ca2);
090                 
091         // set the display title for the frame
092         jmol.setTitle(algorithm.getAlgorithmName() + " : " + name1 + " vs. " + name2);
093
094         // here we open up the alignment - text panel that can interact with the 3D jmol display.
095         DisplayAFP.showAlignmentImage(afpChain, ca1,ca2,jmol);
096         
097         // we can print an XML version 
098         //System.out.println(AFPChainXMLConverter.toXML(afpChain, ca1, ca2));
099
100         // or print the same output as original FATCAT 
101         System.out.println(AfpChainWriter.toFatCat(afpChain, ca1, ca2));
102         
103   
104
105       
106
107      } catch (Exception e){
108         e.printStackTrace();
109      }
110   }
111}
