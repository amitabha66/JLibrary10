045/** a PDB installation that reads all files from one directory in the file system.
046 * and keeps several files with the information parsed from the PDB headers for
047 * easier data access.
048 * <br/>
049 * In particular these files are:
050 * <ul>
051 * <li>pdbinfo.txt - contains resolution, experimetnal technique, deposition and 
052 *     modification date, biological description of the structure</li>
053 * <li>chaininfo.txt - contains information about every chain, like the compound information from the header, the nr of amino acids in chain,
054 *     The alignment to the SEQRES residues </li>
055 * </ul>
056 *     
057 * In order to create these index files for this PDB installation, please see the PrepareIndexFile class.
058 * @see PrepareIndexFile
059 * 
060 * @author Andreas Prlic
061 * @deprecated
062 */

089    /** create a new FlatFile Installation and point it to the directory that contains all PDB files
090     * 
091     * @param filePath
092     */
093    public FlatFileInstallation(File filePath){
094
095        if (! filePath.isDirectory()){
096            throw new IllegalArgumentException("the provided path does not point to a directory!");
097        }
098
099
100        filters = new ArrayList<PDBFilter>();
101        this.filePath = filePath;
102        //default is in the same directory as filePath...
103        indexFile = new File(filePath + File.separator +DEFAULT_INDEX_FILE);
104        chainInfoFile = new File(filePath + File.separator + DEFAUL_CHAIN_FILE);
105        filtersApplied = false;
106        filteredPDBs = new ArrayList<PDBHeader>();
107        filterIterator = null;
108        reader = new PDBFileReader();
109        reader.setPath(filePath.toString());
110
111    }

151    public void applyFilters(){
152
153        filteredPDBs.clear();
154
155        try {
156
157            InputStream in = new FileInputStream(indexFile);
158            BufferedReader      indexReader = new BufferedReader(new InputStreamReader(in));
159
160            String line = null;
161            line = indexReader.readLine();
162            while ( line != null  ) {
163                if (line.startsWith("//")) {
164                    line = indexReader.readLine();
165                    continue;
166                }
167                               
168
169                PDBHeader header = getPDBHeaderFromLine(line);
170                // if no filters have been configured, add all PDBs
171                if ( filters.size() == 0){
172                    filteredPDBs.add(header);
173                } else {
174                    throw new UnsupportedOperationException("can not sort PDBHeaders yet");
175                }
176
177
178                line = indexReader.readLine();
179            }
180        }
181        catch (Exception e) {
182            e.printStackTrace();
183        }
184
185
186        filterIterator = filteredPDBs.iterator();               
187        filtersApplied = true;
188    }
189
190    private String nullCheck(String str){
191        if ( str.equals("null")){
192            str = null;
193        }
194        return str;
195    }
196
197    private PDBHeader getPDBHeaderFromLine(String line){
198        PDBHeader header = new PDBHeader();
199
200        String[] spl = line.split("\t");
201        if ( spl.length != 9){
202            System.err.println("nr tabs length does not match expected length 9 (is "+spl.length+")! " + line);
203        }
204//      pdbId\tnrCAAtoms\ttechnique\tresolution\tdepDate\tmodDate\ttitle\tclassification\ttime");
205        header.setIdCode(spl[0]);
206        
207        header.setTechnique(nullCheck(spl[2]));
208        try {
209            float resolution = Float.parseFloat(spl[3]);
210            header.setResolution(resolution);
211        } catch (NumberFormatException ex){}
212
213        try {
214            
215            Date dep = dateFormat.parse(spl[4]);
216            header.setDepDate(dep);
217            Date mod = dateFormat.parse(spl[5]);
218            header.setModDate(mod);
219            
220        } catch (ParseException e){
221            e.printStackTrace();
222        }
223        
224        
225        header.setTitle(nullCheck(spl[6]));
226        header.setClassification(nullCheck(spl[7]));
227
228        return header;
229    }

231    public List<PDBHeader> getAll() {
232
233        if ( ! filtersApplied)
234            applyFilters();
235
236        throw new UnsupportedOperationException("can not get all PDBHeaders yet");
237
238    }
239
240    public PDBHeader getPDBHeader(String pdbId){
241
242        Iterator<PDBHeader> iter = filteredPDBs.iterator();
243
244        while (iter.hasNext()){
245            PDBHeader header = iter.next();
246            String id = header.getIdCode();
247            if ( pdbId.equals(id)){
248                return header;
249            }    
250        }
251
252        return null;
253    }
254
255    public Structure getStructure(String pdbId) {
256        Structure s = null;
257        try {
258            s = reader.getStructureById(pdbId);
259        } catch (IOException e){
260            e.printStackTrace();
261        }
262        return s;
263    }
264
265    public boolean hasNext() {
266        if ( ! filtersApplied)
267            applyFilters();
268
269        return filterIterator.hasNext(); 
270    }
271
272
273    public Structure next() {
274
275
276        if ( ! filtersApplied )
277            applyFilters();
278
279        PDBHeader header = filterIterator.next();
280        String pdbId = header.getIdCode();
281
282        Structure s = null;
283        try {
284            System.out.println("flatfileinstallation : next:" + pdbId + " " + getFilePath());
285            s = reader.getStructureById(pdbId);
286        } catch (IOException e){
287            e.printStackTrace();
288        }
289        return s;
