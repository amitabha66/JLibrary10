020package org.biojava.bio.structure.align.webstart;
021
022import java.net.MalformedURLException;
023import java.net.URL;
024import java.util.logging.Level;
025import java.util.logging.Logger;
026
027
028public class BrowserOpener {
029    public static Logger logger =  Logger.getLogger("org.biojava.spice");
030   
031    /** open a URL in the browser that was used to launch SPICE
032     * 
033     * @param url URL to be opened
034     * @return true if this was successfull
035     */
036    public static boolean showDocument(URL url) 
037    {
038        if ( url != null ){
039            boolean success = JNLPProxy.showDocument(url); 
040            if ( ! success)
041                logger.info("could not open URL "+url+" in browser. check your config or browser version.");
042        return success;
043        
044        }
045        else
046            return false;
047    }
048    
049
050    /** open a URL in the browser that was used to launch SPICE
051     * 
052     * @param urlstring string represntation of URL to be opened
053     * @return true if this was successfull
054     */
055    public static boolean showDocument(String urlstring){
056        try{
057            URL url = new URL(urlstring);
058            
059            return showDocument(url);
060        } catch (MalformedURLException e){
061            logger.log(Level.WARNING,"malformed URL "+urlstring, e.getCause());
062            return false;
063        }
064    }
065
066}
