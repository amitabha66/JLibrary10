027/** This provider of chemical components can download and cache chemical component definition files from the RCSB PDB web site.
028 *  It is the default way to access these definitions.
029 *  If this provider is called he first time, it will download and install all chemical 
030 *  component definitions in a local directory. 
031 *  Once the definition files have been installed, it has quick startup time and low memory requirements. 
032 *
033 *  An alternative provider, that keeps all definitions in memory is the {@link AllChemCompProvider}. Another provider, that
034 *  does not require any network access, but only can support a limited set of chemical component definitions, is the {@link ReducedChemCompProvider}. 
035 *  
036 * 
037 * @author Andreas Prlic
038 *
039 */
040public class DownloadChemCompProvider implements ChemCompProvider {
041
042        static String path; 
043        private static final String FILE_SEPARATOR = System.getProperty("file.separator");
044        static final String NEWLINE = System.getProperty("line.separator");
045
046        public static String CHEM_COMP_CACHE_DIRECTORY = "chemcomp";
047
048        private static String serverLocation = "http://www.rcsb.org/pdb/files/ligand/";

051        static AtomicBoolean loading = new AtomicBoolean(false);

053        static final List<String> protectedIDs = new ArrayList<String> ();
054        static {
055                protectedIDs.add("CON");
056                protectedIDs.add("PRN");
057                protectedIDs.add("AUX");
058                protectedIDs.add("NUL");
059        }

062         * 
063         */
064        boolean downloadAll = false;

070        /** checks if the chemical components already have been installed into the PDB directory.
071         *  If not, will download the chemical components definitions file and split it up into small
072         *  subfiles.
073         */
074        public void checkDoFirstInstall(){
075
076                if ( ! downloadAll ) {
077                        return;
078                }
079                
080                if ( path == null)
081                        path =  System.getProperty(AbstractUserArgumentProcessor.PDB_DIR);
082                
083                String filename = path +        
084                DownloadChemCompProvider.CHEM_COMP_CACHE_DIRECTORY +
085                FILE_SEPARATOR + 
086                "components.cif.gz";
087
088                File f = new File(filename);
089
090                if ( ! f.exists()) {
091
092                        downloadAllDefinitions();
093
094                } else {
095                        // file exists.. did it get extracted?
096                        String directoryName = path +   
097                        DownloadChemCompProvider.CHEM_COMP_CACHE_DIRECTORY +
098                        FILE_SEPARATOR;
099
100                        File dir = new File(directoryName);
101
102                        FilenameFilter filter =new FilenameFilter() {
103
104                                public boolean accept(File dir, String file) {
105                                        return file.endsWith(".cif.gz");
106                                }
107                        };
108                        String[] files = dir.list(filter);
109                        if ( files.length < 500) {
110                                // not all did get unpacked
111                                split();
112                        }
113                }
114        }

116        private void split(){
117
118                System.out.println("Installing individual chem comp files ...");
119                
120                String filename = path + 
121                DownloadChemCompProvider.CHEM_COMP_CACHE_DIRECTORY + FILE_SEPARATOR +
122                "components.cif.gz";
123
124                int counter = 0;
125                InputStreamProvider prov = new InputStreamProvider();
126                try {
127                        InputStream inStream = prov.getInputStream(filename);
128
129                        BufferedReader buf = new BufferedReader (new InputStreamReader (inStream));
130
131                        String line = null;
132                        line = buf.readLine ();
133                        StringWriter writer = new StringWriter();
134
135                        String currentID = null;
136                        while (line != null){
137
138                                if ( line.startsWith("data_")) {
139                                        // a new record found!
140
141                                        if ( currentID != null) {
142                                                writeID(writer, currentID);
143                                                counter++;
144                                        }
145
146                                        currentID = line.substring(5);
147                                        writer = new StringWriter();
148                                }
149
150                                writer.append(line);
151                                writer.append(NEWLINE);
152
153                                line = buf.readLine ();
154                        }
155
156                        // write the last record...
157                        writeID(writer,currentID);
158                        counter++;
159                } catch (Exception e){
160                        e.printStackTrace();
161                }
162                System.out.println("created " + counter + " chemical component files.");
163        }

165        private void writeID(StringWriter writer, String currentID) throws IOException{
166
167                //System.out.println("writing ID " + currentID);
168
169                
170                
171                String localName = DownloadChemCompProvider.getLocalFileName(currentID);
172
173                FileOutputStream outPut = new FileOutputStream(localName);
174
175                GZIPOutputStream gzOutPut = new GZIPOutputStream(outPut);
176
177                PrintWriter pw = new PrintWriter(gzOutPut);
178
179                pw.print(writer.toString());
180                writer.close();
181                pw.flush();
182                pw.close();
183
184                outPut.close();
185
186        }

188        /** Loads the definitions for this {@link ChemComp} from a local file and instantiates a new object.
189         * 
190         * @param recordName the ID of the {@link ChemComp}
191         * @return a new {@link ChemComp} definition.
192         */
193        public  ChemComp getChemComp(String recordName) {
194
195                checkPath();
196
197                // make sure we work with upper case records            
198                recordName = recordName.toUpperCase().trim();
199
200                if ( recordName.equals("?")){
201                        return null;
202                }
203                try {
204                        if ( ! fileExists(recordName)) {
205                                // check if we should install all components                            
206                                checkDoFirstInstall();
207                        }
208                        if ( ! fileExists(recordName)) {
209                                // we previously have installed already the definitions,
210                                // just do an incrememntal update
211                                downloadChemCompRecord(recordName);
212                        }
213
214                        String filename = getLocalFileName(recordName);
215
216                        System.out.println("reading " + filename);
217                        InputStreamProvider isp = new InputStreamProvider();
218
219                        InputStream inStream = isp.getInputStream(filename);
220
221                        MMcifParser parser = new SimpleMMcifParser();
222
223                        ChemCompConsumer consumer = new ChemCompConsumer();
224
225                        // The Consumer builds up the BioJava - structure object.
226                        // you could also hook in your own and build up you own data model.
227                        parser.addMMcifConsumer(consumer);
228
229                        parser.parse(new BufferedReader(new InputStreamReader(inStream)));
230
231                        ChemicalComponentDictionary dict = consumer.getDictionary();
232
233                        ChemComp chemComp = dict.getChemComp(recordName);
234
235                        return chemComp;
236
237                } catch (Exception e) {
238
239                        e.printStackTrace();
240
241                }
242                return null;
243
244        }

246        private static void checkPath(){
247
248                if ((path == null) || (path.equals("")) || path.equals("null")) {
249
250                        String syspath = System.getProperty(AbstractUserArgumentProcessor.PDB_DIR);
251
252                        if ((syspath != null) && (! syspath.equals("")) && (! syspath.equals("null"))){
253
254                                path = syspath;
255                                return;
256                        }
257
258                        // accessing temp. OS directory:         
259                        String property = "java.io.tmpdir";
260
261                        String tempdir = System.getProperty(property);
262
263                        if ( !(tempdir.endsWith(FILE_SEPARATOR) ) )
264                                tempdir = tempdir + FILE_SEPARATOR;
265
266                        System.err.println("you did not set the path in PDBFileReader, don't know where to write the downloaded file to");
267                        System.err.println("assuming default location is temp directory: " + tempdir);
268                        path = tempdir;
269                }
270        }

272        /** Returns the file name that contains the definition for this {@link ChemComp}
273         *  
274         * @param recordName the ID of the {@link ChemComp}
275         * @return full path to the file
276         */
277        public static String getLocalFileName(String recordName){
278
279                if ( protectedIDs.contains(recordName)){
280                        recordName = "_" + recordName;
281                }
282                
283                String dir = path + CHEM_COMP_CACHE_DIRECTORY + FILE_SEPARATOR;
284
285                File f = new File(dir);
286                if (! f.exists()){
287                        System.out.println("creating directory " + f);
288                        f.mkdir();
289                }
290
291                String fileName = path + CHEM_COMP_CACHE_DIRECTORY + FILE_SEPARATOR + recordName + ".cif.gz";
292
293                return fileName;
294        }
295
296        private static  boolean fileExists(String recordName){
297
298                String fileName = getLocalFileName(recordName);
299
300                File f = new File(fileName);
301
302                return f.exists();
303
304        }

306        private static void downloadChemCompRecord(String recordName) {
307                String path = System.getProperty(AbstractUserArgumentProcessor.PDB_DIR);
308                setPath(path);
309
310
311                String localName = getLocalFileName(recordName);
312
313                String u = serverLocation + recordName + ".cif";
314
315                System.out.println("downloading " + u);
316
317                try {
318
319                        URL url = new URL(u);
320
321                        HttpURLConnection uconn = HTTPConnectionTools.openHttpURLConnection(url);
322
323                        InputStream conn = uconn.getInputStream();
324
325                        FileOutputStream outPut = new FileOutputStream(localName);
326
327                        GZIPOutputStream gzOutPut = new GZIPOutputStream(outPut);
328
329                        PrintWriter pw = new PrintWriter(gzOutPut);
330
331                        BufferedReader fileBuffer = new BufferedReader(new InputStreamReader(conn));
332
333                        String line;
334
335                        while ((line = fileBuffer.readLine()) != null) {
336                                pw.println(line);
337                        }
338
339                        pw.flush();
340                        pw.close();
341
342                        outPut.close();
343                        conn.close();
344
345
346                } catch (Exception e){
347                        e.printStackTrace();
348                }
349
350
351        }

355         * 
356         * @param p path to PDB files.
357         */
358        public static void setPath(String p) {
359                if ( p == null)
360                        return;
361                path = p;
362                if ( ! path.endsWith(FILE_SEPARATOR))
363                        path += FILE_SEPARATOR;
364                
365                System.setProperty(AbstractUserArgumentProcessor.PDB_DIR,path);
366                
367                
368
369        }

371        private void downloadAllDefinitions() {
372
373                if ( loading.get()){
374                        System.out.println("Waiting for other thread to install chemical components...");
375                }
376
377                while ( loading.get() ) {
378
379                        // another thread is already downloading the components definitions
380                        // wait for the other thread to finish...
381
382                        try {
383                                // wait half a second
384
385                                Thread.sleep(500);
386                        } catch (InterruptedException e) {
387                                // TODO Auto-generated catch block
388                                e.printStackTrace();
389                        }
390
391                        System.out.println("Another thread installed the chemical components.");
392                        return;
393
394                }
395
396                loading.set(true);
397                long timeS = System.currentTimeMillis();
398
399                System.out.println("Performing first installation of chemical components.");
400                System.out.println("Downloading components.cif.gz ...");
401
402                AllChemCompProvider.checkPath();
403                try {
404                        AllChemCompProvider.downloadFile();
405                } catch (IOException e){
406                        e.printStackTrace();
407                }
408                
409                split();
410                long timeE = System.currentTimeMillis();                
411                System.out.println("time to install chem comp dictionary: " + (timeE - timeS) / 1000 + " sec.");                
412                loading.set(false);
413
414        }
