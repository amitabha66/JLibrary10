042/** a utility class that listens to Ramsol script commands in the @link {@link BiojavaJmol} class
043 * 
044 * @author Andreas Prlic
045 *
046 */
047public class RasmolCommandListener 
048extends KeyAdapter
049implements ActionListener,
050MouseListener {
051
052        JTextField textfield;
053        JmolPanel jmolPanel;
054
055        List<String> history;
056        int historyPosition;
057
058        public RasmolCommandListener(JmolPanel panel, JTextField field){
059                textfield = field;
060                jmolPanel = panel;
061                history = new ArrayList<String>();
062                historyPosition = -2; // -2 = history = empty;
063        }       
064        
065        public void actionPerformed(ActionEvent event) {
066                /*
067                if ( spice.isLoading() ) {
068                    logger.finest("loading data, please be patient");
069                    return ;
070                }
071                 */
072                String cmd = textfield.getText();
073                jmolPanel.executeCmd(cmd);
074                textfield.setText("");
075
076                // now comes history part:
077
078                // no need for history:
079                if ( cmd.equals("")) return;
080
081                // check last command in history
082                // if equivalent, don't add,
083                // otherwise add               
084                if (history.size()>0){
085                        String txt=(String)history.get(history.size()-1);
086                        if (! txt.equals(cmd)) {
087                                history.add(cmd);  
088                        }
089                } else {             
090                        // the first time always add
091                        history.add(cmd);
092                }
093                historyPosition=history.size();
094
095
096        }
097
098        public void  mouseClicked(MouseEvent e){
099                String cmd = textfield.getText();
100                if ( cmd.equals(StructureAlignmentJmol.COMMAND_LINE_HELP)){
101                        textfield.setText("");
102                        textfield.repaint();
103                }
104        };
105
106
107        public void  mouseExited(MouseEvent e){};
108        public void  mouseReleased(MouseEvent e){};
109        public void  mousePressed(MouseEvent e){};
110
111        public void  mouseEntered(MouseEvent e){};
112
113        /** takes care of the cursor up/down keys. triggers copying of stored 
114         * commands into the current textfield
115         * 
116         */ 
117         
118
119        public void keyReleased(KeyEvent e){
120
121                int code = e.getKeyCode();
122                //String s = e.getKeyText(code);
123                //System.out.println(s);
124                if (( code == KeyEvent.VK_UP ) || 
125                                ( code == KeyEvent.VK_KP_UP)) {
126                        // go one back in history;
127                        if ( historyPosition > 0){
128                                historyPosition= historyPosition-1;              
129                        } 
130                } else if (( code == KeyEvent.VK_DOWN ) || 
131                                ( code == KeyEvent.VK_KP_DOWN)) {            
132                        if ( historyPosition < (history.size()-1) ){
133                                historyPosition++;                
134                        } else {
135                                // clear command if at beginning of history
136                                textfield.setText("");
137                                historyPosition=history.size();
138                                return;
139                        }
140                } else if ( code == KeyEvent.VK_PAGE_UP) {
141                        if ( historyPosition > 0) {
142                                historyPosition = 0;
143                        }
144                } else if ( code == KeyEvent.VK_PAGE_DOWN) {
145                        if ( historyPosition >= 0) {
146                                historyPosition = history.size()-1;
147                        }
148                } else {
149                        // some other key has been pressed, do nothing
150                        return;
151                }
152
153                if ( historyPosition >= 0) {
154                        String txt = (String)history.get(historyPosition);
155                        textfield.setText(txt);
156                }
157
158
159        }
160
161
162}
