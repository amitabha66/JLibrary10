006/**
007 * A list of FeatureI objects implemented using a Java ArrayList; corresponds to a GFF file.
008 * This class is implemented entirely using FeatureI objects, so everything here will work
009 * correctly if you choose to implement your own feature class -- there are no dependencies
010 * on JavaGene's native Feature class.
011 *
012 *
013 * @author Hanno Hinsch
014 */
015public class FeatureList extends ArrayList<FeatureI> {
016
017    Location mLocation;                 //genomic location (union of feature locations)
018
019    /**
020     * Construct an empty list.
021     */
022    public FeatureList() {
023        mLocation = null;
024    }
025
026    /**
027     * Construct a new list containing the same features
028     * as the specified list.
029     *
030     * @param features An existing list or collection of FeatureI objects.
031     */
032    public FeatureList(Collection<FeatureI> features) {
033        super(features);
034
035        mLocation = null;
036    }
037
038    /**
039     * Add specified feature to the end of the list. Updates the bounding location of the
040     * feature list, if needed.
041     *
042     * @param feature The FeatureI object to add.
043     * @return True if the feature was added.
044     */
045    public boolean add(FeatureI feature) {
046        if (mLocation == null) {
047            mLocation = feature.location().plus();
048        } else if (null != feature.location()) {
049            mLocation = mLocation.union(feature.location().plus());
050        }
051
052        return super.add(feature);
053    }
054
055    /**
056     * Add all features in the specified list or collection to this list.
057     *
058     * @param list The collection of FeatureI objects.
059     */
060    public void add(Collection<FeatureI> list) {
061        for (FeatureI f : list) {
062            add(f);
063        }
064    }
065
066    /**
067     * The union of all locations of all features in this list, mapped to the positive strand.
068     * If an added feature is on the negative strand, its positive strand image is added
069     * to the union.
070     * The bounding location is not updated when a feature is removed from the list, so
071     * it is not guaranteed to be the minimal bounding location.
072     *
073     * @return A location that is the union of all feature locations in the list.
074     */
075    public Location bounds() {
076        return mLocation;
077    }
078
079    /**
080     * Check size of gaps between successive features in list. The features in
081     * the list are assumed to be appropriately ordered.
082     *
083     * @param gapLength The minimum gap length to consider. Use a gapLength
084     * of 0 to check if features are contiguous.
085     * @return True if list has any gaps equal to or greater than gapLength.
086     */
087    public boolean hasGaps(int gapLength) {
088        Location last = null;
089        for (FeatureI f : this) {
090            if (last != null && gapLength <= f.location().distance(last)) {
091                return true;
092            } else {
093                last = f.location();
094            }
095        }
096
097        return false;
098    }

103    /**
104     * Concatenate successive portions of the specified sequence
105     * using the feature locations in the list. The list is assumed to be appropriately
106     * ordered.
107     *
108     * @param sequence The source sequence from which portions should be selected.
109     * @return The spliced data.
110     * @throws IllegalStateException Out of order or overlapping FeatureI locations detected.
111     *
112     */
113    public String splice(DNASequence sequence) {
114        String subData = "";            //FIXME should be StringBuffer
115        Location last = null;
116
117        for (FeatureI f : this) {
118            Location loc = f.location();
119
120            if (last == null || loc.startsAfter(last)) {
121                subData += sequence.getSubSequence(loc.start(), loc.end()).toString();
122                last = loc;
123            } else {
124                throw new IllegalStateException("Splice: Feature locations should not overlap.");
125            }
126
127        }
128
129        return subData;
130    }

132    /**
133     * Create a collection of all unique group ids in the list, as defined
134     * by the group() method of the features. For example, if the
135     * features are from a GFF1 file, then each group id identifies a particular gene,
136     * and this method returns a collection of all gene ids.
137     *
138     * @return A collection (suitable for iteration using Java's "for" loop) of all the
139     * group ids found in this list. The order of the values is undefined; it will not match
140     * the order of features in the list.
141     */
142    public Collection<String> groupValues() {
143        LinkedHashMap<String, String> hash = new LinkedHashMap<String, String>();
144        for (FeatureI f : this) {
145            //enter as a key -- removes duplicates
146            hash.put(f.group(), null);
147        }
148
149        return hash.keySet();
150    }
151
152    /**
153     * Create a collection of the unique values for the specified key.
154     * Example: For GTF files, using the "gene_id" key will give the names of all
155     * the genes in this list.
156     *
157     * @return A collection (suitable for iteration using Java's "for" loop) of all the
158     * values found for this key. The order of the values is undefined; it will not match
159     * the order of features in the list.
160     */
161    public Collection<String> attributeValues(String key) {
162        LinkedHashMap<String, String> hash = new LinkedHashMap<String, String>();
163        for (FeatureI f : this) {
164            //enter as a key -- removes duplicates
165            hash.put(f.getAttribute(key), null);
166        }
167
168        return hash.keySet();
169    }

279    /**
280     * Create a list of all features that overlap the specified location on the specified
281     * sequence.
282     *
283     * @param seqname The sequence name. Only features with this sequence name will be checked for overlap.
284     * @param location The location to check.
285     * @param useBothStrands If true, locations are mapped to their positive strand image
286     * before being checked for overlap. If false, only features whose locations are 
287     * on the same strand as the specified location will be considered for inclusion.
288     * @return The new list of features that overlap the location.
289     */
290    public FeatureList selectOverlapping(String seqname, Location location, boolean useBothStrands)
291            throws Exception {
292        FeatureList list = new FeatureList();
293
294        for (FeatureI feature : this) {
295            boolean overlaps = false;
296            if (feature.seqname().equals(seqname)) {
297                if (location.isSameStrand(feature.location())) {
298                    overlaps = feature.location().overlaps(location);
299                } else if (useBothStrands) {
300                    overlaps = feature.location().overlaps(location.opposite());
301                }
302            }
303
304            if (overlaps) {
305                list.add(feature);
306            }
307        }
308
309        return list;
310    }

312    /**
313     * Create a list of all features that do not overlap the specified location on the specified sequence.
314     *
315     * @param seqname The sequence name. Only features with this sequence name will be checked for overlap.
316     * @param location The location to check.
317     * @param useBothStrands If true, locations are mapped to their positive strand image
318     * before being checked for overlap. If false, all features whose locations are
319     * on the opposite strand from the specified location will be considered non-overlapping.
320     * @return The new list of features that do not overlap the location.
321     */
322    public FeatureList omitOverlapping(String seqname, Location location, boolean useBothStrands) {
323        FeatureList list = new FeatureList();
324
325        for (FeatureI feature : this) {
326            boolean overlaps = false;
327            if (feature.seqname().equals(seqname)) {
328                if (location.isSameStrand(feature.location())) {
329                    overlaps = feature.location().overlaps(location);
330                } else if (useBothStrands) {
331                    overlaps = feature.location().overlaps(location.opposite());
332                }
333            }
334
335            if (!overlaps) {
336                list.add(feature);
337            }
338        }
339
340        return list;
341    }

376    /**
377     * Return a string representation of all features in this list.
378     *
379     * @return A string.
380     */
381    public String toString() {
382        String s = "FeatureList: >>\n";
383        for (FeatureI f : this) {
384            s += f.seqname() + ":" + f.toString() + "\n";
385        }
386
387        return s + "\n<<\n";

390    /**
391     * used by sort routine
392     */
393    private class FeatureComparator implements Comparator<FeatureI> {
394
395        public int compare(FeatureI a, FeatureI b) {
396            if (a.seqname().equals(b.seqname()) && a.location().isSameStrand(b.location())) {
397                return a.location().start() - b.location().start();             //sort on start
398            } else {
399                throw new IndexOutOfBoundsException("Cannot compare/sort features whose locations are on opposite strands or with different seqname().\r\n" + a.toString() + "\r\n" + b.toString() );
400            }
401        }
402    }
403
404    /**
405     * Create a new list that is ordered by the starting index of the features' locations. All locations
406     * must be on the same strand of the same sequence.
407     *
408     * @return An ordered list.
409     * @throws IndexOutOfBoundsException Cannot compare/sort features whose locations are on opposite strands, or
410     * whose seqnames differ.
411     */
412    public FeatureList sortByStart() {
413        FeatureI array[] = (FeatureI[]) toArray(new FeatureI[1]);
414
415        Arrays.sort(array, new FeatureComparator());
416
417        return new FeatureList(Arrays.asList(array));
418    }
