074   /**
075     optimize the structural alignment by update the equivalent residues
076      and then run dynamic programming
077      input: len1 the length of structure 1; c1: the structure information of 1
078             len2 the length of structure 2; c2: the structure information of 2
079             iniLen and iniSet is the length and list of initial equivalent residues
080           
081
082    */
083
084   public StructureAlignmentOptimizer(int b1, int end1, Atom[] c1, int b2, int end2, Atom[] c2,
085         int iniLen, int[][] iniSet) throws StructureException{
086      
087      //System.err.println("optimizing range:" + b1 + "-" + end1 + "("+ (end1-b1) + ") b2:  " + b2 + "-" + end2+ "("+ (end2-b2) + ") iniLen " + iniLen);
088      //System.out.println("ca1: " + c1.length + " ca2: " + c2.length);
089
090      int len1 = end1-b1;
091      int len2 = end2-b2;
092           
093      //System.err.println("len1: " + len1 + " len2:" + len2);
094      
095      pro1Len = len1;
096      pro2Len = len2;
097            
098      cod1 = new Atom[len1];
099      cod2 = new Atom[len2];
100      
101      for(int i = 0; i < len1; i ++)      {           
102         Atom a = c1[i+b1];
103         //cod1[i] = (Atom)a.clone();
104         Group parent = (Group)a.getGroup().clone();
105         //cod1[i].setParent(parent);
106         cod1[i] = parent.getAtom(StructureTools.caAtomName);
107         //cod1[i] = c1[i];
108      }
109      for(int i = 0; i < len2; i ++)      {
110         Atom a = c2[i+b2];
111         //cod2[i]= (Atom)a.clone();
112         Group parent = (Group)a.getGroup().clone();
113         //cod2[i].setParent(parent);
114         cod2[i] = parent.getAtom(StructureTools.caAtomName);
115         //cod2[i] = c2[i];
116      }
117
118
119      //initial equivalent sets
120      maxLen = (len1 < len2)?len1:len2;
121
122      equSet = new int[2][maxLen];
123      for(int i = 0; i < iniLen; i ++)    {
124
125         equSet[0][i] = iniSet[0][i];
126         equSet[1][i] = iniSet[1][i];
127         if(iniSet[0][i] > len1 || iniSet[1][i] > len2)  {
128            throw new RuntimeException(String.format("StructureAlignmentOptimizer: focus exceeds the protein 1 or 2 length!"));            
129         }
130      }
131
132      equLen    = iniLen;
133      equLen0   = equLen;
134
135      setParameters();
136
137      sij =  new double[pro1Len][pro2Len];
138
139//      if (showAlig) 
140//         showCurrentAlignment(iniLen, equSet, "initial alignment");
141
142   }

200   /** run the optimization
201    *
202    * @param maxi maximum nr. of iterations
203    * @throws StructureException
204    */
205   public void runOptimization(int maxi) throws StructureException{
206      superimposeBySet();
207      if ( debug)
208         System.err.println("   initial rmsd " + rmsd);
209
210//      if (showAlig) 
211//         showCurrentAlignment(equLen, equSet, "after initial superimposeBySet Len:" +equLen + " rmsd:" +rmsd);
212
213      maxKeepStep = 4;
214      keepStep = 0;
215
216      optimize(maxi);
217   }
218
219   /**
220    * refer CE, similarity = Dc - dij, Dc is increased by 0.5 each cycle,
221  optimization continues until either
222 i)alignment length is less than 95% of alignment length before optimization
223 ii)rmsd is less than 110% of rmsd at the cycle when condition i) was first satisfied
224    */
225   private void setParameters()
226   {
227      Dc = 3.0; //Dc = 2.0
228      increase = 0.5;
229      stopLenPer = 0.95;
230      stopRmsdPer = 1.1;
231      stopRmsd = -1.0;
232      rmsdCut = 3.0;
233      gapIni = 5.0;
234      gapExt = 0.5;
235   }

240    */
241   private void superimposeBySet ()
242   throws StructureException
243   {
244
245      //extract the coordinations of equivalent residues
246      Atom[] tmp1 = new Atom[equLen];
247      Atom[] tmp2 = new Atom[equLen];
248      int     i,  r1, r2;
249
250     
251      for(i = 0; i < equLen; i ++)    {
252         r1 = equSet[0][i];
253         r2 = equSet[1][i];
254
255         tmp1[i] =       cod1[ r1 ];
256         tmp2[i] = (Atom)cod2[ r2 ].clone(); // have to be cloned!
257         //tmp2[i] = cod2[ r2 ];
258         
259
260         /*try {
261            System.out.println("before superimpos: " + equSet[0][i]+"-"+ equSet[1][i]+ " dist:" + Calc.getDistance(tmp1[i], cod2[equSet[1][i]]));
262         } catch (Exception e){
263            e.printStackTrace();
264         }*/
265      }
266
267      //superimpose the equivalent residues
268      SVDSuperimposer svd = new SVDSuperimposer(tmp1, tmp2);
269
270      Matrix    m = svd.getRotation();

284      //System.err.println("rmsd after superimpose by set: " + rmsd);

292      }

309   private void optimize(int maxi) throws StructureException
310   {
311      long optStart = System.currentTimeMillis();
312      if ( debug)
313         System.out.println("Optimizing up to " + maxi + " iterations.. ");
314      boolean ifstop = true;;
315      int     i, alnLen;
316      alnLen = 0;
317
318      int[][]     alnList =  new int[2][maxLen];
319      for(i = 0; i < maxi; i ++)      {
320
321         //if ( debug){
322         //   System.out.println("optimize iteration: " + i);
323         //}
324
325         calMatrix();
326
327         FCAlignHelper aln = new FCAlignHelper(sij,pro1Len,pro2Len,gapIni, gapExt);
328
329         //ALIGN0 *aln = new ALIGN0(sij, pro1Len, pro2Len, gapIni, gapExt);
330         alnLen = aln.getAlignPos(alnList);
331         if(alnLen < 3)  ifstop = true; //very rare, mark by Y.Y on 5/1/03
332         else    ifstop = defineEquPos(alnLen, alnList);
333
334         if(ifstop)      break;
335         Dc += increase;
336
337//         if (showAlig) 
338//            if ( i == 0 )
339//               showCurrentAlignment(alnLen, alnList,  "optimizing alignment - after " + i + " iterations alnLen:" + alnLen + " rmsd " + rmsd);     
340      }
341
342      if  (debug){
343         if(i < maxi)    {
344            System.out.println(String.format("   optimize converged at %d iterations\n", i));
345         }
346         else    System.out.println("   optimize stop without convergence\n");
347         System.out.println("optimization time: " + (System.currentTimeMillis() - optStart) + " ms.");
348      }
349
350//      if (showAlig) 
351//         showCurrentAlignment(alnLen, alnList,  "optimizing alignment - after " + i + " iterations alnLen:" + alnLen + " rmsd " + rmsd);     
352   }

354   //--------------------------------------------------------------------------------------------------------
355   //the definition of matrix between residues:
356   //              Sij = Dc^2 - Dij^2 if Dij <= Dc
357   //                    0            else
358   //--------------------------------------------------------------------------------------------------------
359   private void calMatrix() throws StructureException
360   {
361      int     i, j;
362      double  dis;
363      for(i = 0; i < pro1Len; i ++)   {
364         for(j = 0; j < pro2Len; j ++)   {
365            dis = Calc.getDistance(cod1[i],cod2[j]);
366
367            if(dis < Dc) {
368               sij[i][j] = Dc - dis;               
369            }
370            else  {
371               sij[i][j] = 0;
372            }
373         }
374      }
375   }

377   /**the equivalent residues: residues where Dij <= Dc and i,j is an aligned pair
378 use the previous superimposing
379    */
380
381   private boolean defineEquPos(int alnLen, int[][] alnList)
382   throws StructureException
383   {
384      int     i, r1, r2;
385      int     equLenOld = equLen;
386      int[][]    equSetOld = new int[2][equLenOld];
387      for(i = 0; i < equLen; i ++)    {
388         equSetOld[0][i] = equSet[0][i];
389         equSetOld[1][i] = equSet[1][i];
390      }
391      double  rmsdOld = rmsd;
392      double  dis;
393      equLen = 0;
394      //if (debug)
395      //   System.out.println(String.format(" OPT: Dc %f, equLenOld %d, rmsdOld %f, alnLen %d", Dc, equLenOld, rmsdOld, alnLen));
396      for(i = 0; i < alnLen; i ++)    {
397         r1 = alnList[0][i];
398         r2 = alnList[1][i];
399         dis = Calc.getDistance(cod1[r1],cod2[r2]);         
400         if(dis <= Dc)   {
401            //System.out.println(r1 + "-"  + r2 + " d:" + dis); 
402            equSet[0][equLen] = r1;
403            equSet[1][equLen] = r2;
404            equLen ++;
405         }
406      }
407
408      superimposeBySet();
409
410      //if (debug)
411      //   System.out.println(String.format(" OPT: new equLen %d rmsd %f", equLen, rmsd));
412
413      boolean     ifstop = false;
414
415//      if (debug) {
416//         System.out.print(" OPT: rmsd diff: " + Math.abs(rmsd - rmsdOld) + " equLens: " + equLenOld + ":"+ equLen);
417//         if ( Math.abs(rmsd - rmsdOld) < 1e-10)
418//            System.out.println(" NO DIFF!");
419//         else 
420//            System.out.println(" DIFF!");
421//      }
422
423      if((Math.abs(rmsd - rmsdOld) < 1e-10 ) && (equLenOld == equLen)) keepStep ++;
424      else    keepStep = 0;
425
426      if(keepStep > maxKeepStep)      {
427         ifstop = true; //converge
428      } //allowing up to maxKeepStep instead of 1 is essential for some special cases
429      else if(stopRmsd < 0) {
430         ifstop = false; //condition 1, continue
431      }
432      else if((rmsd <= stopRmsd * stopRmsdPer) || (rmsd < rmsdCut)) {
433         ifstop = false; //condition 2, continue
434      } //rmsdCut is adopted or not? to be tuned
435      else    {
436         ifstop = true; //get worse
437      }
438
439
440      if((stopRmsd <0) && (equLen >= stopLenPer * equLen0))    {
441         //System.err.println("stopRmsd: " + stopRmsd + " setting to rmsd:" + rmsd);
442         stopRmsd = rmsd; //condition 1
443      }
444
445      return ifstop;
446   }

449   public double optimizeResult(int[] optLen, int optLenpos, int[][] list)
450   {
451      optLen[optLenpos] = equLen;
452
453      for(int i = 0; i < equLen; i ++)        {
454         list[0][i] = equSet[0][i];
455         list[1][i] = equSet[1][i];
456      }
457      return rmsd;
458   }
