041    static public LinkedHashMap<String, ChromosomeSequence> loadFastaAddGeneFeaturesFromUpperCaseExonFastaFile(File fastaSequenceFile, File uppercaseFastaFile, boolean throwExceptionGeneNotFound) throws Exception {
042        LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList = new LinkedHashMap<String, ChromosomeSequence>();
043        LinkedHashMap<String, DNASequence> dnaSequenceList = FastaReaderHelper.readFastaDNASequence(fastaSequenceFile);
044        for (String accession : dnaSequenceList.keySet()) {
045            DNASequence contigSequence = dnaSequenceList.get(accession);
046            ChromosomeSequence chromsomeSequence = new ChromosomeSequence(contigSequence.getSequenceAsString());
047            chromsomeSequence.setAccession(contigSequence.getAccession());
048            chromosomeSequenceList.put(accession, chromsomeSequence);
049        }
050
051
052        LinkedHashMap<String, DNASequence> geneSequenceList = FastaReaderHelper.readFastaDNASequence(uppercaseFastaFile);
053        for (DNASequence dnaSequence : geneSequenceList.values()) {
054            String geneSequence = dnaSequence.getSequenceAsString();
055            String lcGeneSequence = geneSequence.toLowerCase();
056            String reverseGeneSequence = dnaSequence.getReverse().getSequenceAsString();
057            String lcReverseGeneSequence = reverseGeneSequence.toLowerCase();
058            Integer bioStart = null;
059            Integer bioEnd = null;
060            Strand strand = Strand.POSITIVE;
061            boolean geneFound = false;
062            String accession = "";
063            DNASequence contigDNASequence = null;
064            for (String id : dnaSequenceList.keySet()) {
065                accession = id;
066                contigDNASequence = dnaSequenceList.get(id);
067                String contigSequence = contigDNASequence.getSequenceAsString().toLowerCase();
068                bioStart = contigSequence.indexOf(lcGeneSequence);
069                if (bioStart != -1) {
070                    bioStart = bioStart + 1;
071                    bioEnd = bioStart + geneSequence.length() - 1;
072                    geneFound = true;
073                    break;
074                } else {
075                    bioStart = contigSequence.indexOf(lcReverseGeneSequence);
076                    if (bioStart != -1) {
077                        bioStart = bioStart + 1;
078                        bioEnd = bioStart - geneSequence.length() - 1;
079                        strand = Strand.NEGATIVE;
080                        geneFound = true;
081                        break;
082                    }
083                }
084            }
085
086            if (geneFound) {
087                System.out.println("Gene " + dnaSequence.getAccession().toString() + " found at " + contigDNASequence.getAccession().toString() + " " + bioStart + " " + bioEnd + " " + strand);
088                ChromosomeSequence chromosomeSequence = chromosomeSequenceList.get(accession);
089
090                ArrayList<Integer> exonBoundries = new ArrayList<Integer>();
091
092                //look for transitions from lowercase to upper case
093                for (int i = 0; i < geneSequence.length(); i++) {
094                    if (i == 0 && Character.isUpperCase(geneSequence.charAt(i))) {
095                        exonBoundries.add(i);
096                    } else if (i == geneSequence.length() - 1) {
097                        exonBoundries.add(i);
098                    } else if (Character.isUpperCase(geneSequence.charAt(i)) && Character.isLowerCase(geneSequence.charAt(i - 1))) {
099                        exonBoundries.add(i);
100                    } else if (Character.isUpperCase(geneSequence.charAt(i)) && Character.isLowerCase(geneSequence.charAt(i + 1))) {
101                        exonBoundries.add(i);
102                    }
103                }
104                if (strand == Strand.NEGATIVE) {
105                    Collections.reverse(exonBoundries);
106                }
107
108
109                String geneaccession = dnaSequence.getAccession().getID();
110                String note = geneaccession;
111                String[] values = geneaccession.split(" ");
112                geneaccession = values[0];
113
114
115
116                GeneSequence geneSeq = chromosomeSequence.addGene(new AccessionID(geneaccession), bioStart, bioEnd, strand);
117                geneSeq.addNote(note);
118                geneSeq.setSource(uppercaseFastaFile.getName());
119                String transcriptName = geneaccession + "-transcript";
120                TranscriptSequence transcriptSequence = geneSeq.addTranscript(new AccessionID(transcriptName), bioStart, bioEnd);
121
122                int runningFrameLength = 0;
123                for (int i = 0; i < exonBoundries.size() - 1; i = i + 2) {
124                    int cdsBioStart = exonBoundries.get(i) + bioStart;
125                    int cdsBioEnd = exonBoundries.get(i + 1) + bioStart;
126                    runningFrameLength = runningFrameLength + Math.abs(cdsBioEnd - cdsBioStart) + 1;
127                    String cdsName = transcriptName + "-cds-" + cdsBioStart + "-" + cdsBioEnd;
128
129                    AccessionID cdsAccessionID = new AccessionID(cdsName);
130                    ExonSequence exonSequence = geneSeq.addExon(cdsAccessionID, cdsBioStart, cdsBioEnd);
131                    int remainder = runningFrameLength % 3;
132                    int frame = 0;
133                    if (remainder == 1) {
134                        frame = 2; // borrow 2 from next CDS region
135                    } else if (remainder == 2) {
136                        frame = 1;
137                    }
138                    CDSSequence cdsSequence = transcriptSequence.addCDS(cdsAccessionID, cdsBioStart, cdsBioEnd, frame);
139                }
140
141
142            } else {
143                if (throwExceptionGeneNotFound) {
144                    throw new Exception(dnaSequence.getAccession().toString() + " not found");
145                }
146                System.out.println("Gene not found " + dnaSequence.getAccession().toString());
147            }
148
149        }
150        return chromosomeSequenceList;
151    }

153    /**
154     * Output a gff3 feature file that will give the length of each scaffold/chromosome in the fasta file.
155     * Used for gbrowse so it knows length.
156     * @param fastaSequenceFile
157     * @param gffFile
158     * @throws Exception
159     */
160    static public void outputFastaSequenceLengthGFF3(File fastaSequenceFile, File gffFile) throws Exception {
161        LinkedHashMap<String, DNASequence> dnaSequenceList = FastaReaderHelper.readFastaDNASequence(fastaSequenceFile);
162        String fileName = fastaSequenceFile.getName();
163        FileWriter fw = new FileWriter(gffFile);
164        fw.write("##gff-version 3\n");
165        for (DNASequence dnaSequence : dnaSequenceList.values()) {
166
167            String gff3line = dnaSequence.getAccession().getID() + "\t" + fileName + "\t" + "contig" + "\t" + "1" + "\t" + dnaSequence.getBioEnd() + "\t.\t.\t.\tName=" + dnaSequence.getAccession().getID() + "\n";
168            fw.write(gff3line);
169        }
170        fw.close();
171    }

189    /**
190     * Load GFF2 feature file generated from the geneid prediction algorithm and map features onto the chromosome sequences
191     *
192     * @param chromosomeSequenceList
193     * @param listGenes
194     * @throws Exception
195     */
196    static public void addGeneIDGFF2GeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
197        Collection<String> geneIds = listGenes.attributeValues("gene_id");
198        for (String geneid : geneIds) {
199            FeatureList gene = listGenes.selectByAttribute("gene_id", geneid);
200            FeatureI geneFeature = gene.get(0);
201            ChromosomeSequence seq = chromosomeSequenceList.get(geneFeature.seqname());
202            geneid = geneid.replaceAll("_", ".G");
203            AccessionID geneAccessionID = new AccessionID(geneid);
204            GeneSequence geneSequence = null;
205            Collection<String> transcriptids = gene.attributeValues("gene_id");
206            for (String transcriptid : transcriptids) {
207                // get all the individual features (exons, CDS regions, etc.) of this gene
208                FeatureList transcriptFeature = listGenes.selectByAttribute("gene_id", transcriptid);
209                transcriptid = transcriptid.replaceAll("_", ".G");
210
211
212
213
214                //      String seqName = feature.seqname();
215                FeatureI startCodon = null;
216                FeatureI stopCodon = null;
217                Integer startCodonBegin = null;
218                Integer stopCodonEnd = null;
219                String startCodonName = "";
220                String stopCodonName = "";
221
222
223                // now select only the coding regions of this gene
224                FeatureList firstFeatures = transcriptFeature.selectByType("First");
225                FeatureList terminalFeatures = transcriptFeature.selectByType("Terminal");
226                FeatureList internalFeatures = transcriptFeature.selectByType("Internal");
227                FeatureList singleFeatures = transcriptFeature.selectByType("Single");
228                FeatureList cdsFeatures = new FeatureList();
229                cdsFeatures.add(firstFeatures);
230                cdsFeatures.add(terminalFeatures);
231                cdsFeatures.add(internalFeatures);
232                cdsFeatures.add(singleFeatures);
233                // sort them
234                cdsFeatures = cdsFeatures.sortByStart();
235                Strand strand = Strand.POSITIVE;
236                FeatureI feature = cdsFeatures.get(0);
237                if (feature.location().isNegative()) {
238                    strand = strand.NEGATIVE;
239                }
240                if (startCodonBegin == null) {
241                    FeatureI firstFeature = cdsFeatures.get(0);
242                    if (strand == strand.NEGATIVE) {
243                        startCodonBegin = firstFeature.location().bioEnd();
244                    } else {
245                        startCodonBegin = firstFeature.location().bioStart();
246                    }
247                }
248
249                if (stopCodonEnd == null) {
250
251                    FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
252                    if (strand == strand.NEGATIVE) {
253                        stopCodonEnd = lastFeature.location().bioStart();
254                    } else {
255                        stopCodonEnd = lastFeature.location().bioEnd();
256                    }
257                }
258                //for gtf ordering can be strand based so first is last and last is first
259                if (startCodonBegin > stopCodonEnd) {
260                    int temp = startCodonBegin;
261                    startCodonBegin = stopCodonEnd;
262                    stopCodonEnd = temp;
263                }
264
265                AccessionID transcriptAccessionID = new AccessionID(transcriptid);
266                if (geneSequence == null) {
267                    geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
268                    geneSequence.setSource(((Feature) feature).source());
269                } else {
270                    //if multiple transcripts for one gene make sure the gene is defined as the min and max start/end
271
272                    if (startCodonBegin < geneSequence.getBioBegin()) {
273                        geneSequence.setBioBegin(startCodonBegin);
274                    }
275                    if (stopCodonEnd > geneSequence.getBioBegin()) {
276                        geneSequence.setBioEnd(stopCodonEnd);
277                    }
278
279                }
280                TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
281                if (startCodon != null) {
282                    if (startCodonName == null || startCodonName.length() == 0) {
283                        startCodonName = transcriptid + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
284                    }
285                    transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
286                }
287                if (stopCodon != null) {
288                    if (stopCodonName == null || stopCodonName.length() == 0) {
289                        stopCodonName = transcriptid + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
290                    }
291                    transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
292                }
293
294                for (FeatureI cdsFeature : cdsFeatures) {
295                    Feature cds = (Feature) cdsFeature;
296                    String cdsName = cds.getAttribute("transcript_name");
297                    if (cdsName == null || cdsName.length() == 0) {
298                        cdsName = transcriptid + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
299                    }
300                    AccessionID cdsAccessionID = new AccessionID(cdsName);
301                    ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
302                    CDSSequence cdsSequence = transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), cds.frame());
303                    cdsSequence.setSequenceScore(cds.score());
304                }
305            }
306        }

310    static public LinkedHashMap<String, ChromosomeSequence> getChromosomeSequenceFromDNASequence(LinkedHashMap<String, DNASequence> dnaSequenceList) {
311        LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList = new LinkedHashMap<String, ChromosomeSequence>();
312        for (String key : dnaSequenceList.keySet()) {
313            DNASequence dnaSequence = dnaSequenceList.get(key);
314            ChromosomeSequence chromosomeSequence = new ChromosomeSequence(dnaSequence.getProxySequenceReader()); //we want the underlying sequence but don't need storage
315            chromosomeSequence.setAccession(dnaSequence.getAccession());
316            chromosomeSequenceList.put(key, chromosomeSequence);
317        }
318        return chromosomeSequenceList;
319    }

345    static public void addGmodGFF3GeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
346
347
348        // key off mRNA as being a known feature that may or may not have a parent gene
349
350
351        FeatureList mRNAFeatures = listGenes.selectByType("mRNA");
352        LinkedHashMap<String,FeatureList> featureIDHashMap = FeatureHelper.buildFeatureAtrributeIndex("ID", listGenes);

355        for (FeatureI f : mRNAFeatures) {
356            String geneID;
357            String geneNote = null;
358            String geneSource = null;
359            String sequenceName = null;
360            ChromosomeSequence seq = null;
361            GeneSequence geneSequence = null;
362
363            Feature mRNAFeature = (Feature) f;
364            String mRNAID = mRNAFeature.getAttribute("ID");
365            String mRNAsource = mRNAFeature.source();
366            String mRNANote = mRNAFeature.getAttribute("Note");
367            String mRNAParent = mRNAFeature.getAttribute("Parent");
368            if (mRNAParent != null && mRNAParent.length() > 0) {
369               // FeatureList geneFeatureList = listGenes.selectByAttribute("ID", mRNAParent);
370                FeatureList geneFeatureList = featureIDHashMap.get(mRNAParent);
371                Feature geneFeature = (Feature) geneFeatureList.get(0);
372                geneID = geneFeature.getAttribute("ID");
373                geneNote = geneFeature.getAttribute("Note");
374                geneSource = ((Feature) geneFeature).source();
375                sequenceName = geneFeature.seqname();
376
377                //
378            } else {
379                //deal with cases where no parent gene is given
380                geneID = mRNAID;
381                geneSource = mRNAsource;
382                sequenceName = mRNAFeature.seqname();
383            }
384
385            seq = chromosomeSequenceList.get(sequenceName);
386
387            AccessionID geneAccessionID = new AccessionID(geneID);
388
389          //  FeatureList mRNAChildren = listGenes.selectByAttribute("Parent", mRNAID);
390            FeatureList mRNAChildren = featureParentHashMap.get(mRNAID);
391            FeatureList cdsFeatures = mRNAChildren.selectByType("CDS");
392            FeatureI feature = cdsFeatures.get(0);
393            Strand strand = Strand.POSITIVE;
394
395            if (feature.location().isNegative()) {
396                strand = strand.NEGATIVE;
397            }
398            cdsFeatures = cdsFeatures.sortByStart();
399
400
401
402
403
404
405
406            String seqName = feature.seqname();
407            FeatureI startCodon = null;
408            FeatureI stopCodon = null;
409            Integer startCodonBegin = null;
410            Integer stopCodonEnd = null;
411            String startCodonName = "";
412            String stopCodonName = "";
413            FeatureList startCodonList = mRNAChildren.selectByType("five_prime_UTR");
414            if (startCodonList != null && startCodonList.size() > 0) {
415                startCodon = startCodonList.get(0);
416                if (strand == Strand.NEGATIVE) {
417                    startCodonBegin = startCodon.location().bioEnd();
418                } else {
419                    startCodonBegin = startCodon.location().bioStart();
420                }
421                startCodonName = startCodon.getAttribute("ID");
422            }
423
424            FeatureList stopCodonList = mRNAChildren.selectByType("three_prime_UTR");
425
426            if (stopCodonList != null && stopCodonList.size() > 0) {
427                stopCodon = stopCodonList.get(0);
428                if (strand == Strand.NEGATIVE) {
429                    stopCodonEnd = stopCodon.location().bioStart();
430                } else {
431                    stopCodonEnd = stopCodon.location().bioEnd();
432                }
433                stopCodonName = stopCodon.getAttribute("ID");
434
435            }
436
437
438
439
440            if (startCodonBegin == null) {
441                if (strand == Strand.NEGATIVE) {
442                    FeatureI firstFeature = cdsFeatures.get(0);
443
444                    startCodonBegin = firstFeature.location().bioEnd();
445                } else {
446                    FeatureI firstFeature = cdsFeatures.get(0);
447
448                    startCodonBegin = firstFeature.location().bioStart();
449                }
450            }
451
452            if (stopCodonEnd == null) {
453                if (strand == Strand.NEGATIVE) {
454                    FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
455                    stopCodonEnd = lastFeature.location().bioStart();
456                } else {
457                    FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
458                    stopCodonEnd = lastFeature.location().bioEnd();
459                }
460            }
461            //for gtf ordering can be strand based so first is last and last is first
462            if (startCodonBegin > stopCodonEnd) {
463                int temp = startCodonBegin;
464                startCodonBegin = stopCodonEnd;
465                stopCodonEnd = temp;
466            }
467
468
469
470            AccessionID transcriptAccessionID = new AccessionID(mRNAID);
471            geneSequence = seq.getGene(geneID);
472            if (geneSequence == null) {
473                geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
474                geneSequence.setSource(geneSource);
475                if (geneNote != null && geneNote.length() > 0) {
476                    geneSequence.addNote(geneNote);
477                }
478            } else {
479
480                if (startCodonBegin < geneSequence.getBioBegin()) {
481                    geneSequence.setBioBegin(startCodonBegin);
482                }
483                if (stopCodonEnd > geneSequence.getBioBegin()) {
484                    geneSequence.setBioEnd(stopCodonEnd);
485                }
486
487            }
488            TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
489            transcriptSequence.setSource(mRNAsource);
490            if (mRNANote != null && mRNANote.length() > 0) {
491                transcriptSequence.addNote(mRNANote);
492
493            }
494            if (startCodon != null) {
495                if (startCodonName == null || startCodonName.length() == 0) {
496                    startCodonName = mRNAID + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
497                }
498                transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
499            }
500            if (stopCodon != null) {
501                if (stopCodonName == null || stopCodonName.length() == 0) {
502                    stopCodonName = mRNAID + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
503                }
504                transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
505            }
506
507            for (FeatureI cdsFeature : cdsFeatures) {
508                Feature cds = (Feature) cdsFeature;
509                String cdsNote = cdsFeature.getAttribute("Note");
510                String cdsSource = cds.source();
511                String cdsName = cds.getAttribute("ID");
512                if (cdsName == null || cdsName.length() == 0) {
513                    cdsName = mRNAID + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
514                }
515                AccessionID cdsAccessionID = new AccessionID(cdsName);
516                ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
517                exonSequence.setSource(cdsSource);
518                if (cdsNote != null && cdsNote.length() > 0) {
519                    exonSequence.addNote(cdsNote);
520                }
521                transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), cds.frame());
522            }
523            geneSequence.addIntronsUsingExons();
524
525        }

537    static public void addGlimmerGFF3GeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
538        FeatureList mRNAFeatures = listGenes.selectByType("mRNA");
539        for (FeatureI f : mRNAFeatures) {
540            Feature mRNAFeature = (Feature) f;
541            String geneid = mRNAFeature.getAttribute("ID");
542            String source = mRNAFeature.source();
543
544            FeatureList gene = listGenes.selectByAttribute("Parent", geneid);
545            FeatureI geneFeature = gene.get(0);
546            ChromosomeSequence seq = chromosomeSequenceList.get(geneFeature.seqname());
547            AccessionID geneAccessionID = new AccessionID(geneid);
548            GeneSequence geneSequence = null;
549
550            FeatureList cdsFeatures = gene.selectByType("CDS");
551            FeatureI feature = cdsFeatures.get(0);
552            Strand strand = Strand.POSITIVE;
553
554            if (feature.location().isNegative()) {
555                strand = strand.NEGATIVE;
556            }
557            cdsFeatures = cdsFeatures.sortByStart();
558
559
560
561
562
563
564
565            String seqName = feature.seqname();
566            FeatureI startCodon = null;
567            FeatureI stopCodon = null;
568            Integer startCodonBegin = null;
569            Integer stopCodonEnd = null;
570            String startCodonName = "";
571            String stopCodonName = "";
572            FeatureList startCodonList = gene.selectByAttribute("Note", "initial-exon");
573            if (startCodonList != null && startCodonList.size() > 0) {
574                startCodon = startCodonList.get(0);
575                if (strand == Strand.NEGATIVE) {
576                    startCodonBegin = startCodon.location().bioEnd();
577                } else {
578                    startCodonBegin = startCodon.location().bioStart();
579                }
580                startCodonName = startCodon.getAttribute("ID");
581            }
582
583            FeatureList stopCodonList = gene.selectByAttribute("Note", "final-exon");
584
585            if (stopCodonList != null && stopCodonList.size() > 0) {
586                stopCodon = stopCodonList.get(0);
587                if (strand == Strand.NEGATIVE) {
588                    stopCodonEnd = stopCodon.location().bioStart();
589                } else {
590                    stopCodonEnd = stopCodon.location().bioEnd();
591                }
592                stopCodonName = stopCodon.getAttribute("ID");
593
594            }
595
596
597
598
599            if (startCodonBegin == null) {
600                if (strand == Strand.NEGATIVE) {
601                    FeatureI firstFeature = cdsFeatures.get(0);
602
603                    startCodonBegin = firstFeature.location().bioEnd();
604                } else {
605                    FeatureI firstFeature = cdsFeatures.get(0);
606
607                    startCodonBegin = firstFeature.location().bioStart();
608                }
609            }
610
611            if (stopCodonEnd == null) {
612                if (strand == Strand.NEGATIVE) {
613                    FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
614                    stopCodonEnd = lastFeature.location().bioStart();
615                } else {
616                    FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
617                    stopCodonEnd = lastFeature.location().bioEnd();
618                }
619            }
620            //for gtf ordering can be strand based so first is last and last is first
621            if (startCodonBegin > stopCodonEnd) {
622                int temp = startCodonBegin;
623                startCodonBegin = stopCodonEnd;
624                stopCodonEnd = temp;
625            }
626
627
628
629            AccessionID transcriptAccessionID = new AccessionID(geneid);
630            if (geneSequence == null) {
631                geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
632                geneSequence.setSource(source);
633            } else {
634
635                if (startCodonBegin < geneSequence.getBioBegin()) {
636                    geneSequence.setBioBegin(startCodonBegin);
637                }
638                if (stopCodonEnd > geneSequence.getBioBegin()) {
639                    geneSequence.setBioEnd(stopCodonEnd);
640                }
641
642            }
643            TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
644            if (startCodon != null) {
645                if (startCodonName == null || startCodonName.length() == 0) {
646                    startCodonName = geneid + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
647                }
648                transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
649            }
650            if (stopCodon != null) {
651                if (stopCodonName == null || stopCodonName.length() == 0) {
652                    stopCodonName = geneid + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
653                }
654                transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
655            }
656
657            for (FeatureI cdsFeature : cdsFeatures) {
658                Feature cds = (Feature) cdsFeature;
659                String cdsName = cds.getAttribute("ID");
660                if (cdsName == null || cdsName.length() == 0) {
661                    cdsName = geneid + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
662                }
663                AccessionID cdsAccessionID = new AccessionID(cdsName);
664                ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
665                transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), cds.frame());
666            }
667
668        }
669
670    }

680    static public void addGeneMarkGTFGeneFeatures(LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList, FeatureList listGenes) throws Exception {
681        Collection<String> geneIds = listGenes.attributeValues("gene_id");
682        for (String geneid : geneIds) {
683            //       if(geneid.equals("45_g")){
684            //           int dummy =1;
685            //       }
686            FeatureList gene = listGenes.selectByAttribute("gene_id", geneid);
687            FeatureI geneFeature = gene.get(0);
688            ChromosomeSequence seq = chromosomeSequenceList.get(geneFeature.seqname());
689            AccessionID geneAccessionID = new AccessionID(geneid);
690            GeneSequence geneSequence = null;
691            Collection<String> transcriptids = gene.attributeValues("transcript_id");
692            for (String transcriptid : transcriptids) {
693                // get all the individual features (exons, CDS regions, etc.) of this gene
694
695
696                FeatureList transcriptFeature = listGenes.selectByAttribute("transcript_id", transcriptid);
697                // now select only the coding regions of this gene
698                FeatureList cdsFeatures = transcriptFeature.selectByType("CDS");
699                // sort them
700                cdsFeatures = cdsFeatures.sortByStart();
701
702                FeatureI feature = cdsFeatures.get(0);
703                Strand strand = Strand.POSITIVE;
704
705                if (feature.location().isNegative()) {
706                    strand = strand.NEGATIVE;
707                }
708
709                String seqName = feature.seqname();
710                FeatureI startCodon = null;
711                FeatureI stopCodon = null;
712                Integer startCodonBegin = null;
713                Integer stopCodonEnd = null;
714                String startCodonName = "";
715                String stopCodonName = "";
716                FeatureList startCodonList = transcriptFeature.selectByType("start_codon");
717                if (startCodonList != null && startCodonList.size() > 0) {
718                    startCodon = startCodonList.get(0);
719                    if (strand == Strand.POSITIVE) {
720                        startCodonBegin = startCodon.location().bioStart();
721                    } else {
722                        startCodonBegin = startCodon.location().bioEnd();
723                    }
724                    startCodonName = startCodon.getAttribute("transcript_name");
725                }
726
727                FeatureList stopCodonList = transcriptFeature.selectByType("stop_codon");
728
729                if (stopCodonList != null && stopCodonList.size() > 0) {
730                    stopCodon = stopCodonList.get(0);
731                    if (strand == Strand.POSITIVE) {
732                        stopCodonEnd = stopCodon.location().bioEnd();
733                    } else {
734                        stopCodonEnd = stopCodon.location().bioStart();
735                    }
736
737                    stopCodonName = stopCodon.getAttribute("transcript_name");
738
739                }
740
741
742
743
744                if (startCodonBegin == null) {
745                    if (strand == Strand.NEGATIVE) {
746                        FeatureI firstFeature = cdsFeatures.get(0);
747
748                        startCodonBegin = firstFeature.location().bioEnd();
749                    } else {
750                        FeatureI firstFeature = cdsFeatures.get(0);
751
752                        startCodonBegin = firstFeature.location().bioStart();
753                    }
754                }
755
756                if (stopCodonEnd == null) {
757                    if (strand == Strand.NEGATIVE) {
758                        FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
759                        stopCodonEnd = lastFeature.location().bioStart();
760                    } else {
761                        FeatureI lastFeature = cdsFeatures.get(cdsFeatures.size() - 1);
762                        stopCodonEnd = lastFeature.location().bioEnd();
763                    }
764                }
765                //for gtf ordering can be strand based so first is last and last is first
766                if (startCodonBegin > stopCodonEnd) {
767                    int temp = startCodonBegin;
768                    startCodonBegin = stopCodonEnd;
769                    stopCodonEnd = temp;
770                }
771
772                AccessionID transcriptAccessionID = new AccessionID(transcriptid);
773                if (geneSequence == null) {
774                    geneSequence = seq.addGene(geneAccessionID, startCodonBegin, stopCodonEnd, strand);
775                    geneSequence.setSource(((Feature) feature).source());
776                } else {
777                    //if multiple transcripts for one gene make sure the gene is defined as the min and max start/end
778
779                    if (startCodonBegin < geneSequence.getBioBegin()) {
780                        geneSequence.setBioBegin(startCodonBegin);
781                    }
782                    if (stopCodonEnd > geneSequence.getBioBegin()) {
783                        geneSequence.setBioEnd(stopCodonEnd);
784                    }
785
786                }
787                TranscriptSequence transcriptSequence = geneSequence.addTranscript(transcriptAccessionID, startCodonBegin, stopCodonEnd);
788                if (startCodon != null) {
789                    if (startCodonName == null || startCodonName.length() == 0) {
790                        startCodonName = transcriptid + "-start_codon-" + startCodon.location().bioStart() + "-" + startCodon.location().bioEnd();
791                    }
792                    transcriptSequence.addStartCodonSequence(new AccessionID(startCodonName), startCodon.location().bioStart(), startCodon.location().bioEnd());
793                }
794                if (stopCodon != null) {
795                    if (stopCodonName == null || stopCodonName.length() == 0) {
796                        stopCodonName = transcriptid + "-stop_codon-" + stopCodon.location().bioStart() + "-" + stopCodon.location().bioEnd();
797                    }
798                    transcriptSequence.addStopCodonSequence(new AccessionID(stopCodonName), stopCodon.location().bioStart(), stopCodon.location().bioEnd());
799                }
800
801                for (FeatureI cdsFeature : cdsFeatures) {
802                    Feature cds = (Feature) cdsFeature;
803                    // for genemark it appears frame of 2 =1 and frame of 1 = 2
804                    // doesn't matter when you string cds regions together as one block
805                    // but does make a difference when you try to make a protein sequence for each CDS region where
806                    // you give up or borrow based on the frame value
807                    // compared with gff like files and docs for geneid and glimmer where geneid and glimmer both do it the same
808                    // way that appears to match the gff3 docs.
809                    int frame = cds.frame();
810                    if (frame == 1) {
811                        frame = 2;
812                    } else if (frame == 2) {
813                        frame = 1;
814                    } else {
815                        frame = 0;
816                    }
817                    String cdsName = cds.getAttribute("transcript_name");
818                    if (cdsName == null || cdsName.length() == 0) {
819                        cdsName = transcriptid + "-cds-" + cds.location().bioStart() + "-" + cds.location().bioEnd();
820                    }
821                    AccessionID cdsAccessionID = new AccessionID(cdsName);
822                    ExonSequence exonSequence = geneSequence.addExon(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd());
823                    transcriptSequence.addCDS(cdsAccessionID, cdsFeature.location().bioStart(), cdsFeature.location().bioEnd(), frame);
824                }
825            }
826        }

830    static public LinkedHashMap<String, ProteinSequence> getProteinSequences(Collection<ChromosomeSequence> chromosomeSequences) throws Exception {
831        LinkedHashMap<String, ProteinSequence> proteinSequenceHashMap = new LinkedHashMap<String, ProteinSequence>();
832        for (ChromosomeSequence dnaSequence : chromosomeSequences) {
833            for (GeneSequence geneSequence : dnaSequence.getGeneSequences().values()) {
834                for (TranscriptSequence transcriptSequence : geneSequence.getTranscripts().values()) {
835                    //TODO remove?
836//                    DNASequence dnaCodingSequence = transcriptSequence.getDNACodingSequence();
837//                    System.out.println("CDS=" + dnaCodingSequence.getSequenceAsString());
838
839                    try {
840                        ProteinSequence proteinSequence = transcriptSequence.getProteinSequence();
841                       
842//                        System.out.println(proteinSequence.getAccession().getID() + " " + proteinSequence);
843                        if (proteinSequenceHashMap.containsKey(proteinSequence.getAccession().getID())) {
844                            throw new Exception("Duplicate protein sequence id=" + proteinSequence.getAccession().getID() + " found at Gene id=" + geneSequence.getAccession().getID());
845                        } else {
846                            proteinSequenceHashMap.put(proteinSequence.getAccession().getID(), proteinSequence);
847                        }
848                    } catch (Exception e) {
849                        e.printStackTrace();
850                    }
851
852                }
853
854            }
855        }
856        return proteinSequenceHashMap;
