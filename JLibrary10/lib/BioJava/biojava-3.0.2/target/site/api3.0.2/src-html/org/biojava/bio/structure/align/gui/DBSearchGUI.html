124        private JTabbedPane createListPane() {
125                JTabbedPane tabP = new JTabbedPane();
126                
127                JPanel dir = new JPanel();
128
129                
130                outFileLocation = new JTextField(20);
131                JButton chB = new JButton("Select");
132                
133                Box hBox = Box.createHorizontalBox();
134                hBox.add(outFileLocation);
135                hBox.add(Box.createGlue());
136                hBox.add(chB);
137                
138                dir.add(hBox);
139                
140                chB.addActionListener(new ActionListener() {
141                        
142                        public void actionPerformed(ActionEvent e) {
143                                JFileChooser chooser = new JFileChooser();
144                                chooser.setDialogTitle("Select Output Directory");
145                                chooser.setFileSelectionMode(JFileChooser.DIRECTORIES_ONLY);
146                                //
147                                // disable the "All files" option.
148                                //
149                                chooser.setAcceptAllFileFilterUsed(false);
150                                //    
151
152
153//                              In response to a button click:
154                                int returnVal = chooser.showSaveDialog(null);
155                                if ( returnVal == JFileChooser.APPROVE_OPTION) {
156                                        File file = chooser.getSelectedFile();
157                                        outFileLocation.setText(file.getAbsolutePath());
158                                        outFileLocation.repaint();
159                                }
160                                
161                        }
162                });
163                
164                tabP.addTab("Select Output Directory", null, dir,
165                "Configure the folder that will contain the results.");
166                
167                return tabP;
168        }
