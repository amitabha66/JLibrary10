078        /** align a string to the right
079         * length is the total length the new string should take, inlcuding spaces on the left
080         * incredible that this tool is missing in java !!!
081         */
082        private static String alignRight(String input, int length){
083
084
085                int n = input.length();
086                if ( n >= length)
087                        return input;
088
089                String spaces = "                           " ;
090                int diff = length - n ;
091                StringBuffer s = new StringBuffer();
092
093                s.append(spaces.substring(0,diff));
094                s.append(input);
095
096                return s.toString();
097        }
098
099        private static String alignLeft(String input, int length){
100                if (input.length() >= length) {
101                        return input;
102                }
103
104                String spaces = "                           " ;
105                input += spaces.substring(0, length - input.length() );
106                return input;
107
108        }

127        /** prints the connections in PDB style
128         *
129         * Thanks to Tamas Horvath for this one
130         */
131        private String printPDBConnections(){
132
133
134                StringBuffer str = new StringBuffer();
135
136                List<Map<String, Integer>> cons = structure.getConnections();
137                for (int cnr = 0; cnr<cons.size();cnr++){
138                        Map<String,Integer> con =  cons.get(cnr);
139                        Integer as = (Integer) con.get("atomserial");
140
141                        String atomserial =  "";
142
143                        String bond1 = "";
144                        String bond2 = "";
145                        String bond3 = "";
146                        String bond4 = "";
147                        String hyd1  = "";
148                        String hyd2  = "";
149                        String salt1 = "";
150                        String hyd3  = "";
151                        String hyd4  = "";
152                        String salt2 = "";
153
154
155
156                        if (con.containsKey("bond1"))  bond1 = con.get("bond1").toString();
157                        if (con.containsKey("bond2"))  bond2 = con.get("bond2").toString();
158                        if (con.containsKey("bond3"))  bond3 = con.get("bond3").toString();
159                        if (con.containsKey("bond4"))  bond4 = con.get("bond4").toString();
160                        if (con.containsKey("hyd1"))   hyd1  = con.get("hyd1").toString();
161                        if (con.containsKey("hyd2"))   hyd2  = con.get("hyd2").toString();
162                        if (con.containsKey("salt1"))  salt1 = con.get("salt1").toString();
163                        if (con.containsKey("hyd3"))   hyd3  = con.get("hyd3").toString();
164                        if (con.containsKey("hyd4"))   hyd4  = con.get("hyd4").toString();
165                        if (con.containsKey("salt2"))  salt2 = con.get("salt2").toString();
166
167                        atomserial = alignRight(""+as,5) ;
168                        bond1      = alignRight(bond1,5) ;
169                        bond2      = alignRight(bond2,5) ;
170                        bond3      = alignRight(bond3,5) ;
171                        bond4      = alignRight(bond4,5) ;
172                        hyd1       = alignRight(hyd1,5)  ;
173                        hyd2       = alignRight(hyd2,5)  ;
174                        salt1      = alignRight(salt1,5) ;
175                        hyd3       = alignRight(hyd3,5)  ;
176                        hyd4       = alignRight(hyd4,5)  ;
177                        salt2      = alignRight(salt2,5) ;
178
179                        String connectLine = "CONECT" + atomserial + bond1 + bond2 + bond3 +
180                        bond4 + hyd1 + hyd2 + salt1 + hyd3 + hyd4 + salt2;
181
182                        str.append(connectLine + newline);
183                }
184                return str.toString();
185        }

187        /** Convert a structure into a PDB file.
188         * @return a String representing a PDB file.
189         */
190        public String toPDB() {
191
192
193                StringBuffer str = new StringBuffer();
194                //int i = 0 ;
195
196
197
198                // TODO: print all the PDB header informaton in PDB style
199                // some objects (PDBHeader, Compound) are still missing
200                //
201
202                PDBHeader header = structure.getPDBHeader();
203                header.toPDB(str);
204
205
206                //REMARK 800
207                if (!structure.getSites().isEmpty()) {
208                        str.append("REMARK 800                                                                      " + newline);
209                        str.append("REMARK 800 SITE                                                                 " + newline);
210                        for (Site site : structure.getSites()) {
211                                site.remark800toPDB(str);
212                        }
213                }
214                //DBREF
215                for (DBRef dbref : structure.getDBRefs()){
216                        dbref.toPDB(str);
217                        str.append(newline);
218                }
219                //SSBOND
220                for (SSBond ssbond : structure.getSSBonds()){
221                        ssbond.toPDB(str);
222                        str.append(newline);
223                }
224                //SITE
225                for (Site site : structure.getSites()) {
226                        try {
227                                site.toPDB(str);
228                        } catch (Exception e){
229                                e.printStackTrace();
230                        }
231                }
232
233                //
234                // print the atom records
235                //
236
237                // do for all models
238                int nrModels = structure.nrModels() ;
239                if ( structure.isNmr()) {
240                        str.append("EXPDTA    NMR, "+ nrModels+" STRUCTURES"+newline) ;
241                }
242                for (int m = 0 ; m < nrModels ; m++) {
243                        List<Chain> model = structure.getModel(m);
244                        // todo support NMR structures ...
245                        if ( structure.isNmr()) {
246                                str.append("MODEL      " + (m+1)+ newline);
247                        }
248                        // do for all chains
249                        int nrChains = model.size();
250                        for ( int c =0; c<nrChains;c++) {
251                                Chain  chain   = model.get(c);
252                                //String chainID = chain.getChainID();
253                                //if ( chainID.equals(DEFAULTCHAIN) ) chainID = " ";
254                                // do for all groups
255                                int nrGroups = chain.getAtomLength();
256                                for ( int h=0; h<nrGroups;h++){
257
258                                        Group g= chain.getAtomGroup(h);
259
260                                        
261                                        toPDB(g,str);
262                                        
263                                        
264                                }
265                        }
266
267                        if ( structure.isNmr()) {
268                                str.append("ENDMDL"+newline);
269                        }
270
271
272
273                }
274
275                if ( doPrintConnections() )
276                        str.append(printPDBConnections());
277
278                return str.toString() ;
279        }

327        public static void toPDB(Atom a, StringBuffer str) {
328
329                Group g = a.getGroup();
330                Chain c = g.getChain();
331                String chainID = c.getChainID();
332
333                String type = g.getType() ;
334
335                String record = "" ;
336                if ( type.equals("hetatm") ) {
337                        record = "HETATM";
338                } else {
339                        record = "ATOM  ";
340                }
341
342
343                // format output ...
344                //int groupsize  = g.size();
345                String resName = g.getPDBName(); 
346                String pdbcode = g.getResidueNumber().toString();
347                //String line    = "" ;
348
349
350                int    seri       = a.getPDBserial()        ;
351                String serial     = alignRight(""+seri,5)   ;
352                String fullname   = a.getFullName()         ;
353
354                // System.out.println(" fullname: " + fullname + " : " + a.getAltLoc() + " : " + pdbcode);
355
356                Character  altLoc = a.getAltLoc()           ;
357                String resseq = "" ;
358                if ( hasInsertionCode(pdbcode) )
359                        resseq     = alignRight(""+pdbcode,5);
360                else
361                        resseq     = alignRight(""+pdbcode,4)+" ";
362                String x          = alignRight(""+d3.format(a.getX()),8);
363                String y          = alignRight(""+d3.format(a.getY()),8);
364                String z          = alignRight(""+d3.format(a.getZ()),8);
365                String occupancy  = alignRight(""+d2.format(a.getOccupancy()),6) ;
366                String tempfactor = alignRight(""+d2.format(a.getTempFactor()),6);
367
368                //System.out.println("fullname,zise:" + fullname + " " + fullname.length());
369
370                String leftResName = alignLeft(resName,3);
371
372                str.append(record);
373                str.append(serial);
374                str.append(" ");
375                str.append(fullname);
376                str.append(altLoc);
377                str.append(leftResName);
378                str.append(" ");
379                str.append(chainID);
380                str.append(resseq);
381                str.append("   ");
382                str.append(x);
383                str.append(y);
384                str.append(z);
385                str.append(occupancy);
386                str.append(tempfactor);
387
388                str.append(" ");
389                str.append(newline);
390
391        }

393        /** test if pdbserial has an insertion code */
394        private static boolean hasInsertionCode(String pdbserial) {
395                try {
396                        Integer.parseInt(pdbserial) ;
397                } catch (NumberFormatException e) {
398                        return true ;
399                }
400                return false ;
401        }

404        /** convert a protein Structure to a DAS Structure XML response .
405         * @param xw  a XMLWriter object
406         * @throws IOException ...
407         *
408         */
409        @SuppressWarnings("deprecation")
410        public void toDASStructure(XMLWriter xw)
411        throws IOException
412        {
413
414                /*xmlns="http://www.sanger.ac.uk/xml/das/2004/06/17/dasalignment.xsd" xmlns:align="http://www.sanger.ac.uk/xml/das/2004/06/17/alignment.xsd" xmlns:xsd="http://www.w3.org/2001/XMLSchema-instance" xsd:schemaLocation="http://www.sanger.ac.uk/xml/das/2004/06/17/dasalignment.xsd http://www.sanger.ac.uk/xml/das//2004/06/17/dasalignment.xsd"*/
415
416                if ( structure == null){
417                        System.err.println("can not convert structure null");
418                        return;
419                }
420
421                Map<String,Object> header = structure.getHeader();
422
423                xw.openTag("object");
424                xw.attribute("dbAccessionId",structure.getPDBCode());
425                xw.attribute("intObjectId"  ,structure.getPDBCode());
426                // missing modification date
427                String modificationDate = (String)header.get("modDate") ;
428                xw.attribute("objectVersion",modificationDate);
429                xw.attribute("type","protein structure");
430                xw.attribute("dbSource","PDB");
431                xw.attribute("dbVersion","20070116");
432                xw.attribute("dbCoordSys","PDBresnum,Protein Structure");
433
434                // do we need object details ???
435                xw.closeTag("object");
436
437
438                // do for all models
439                for (int modelnr = 0;modelnr<structure.nrModels();modelnr++){
440
441                        // do for all chains:
442                        for (int chainnr = 0;chainnr<structure.size(modelnr);chainnr++){
443                                Chain chain = (Chain)structure.getChain(modelnr,chainnr);
444                                xw.openTag("chain");
445                                xw.attribute("id",chain.getChainID());
446                                xw.attribute("SwissprotId",chain.getSwissprotId() );
447                                if (structure.isNmr()){
448                                        xw.attribute("model",Integer.toString(modelnr+1));
449                                }
450
451                                //do for all groups:
452                                for (int groupnr =0;
453                                groupnr<chain.getAtomLength()
454                                ;groupnr++){
455                                        Group gr = chain.getAtomGroup(groupnr);
456                                        xw.openTag("group");
457                                        xw.attribute("name",gr.getPDBName());
458                                        xw.attribute("type",gr.getType());
459                                        xw.attribute("groupID",gr.getResidueNumber().toString());
460
461
462                                        // do for all atoms:
463                                        //Atom[] atoms  = gr.getAtoms();
464                                        List<Atom> atoms =  gr.getAtoms();
465                                        for (int atomnr=0;atomnr<atoms.size();atomnr++){
466                                                Atom atom = (Atom)atoms.get(atomnr);
467                                                xw.openTag("atom");
468                                                xw.attribute("atomID",Integer.toString(atom.getPDBserial()));
469                                                xw.attribute("atomName",atom.getFullName());
470                                                xw.attribute("x",Double.toString(atom.getX()));
471                                                xw.attribute("y",Double.toString(atom.getY()));
472                                                xw.attribute("z",Double.toString(atom.getZ()));
473                                                xw.closeTag("atom");
474                                        }
475                                        xw.closeTag("group") ;
476                                }
477
478                                xw.closeTag("chain");
479                        }
480                }
481
482
483                if ( doPrintConnections() ) {
484                        // do connectivity for all chains:
485
486                        List<Map<String,Integer>> cons = structure.getConnections();
487                        for (int cnr = 0; cnr<cons.size();cnr++){
488
489
490                                /*
491                 the HashMap for a single CONECT line contains the following fields:
492                 <ul>
493                 <li>atomserial (mandatory) : Atom serial number
494                 <li>bond1 .. bond4 (optional): Serial number of bonded atom
495                 <li>hydrogen1 .. hydrogen4 (optional):Serial number of hydrogen bonded atom
496                 <li>salt1 .. salt2 (optional): Serial number of salt bridged atom
497                 </ul>
498                                 */
499
500                                Map<String, Integer> con = (Map<String, Integer>)cons.get(cnr);
501                                Integer as = (Integer)con.get("atomserial");
502                                int atomserial = as.intValue();
503
504
505                                List<Integer> atomids = new ArrayList<Integer>() ;
506
507                                // test salt and hydrogen first //
508                                if (con.containsKey("salt1")) atomids.add(con.get("salt1"));
509                                if (con.containsKey("salt2")) atomids.add(con.get("salt2"));
510
511                                if (atomids.size()!=0){
512                                        addConnection(xw,"salt",atomserial,atomids);
513                                        atomids = new ArrayList<Integer>() ;
514                                }
515                                if (con.containsKey("hydrogen1")) atomids.add(con.get("hydrogen1"));
516                                if (con.containsKey("hydrogen2")) atomids.add(con.get("hydrogen2"));
517                                if (con.containsKey("hydrogen3")) atomids.add(con.get("hydrogen3"));
518                                if (con.containsKey("hydrogen4")) atomids.add(con.get("hydrogen4"));
519                                if (atomids.size()!=0){
520                                        addConnection(xw,"hydrogen",atomserial,atomids);
521                                        atomids = new ArrayList<Integer>() ;
522                                }
523
524                                if (con.containsKey("bond1")) atomids.add(con.get("bond1"));
525                                if (con.containsKey("bond2")) atomids.add(con.get("bond2"));
526                                if (con.containsKey("bond3")) atomids.add(con.get("bond3"));
527                                if (con.containsKey("bond4")) atomids.add(con.get("bond4"));
528
529                                if (atomids.size()!=0){
530                                        addConnection(xw,"bond",atomserial,atomids);
531                                }
532                        }
533                }
534        }

536        private void addConnection(XMLWriter xw,String connType, int atomserial, List<Integer> atomids){
537                try{
538                        xw.openTag("connect");
539                        xw.attribute("atomSerial",Integer.toString(atomserial));
540                        xw.attribute("type",connType);
541                        for (int i=0;i<atomids.size();i++){
542                                Integer atomid = atomids.get(i);
543                                if ( atomid == null)
544                                        continue;
545                                int aid = atomid.intValue();
546                                xw.openTag("atomID");
547                                xw.attribute("atomID",Integer.toString(aid));
548                                xw.closeTag("atomID");
549                        }
550                        xw.closeTag("connect");
551                } catch( Exception e) {
552                        e.printStackTrace();
553                }
554        }
