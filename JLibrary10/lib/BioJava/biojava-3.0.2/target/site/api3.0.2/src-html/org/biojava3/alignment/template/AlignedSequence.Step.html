024package org.biojava3.alignment.template;
025
026import org.biojava3.core.sequence.location.template.Location;
027import org.biojava3.core.sequence.location.template.Point;
028import org.biojava3.core.sequence.template.Compound;
029import org.biojava3.core.sequence.template.Sequence;
030
031/**
032 * Defines a data structure for a {@link Sequence} within an alignment.
033 *
034 * @author Mark Chapman
035 * @param <C> each element of the {@link Sequence} is a {@link Compound} of type C
036 */
037public interface AlignedSequence<S extends Sequence<C>, C extends Compound> extends Sequence<C> {
038
039    /**
040     * Defines an alignment step in order to pass alignment information from an {@link Aligner} to a constructor.
041     */
042    enum Step { COMPOUND, GAP }
043
044    /**
045     * Nullifies cached arrays/objects.
046     */
047    void clearCache();
048
049    /**
050     * Returns the column index within an alignment corresponding to the given index in the original {@link Sequence}.
051     * Both indices are 1-indexed and inclusive.
052     *
053     * @param sequenceIndex index in the original {@link Sequence}
054     * @return column index within an alignment
055     * @throws IndexOutOfBoundsException if sequenceIndex < 1 or sequenceIndex >
056     *         {@link #getOriginalSequence()}.{@link #getLength()}
057     */
058    int getAlignmentIndexAt(int sequenceIndex);
059
060    /**
061     * Returns the {@link Point} within an alignment of the last element of the original {@link Sequence}.
062     *
063     * @return position within an alignment of final original {@link Sequence} element
064     */
065    Point getEnd();
066
067    /**
068     * Returns the {@link Location} of the original {@link Sequence} within an alignment.  This provides access to
069     * additional substructure beyond start and end points.
070     *
071     * @return location within an alignment
072     */
073    Location getLocationInAlignment();
074
075    /**
076     * Returns number of gaps in the sequence.  This could be determined from the {@link Location} information or from
077     * gap {@link Compound}s, which may not necessarily result in the same number.
078     *
079     * @return number of gaps in the sequence
080     */
081    int getNumGaps();
082
083    /**
084     * Returns the original {@link Sequence} before alignment.
085     *
086     * @return the original sequence
087     */
088    S getOriginalSequence();
089
090    /**
091     * Returns the maximum number of elements contributed to a column of an alignment by this {@link Sequence}.  If
092     * this {@link Sequence} is circular, this number is >= 1.  If not, this overlap count is definitely 1.
093     *
094     * @return the most elements contributed to any alignment column
095     */
096    int getOverlapCount();
097
098    /**
099     * Returns the index in the original {@link Sequence} corresponding to the given index within an alignment.  Both
100     * indices are 1-indexed and inclusive.
101     *
102     * @param alignmentIndex column index within an alignment
103     * @return index in the original {@link Sequence}
104     * @throws IndexOutOfBoundsException if alignmentIndex < 1 or alignmentIndex > {@link #getLength()}
105     */
106    int getSequenceIndexAt(int alignmentIndex);
107
108    /**
109     * Returns the {@link Point} within an alignment of the first element of the original {@link Sequence}.
110     *
111     * @return position within an alignment of first original {@link Sequence} element
112     */
113    Point getStart();
114
115    /**
116     * Returns true if this {@link Sequence} wraps around from the last alignment column back to the first.  This makes
117     * overlap possible, but does not require an overlap count > 1.
118     *
119     * @return true for circular alignment elements
120     */
121    boolean isCircular();
122
123    /**
124     * Returns true if this {@link Sequence} has a gap at a particular alignment column.
125     *
126     * @param alignmentIndex column index within an alignment
127     * @return true if this column has a gap
128     * @throws IndexOutOfBoundsException if alignmentIndex < 1 or alignmentIndex > {@link #getLength()}
129     */
130    boolean isGap(int alignmentIndex);
131
132}
