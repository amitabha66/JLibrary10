001package org.biojava3.aaproperties.profeat.convertor;
002
003public class Convert2SolventAccessibility extends Convertor{
004        /**
005         * Class for the conversion of protein sequence into solvent accessibility
006         */
007        @Override
008        public char convert(char c){
009                switch(c){
010                case 'A':case 'L':case 'F':case 'C':case 'G':case 'I':case 'V':case 'W':
011                        return group1;//Buried
012                case 'R':case 'K':case 'Q':case 'E':case 'N':case 'D':
013                        return group2;//Exposed
014                case 'M':case 'P':case 'S':case 'T':case 'H':case 'Y':
015                        return group3;//Intermediate
016                default:
017                        return unknownGroup;//Non-standard AA
018                }
019        }
020
021        private static String[] subCategory = {"Buried", "Exposed", "Intermediate"};
022        @Override
023        public String[] getGrouping(){return subCategory;}
024        @Override
025        public String getAttribute(){return "Solvent Accessibility";}
026}
