265        /**
266         * Identify a set of modifications in a a list of chains.
267         * @param chains query {@link Chain}s.
268         * @param potentialModifications query {@link ProteinModification}s.
269         */
270        public void identify(final List<Chain> chains,
271                        final Set<ProteinModification> potentialModifications) {
272                
273                if (chains==null) {
274                        throw new IllegalArgumentException("Null structure.");
275                }
276                
277                if (potentialModifications==null) {
278                        throw new IllegalArgumentException("Null potentialModifications.");
279                }
280                
281                        
282                reset();
283                
284                if (potentialModifications.isEmpty()) {
285                        return;
286                }
287                
288                Map<String, Chain> mapChainIdChain = new HashMap<String, Chain>(chains.size());
289                residues = new ArrayList<Group>();
290                List<Group> ligands = new ArrayList<Group>();
291                Map<Component, Set<Group>> mapCompGroups = new HashMap<Component, Set<Group>>();
292                
293                for (Chain chain : chains) {
294                        mapChainIdChain.put(chain.getChainID(), chain);
295                                        
296                        List<Group> ress = StructureUtil.getAminoAcids(chain);
297                        List<Group> ligs = chain.getAtomLigands();
298                        residues.addAll(ress);
299                        residues.removeAll(ligs);
300                        ligands.addAll(ligs);
301                        addModificationGroups(potentialModifications, ress, ligs, mapCompGroups);
302                }
303                
304                if (residues.isEmpty()) {
305                        String pdbId = "?";
306                        if ( chains.size() > 0) {
307                                Structure struc = chains.get(0).getParent();
308                                if ( struc != null)
309                                        pdbId = struc.getPDBCode(); 
310                        }
311                        System.err.println("WARNING: no amino acids found for "+ pdbId + ". Either you did not parse the PDB file with alignSEQRES records, or this record does not contain any amino acids.");
312                }
313                List<ModifiedCompound> modComps = new ArrayList<ModifiedCompound>();
314                
315                for (ProteinModification mod : potentialModifications) {
316                        ModificationCondition condition = mod.getCondition();
317                        List<Component> components = condition.getComponents();
318                        if (!mapCompGroups.keySet().containsAll(components)) {
319                                // not all components exist for this mod.
320                                continue;
321                        }
322                        
323                        int sizeComps = components.size();
324                        if (sizeComps==1) {
325                                
326                                processCrosslink1(mapCompGroups, modComps, mod, components);
327                        
328                        } else {
329                                
330                                processMultiCrosslink(mapCompGroups, modComps, mod, condition);
331                        }
332                }
333
334                if (recordAdditionalAttachments) {
335                        // identify additional groups that are not directly attached to amino acids.
336                        for (ModifiedCompound mc : modComps) {
337                                identifyAdditionalAttachments(mc, ligands, mapChainIdChain);
338                        }
339                }
340                
341                mergeModComps(modComps);
342                
343                identifiedModifiedCompounds.addAll(modComps);
344                
345                
346                // record unidentifiable linkage
347                if (recordUnidentifiableModifiedCompounds) {
348                        recordUnidentifiableAtomLinkages(modComps, ligands);
349                        recordUnidentifiableModifiedResidues(modComps);
350                }
351        }

353        private void reset() {
354                identifiedModifiedCompounds = new LinkedHashSet<ModifiedCompound>();
355                if (recordUnidentifiableModifiedCompounds) {
356                        unidentifiableAtomLinkages = new LinkedHashSet<StructureAtomLinkage>();
357                        unidentifiableModifiedResidues = new LinkedHashSet<StructureGroup>();
358                }
359                
360        }

362        private void processMultiCrosslink(
363                        Map<Component, Set<Group>> mapCompGroups,
364                        List<ModifiedCompound> modComps, ProteinModification mod,
365                        ModificationCondition condition) {
366                // for multiple components
367                
368                // find linkages first
369                List<List<Atom[]>> matchedAtomsOfLinkages =
370                                getMatchedAtomsOfLinkages(condition, mapCompGroups);
371                
372                if (matchedAtomsOfLinkages.size() != condition.getLinkages().size()) {
373                        return;
374                } 
375                
376                assembleLinkages(matchedAtomsOfLinkages, mod, modComps);
377                
378        }

380        private void processCrosslink1(Map<Component, Set<Group>> mapCompGroups,
381                        List<ModifiedCompound> modComps, ProteinModification mod,
382                        List<Component> components) {
383                // modified residue
384                // TODO: is this the correct logic for CROSS_LINK_1?
385                Set<Group> modifiedResidues = mapCompGroups.get(components.get(0));
386                if (modifiedResidues != null) {
387                        for (Group residue : modifiedResidues) {
388                                StructureGroup strucGroup = StructureUtil.getStructureGroup(residue, true);
389                                ModifiedCompound modRes = new ModifiedCompoundImpl(mod, strucGroup);
390                                modComps.add(modRes);
391                        }
392                }
393        }

395        /**
396         * identify additional groups that are not directly attached to amino acids.
397         * @param mc {@link ModifiedCompound}.
398         * @param chain a {@link Chain}.
399         * @return a list of added groups.
400         */
401        private void identifyAdditionalAttachments(ModifiedCompound mc, 
402                        List<Group> ligands, Map<String, Chain> mapChainIdChain) {
403                if (ligands.isEmpty()) {
404                        return;
405                }
406                
407                // TODO: should the additional groups only be allowed to the identified 
408                // ligands or both amino acids and ligands? Currently only on ligands
409                // ligands to amino acid bonds for same modification of unknown category
410                // will be combined in mergeModComps()
411                // TODO: how about chain-chain links?
412                List<Group> identifiedGroups = new ArrayList<Group>();
413                for (StructureGroup num : mc.getGroups(false)) {
414                        Group group;
415                        try {
416                                //String numIns = "" + num.getResidueNumber();
417                                //if (num.getInsCode() != null) {
418                                //      numIns += num.getInsCode();
419                                //}
420                                ResidueNumber resNum = new ResidueNumber();
421                                resNum.setChainId(num.getChainId());
422                                resNum.setSeqNum(num.getResidueNumber());
423                                resNum.setInsCode(num.getInsCode());
424                                //group = chain.getGroupByPDB(numIns);
425                                group = mapChainIdChain.get(num.getChainId()).getGroupByPDB(resNum);
426                        } catch (StructureException e) {
427                                e.printStackTrace();
428                                // should not happen
429                                continue;
430                        }
431                        identifiedGroups.add(group);
432                }
433                
434                int start = 0;
435                
436                int n = identifiedGroups.size();
437                while (n > start) {
438                        for (Group group1 : ligands) {
439                                for (int i=start; i<n; i++) {
440                                        Group group2 = identifiedGroups.get(i);
441                                        if (!identifiedGroups.contains(group1)) {
442                                                List<Atom[]> linkedAtoms = StructureUtil.findAtomLinkages(
443                                                                group1, group2, false, bondLengthTolerance);
444                                                if (!linkedAtoms.isEmpty()) {
445                                                        for (Atom[] atoms : linkedAtoms) {
446                                                                mc.addAtomLinkage(StructureUtil.getStructureAtomLinkage(atoms[0], 
447                                                                                false, atoms[1], false));
448                                                        }
449                                                        identifiedGroups.add(group1);
450                                                        break;
451                                                }
452                                        }
453                                }
454                        }
455                        
456                        start = n;
457                        n = identifiedGroups.size();
458                }
459        }

461        /**
462         * Merge identified modified compounds if linked.
463         */
464        private void mergeModComps(List<ModifiedCompound> modComps) {
465                TreeSet<Integer> remove = new TreeSet<Integer>();
466                int n = modComps.size();
467                for (int icurr=1; icurr<n; icurr++) {
468                        ModifiedCompound curr = modComps.get(icurr);
469                        
470                        String id = curr.getModification().getId();
471                        if (ProteinModificationRegistry.getById(id).getCategory()
472                                        !=ModificationCategory.UNDEFINED)
473                                continue;
474                        
475                        // find linked compounds that before curr
476                        //List<Integer> merging = new ArrayList<Integer>();
477                        int ipre = 0;
478                        for (; ipre<icurr; ipre++) {
479                                if (remove.contains(ipre))      continue;
480                                ModifiedCompound pre = modComps.get(ipre);
481                                if (!Collections.disjoint(pre.getGroups(false),
482                                                curr.getGroups(false))) {
483                                        break;
484                                }
485                        }
486                        
487                        if (ipre<icurr) {                            
488                                ModifiedCompound mcKeep = modComps.get(ipre);
489                                
490                                // merge modifications of the same type
491                                if (mcKeep.getModification().getId().equals(id)) {
492                                        // merging the current one to the previous one
493                                        mcKeep.addAtomLinkages(curr.getAtomLinkages());
494                                        remove.add(icurr);
495                                }
496                        }
497                }
498                
499                Iterator<Integer> it = remove.descendingIterator();
500                while (it.hasNext()) {
501                        modComps.remove(it.next().intValue());
502                }
503        }

505        /**
506         * Record unidentifiable atom linkages in a chain. Only linkages between two
507         * residues or one residue and one ligand will be recorded.
508         */
509        private void recordUnidentifiableAtomLinkages(List<ModifiedCompound> modComps,
510                        List<Group> ligands) {
511                
512                // first put identified linkages in a map for fast query
513                Set<StructureAtomLinkage> identifiedLinkages = new HashSet<StructureAtomLinkage>();
514                for (ModifiedCompound mc : modComps) {
515                        identifiedLinkages.addAll(mc.getAtomLinkages());
516                }
517                
518                // record
519                // cross link
520                int nRes = residues.size();
521                for (int i=0; i<nRes-1; i++) {
522                        Group group1 = residues.get(i);
523                        for (int j=i+1; j<nRes; j++) {
524                                Group group2 = residues.get(j);
525                                List<Atom[]> linkages = StructureUtil.findAtomLinkages(
526                                                group1, group2, true, bondLengthTolerance);
527                                for (Atom[] atoms : linkages) {
528                                        StructureAtomLinkage link = StructureUtil.getStructureAtomLinkage(atoms[0], 
529                                                        true, atoms[1], true);
530                                        unidentifiableAtomLinkages.add(link);
531                                }
532                        }
533                }
534                
535                // attachment
536                int nLig = ligands.size();
537                for (int i=0; i<nRes; i++) {
538                        Group group1 = residues.get(i);
539                        for (int j=0; j<nLig; j++) {
540                                Group group2 = ligands.get(j);
541                                if (group1.equals(group2)) { // overlap between residues and ligands
542                                        continue;
543                                }
544                                List<Atom[]> linkages = StructureUtil.findAtomLinkages(
545                                                group1, group2, false, bondLengthTolerance);
546                                for (Atom[] atoms : linkages) {
547                                        StructureAtomLinkage link = StructureUtil.getStructureAtomLinkage(atoms[0], 
548                                                        true, atoms[1], false);
549                                        unidentifiableAtomLinkages.add(link);
550                                }
551                        }
552                }
553        }

555        private void recordUnidentifiableModifiedResidues(List<ModifiedCompound> modComps) {
556                Set<StructureGroup> identifiedComps = new HashSet<StructureGroup>();
557                for (ModifiedCompound mc : modComps) {
558                        identifiedComps.addAll(mc.getGroups(true));
559                }
560                
561                // TODO: use the ModifiedAminoAcid after Andreas add that.
562                for (Group group : residues) {
563                        if (group.getType().equals(GroupType.HETATM)) {
564                                StructureGroup strucGroup = StructureUtil.getStructureGroup(
565                                                group, true);
566                                if (!identifiedComps.contains(strucGroup)) {
567                                        unidentifiableModifiedResidues.add(strucGroup);
568                                }
569                        }
570                }
571        }

573        /**
574         * 
575         * @param modifications a set of {@link ProteinModification}s.
576         * @param residues
577         * @param ligands 
578         * @param saveTo save result to
579         * @return map from component to list of corresponding residues
580         *  in the chain.
581         */
582        private void addModificationGroups(
583                        final Set<ProteinModification> modifications,
584                        final List<Group> residues,
585                        final List<Group> ligands,
586                        final Map<Component, Set<Group>> saveTo) {
587                if (residues==null || ligands==null || modifications==null) {
588                        throw new IllegalArgumentException("Null argument(s).");
589                }
590                
591                Map<Component,Set<Component>> mapSingleMultiComps = new HashMap<Component,Set<Component>>();
592                for (ProteinModification mod : modifications) {
593                        ModificationCondition condition = mod.getCondition();
594                        for (Component comp : condition.getComponents()) {
595                                for (String pdbccId : comp.getPdbccIds()) {
596                                        Component single = Component.of(Collections.singleton(pdbccId), 
597                                                        comp.isNTerminal(), comp.isCTerminal());
598                                        Set<Component> mult = mapSingleMultiComps.get(single);
599                                        if (mult == null) {
600                                                mult = new HashSet<Component>();
601                                                mapSingleMultiComps.put(single, mult);
602                                        }
603                                        mult.add(comp);
604                                }
605                        }
606                }
607                
608                {
609                        // ligands
610                        Set<Component> ligandsWildCard = mapSingleMultiComps.get(
611                                        Component.of("*"));
612                        for (Group group : ligands) {
613                                String pdbccId = group.getPDBName().trim();
614                                Set<Component> comps = mapSingleMultiComps.get(
615                                                Component.of(pdbccId));
616                                
617                                for (Component comp : unionComponentSet(ligandsWildCard, comps)) {
618                                        Set<Group> gs = saveTo.get(comp);
619                                        if (gs==null) {
620                                                gs = new LinkedHashSet<Group>();
621                                                saveTo.put(comp, gs);
622                                        }
623                                        gs.add(group);
624                                }
625                        }
626                }
627                
628                {
629                        // residues
630                        if (residues.isEmpty()) {
631                                return;
632                        }
633                        
634                        Set<Component> residuesWildCard = mapSingleMultiComps.get(
635                                        Component.of("*"));
636                        
637                        // for all residues
638                        for (Group group : residues) {
639                                String pdbccId = group.getPDBName().trim();
640                                Set<Component> comps = mapSingleMultiComps.get(
641                                                Component.of(pdbccId));
642                                
643                                for (Component comp : unionComponentSet(residuesWildCard, comps)) {
644                                        Set<Group> gs = saveTo.get(comp);
645                                        if (gs==null) {
646                                                gs = new LinkedHashSet<Group>();
647                                                saveTo.put(comp, gs);
648                                        }
649                                        gs.add(group);
650                                }
651                        }
652
653                        // for N-terminal
654                        int nRes = residues.size();
655                        int iRes = 0;
656                        Group res;
657                        do {
658                                // for all ligands on N terminal and the first residue
659                                res = residues.get(iRes++);
660
661                                Set<Component> nTermWildCard = mapSingleMultiComps.get(
662                                                Component.of("*", true, false));
663
664                                Set<Component> comps = mapSingleMultiComps.get(
665                                                Component.of(res.getPDBName(), true, false));
666                                
667                                for (Component comp : unionComponentSet(nTermWildCard, comps)) {
668                                        Set<Group> gs = saveTo.get(comp);
669                                        if (gs==null) {
670                                                gs = new LinkedHashSet<Group>();
671                                                saveTo.put(comp, gs);
672                                        }
673                                        gs.add(res);
674                                }
675                        } while (iRes<nRes && ligands.contains(res));
676                        
677                        // for C-terminal
678                        iRes = residues.size()-1;
679                        do {
680                                // for all ligands on C terminal and the last residue
681                                res = residues.get(iRes--);
682
683                                Set<Component> cTermWildCard = mapSingleMultiComps.get(
684                                                Component.of("*", false, true));
685                                
686                                Set<Component> comps = mapSingleMultiComps.get(
687                                                Component.of(res.getPDBName(), false, true));
688
689                                for (Component comp : unionComponentSet(cTermWildCard, comps)) {
690                                        Set<Group> gs = saveTo.get(comp);
691                                        if (gs==null) {
692                                                gs = new LinkedHashSet<Group>();
693                                                saveTo.put(comp, gs);
694                                        }
695                                        gs.add(res);
696                                }
697                        } while (iRes>=0 && ligands.contains(res));
698                }
699        }

701        private Set<Component> unionComponentSet(Set<Component> set1, Set<Component> set2) {
702                if (set1 == null && set2 == null)
703                        return Collections.emptySet();
704                
705                if (set1 == null)
706                        return set2;
707                
708                if (set2 == null)
709                        return set1;
710                
711                Set<Component> set = new HashSet<Component>(set1.size()+set2.size());
712                set.addAll(set1);
713                set.addAll(set2);
714                
715                return set;
716        }

718        /**
719         * Get matched atoms for all linkages.  
720         */
721        private List<List<Atom[]>> getMatchedAtomsOfLinkages(
722                        ModificationCondition condition, Map<Component, Set<Group>> mapCompGroups) {
723                List<ModificationLinkage> linkages = condition.getLinkages();
724                int nLink = linkages.size();
725
726                List<List<Atom[]>> matchedAtomsOfLinkages = 
727                                new ArrayList<List<Atom[]>>(nLink);
728                
729                for (int iLink=0; iLink<nLink; iLink++) {
730                        ModificationLinkage linkage = linkages.get(iLink);
731                        Component comp1 = linkage.getComponent1();
732                        Component comp2 = linkage.getComponent2();
733
734//                      boolean isAA1 = comp1.;
735//                      boolean isAA2 = comp2.getType()==true;
736                        
737                        Set<Group> groups1 = mapCompGroups.get(comp1);
738                        Set<Group> groups2 = mapCompGroups.get(comp2);
739                        
740                        List<Atom[]> list = new ArrayList<Atom[]>();
741
742                        List<String> potentialNamesOfAtomOnGroup1 = linkage.getPDBNameOfPotentialAtomsOnComponent1();
743                        for (String name : potentialNamesOfAtomOnGroup1) {
744                                if (name.equals("*")) {
745                                        // wildcard
746                                        potentialNamesOfAtomOnGroup1 = null; // search all atoms
747                                        break;
748                                }
749                        }
750                        
751                        List<String> potentialNamesOfAtomOnGroup2 = linkage.getPDBNameOfPotentialAtomsOnComponent2();
752                        for (String name : potentialNamesOfAtomOnGroup2) {
753                                if (name.equals("*")) {
754                                        // wildcard
755                                        potentialNamesOfAtomOnGroup2 = null; // search all atoms
756                                        break;
757                                }
758                        }
759
760                        for (Group g1 : groups1) {
761                                for (Group g2 : groups2) {
762                                        if (g1.equals(g2)) {
763                                                continue;
764                                        }
765
766                                        // only for wildcard match of two residues
767                                        boolean ignoreNCLinkage = 
768                                                potentialNamesOfAtomOnGroup1 == null &&
769                                                potentialNamesOfAtomOnGroup2 == null &&
770                                                residues.contains(g1) &&
771                                                residues.contains(g2);
772                
773                                        Atom[] atoms = StructureUtil.findNearestAtomLinkage(
774                                                        g1, g2, 
775                                                        potentialNamesOfAtomOnGroup1,
776                                                        potentialNamesOfAtomOnGroup2,
777                                                        ignoreNCLinkage,
778                                                        bondLengthTolerance);
779                                        if (atoms!=null) {
780                                                list.add(atoms);
781                                        }
782                                }
783                        }
784                                
785                        if (list.isEmpty()) {
786                                // broken linkage
787                                break;
788                        }
789        
790                        matchedAtomsOfLinkages.add(list);
791                }
792                
793                return matchedAtomsOfLinkages;
794        }

796        /**
797         * Assembly the matched linkages.
798         * @param matchedAtomsOfLinkages
799         * @param mod
800         * @param condition
801         * @param ret ModifiedCompound will be stored here.
802         */
803        private void assembleLinkages(List<List<Atom[]>> matchedAtomsOfLinkages,
804                        ProteinModification mod, List<ModifiedCompound> ret) {
805                ModificationCondition condition = mod.getCondition();
806                List<ModificationLinkage> modLinks = condition.getLinkages();
807                
808                int nLink = matchedAtomsOfLinkages.size();
809                int[] indices = new int[nLink];
810                Set<ModifiedCompound> identifiedCompounds = new HashSet<ModifiedCompound>();
811                while (indices[0]<matchedAtomsOfLinkages.get(0).size()) {
812                        List<Atom[]> atomLinkages = new ArrayList<Atom[]>(nLink);
813                        for (int iLink=0; iLink<nLink; iLink++) {
814                                Atom[] atoms = matchedAtomsOfLinkages.get(iLink).get(indices[iLink]);
815                                atomLinkages.add(atoms);
816                        }
817                        if (matchLinkages(modLinks, atomLinkages)) {
818                                // matched
819                                
820                                int n = atomLinkages.size();
821                                List<StructureAtomLinkage> linkages = new ArrayList<StructureAtomLinkage>(n);
822                                for (int i=0; i<n; i++) {
823                                        Atom[] linkage = atomLinkages.get(i);
824                                        StructureAtomLinkage link = StructureUtil.getStructureAtomLinkage(
825                                                        linkage[0], residues.contains(linkage[0].getGroup()),
826                                                        linkage[1], residues.contains(linkage[1].getGroup()));
827                                        linkages.add(link);
828                                }
829                                
830                                ModifiedCompound mc = new ModifiedCompoundImpl(mod, linkages);
831                                if (!identifiedCompounds.contains(mc)) {
832                                        ret.add(mc);
833                                        identifiedCompounds.add(mc);
834                                }
835                        }
836                        
837                        // indices++ (e.g. [0,0,1]=>[0,0,2]=>[1,2,0])
838                        int i = nLink-1;
839                        while (i>=0) {
840                                if (i==0 || indices[i]<matchedAtomsOfLinkages.get(i).size()-1) {
841                                        indices[i]++;
842                                        break;
843                                } else {
844                                        indices[i] = 0;
845                                        i--;
846                                }
847                        }
848                }
849        }

851        /**
852         * 
853         * @param linkages
854         * @param atomLinkages
855         * @return true if atomLinkages satisfy the condition; false, otherwise.
856         */
857        private boolean matchLinkages(List<ModificationLinkage> linkages, 
858                        List<Atom[]> atomLinkages) {
859                int nLink = linkages.size();
860                if (nLink != atomLinkages.size()) {
861                        return false;
862                }
863                for (int i=0; i<nLink-1; i++) {
864                        ModificationLinkage link1 = linkages.get(i);
865                        Atom[] atoms1 = atomLinkages.get(i);
866                        for (int j=i+1; j<nLink; j++) {
867                                ModificationLinkage link2 = linkages.get(j);
868                                Atom[] atoms2 = atomLinkages.get(j);
869                                
870                                // check components
871                                if (((link1.getIndexOfComponent1()==link2.getIndexOfComponent1())
872                                                        != (atoms1[0].getGroup().equals(atoms2[0].getGroup())))
873                                        || ((link1.getIndexOfComponent1()==link2.getIndexOfComponent2())
874                                                        != (atoms1[0].getGroup().equals(atoms2[1].getGroup())))
875                                        || ((link1.getIndexOfComponent2()==link2.getIndexOfComponent1())
876                                                        != (atoms1[1].getGroup().equals(atoms2[0].getGroup())))
877                                        || ((link1.getIndexOfComponent2()==link2.getIndexOfComponent2())
878                                                        != (atoms1[1].getGroup().equals(atoms2[1].getGroup())))) {
879                                        return false;
880                                }
881                                
882                                // check atoms
883                                String label11 = link1.getLabelOfAtomOnComponent1();
884                                String label12 = link1.getLabelOfAtomOnComponent2();
885                                String label21 = link2.getLabelOfAtomOnComponent1();
886                                String label22 = link2.getLabelOfAtomOnComponent2();
887                                if ((label11!=null && label21!=null && label11.equals(label21))
888                                                        != (atoms1[0].equals(atoms2[0]))
889                                         || (label11!=null && label22!=null && label11.equals(label22))
890                                                        != (atoms1[0].equals(atoms2[1]))
891                                         || (label12!=null && label21!=null && label12.equals(label21))
892                                                        != (atoms1[1].equals(atoms2[0]))
893                                         || (label12!=null && label22!=null && label12.equals(label22))
894                                                        != (atoms1[1].equals(atoms2[1]))) {
895                                        return false;
896                                }
897                        }
898                }
899                
900                return true;
901        }
