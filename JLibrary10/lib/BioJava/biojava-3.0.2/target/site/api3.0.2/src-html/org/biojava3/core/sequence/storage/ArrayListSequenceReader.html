045/**
046 * Stores a Sequence as a collection of compounds in an ArrayList
047 *
048 * @param <C>
049 */
050public class ArrayListSequenceReader<C extends Compound> implements SequenceReader<C> {
051
052    private CompoundSet<C> compoundSet;
053    private ArrayList<C> parsedCompounds = new ArrayList<C>();
054
055    private volatile Integer hashcode = null;
056
057    /**
058     *
059     */
060    public ArrayListSequenceReader() {
061        //Do nothing
062    }
063
064    /**
065     *
066     * @param compounds
067     * @param compoundSet
068     */
069    public ArrayListSequenceReader(List<C> compounds, CompoundSet<C> compoundSet) {
070        setCompoundSet(compoundSet);
071        setContents(compounds);
072    }
073
074    /**
075     *
076     * @param sequence
077     * @param compoundSet
078     */
079    public ArrayListSequenceReader(String sequence, CompoundSet<C> compoundSet) {
080        setCompoundSet(compoundSet);
081        setContents(sequence);
082    }
083
084    /**
085     *
086     * @return
087     */
088    public String getSequenceAsString() {
089        return getSequenceAsString(1, getLength(), Strand.POSITIVE);
090    }
091
092    /**
093     *
094     * @param begin
095     * @param end
096     * @param strand
097     * @return
098     */
099    public String getSequenceAsString(Integer begin, Integer end, Strand strand) {
100        // TODO Optimise/cache.
101        SequenceAsStringHelper<C> sequenceAsStringHelper = new SequenceAsStringHelper<C>();
102        return sequenceAsStringHelper.getSequenceAsString(this.parsedCompounds, compoundSet, begin, end, strand);
103    }
104
105    /**
106     *
107     * @return
108     */
109    public List<C> getAsList() {
110        return this.parsedCompounds;
111    }
112
113    /**
114     *
115     * @param position
116     * @return
117     */
118    public C getCompoundAt(int position) {
119        return this.parsedCompounds.get(position - 1);
120    }
121
122    /**
123     *
124     * @param compound
125     * @return
126     */
127    public int getIndexOf(C compound) {
128        return this.parsedCompounds.indexOf(compound) + 1;
129    }
130
131    /**
132     *
133     * @param compound
134     * @return
135     */
136    public int getLastIndexOf(C compound) {
137        return this.parsedCompounds.lastIndexOf(compound) + 1;
138    }
139
140    /**
141     *
142     * @return
143     */
144    public int getLength() {
145        return this.parsedCompounds.size();
146    }
147
148    /**
149     *
150     * @return
151     */
152    public Iterator<C> iterator() {
153        return this.parsedCompounds.iterator();
154    }
155
156    /**
157     *
158     * @param compoundSet
159     */
160    public void setCompoundSet(CompoundSet<C> compoundSet) {
161        this.compoundSet = compoundSet;
162    }
163
164    /**
165     *
166     * @return
167     */
168    public CompoundSet<C> getCompoundSet() {
169        return compoundSet;
170    }
171
172    /**
173     *
174     * @param sequence
175     */
176    public void setContents(String sequence) {
177        // Horrendously inefficient - pretty much the way the old BJ did things.
178        // TODO Should be optimised.
179        this.parsedCompounds.clear();
180        hashcode = null;
181        int maxCompoundLength = compoundSet.getMaxSingleCompoundStringLength();
182        boolean maxCompundLengthEqual1 = true;
183        if (maxCompoundLength > 1) {
184            maxCompundLengthEqual1 = false;
185        }
186        int length = sequence.length();
187        parsedCompounds.ensureCapacity(length); //get the array size correct
188        for (int i = 0; i < length;) {
189            String compoundStr = null;
190            C compound = null;
191            if (maxCompundLengthEqual1) { // trying to save some steps where typically the answer is 1 so avoid complicated for loop
192                compoundStr = sequence.substring(i, i + 1);
193                compound = compoundSet.getCompoundForString(compoundStr);
194            } else {
195                for (int compoundStrLength = 1; compound == null && compoundStrLength <= maxCompoundLength; compoundStrLength++) {
196                    compoundStr = sequence.substring(i, i + compoundStrLength);
197                    compound = compoundSet.getCompoundForString(compoundStr);
198                }
199            }
200            if (compound == null) {
201                throw new CompoundNotFoundError("Cannot find compound for: " + compoundStr);
202            } else {
203                i += compoundStr.length();
204            }
205            this.parsedCompounds.add(compound);
206        }
207        parsedCompounds.trimToSize(); // just in case it increases capacity free up extra memory
208    }
209
210    /**
211     *
212     * @param list
213     */
214    public void setContents(List<C> list) {
215        parsedCompounds.clear();
216        for (C c : list) {
217            parsedCompounds.add(c);
218        }
219    }
220
221    /**
222     *
223     * @param bioBegin
224     * @param bioEnd
225     * @return
226     */
227    public SequenceView<C> getSubSequence(final Integer bioBegin, final Integer bioEnd) {
228        return new SequenceProxyView<C>(ArrayListSequenceReader.this, bioBegin, bioEnd);
229    }
230
231    /**
232     *
233     * @return
234     */
235    public AccessionID getAccession() {
236        throw new UnsupportedOperationException("Not supported yet.");
237    }
238
239    /**
240     *
241     * @param compounds
242     * @return
243     */
244    public int countCompounds(C... compounds) {
245        return SequenceMixin.countCompounds(this, compounds);
246    }
247
248    /**
249     *
250     * @return
251     */
252    @Override
253    public SequenceView<C> getInverse() {
254        return SequenceMixin.inverse(this);
255    }
256
257    @Override
258    public int hashCode() {
259        if(hashcode == null) {
260            int s = Hashcoder.SEED;
261            s = Hashcoder.hash(s, parsedCompounds);
262            s = Hashcoder.hash(s, compoundSet);
263            hashcode = s;
264        }
265        return hashcode;
266    }
267
268    @Override
269    @SuppressWarnings("unchecked")
270    public boolean equals(Object o) {
271        if(Equals.classEqual(this, o)) {
272            ArrayListSequenceReader<C> that = (ArrayListSequenceReader<C>)o;
273            return  Equals.equal(parsedCompounds, that.parsedCompounds) &&
274                    Equals.equal(compoundSet, that.compoundSet);
275        }
276        return false;
277    }
278}
