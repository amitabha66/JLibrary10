001package org.biojava.bio.structure.align.model;
002
003import java.io.Serializable;
004
005import org.biojava.bio.structure.jama.Matrix;
006
007/** A class to represent a FATCAT AFP
008 *
009 * @author Andreas Prlic
010 *
011 */
012
013public class AFP implements Serializable {
014
015   /**
016    * 
017    */
018   private static final long serialVersionUID = 3901209995477111829L;
019   int p1;
020   int p2;
021   int fragLen;
022   double rmsd;
023   Matrix m;
024   double[] t;
025   double score;
026
027   long id;
028
029   public String toString(){
030
031      // we use the metric of
032
033      // Manfred J. Sippl
034      // On Distance and Similarity in Fold Space
035      // Bioinformatics, 24, pp. 872-873  (2008)
036
037
038      StringBuffer buf = new StringBuffer();
039      buf.append("AFP: p1:");
040      buf.append(p1);
041      buf.append(" p2: ");
042      buf.append(p2);
043      buf.append(" len " );
044      buf.append(fragLen);
045      buf.append(" rmsd ");
046      buf.append(rmsd);
047      buf.append(" score ");
048      buf.append(score);
049      return buf.toString();
050   }
051
052 
053   public long getId()
054   {
055      return id;
056   }
057
058   public void setId(long id)
059   {
060      this.id = id;
061   }
062
063   public int getP1() {
064      return p1;
065   }
066   public void setP1(int p1) {
067      this.p1 = p1;
068   }
069   public int getP2() {
070      return p2;
071   }
072   public void setP2(int p2) {
073      this.p2 = p2;
074   }
075   public int getFragLen() {
076      return fragLen;
077   }
078   public void setFragLen(int fragLen) {
079      this.fragLen = fragLen;
080   }
081   public double getRmsd() {
082      return rmsd;
083   }
084   public void setRmsd(double rmsd) {
085      this.rmsd = rmsd;
086   }
087   public Matrix getM() {
088      return m;
089   }
090   public void setM(Matrix m) {
091      this.m = m;
092   }
093   public double[] getT() {
094      return t;
095   }
096   public void setT(double[] t) {
097      this.t = t;
098   }
099   public double getScore() {
100      return score;
101   }
102   public void setScore(double score) {
103      this.score = score;
104   }
105
106
107
108
109}
