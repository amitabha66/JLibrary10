022package org.biojava.bio.structure.gui.util.color;
023
024import java.awt.BasicStroke;
025import java.awt.Dimension;
026import java.awt.Graphics;
027import java.awt.Graphics2D;
028
029import javax.swing.JPanel;
030
031public class GradientPanel extends JPanel {
032        private static final long serialVersionUID = -6387922432121206731L;
033        private ContinuousColorMapper mapper;
034        private double min, max;
035        
036        
037        public GradientPanel(ContinuousColorMapper mapper, double min, double max) {
038                this.min = min; 
039                this.max = max;
040                this.mapper = mapper;
041                this.setPreferredSize(new Dimension(100,20));
042        }
043        
044        public void paintComponent(Graphics g) {
045                Graphics2D g2 = (Graphics2D) g;
046                int w = getWidth();
047                int h = getHeight();
048                
049                g2.setStroke(new BasicStroke(1.0f));
050                for(int i=0;i<w;i++) {
051                        double val = (max-min)*i/w+min;
052                        g2.setColor(mapper.getColor(val));
053                        g2.drawLine(i, 0, i, h);
054                }
055        }
056        
057}
