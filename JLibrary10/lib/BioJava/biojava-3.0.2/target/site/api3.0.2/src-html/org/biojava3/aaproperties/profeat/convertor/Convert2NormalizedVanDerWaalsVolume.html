001package org.biojava3.aaproperties.profeat.convertor;
002
003public class Convert2NormalizedVanDerWaalsVolume extends Convertor{
004        /**
005         * Class for the conversion of protein sequence into normalized van der waals volume
006         */
007        
008        @Override
009        public char convert(char c){
010                switch(c){
011                case 'G':case 'A':case 'S':case 'T':case 'P':case 'D': case 'C':
012                        return group1;//Volume Range 0-2.78
013                case 'N':case 'V':case 'E':case 'Q':case 'I':case 'L':
014                        return group2;//Volume Range 2.95-4.0
015                case 'M':case 'H':case 'K':case 'F':case 'R':case 'Y':case 'W':
016                        return group3;//Volume Range 4.03-8.08
017                default:
018                        return unknownGroup;//Non-standard AA
019                }
020        }
021        private static String[] subCategory = {"Range_0-2.78", "Range_2.95-4.0", "Range_4.03-8.08"};
022        @Override
023        public String[] getGrouping(){return subCategory;}
024        @Override
025        public String getAttribute(){return "Normalized van der waals Volume";}
026}
