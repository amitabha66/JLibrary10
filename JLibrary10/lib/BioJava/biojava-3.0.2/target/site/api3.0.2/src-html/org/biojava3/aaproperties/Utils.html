001package org.biojava3.aaproperties;
002
003import java.util.HashSet;
004import java.util.Set;
005
006/**
007 * This is a utility class that contains utility methods which will facilitates the coding of other methods
008 * 
009 * @author kohchuanhock
010 * @version 2011.08.22
011 * @since 3.0.2
012 */
013public class Utils {
014
015        /**
016         * Returns a value with the desired number of decimal places.
017         * 
018         * @param d
019         *              value to round
020         * @param c
021         *              number of decimal places desired. 
022         *              Must be greater or equal to zero, otherwise, the given value d would be returned without any modification.
023         * @return
024         *              a value with the given number of decimal places.
025         */
026        public final static double roundToDecimals(double d, int c) {
027                if(c < 0) return d;
028                double p = Math.pow(10,c);
029                d = d * p;
030                double tmp = Math.round(d);
031                return tmp/p;
032        }
033        
034        /**
035         * Checks if given sequence contains invalid characters. Returns true if invalid characters are found, else return false.
036         * Note that any characters are deemed as valid only if it is found in cSet.
037         * 
038         * @param sequence
039         *              protein sequence to be check.
040         * @param cSet
041         *              the set of characters that are deemed valid.
042         * @return
043         *              true if invalid characters are found, else return false.
044         */
045        public final static boolean doesSequenceContainInvalidChar(String sequence, Set<Character> cSet){
046                for(char c:sequence.toCharArray()){
047                        if(cSet.contains(c) == false) return true;
048                }
049                return false;
050        }
051        
052        /**
053         * Return the number of invalid characters in sequence.
054         * 
055         * @param sequence
056         *              protein sequence to count for invalid characters.
057         * @param cSet
058         *              the set of characters that are deemed valid.
059         * @param ignoreCase
060         *              indicates if cases should be ignored
061         * @return
062         *              the number of invalid characters in sequence.
063         */
064        public final static int getNumberOfInvalidChar(String sequence, Set<Character> cSet, boolean ignoreCase){
065                int total = 0;
066                char[] cArray;
067                if(ignoreCase) cArray = sequence.toUpperCase().toCharArray();
068                else cArray = sequence.toCharArray();
069                if(cSet == null) cSet = PeptideProperties.standardAASet;
070                for(char c:cArray){
071                        if(cSet.contains(c) == false) total++;
072                }
073                return total;
074        }
075        
076        /**
077         * Returns a new sequence with all invalid characters being replaced by '-'.
078         * Note that any character outside of the 20 standard protein amino acid codes are considered as invalid.
079         * 
080         * @param sequence
081         *              protein sequence to be clean
082         * @param cSet
083         *              user defined characters that are valid. Can be null. If null, then 20 standard protein amino acid codes will be considered as valid.
084         * @return
085         *              a new sequence with all invalid characters being replaced by '-'.
086         */
087        public final static String cleanSequence(String sequence, Set<Character> cSet){
088                Set<Character> invalidCharSet = new HashSet<Character>();
089                String cleanSeq = "";
090                if(cSet == null) cSet = PeptideProperties.standardAASet; 
091                for(char c:sequence.toCharArray()){
092                        if(cSet.contains(c) == false){
093                                cleanSeq += "-";
094                                invalidCharSet.add(c);
095                        }else{
096                                cleanSeq += c;
097                        }
098                }
099                int count = 0;
100                for(char c:invalidCharSet){
101                        if(count != 0) System.err.print(",");
102                        System.err.print("\'" + c + "\'");
103                        count++;
104                }
105                System.err.println(" are being replaced with '-'");
106                return cleanSeq;
107        }
108        
109        /**
110         * Checks if the sequence contains invalid characters. 
111         * Note that any character outside of the 20 standard protein amino acid codes are considered as invalid.
112         * If yes, it will return a new sequence where invalid characters are replaced with '-'.
113         * If no, it will simply return the input sequence. 
114         * 
115         * @param sequence
116         *              protein sequence to be check for invalid characters.
117         * @return
118         *              a sequence with no invalid characters.
119         */
120        public static final String checkSequence(String sequence){
121                return checkSequence(sequence, null);
122        }
123        
124        /**
125         * Checks if the sequence contains invalid characters. 
126         * Note that any character outside of the 20 standard protein amino acid codes are considered as invalid.
127         * If yes, it will return a new sequence where invalid characters are replaced with '-'.
128         * If no, it will simply return the input sequence. 
129         * 
130         * @param sequence
131         *              protein sequence to be check for invalid characters.
132         * @param cSet
133         *              character set which define the valid characters.
134         * @return
135         *              a sequence with no invalid characters.
136         */
137        public static final String checkSequence(String sequence, Set<Character> cSet){
138                boolean containInvalid = false;
139                if(cSet != null){
140                        containInvalid = sequence != null && doesSequenceContainInvalidChar(sequence, cSet);
141                }else{
142                        containInvalid = sequence != null && doesSequenceContainInvalidChar(sequence, PeptideProperties.standardAASet);
143                }
144                if(containInvalid){
145                        String cSeq = cleanSequence(sequence, cSet);
146                        System.err.println("Warning: There exists invalid characters in the sequence. Computed results might not be precise.");
147                        System.err.println("To remove this warning: Please use org.biojava3.aaproperties.Utils.cleanSequence to clean sequence.");
148                        return cSeq;
149                }
150                else{
151                        return sequence;
152                }
153        }
154}
