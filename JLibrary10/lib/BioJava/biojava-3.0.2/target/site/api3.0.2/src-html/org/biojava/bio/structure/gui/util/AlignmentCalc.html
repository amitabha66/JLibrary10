023package org.biojava.bio.structure.gui.util;
024
025
026import java.util.logging.Logger;
027
028import org.biojava.bio.structure.Structure;
029import org.biojava.bio.structure.StructureException;
030import org.biojava.bio.structure.align.ClusterAltAligs;
031import org.biojava.bio.structure.align.StructurePairAligner;
032import org.biojava.bio.structure.align.gui.AlignmentGui;
033import org.biojava.bio.structure.align.pairwise.AlternativeAlignment;
034
035
036
037/** A class that obtains two structures via DAS and aligns them
038 *  This is done in a separate thread.
039 *  It is possible to register Event listeners to get notification of when the download has finished.
040 *  
041 * @author Andreas Prlic
042 * @since 1.7
043 * @version %I% %G%
044 */
045public class AlignmentCalc implements Runnable {
046
047
048        public static Logger logger =  Logger.getLogger("org.biojava");
049
050        boolean interrupted = false;
051
052        String pdb1;
053        String pdb2;
054        String chain1;
055        String chain2;
056
057        Structure structure1;
058        Structure structure2;
059
060        AlignmentGui parent;
061
062
063        /** requests an alignment of pdb1 vs pdb 2.
064         * Chain 1 and chain2 are optional.
065         * If they are empty strings, they are ignored
066         * @param parent the alignment gui frame that interacts with this class          
067         * @param s1 structure 1
068         * @param s2 structure 2
069         */
070        public AlignmentCalc(AlignmentGui parent, Structure s1, Structure s2 ) {
071
072                this.parent= parent;
073
074                structure1 = s1;
075                structure2 = s2;
076
077        }
078
079        public void run() {
080
081                // both structure have been downloaded, now calculate the alignment ...
082
083
084                StructurePairAligner aligner = new StructurePairAligner();
085                aligner.setDebug(true);
086                try {
087                        aligner.align(structure1,structure2);
088                } catch (StructureException e){
089                        logger.warning(e.getMessage());
090
091                }
092
093
094
095                AlternativeAlignment[] aligs = aligner.getAlignments();
096                //cluster similar results together 
097                ClusterAltAligs.cluster(aligs);
098                showAlignment(aligner,aligs);
099
100                //logger.info("done!");
101
102                parent.notifyCalcFinished();
103
104        }
105
106
107
108        private void showAlignment(StructurePairAligner alignment, AlternativeAlignment[] aligs) {
109                AlternativeAlignmentFrame frame = new AlternativeAlignmentFrame(structure1, structure2);
110                frame.setStructurePairAligner(alignment);
111                frame.setAlternativeAlignments(aligs);
112                frame.pack();
113                frame.setVisible(true);
114
115
116
117        }
118
119        /** stops what is currently happening and does not continue
120         * 
121         *
122         */
123        public void interrupt() {
124                interrupted = true;
125        }
126
127        public void cleanup() {
128
129                parent.notifyCalcFinished();
130
131                parent=null;
132                // cleanup...
133
134                structure1 = null;
135                structure2 = null;
136
137        }
138
139
140
141}
