040    /**
041     * Defines a traceback pointer for the three edit operations: substitution (match/replacement of a query compound
042     * with a target compound), deletion (removal of a query compound leaving a gap in the target sequence), and
043     * insertion (addition of a target compound opening a gap in the query sequence).
044     */
045    public enum Last {
046        SUBSTITUTION,
047        DELETION,
048        INSERTION
049    }

051    /**
052     * Defines a 'cut' row for divide-and-conquer alignment in which a new anchor is found.
053     */
054    public static class Cut {
055
056        private int queryIndex;
057        private int[][] targetIndices, tiLast, ti1, ti2;
058
059        public Cut(int queryIndex, int[] dim) {
060            this.queryIndex = queryIndex;
061            targetIndices = ti1 = new int[dim[1]][dim[2]];
062            ti2 = new int[dim[1]][dim[2]];
063        }
064
065        public int getQueryIndex() {
066            return queryIndex;
067        }
068
069        public int getTargetIndex(int z) {
070            return targetIndices[targetIndices.length - 1][z];
071        }
072
073        public void update(int x, int[] subproblem, Last[][] pointers) {
074            if (pointers[subproblem[1]].length == 1) {
075                if (queryIndex == x - 1) {
076                    updateLinearInitial(subproblem, pointers);
077                } else if (queryIndex < x) {
078                    updateLinearAdvance(subproblem, pointers);
079                }
080            } else {
081                if (queryIndex == x - 1) {
082                    updateInitial(subproblem, pointers);
083                } else if (queryIndex < x) {
084                    updateAdvance(subproblem, pointers);
085                }
086            }
087        }
088
089        private void updateAdvance(int[] subproblem, Last[][] pointers) {
090            tiLast = targetIndices;
091            targetIndices = (targetIndices == ti2) ? ti1 : ti2;
092            for (int y = subproblem[1]; y <= subproblem[3]; y++) {
093                if (pointers[y][0] != null) {
094                    targetIndices[y][0] = tiLast[y - 1][pointers[y][0].ordinal()];
095                }
096                if (pointers[y][1] != null) {
097                    targetIndices[y][1] = tiLast[y][pointers[y][1].ordinal()];
098                }
099                if (pointers[y][2] != null) {
100                    targetIndices[y][2] = targetIndices[y - 1][pointers[y][2].ordinal()];
101                }
102            }
103        }
104
105        private void updateInitial(int[] subproblem, Last[][] pointers) {
106            for (int y = subproblem[1]; y <= subproblem[3]; y++) {
107                if (pointers[y][0] != null) {
108                    targetIndices[y][0] = y - 1;
109                }
110                if (pointers[y][1] != null) {
111                    targetIndices[y][1] = y;
112                }
113                if (pointers[y][2] != null) {
114                    targetIndices[y][2] = targetIndices[y - 1][2];
115                }
116            }
117        }
118
119        private void updateLinearAdvance(int[] subproblem, Last[][] pointers) {
120            tiLast = targetIndices;
121            targetIndices = (targetIndices == ti2) ? ti1 : ti2;
122            for (int y = subproblem[1]; y <= subproblem[3]; y++) {
123                switch (pointers[y][0]) {
124                case DELETION:
125                    targetIndices[y][0] = tiLast[y][0];
126                    break;
127                case SUBSTITUTION:
128                    targetIndices[y][0] = tiLast[y - 1][0];
129                    break;
130                case INSERTION:
131                    targetIndices[y][0] = targetIndices[y - 1][0];
132                }
133            }
134        }
135
136        private void updateLinearInitial(int[] subproblem, Last[][] pointers) {
137            for (int y = subproblem[1]; y <= subproblem[3]; y++) {
138                if (pointers[y][0] != null) {
139                    switch (pointers[y][0]) {
140                    case DELETION:
141                        targetIndices[y][0] = y;
142                        break;
143                    case SUBSTITUTION:
144                        targetIndices[y][0] = y - 1;
145                        break;
146                    case INSERTION:
147                        targetIndices[y][0] = targetIndices[y - 1][0];
148                    }
149                }
150            }
151        }
152
153    }

157    public static short addAnchors(Cut[] cuts, short[] scores, boolean addScore, int[] anchors) {
158        int zMax = 0, subscore = scores[0];
159        for (int z = 1; z < scores.length; z++) {
160            if (scores[z] > subscore) {
161                zMax = z;
162                subscore = scores[z];
163            }
164        }
165        for (Cut c : cuts) {
166            anchors[c.getQueryIndex()] = c.getTargetIndex(zMax);
167        }
168        return addScore ? (short) subscore : 0;
169    }

171    public static Cut[] getCuts(int k, int[] subproblem, int[] dim, boolean anchor0) {
172        Cut[] cuts;
173        int m = subproblem[2] - subproblem[0] - (anchor0 ? 1 : 0);
174        if (k < m) {
175            cuts = new Cut[k];
176            for (int i = 0; i < k; i++) {
177                cuts[i] = new Cut(subproblem[0] + ((i + 1) * (m + 1) * k / (k + 1)), dim);
178            }
179        } else {
180            cuts = new Cut[m];
181            for (int i = 0, x = subproblem[0] + (anchor0 ? 1 : 0); i < m; i++, x++) {
182                cuts[i] = new Cut(x, dim);
183            }
184        }
185        return cuts;
186    }

188    /**
189     * Returns the coordinates for the next subproblem.  The first element is the starting index in the query sequence.
190     * The second element is the starting index in the target sequence.  Similarly, the third and fourth elements are
191     * the ending indices of the query and target sequences, respectively.  When the alignment is finished, there is no
192     * next subproblem and this method returns {@code null}.
193     *
194     * @param anchors current list of anchors
195     * @return the coordinates for the next subproblem
196     */
197    public static int[] getNextSubproblem(int[] anchors) {
198        int[] subproblem = new int[4];
199        int x = 0;
200
201        // find first unanchored x (query sequence index)
202        while (x < anchors.length && anchors[x] >= 0) {
203            x++;
204        }
205        if (x == anchors.length) {
206            return null; // no unanchored x, therefore alignment is complete
207        }
208
209        // save first point or last anchor as starting point
210        if (x == 0) {
211            subproblem[0] = 0;
212            subproblem[1] = 0;
213            x++;
214        } else {
215            subproblem[0] = x - 1;
216            subproblem[1] = anchors[x - 1];
217        }
218
219        // find next anchored x
220        while (x < anchors.length && anchors[x] < 0) {
221            x++;
222        }
223        subproblem[2] = x;
224        subproblem[3] = anchors[x];
225
226        return subproblem;
227    }

236    // scores alignment for a given position in both sequences
237    public static Last[] setScorePoint(int x, int y, short gop, short gep, short sub, short[][][] scores) {
238        Last[] pointers = new Last[3];
239
240        // substitution
241        if (scores[x - 1][y - 1][1] >= scores[x - 1][y - 1][0] && scores[x - 1][y - 1][1] >= scores[x - 1][y - 1][2]) {
242            scores[x][y][0] = (short) (scores[x - 1][y - 1][1] + sub);
243            pointers[0] = Last.DELETION;
244        } else if (scores[x - 1][y - 1][0] >= scores[x - 1][y - 1][2]) {
245            scores[x][y][0] = (short) (scores[x - 1][y - 1][0] + sub);
246            pointers[0] = Last.SUBSTITUTION;
247        } else {
248            scores[x][y][0] = (short) (scores[x - 1][y - 1][2] + sub);
249            pointers[0] = Last.INSERTION;
250        }
251
252        // deletion
253        if (scores[x - 1][y][1] >= scores[x - 1][y][0] + gop) {
254            scores[x][y][1] = (short) (scores[x - 1][y][1] + gep);
255            pointers[1] = Last.DELETION;
256        } else {
257            scores[x][y][1] = (short) (scores[x - 1][y][0] + gop + gep);
258            pointers[1] = Last.SUBSTITUTION;
259        }
260
261        // insertion
262        if (scores[x][y - 1][0] + gop >= scores[x][y - 1][2]) {
263            scores[x][y][2] = (short) (scores[x][y - 1][0] + gop + gep);
264            pointers[2] = Last.SUBSTITUTION;
265        } else {
266            scores[x][y][2] = (short) (scores[x][y - 1][2] + gep);
267            pointers[2] = Last.INSERTION;
268        }
269
270        return pointers;
271    }

273    // scores alignment for a given position in both sequences for linear gap penalty
274    public static Last setScorePoint(int x, int y, short gep, short sub, short[][][] scores) {
275        int d = scores[x - 1][y][0] + gep, i = scores[x][y - 1][0] + gep, s = scores[x - 1][y - 1][0] + sub;
276        if (d >= s && d >= i) {
277            scores[x][y][0] = (short) d;
278            return Last.DELETION;
279        } else if (s >= i) {
280            scores[x][y][0] = (short) s;
281            return Last.SUBSTITUTION;
282        } else {
283            scores[x][y][0] = (short) i;
284            return Last.INSERTION;
285        }
286    }

300    // scores global alignment for a given position in the query sequence
301    public static Last[][] setScoreVector(int x, int xb, int yb, int ye, short gop, short gep, short[] subs,
302            boolean storing, short[][][] scores) {
303        Last[][] pointers = new Last[ye + 1][];
304        short min = (short) (Short.MIN_VALUE - gop - gep);
305        if (x == xb) {
306            scores[xb][yb][1] = scores[xb][yb][2] = gop;
307            pointers[yb] = new Last[] {null, null, null};
308            Last[] insertion = new Last[] { null, null, Last.INSERTION };
309            for (int y = yb + 1; y <= ye; y++) {
310                scores[xb][y][0] = scores[xb][y][1] = min;
311                scores[xb][y][2] = (short) (scores[xb][y - 1][2] + gep);
312                pointers[y] = insertion;
313            }
314        } else {
315            if (!storing && x > xb + 1) {
316                scores[x] = scores[x - 2];
317            }
318            scores[x][yb][0] = scores[x][yb][2] = min;
319            scores[x][yb][1] = (short) (scores[x - 1][yb][1] + gep);
320            pointers[yb] = new Last[] { null, Last.DELETION, null };
321            for (int y = yb + 1; y <= ye; y++) {
322                pointers[y] = setScorePoint(x, y, gop, gep, subs[y], scores);
323            }
324        }
325        return pointers;
326    }

355                pointers[y][0] = setScorePoint(x, y, gep, subs[y], scores);

367    // scores local alignment for a given position in the query sequence
368    public static Last[][] setScoreVector(int x, int xb, int yb, int ye, short gop, short gep, short[] subs,
369            boolean storing, short[][][] scores, int[] xyMax, int score) {
370        Last[][] pointers;
371        if (x == xb) {
372            pointers = new Last[ye + 1][scores[0][0].length];
373        } else {
374            pointers = new Last[ye + 1][];
375            pointers[0] = new Last[scores[0][0].length];
376            if (!storing && x > 1) {
377                scores[x] = scores[x - 2];
378            }
379            for (int y = 1; y < scores[0].length; y++) {
380                pointers[y] = setScorePoint(x, y, gop, gep, subs[y], scores);
381                for (int z = 0; z < scores[0][0].length; z++) {
382                    if (scores[x][y][z] <= 0) {
383                        scores[x][y][z] = 0;
384                        pointers[y][z] = null;
385                    }
386                }
387                if (scores[x][y][0] > score) {
388                    xyMax[0] = x;
389                    xyMax[1] = y;
390                    score = scores[x][y][0];
391                }
392            }
393        }
394        return pointers;
395    }

404    public static Last[][] setScoreVector(int x, int xb, int yb, int ye, short gep, short[] subs, boolean storing,
405            short[][][] scores, int[] xyMax, int score) {
406        Last[][] pointers;
407        if (x == xb) {
408            pointers = new Last[ye + 1][1];
409        } else {
410            pointers = new Last[ye + 1][];
411            pointers[0] = new Last[1];
412            if (!storing && x > 1) {
413                scores[x] = scores[x - 2];
414            }
415            for (int y = 1; y < scores[x].length; y++) {
416                pointers[y][0] = setScorePoint(x, y, gep, subs[y], scores);
417                if (scores[x][y][0] <= 0) {
418                    scores[x][y][0] = 0;
419                    pointers[y][0] = null;
420                } else if (scores[x][y][0] > score) {
421                    xyMax[0] = x;
422                    xyMax[1] = y;
423                    score = scores[x][y][0];
424                }
425            }
426        }
427        return pointers;
428    }

430    /**
431     * Sets the alignment path following the list of anchors.
432     *
433     * @param anchors a complete list of anchors (input)
434     * @param sx steps for the query sequence (output)
435     * @param sy steps for the target sequence (output)
436     */
437    public static int[] setSteps(int[] anchors, List<Step> sx, List<Step> sy) {
438        for (int gap = anchors[0]; gap > 0; gap--) {
439            sx.add(Step.GAP);
440            sy.add(Step.COMPOUND);
441        }
442        for (int x = 1; x < anchors.length; x++) {
443            int change = anchors[x] - anchors[x - 1];
444            if (change == 0) {
445                sx.add(Step.COMPOUND);
446                sy.add(Step.GAP);
447            } else {
448                sx.add(Step.COMPOUND);
449                sy.add(Step.COMPOUND);
450                for (change--; change > 0; change--) {
451                    sx.add(Step.GAP);
452                    sy.add(Step.COMPOUND);
453                }
454            }
455        }
456        return new int[] {0, 0};
457    }

459    // finds alignment path through traceback matrix
460    public static int[] setSteps(Last[][][] traceback, boolean local, int[] xyMax, Last last, List<Step> sx,
461            List<Step> sy) {
462        int x = xyMax[0], y = xyMax[1];
463        boolean linear = (traceback[x][y].length == 1);
464        while (local ? (linear ? last : traceback[x][y][last.ordinal()]) != null : x > 0 || y > 0) {
465            switch (last) {
466            case DELETION:
467                sx.add(Step.COMPOUND);
468                sy.add(Step.GAP);
469                last = linear ? traceback[--x][y][0] : traceback[x--][y][1];
470                break;
471            case SUBSTITUTION:
472                sx.add(Step.COMPOUND);
473                sy.add(Step.COMPOUND);
474                last = linear ? traceback[--x][--y][0] : traceback[x--][y--][0];
475                break;
476            case INSERTION:
477                sx.add(Step.GAP);
478                sy.add(Step.COMPOUND);
479                last = linear ? traceback[x][--y][0] : traceback[x][y--][2];
480            }
481        }
482        Collections.reverse(sx);
483        Collections.reverse(sy);
484        return new int[] {x, y};
485    }

487    // finds global alignment path through traceback matrix
488    public static int[] setSteps(Last[][][] traceback, short[][][] scores, List<Step> sx, List<Step> sy) {
489        int xMax = scores.length - 1, yMax = scores[xMax].length - 1;
490        boolean linear = (traceback[xMax][yMax].length == 1);
491        Last last = linear ? traceback[xMax][yMax][0] : (scores[xMax][yMax][1] > scores[xMax][yMax][0] &&
492                scores[xMax][yMax][1] > scores[xMax][yMax][2]) ? Last.DELETION : (scores[xMax][yMax][0] >
493                scores[xMax][yMax][2]) ? Last.SUBSTITUTION : Last.INSERTION;
494        return setSteps(traceback, false, new int[] {xMax, yMax}, last, sx, sy);
495    }

497    // finds local alignment path through traceback matrix
498    public static int[] setSteps(Last[][][] traceback, int[] xyMax, List<Step> sx, List<Step> sy) {
499        return setSteps(traceback, true, xyMax, Last.SUBSTITUTION, sx, sy);
