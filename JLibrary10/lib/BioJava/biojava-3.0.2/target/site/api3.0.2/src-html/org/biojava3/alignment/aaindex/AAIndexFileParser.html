022public class AAIndexFileParser {
023
024
025        int scale = -1;
026
027        Map<String,SubstitutionMatrix<AminoAcidCompound>> matrices;
028
029        ScaledSubstitutionMatrix currentMatrix;
030        String currentRows;
031        String currentCols;
032        int currentRowPos;
033        private short[][] matrix;
034        short max;
035        short min;
036        private List<AminoAcidCompound> rows, cols;
037        boolean inMatrix;
038        boolean symmetricMatrix ;
039
040
041        public AAIndexFileParser(){
042                matrices  = new HashMap<String, SubstitutionMatrix<AminoAcidCompound>>();
043        }
044
045        /** parse an inputStream that points to an AAINDEX database file
046         * 
047         * @param inputStream
048         * @throws IOException 
049         */
050        public void parse(InputStream inputStream) throws IOException {
051
052                currentMatrix = null;
053                currentRows = "";
054                currentCols = "";
055                max = Short.MIN_VALUE;
056                min = Short.MAX_VALUE;
057                inMatrix = false;
058                
059                BufferedReader buf = new BufferedReader (new InputStreamReader (inputStream));
060                String line = null;
061                line = buf.readLine();
062
063                while (  line != null ) {
064
065                        if ( line.startsWith("//")) {
066                                finalizeMatrix();
067                                inMatrix = false;
068
069                        } else if ( line.startsWith("H ")){
070                                // a new matric!
071                                newMatrix(line);
072                        } else if ( line.startsWith("D ")) {
073                                currentMatrix.setDescription(line.substring(2));
074                        } else if ( line.startsWith("M ")) {
075                                initMatrix(line);
076                                inMatrix = true;
077                        } else if ( line.startsWith("  ")){
078                                if ( inMatrix)
079                                        processScores(line);
080                        }
081
082                        line = buf.readLine();
083                }
084
085        }
086
087
088        //  process a line such as >    -0.3     1.6     0.7     0.8    -2.6     3.0<
089        private void processScores(String line) {
090
091                String[] values = line.trim().split(" +");
092
093                // increment the current row we are talking about
094                currentRowPos++;
095
096                
097                
098                for ( int i =0 ; i < values.length ; i++){
099                
100                        if ( values[i].endsWith(".")) {
101                                values[i] = values[i] + "0";
102                        }
103
104                        // special case: MEHP950101
105                        if (values[i].equals("-")) {
106                                values[i] = "0";
107                        }
108                
109                        if ( scale == -1 ) {
110                                scale = determineScale(values[0]);                              
111                        }
112                        
113                        
114                        Float score = Float.parseFloat(values[i]);
115                        score = scale * score;
116
117                        Short s = (short) Math.round(score);
118                        
119                        matrix[currentRowPos][i] = s;
120
121                        if ( values.length < cols.size() || ( symmetricMatrix)){
122                                //System.out.println(values.length + " " + cols.size() + " " + currentRowPos + " " + i + " " +  line);
123                                
124                                matrix[i][currentRowPos] = s;
125                                
126                                symmetricMatrix = true;
127                                
128                        }
129
130                        if ( score > max)
131                                max = s;
132                        if ( score < min)
133                                min = s;
134
135
136                }
137        }
138
139        private int determineScale(String value) {
140                
141                String[] spl = value.split("\\.");
142                
143                if (spl.length <= 1)
144                        return 1;
145                
146                String digits = spl[1];
147                
148                return (int)Math.round(Math.pow(10, digits.length()));
149                
150        }
151
152        // process a line of type >M rows = ARNDCQEGHILKMFPSTWYV, cols = ARNDCQEGHILKMFPSTWYV<
153        private void initMatrix(String line) {
154                String[] spl = line.split(" ");
155
156                // trim off the final , character 
157                currentRows = spl[3].substring(0, spl[3].length()-1);
158                currentCols = spl[6];
159                currentRowPos = -1;
160
161                int nrRows = currentRows.length();
162                int nrCols = currentCols.length();
163
164                matrix = new short[nrRows][nrCols];
165
166                rows = new ArrayList<AminoAcidCompound>();
167                cols = new ArrayList<AminoAcidCompound>();
168
169
170                //System.out.println(">" + currentRows+"<");
171                AminoAcidCompoundSet compoundSet = AminoAcidCompoundSet.getAminoAcidCompoundSet();
172                for ( int i = 0 ; i < currentRows.length() ; i ++){
173                        char c = currentRows.charAt(i);
174                        AminoAcidCompound aa = compoundSet.getCompoundForString(c+""); 
175
176                        rows.add(aa);
177                }
178
179                for ( int i = 0 ; i < currentCols.length() ; i ++){
180                        char c = currentRows.charAt(i);
181                        AminoAcidCompound aa = compoundSet.getCompoundForString(c+""); 
182
183                        cols.add(aa);
184                }
185
186
187
188        
189
190                currentMatrix.setScale(scale);
191        }
192
193        
194        private void newMatrix(String line) {
195                symmetricMatrix = false;
196                scale = -1;
197                
198                currentMatrix = new ScaledSubstitutionMatrix();
199                currentMatrix.setName(line.substring(2));
200                
201                
202                //System.out.println("new Matrix " + currentMatrix.getName());
203        }
204
205        // 
206        private SubstitutionMatrix<AminoAcidCompound> finalizeMatrix() {
207
208                currentMatrix.setMatrix(matrix);
209                currentMatrix.setMax(max);
210                currentMatrix.setMin(min);
211                currentMatrix.setCols(cols);
212                currentMatrix.setRows(rows);
213                currentMatrix.setScale(scale);
214                matrices.put(currentMatrix.getName(), currentMatrix);
215
216                return currentMatrix;
217
218        }
219
220        public Map<String, SubstitutionMatrix<AminoAcidCompound>> getMatrices() {
221                return matrices;
222        }
223}
