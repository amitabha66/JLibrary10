037/** a mouse listener for the AbstractChainRenderer class
038 * it listens to all mouse events and triggeres appropriate
039 * SequenceListener and FeatureListener events
040 * 
041 * @author Andreas Prlic
042 *
043 */
044public class SequenceMouseListener implements
045MouseListener,
046MouseMotionListener
047{
048
049
050        boolean selectionLocked;
051        boolean dragging;
052
053        int selectionStart;
054        int selectionEnd;
055        int draggingStart;
056        int oldSelectionStart;
057
058        int chainLength;
059
060
061        CoordManager coordManager;
062
063        static Logger logger = Logger.getLogger("org.biojava");
064
065        SequenceDisplay parent;
066
067        List<AlignmentPositionListener> alignmentPositionListeners;
068        
069        public SequenceMouseListener(SequenceDisplay parent) {
070                super();
071
072                this.parent = parent;
073
074                selectionLocked = false;
075                dragging = false;
076
077                selectionStart = -1 ;
078                selectionEnd = -1;
079                oldSelectionStart = -1;
080                draggingStart = -1;
081                chainLength = 0;
082                //clearSequenceListeners();
083
084                coordManager = new CoordManager();
085
086                alignmentPositionListeners = new ArrayList<AlignmentPositionListener>();
087                //renderer.getLayeredPane().addMouseListener(popupFrame);
088
089        }
090
091        public void clearListeners(){
092
093                alignmentPositionListeners.clear();
094        }
095        public void addAlignmentPositionListener(AlignmentPositionListener li){
096                alignmentPositionListeners.add(li);
097        }
098
099        public void mousePressed(MouseEvent event) {
100
101                int pos  = getSeqPos(event);
102
103                draggingStart=pos;
104                selectionStart = pos ;
105                //selectionEnd   = pos ;
106                //triggerClearSelection();
107                // triggerSelectionLocked(false);
108                triggerMouseOverPosition(pos,event.getY());
109
110
111
112
113        }
114
115        // mouse motion part
116
117        /** get the sequence position of the current mouse event
118         * 
119         */
120        private int getSeqPos(MouseEvent e) {
121
122                int x = e.getX();
123                //int y = e.getY();
124                //float scale = seqScale.getScale();
125                //int DEFAULT_X_START = SequenceScalePanel.DEFAULT_X_START;
126
127                float scale = parent.getScale();
128
129                coordManager.setScale(scale);
130                int seqpos = coordManager.getSeqPos(x-2);
131
132
133                return seqpos  ;
134        }   
135
136
137
138        public void setChain(Chain c){
139
140                chainLength = c.getAtomLength();
141                coordManager.setLength(chainLength);
142        }
143
144        private void setSelectionStart(int start){
145                if ( start < 0 )
146                        start = 0;
147                if ( start > chainLength)
148                        start = chainLength;
149                selectionStart = start;
150        }
151
152
153        private void setSelectionEnd(int end){
154                //if ( end < 0 )
155                        //end = 0;
156                if ( end > chainLength)
157                        end = chainLength;
158                selectionEnd = end;
159        }
160
161
162        public void mouseDragged(MouseEvent e) {
163                dragging = true;
164
165                int pos = getSeqPos(e) ;
166
167
168                if (( pos < 0)|| ( pos> chainLength)){
169                        return;
170                }
171
172                if (pos == oldSelectionStart){
173                        return;
174                }
175                oldSelectionStart = pos;
176
177                if ( pos > draggingStart){
178                        selectionStart = draggingStart;
179                        selectionEnd = pos ;
180                } else {
181                        selectionStart = pos;
182                        selectionEnd = draggingStart;
183                }
184                //triggerNewSequenceRange(selectionStart,selectionEnd);
185
186        }
187
188
189
190
191        public void mouseMoved(MouseEvent e) {
192        
193
194                if ( selectionLocked )
195                        return;
196
197
198
199                int pos = getSeqPos(e) ;
200                //System.out.println("mouse moved " + pos);
201
202                
203                
204                oldSelectionStart = pos;
205
206
207                this.setSelectionStart(pos);
208                this.setSelectionEnd(pos);
209
210                triggerMouseOverPosition(pos,e.getY());
211
212
213
214        }
215
216
217
218        public void mouseClicked(MouseEvent arg0) {
219
220
221        }
222
223        public void mouseEntered(MouseEvent arg0) {
224
225
226        }
227
228        public void mouseExited(MouseEvent arg0) {
229
230        }
231
232
233        public void mouseReleased(MouseEvent event) {
234                // logger.info("mouse released");
235
236
237
238                draggingStart = -1;
239
240
241                if ( dragging) {
242                        if  ( ! selectionLocked) {
243                                // triggerSelectionLocked(true);
244                        }
245                } else {}
246
247                dragging = false ;
248        }
249
250
251
252        protected void triggerMouseOverPosition(int pos, int mouseY){
253                if ( selectionLocked )
254                        return;
255                List<AlignedPosition> apos = parent.getAligMap();
256                if (pos > apos.size()-1)
257                        return;
258                //System.out.println(parent.getAligMap().get(pos));
259
260                for (AlignmentPositionListener li : alignmentPositionListeners) {
261                        li.mouseOverPosition(apos.get(pos));
262                }
263                
264
265
266        }
267
268}
