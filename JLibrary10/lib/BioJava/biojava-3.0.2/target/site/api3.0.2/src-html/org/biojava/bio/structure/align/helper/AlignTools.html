033    /** get a subset of Atoms based by their positions
034     *  
035     * @param caall
036     * @param idx an array where each element is a position of all the Atoms to return
037     * @return at Atom[] array
038     */
039    public static Atom[] getFragmentFromIdxList(Atom[] caall, int[] idx){
040        Atom[] subset = new Atom[idx.length];
041        
042        for (int p = 0 ; p < idx.length;p++){
043        
044            int pos1 =  idx[p];
045            subset[p] =  (Atom) caall[pos1].clone();            
046        }
047        return subset;
048    }
049    
050    /** get a continue subset of Atoms based by the starting position and the length
051     * 
052     * @param caall
053     * @param pos ... the start position
054     * @param fragmentLength .. the length of the subset to extract.
055     * @return an Atom[] array
056     */
057    public static Atom[] getFragment(Atom[] caall, int pos, int fragmentLength){
058     
059        if ( pos+fragmentLength > caall.length)
060            return null;
061        
062        Atom[] tmp = new Atom[fragmentLength];
063        
064        for (int i=0;i< fragmentLength;i++){
065            tmp[i] = (Atom)caall[i+pos].clone();
066        }
067        return tmp;
068        
069    }
070    
071    
072    /** get a continue subset of Atoms based by the starting position and the length
073     * does not clone the original atoms.
074     * 
075     * @param caall
076     * @param pos ... the start position
077     * @param fragmentLength .. the length of the subset to extract.
078     * @return an Atom[] array
079     */
080    public static Atom[] getFragmentNoClone(Atom[] caall, int pos, int fragmentLength){
081     
082        if ( pos+fragmentLength > caall.length)
083            return null;
084        
085        Atom[] tmp = new Atom[fragmentLength];
086        
087        for (int i=0;i< fragmentLength;i++){
088            tmp[i] = (Atom)caall[i+pos];
089        }
090        return tmp;
091        
092    }
093    
094    /** get the centroid for the set of atoms starting fromposition pos, length fragmentLenght
095     * 
096     * @param ca
097     * @param pos
098     * @param fragmentLength
099     * @return an Atom
100     */
101    public static Atom getCenter(Atom[] ca, int pos, int fragmentLength){
102        Atom center = new AtomImpl();
103        
104        if ( pos+fragmentLength > ca.length) {
105            System.err.println("pos ("+pos+"), fragL ("+fragmentLength +") > ca.length"+ca.length);
106            return center;
107        }
108        
109        Atom[] tmp = getFragmentNoClone(ca,pos,fragmentLength);
110        
111        return Calc.getCentroid(tmp);
112    }

115    /* Get distances along diagonal k from coordinate array coords.
116     * 
117     * @param atoms set of atoms to be used
118     * @param k number of diagonal to be used
119     */
120    public static double[] getDiagonalAtK (Atom[] atoms,int k) {
121        
122        int l = atoms.length;
123                
124        double[] dk = new double[(l-k)];
125        
126        for ( int i = 0 ; i< (l-k); i++){
127            try {
128                double dist = Calc.getDistance(atoms[i],atoms[i+k]);                
129                dk[i] = dist; 
130            } catch (StructureException e){
131                // this should not happen ...
132                // atoms should never be null here ...
133            }
134        }
135        
136        return dk;
137    }

139    /**
140     * Given distance matrix diagonals dk1, dk2, get the rmsd of a fpair.
141     * i,j is the fpair start in mol1 and mol2, l is the length of the fragment
142     * 
143     * @param dk1 distances of structure 1
144     * @param dk2 distance of structure 2
145     * @param i position in structure 1
146     * @param j position in structure 2
147     * @param l length of the fragments
148     * @param k diagonal used
149     * @return a double
150     */
151    public static double rms_dk_diag(double[] dk1, double[] dk2, int i, int j, int l, int k) {
152//        dk = 0
153//        for x in range(l-k):
154//            dk += (dk1[x+i]-dk2[x+j])*(dk1[x+i]-dk2[x+j])
155//        dk /= (l-k)
156//        return math.sqrt(dk)
157        
158        double dk = 0.0;
159        for (int x = 0 ; x < (l-k) ; x++)
160            dk += (dk1[x+i]-dk2[x+j])*(dk1[x+i]-dk2[x+j]);
161        
162        dk /= ( l-k);
163        return Math.sqrt(dk);
164        
165    }

167    /** matrix of all distances between two sets of 3d coords"
168     * 
169     * @param ca1
170     * @param ca2
171     * @return a Matrixd
172     */
173    public static Matrix getDistanceMatrix(Atom[] ca1, Atom[]ca2){
174        
175      int r = ca1.length;
176      int c = ca2.length;
177      
178      Matrix out = new Matrix(r,c);
179      
180      for (int i=0; i<r; i++) {
181          Atom a1 = ca1[i];
182          for (int j=0;j<c;j++){
183              Atom b1 = ca2[j];
184            
185//              double d = Math.sqrt(
186//                      (a1.getX()-b1.getX()) * (a1.getX()-b1.getX()) + 
187//                      (a1.getY()-b1.getY()) * (a1.getY()-b1.getY()) +
188//                      (a1.getZ()-b1.getZ()) * (a1.getZ()-b1.getZ()));    
189              try {
190                  double d = Calc.getDistance(a1,b1);
191                  out.set(i,j,d);
192              } catch (StructureException e) {
193                  e.printStackTrace();
194                  out.set(i,j,999);
195              }
196          }
197      }
198      return out;  
199    }
