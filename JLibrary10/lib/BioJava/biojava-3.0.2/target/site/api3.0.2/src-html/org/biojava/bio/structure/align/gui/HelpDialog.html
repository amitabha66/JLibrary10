025package org.biojava.bio.structure.align.gui;
026
027import java.awt.BorderLayout;
028import java.awt.Container;
029import java.awt.Cursor;
030import java.awt.Dimension;
031import java.awt.event.ActionEvent;
032import java.awt.event.ActionListener;
033
034import javax.swing.Box;
035import javax.swing.JButton;
036import javax.swing.JDialog;
037import javax.swing.JEditorPane;
038import javax.swing.JScrollPane;
039import javax.swing.event.HyperlinkEvent;
040import javax.swing.event.HyperlinkListener;
041
042import org.biojava.bio.structure.align.util.ResourceManager;
043import org.biojava.bio.structure.align.webstart.BrowserOpener;
044
045public class HelpDialog {
046        Box vBox;
047        
048        public void showDialog(){
049
050              JDialog dialog = new JDialog();
051
052              dialog.setSize(new Dimension(500,600));
053
054              ResourceManager mgr = ResourceManager.getResourceManager("ce");
055
056              String msg = mgr.getString("ce.help");
057                      
058              JEditorPane txt = new JEditorPane("text/html", msg);
059              txt.setEditable(false);
060
061              JScrollPane scroll = new JScrollPane(txt);
062              scroll.setSize(new Dimension(300,500));
063              vBox= Box.createVerticalBox();
064              vBox.add(scroll);
065              
066              txt.addHyperlinkListener(new HyperlinkListener(){
067                 
068                 public void hyperlinkUpdate(HyperlinkEvent e) {
069                     
070                     if (e.getEventType() == HyperlinkEvent.EventType.ACTIVATED) {
071                         String href = e.getDescription();
072                         BrowserOpener.showDocument(href);
073                     }
074                     if ( e.getEventType() == HyperlinkEvent.EventType.ENTERED) {
075                         // change the mouse curor
076                         vBox.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
077                     }
078                     if (e.getEventType() == HyperlinkEvent.EventType.EXITED) { 
079                         vBox.setCursor(Cursor.getPredefinedCursor(Cursor.DEFAULT_CURSOR));
080                     }
081                 }
082             });
083
084
085              JButton close = new JButton("Close");
086
087              close.addActionListener(new ActionListener(){
088                 public void actionPerformed(ActionEvent event) {
089                    Object source = event.getSource();
090
091                    JButton but = (JButton)source;
092                    Container parent = but.getParent().getParent().getParent().getParent().getParent().getParent() ;
093
094                    JDialog dia = (JDialog) parent;
095                    dia.dispose();
096                 }
097              });
098
099              Box hBoxb = Box.createHorizontalBox();
100              hBoxb.add(Box.createGlue());
101              hBoxb.add(close,BorderLayout.EAST);
102
103              vBox.add(hBoxb);
104
105              dialog.getContentPane().add(vBox);
106              dialog.setVisible(true);
107              
108              
109           
110        }
111}
