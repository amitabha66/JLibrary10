018/**
019 * An implementation of the popular bit encodings. This class provides the
020 * Sequence view over what is actually carried out in the {@link BitArrayWorker}
021 * instances. These are the objects that carry out array storage as well as
022 * indexing into those arrays. New bit encodings can be written by extending
023 * this class and a worker class. There are a number of issues with this
024 * type of storage engine:
025 *
026 * <ul>
027 * <li>We can only support a finite number of {@link Compound}s; 2 bit allows no N compounds</li>
028 * <li>For real savings you must read the sequence in using your own
029 * Reader and a {@link BitArrayWorker} instance</li>
030 * </ul>
031 *
032 * @author ayates
033 *
034 * @param <C> Type of compound; must extend {@link NucleotideCompound}
035 */
036public class BitSequenceReader<C extends Compound> implements ProxySequenceReader<C> {
037
038    private final AccessionID accession;
039    private final BitArrayWorker<C> worker;
040
041    /**
042     * Instance which allows you to supply a different @{BitArrayWorker}
043     * object.
044     */
045    public BitSequenceReader(BitArrayWorker<C> worker, AccessionID accession) {
046        this.accession = accession;
047        this.worker = worker;
048    }
049

056    }
057
058    /**
059     * Class is immutable & so this is unsupported
060     */
061    
062    public void setContents(String sequence) {
063        throw new UnsupportedOperationException(getClass().getSimpleName() + " is an immutable data structure; cannot reset contents");
064    }

169    @Override

171        if(Equals.classEqual(this, o)) {
172            @SuppressWarnings("unchecked")

175                    Equals.equal(this.worker, that.worker);
176        }

178    }

180    /**
181     * The logic of working with a bit has been separated out into this class
182     * to help developers create the bit data structures without having to
183     * put the code into an intermediate format and to also use the format
184     * without the need to copy this code.
185     *
186     * This class behaves just like a {@link Sequence} without the interface
187     *
188     * @author ayates
189     *
190     * @param <C> The {@link Compound} to use
191     */
192    public static abstract class BitArrayWorker<C extends Compound> {
193
194        private final CompoundSet<C> compoundSet;
195        private final int length;
196        private final int[] sequence;
197        private transient List<C> indexToCompoundsLookup = null;
198        private transient Map<C, Integer> compoundsToIndexLookup = null;
199        public static final int BYTES_PER_INT = 32;
200
201        private volatile Integer hashcode = null;
202
203        public BitArrayWorker(Sequence<C> sequence) {
204            this(sequence.getCompoundSet(), sequence.getLength());
205            populate(sequence);
206        }
207
208        public BitArrayWorker(String sequence, CompoundSet<C> compoundSet) {
209            this(compoundSet, sequence.length());
210            populate(sequence);
211        }
212
213        public BitArrayWorker(CompoundSet<C> compoundSet, int length) {
214            this.compoundSet = compoundSet;
215            this.length = length;
216            this.sequence = new int[seqArraySize(length)];
217        }
218
219        public BitArrayWorker(CompoundSet<C> compoundSet, int[] sequence) {
220            this.compoundSet = compoundSet;
221            this.sequence = sequence;
222            this.length = sequence.length;
223        }
224
225        /**
226         * This method should return the bit mask to be used to extract the
227         * bytes you are interested in working with. See solid implementations
228         * on how to create these
229         */
230        protected abstract byte bitMask();
231
232        /**
233         * Should return the maximum amount of compounds we can encode per int
234         */
235        protected abstract int compoundsPerDatatype();
236
237        /**
238         * Should return the inverse information that {@link #generateCompoundsToIndex() }
239         * returns i.e. if the Compound C returns 1 from compoundsToIndex then we
240         * should find that compound here in position 1
241         */
242        protected abstract List<C> generateIndexToCompounds();
243
244        /**
245         * Returns what the value of a compound is in the backing bit storage i.e.
246         * in 2bit storage the value 0 is encoded as 00 (in binary).
247         */
248        protected abstract Map<C, Integer> generateCompoundsToIndex();
249
250        /**
251         * Returns how many bits are used to represent a compound e.g. 2 if using
252         * 2bit encoding.
253         */
254        protected int bitsPerCompound() {
255            return BYTES_PER_INT / compoundsPerDatatype();
256        }
257
258        public int seqArraySize(int length) {
259            return (int) Math.ceil((double) length / (double) compoundsPerDatatype());
260        }
261
262        /**
263         * Loops through the Compounds in a Sequence and passes them onto
264         * {@link #setCompoundAt(Compound, int)}
265         */
266        public void populate(Sequence<C> sequence) {
267            int position = 1;
268            for (C c : sequence) {
269                setCompoundAt(c, position++);
270            }
271        }
272
273        /**
274         * Loops through the chars in a String and passes them onto
275         * {@link #setCompoundAt(char, int)}
276         */
277        public void populate(String sequence) {
278            for (int index = 0; index < getLength(); index++) {
279                setCompoundAt(sequence.charAt(index), index + 1);
280            }
281        }
282
283        /**
284         * Converts from char to Compound and sets it at the given biological index
285         */
286        public void setCompoundAt(char base, int position) {
287            C compound = getCompoundSet().getCompoundForString(Character.toString(base));
288            setCompoundAt(compound, position);
289        }
290
291        /**
292         * Sets the compound at the specified biological index
293         */
294        public void setCompoundAt(C compound, int position) {
295            hashcode = null;
296            int arrayIndex = biologicalIndexToArrayIndex(position);
297            int currentInt = sequence[arrayIndex];
298            int shiftBy = shiftBy(position);
299            Integer integerValue = getCompoundsToIndexLookup().get(compound);
300
301            //If we got nothing then throw an error as it's wrong
302            if (integerValue == null) {
303                processUnknownCompound(compound, position);
304            }
305
306            int shiftedValue = integerValue << shiftBy;
307
308            sequence[arrayIndex] = currentInt | shiftedValue;
309        }
310
311        /**
312         * Returns the compound at the specified biological index
313         */
314        public C getCompoundAt(int position) {
315            //Avoids asking for something which is not encoded by a bit-pair
316            if (position > getLength()) {
317                throw new IllegalArgumentException(position + " is greater than length. Cannot access this position");
318            }
319            //Just stops us from using 0 indexing
320            if (position < 1) {
321                throw new IllegalArgumentException(position + " is less than 1; you must use biological indexing (indexing from 1)");
322            }
323
324            int arrayIndex = biologicalIndexToArrayIndex(position);
325            int currentByte = sequence[arrayIndex];
326            int shiftBy = shiftBy(position);
327            int shifted = (int) (currentByte >>> shiftBy);
328            int masked = (int) (shifted & bitMask());
329
330            //If we could encode 4 compounds then our max masked value is 3
331            if (masked > (compoundsPerDatatype() - 1)) {
332                throw new IllegalStateException("Got a masked value of " + masked + "; do not understand values greater than " + (compoundsPerDatatype() - 1));
333            }
334            return getIndexToCompoundsLookup().get(masked);
335        }
336
337        /**
338         * Since bit encoding only supports a finite number of bases
339         * it is more than likely when processing sequence you will encounter a
340         * compound which is not covered by the encoding e.g. N in a 2bit sequence.
341         * You can override this to convert the unknown base into one you can
342         * process or store locations of unknown bases for a level of post processing
343         * in your subclass.
344         *
345         * @param compound Compound process
346         * @return Byte representation of the compound
347         * @throws IllegalStateException Done whenever this method is invoked
348         */
349        protected byte processUnknownCompound(C compound, int position) throws IllegalStateException {
350            throw new IllegalStateException("Do not know how to translate the compound " + compound + " to a " + bitsPerCompound() + "bit representation");
351        }
352
353        /**
354         * Returns a list of compounds the index position of which is used
355         * to translate from the byte representation into a compound.
356         */
357        protected List<C> getIndexToCompoundsLookup() {
358            if (indexToCompoundsLookup == null) {
359                indexToCompoundsLookup = generateIndexToCompounds();
360            }
361            return indexToCompoundsLookup;
362        }
363
364        /**
365         * Returns a map which converts from compound to an integer representation
366         */
367        protected Map<C, Integer> getCompoundsToIndexLookup() {
368            if (compoundsToIndexLookup == null) {
369                compoundsToIndexLookup = generateCompoundsToIndex();
370            }
371            return compoundsToIndexLookup;
372        }
373
374        /**
375         * Converting a biological index to the int which is used to store that
376         * position's data.
377         *
378         * <ul>
379         * <li>Assuming 2bit encoding using the 17 base in a sequence</li>
380         * <li>Convert into 0 index; 17 - 1 = 16</li>
381         * <li>Divide by the number of compounds per int; 16 / 16</li>
382         * <li>Floor the value; floor(1) = 1</li>
383         * </ul>
384         *
385         * Running this for position 13
386         *
387         * <ul>
388         * <li>13 - 1 = 12</li>
389         * <li>12 / 16 = 0.75</li>
390         * <li>floor(0.75) = 0</li>
391         * </ul>
392         */
393        private int biologicalIndexToArrayIndex(int index) {
394            return ((index - 1) / compoundsPerDatatype());
395        }
396
397        /**
398         * Convert from bio to 0 index, remainder & then multiply by 2
399         * <ul>
400         * <li>Using 2bit encoding and working with position 19</li>
401         * <li>19 is the 3rd position in the second int</li>
402         * <li>Means a shift of 4 into that int to get the right data out</li>
403         * <li>Also must convert into the non-bio index</li>
404         * <li>19 - 1 = 18</li>
405         * <li>18 % compoundsPerDatatype() (16) = 2</li>
406         * <li>2 * bits per compound (2) = 4</li>
407         * </ul>
408         */
409        private byte shiftBy(int index) {
410            return (byte) (((index - 1) % compoundsPerDatatype()) * bitsPerCompound());
411        }
412
413        /**
414         * Returns the compound set backing this store
415         */
416        public CompoundSet<C> getCompoundSet() {
417            return compoundSet;
418        }
419
420        public int getLength() {
421            return length;
422        }
423
424        @Override
425        public int hashCode() {
426            if(hashcode == null) {
427                int s = Hashcoder.SEED;
428                s = Hashcoder.hash(s, sequence);
429                s = Hashcoder.hash(s, indexToCompoundsLookup);
430                s = Hashcoder.hash(s, compoundSet);
431                hashcode = s;
432            }
433            return hashcode;
434        }
435
436        @Override
437        @SuppressWarnings("unchecked")
438        public boolean equals(Object o) {
439            if(Equals.classEqual(this, o)) {
440                BitArrayWorker<C> that = (BitArrayWorker<C>)o;
441                return  Equals.equal(compoundSet, that.compoundSet) &&
442                        Equals.equal(indexToCompoundsLookup, that.indexToCompoundsLookup) &&
443                        Equals.equal(sequence, that.sequence);
444            }
445            return false;
446        }
447    }
