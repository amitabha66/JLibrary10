099        public static final Pattern pdbNumRangeRegex = Pattern.compile(
100                        "^\\s*(\\w)" + //chain ID
101                        "(?:" + //begin range, this is a "non-capturing group"
102                          "(?::|_|:$|_$|$)" + //colon or underscore, could be at the end of a line, another non-capt. group.
103                                "(?:"+ // another non capturing group for the residue range
104                                 "([-+]?[0-9]+[A-Za-z]?)" + // first residue
105                                 "\\s*-\\s*" + // -
106                                 "([-+]?[0-9]+[A-Za-z]?)" + // second residue

108                        ")?" + //end range

190        /** Count how many number of Atoms are contained within a Structure object.
191         *
192         * @param s the structure object
193         * @return the number of Atoms in this Structure
194         */
195        public static final int getNrAtoms(Structure s){
196
197                int nrAtoms = 0;
198
199                Iterator<Group> iter = new GroupIterator(s);
200
201                while ( iter.hasNext()){
202                        Group g = (Group) iter.next();
203                        nrAtoms += g.size();
204                }
205
206                return nrAtoms;
207        }

213         * @return the number of groups in the structure
214         */
215        public static final int getNrGroups(Structure s){
216                int nrGroups = 0;
217
218                List<Chain> chains = s.getChains(0);

220                while (iter.hasNext()){
221                        Chain c = (Chain) iter.next();

272        private static void extractCAatoms(String[] atomNames, List<Chain> chains,
273                        List<Atom> atoms) {
274                for ( Chain c : chains) {
275                        
276                        for ( Group g : c.getAtomGroups()) {
277
278                                // a temp container for the atoms of this group
279                                List<Atom> thisGroupAtoms = new ArrayList<Atom>();
280                                // flag to check if this group contains all the requested atoms.
281                                boolean thisGroupAllAtoms = true;
282                                for ( int i = 0 ; i < atomNames.length; i++){
283                                        String atomName = atomNames[i];
284                                        try {
285                                                Atom a = g.getAtom(atomName);
286                                                thisGroupAtoms.add(a);
287                                        } catch (StructureException e){
288                                                // this group does not have a required atom, skip it...
289                                                thisGroupAllAtoms = false;
290                                                break;
291                                        }
292                                }
293                                if ( thisGroupAllAtoms){
294                                        // add the atoms of this group to the array.
295                                        Iterator<Atom> aIter = thisGroupAtoms.iterator();
296                                        while(aIter.hasNext()){
297                                                Atom a = (Atom) aIter.next();
298                                                atoms.add(a);
299                                        }
300                                }
301
302                        }
303                }
304        }

361        /** Provides an equivalent copy of Atoms in a new array. Clones everything, starting with parent 
362         * groups and chains. The chain will only contain groups that are part of the CA array.
363         * 
364         * @param ca array of CA atoms
365         * @return Atom array
366         */
367        public static final Atom[] cloneCAArray(Atom[] ca) throws StructureException{
368                Atom[] newCA = new Atom[ca.length];
369
370                List<Chain> model = new ArrayList<Chain>();
371                int apos = -1;
372                for(Atom a: ca){
373                        apos++;
374                        Group parentG = a.getGroup();
375                        Chain parentC = parentG.getChain();
376
377                        Chain newChain = null;
378                        for ( Chain c : model){
379                                if ( c.getChainID().equals(parentC.getChainID())){
380                                        newChain = c;
381                                        break;
382                                }
383                        }
384                        if ( newChain == null){
385                                newChain = new ChainImpl();
386                                newChain.setChainID(parentC.getChainID());
387                                model.add(newChain);
388                        }
389
390                        Group parentN = (Group)parentG.clone();
391
392                        newCA[apos] = parentN.getAtom(" CA ");
393                        newChain.addGroup(parentN);
394                }
395                return newCA;
396        }

398        /** Clone a set of CA Atoms, but returns the parent groups
399         *  
400         * @param ca Atom array
401         * @return Group array
402         */
403        public static Group[] cloneGroups(Atom[] ca) {
404                Group[] newGroup = new Group[ca.length]; 
405
406                List<Chain> model = new ArrayList<Chain>();
407                int apos = -1;
408                for(Atom a: ca){
409                        apos++;
410                        Group parentG = a.getGroup();
411                        Chain parentC = parentG.getChain();
412
413                        Chain newChain = null;
414                        for ( Chain c : model){
415                                if ( c.getChainID().equals(parentC.getChainID())){
416                                        newChain = c;
417                                        break;
418                                }
419                        }
420                        if ( newChain == null){
421                                newChain = new ChainImpl();
422                                newChain.setChainID(parentC.getChainID());
423                                model.add(newChain);
424                        }
425
426                        Group ng = (Group)parentG.clone();
427                        newGroup[apos] = ng;
428                        newChain.addGroup(ng);
429                }
430                return newGroup;
431        }

478        /** convert a three letter code into single character.
479         * catches for unusual characters
480         *
481         * @param groupCode3 three letter representation
482         * @return null if group is a nucleotide code
483         */
484        public static final Character get1LetterCode(String groupCode3){
485
486                Character aminoCode1 = null;
487                try {
488                        // is it a standard amino acid ?
489                        aminoCode1 = convert_3code_1code(groupCode3);
490                } catch (UnknownPdbAminoAcidException e){
491                        // hm groupCode3 is not standard
492                        // perhaps it is an nucleotide?
493                        if ( isNucleotide(groupCode3) ) {
494                                //System.out.println("nucleotide, aminoCode1:"+aminoCode1);
495                                aminoCode1= null;
496                        } else {
497                                // does not seem to be so let's assume it is
498                                //  nonstandard aminoacid and label it "X"
499                                //logger.warning("unknown group name "+groupCode3 );
500                                aminoCode1 = UNKNOWN_GROUP_LABEL;
501                        }
502                }
503
504                return aminoCode1;
505
506        }

515
516                String code = groupCode3.trim();

538                // since we deal here with structure alignments,

540

557                        chainId = chainId.trim();

574                        c = s.getChainByPDB(chainId.toUpperCase());
575
576                }
577                if ( c != null)
578                        newS.addChain(c);
579
580
581                return newS;
582        }

646         *  1CDG (A:407-495,A:582-686)
647         *  1CDG (A_407-495,A_582-686)
648         * 
649         * 
650         * @param s The full structure
651         * @param ranges A comma-seperated list of ranges, optionally surrounded by parentheses
652         * @return Substructure of s specified by ranges
653         */
654        @SuppressWarnings("deprecation")
655        public static final Structure getSubRanges(Structure s, String ranges ) 
656        throws StructureException
657        {
658                Structure struc = getReducedStructure(s, null);
659
660                if ( ranges == null || ranges.equals(""))
661                        throw new IllegalArgumentException("ranges can't be null or empty");
662
663                ranges = ranges.trim();
664
665                if ( ranges.startsWith("("))
666                        ranges = ranges.substring(1);
667                if ( ranges.endsWith(")")) {
668                        ranges = ranges.substring(0,ranges.length()-1);
669                }
670
671                Structure newS = new StructureImpl();
672                newS.setHeader(s.getHeader());
673                newS.setPDBCode(s.getPDBCode());
674                newS.setPDBHeader(s.getPDBHeader());
675                newS.setName(s.getName());
676                newS.setDBRefs(s.getDBRefs());
677                newS.setNmr(s.isNmr());
678                newS.setBiologicalAssembly(s.isBiologicalAssembly());
679
680                // TODO The following should be only copied for atoms which are present in the range.
681                //newS.setCompounds(s.getCompounds());
682                //newS.setConnections(s.getConnections());
683                //newS.setSSBonds(s.getSSBonds());
684                //newS.setSites(s.getSites());
685                
686                
687                String[] rangS =ranges.split(",");
688
689
690                String prevChainId = null;
691
692                // parse the ranges, adding the specified residues to newS
693                for ( String r: rangS){
694                        //System.out.println(">"+r+"<");
695                        // Match a single range, eg "A_4-27"
696                        
697                        Matcher matcher = pdbNumRangeRegex.matcher(r);
698                        if( ! matcher.matches() ){
699                                throw new StructureException("wrong range specification, should be provided as chainID_pdbResnum1-pdbRensum2");
700                        }
701                        String chainId = matcher.group(1);
702                        
703                        Chain chain = struc.getChainByPDB(chainId);
704                        Group[] groups;
705                        
706                        String pdbresnumStart = matcher.group(2);
707                        String pdbresnumEnd   = matcher.group(3);
708                        
709
710                        if( pdbresnumStart != null && pdbresnumEnd != null) {
711                                // not a full chain
712                                //since Java doesn't allow '+' before integers, fix this up.
713                                if(pdbresnumStart.charAt(0) == '+')
714                                        pdbresnumStart = pdbresnumStart.substring(1);
715                                if(pdbresnumEnd.charAt(0) == '+')
716                                        pdbresnumEnd = pdbresnumEnd.substring(1);
717                                groups = chain.getGroupsByPDB(pdbresnumStart, pdbresnumEnd);
718                        } else {
719                                // full chain
720                                groups = chain.getAtomGroups().toArray(new Group[chain.getAtomGroups().size()]);
721                        }
722                        
723                        
724                        // Create new chain, if needed
725                        Chain c = null;
726                        if ( prevChainId == null) {
727                                // first chain...
728                                c = new ChainImpl();
729                                c.setName(chain.getName());
730                                newS.addChain(c);
731                        } else if ( prevChainId.equals(chain.getName())) {
732                                c = newS.getChainByPDB(prevChainId);
733
734                        } else {
735                                try {
736                                        c = newS.getChainByPDB(chain.getName());
737                                } catch (StructureException e){
738                                        // chain not in structure yet...
739                                        c = new ChainImpl();
740                                        c.setName(chain.getName());
741                                        newS.addChain(c);
742                                }
743                        }
744
745                        // add the groups to the chain:
746                        for ( Group g: groups) {
747                                c.addGroup(g);
748                        }
749
750                        prevChainId = c.getName();
751                }
752
753                return newS;
754        }

756        public static final String convertAtomsToSeq(Atom[] atoms) {
757
758                StringBuffer buf = new StringBuffer();
759                Group prevGroup  = null;
760                for (Atom a : atoms){
761                        Group g = a.getGroup();
762                        if ( prevGroup != null) {
763                                if ( prevGroup.equals(g)) {
764                                        // we add each group only once.
765                                        continue;
766                                }
767                        }
768                        String code3 = g.getPDBName();
769                        try {
770                                buf.append(convert_3code_1code(code3) );
771                        } catch (UnknownPdbAminoAcidException e){
772                                buf.append('X');
773                        }
774                        prevGroup = g;
775
776                }
777                return buf.toString();
778        }

816        /*
817         * Returns a List of Groups in a structure within the distance specified of a given group.
818         */
819        public static List<Group> getGroupsWithinShell(Structure structure, Group group, double distance, boolean includeWater) {
820            Set<Group> returnSet = new LinkedHashSet<Group>();
821
822            //square the distance to use as a comparison against getDistanceFast which returns the square of a distance.
823            distance = distance * distance;
824
825            for (Atom atomA : group.getAtoms()) {
826                for (Chain chain : structure.getChains()) {
827                    for (Group chainGroup : chain.getAtomGroups()) {
828//                        System.out.println("Checking group: " + chainGroup);
829                        if (chainGroup.getResidueNumber().equals(group.getResidueNumber())) {
830                            continue;
831                        }
832                        else if (!includeWater && chainGroup.getPDBName().equals("HOH")) {
833                            continue;
834                        }
835                        else {
836                            for (Atom atomB : chainGroup.getAtoms()) {
837                            try {
838                                //use getDistanceFast as we are doing a lot of comparisons
839                                double dist = Calc.getDistanceFast(atomA, atomB);
840                                if (dist <= distance) {
841                                    returnSet.add(chainGroup);
842//                                    System.out.println(String.format("%s within %s of %s", atomB, dist, atomA));
843                                    break;
844                                }
845                                
846                            } catch (StructureException ex) {
847                                Logger.getLogger(StructureTools.class.getName()).log(Level.SEVERE, null, ex);
848                            }
849                                
850                            }
851                        }
852                    }
853                }
854            }
855            List<Group> returnList = new ArrayList<Group>();
856            returnList.addAll(returnSet);
857            return returnList;
858        }

860        /*
861         * Very simple distance-based bond calculator. Will give approximations,
862         * but do not rely on this to be chemically correct.
863         */
864        public static List<Bond> findBonds(Group group, List<Group> groups) {
865            List<Bond> bondList = new ArrayList<Bond>();
866            for (Atom atomA : group.getAtoms()) {
867                for (Group groupB : groups) {
868                    if (groupB.getType().equals(GroupType.HETATM)) {
869                        continue;
870                    }
871                    for (Atom atomB : groupB.getAtoms()) {
872                        try {
873                            double dist = Calc.getDistance(atomA, atomB);
874                            BondType bondType = BondType.UNDEFINED;
875                            if (dist <= 2) {
876                                bondType = BondType.COVALENT;
877                                Bond bond = new Bond(dist, bondType, group, atomA, groupB, atomB);
878                                bondList.add(bond);
879//                                    System.out.println(String.format("%s within %s of %s", atomB, dist, atomA));
880                            }
881                            else if (dist <= 3.25) {
882                                
883                                if (isHbondDonorAcceptor(atomA) && isHbondDonorAcceptor(atomB)) {
884                                    bondType = BondType.HBOND;
885                                }
886                                else if (atomA.getElement().isMetal() && isHbondDonorAcceptor(atomB)) {
887                                    bondType = BondType.METAL;
888                                }
889                                else if (atomA.getElement().equals(Element.C) && atomB.getElement().equals(Element.C)) {
890                                    bondType = BondType.HYDROPHOBIC;
891                                }
892                                //not really interested in 'undefined' types
893                                if (bondType != BondType.UNDEFINED) {
894                                    Bond bond = new Bond(dist, bondType, group, atomA, groupB, atomB);
895                                    bondList.add(bond);
896                                }
897//                                    System.out.println(String.format("%s within %s of %s", atomB, dist, atomA));
898                            } else if (dist <= 3.9) {
899                                if (atomA.getElement().equals(Element.C) && atomB.getElement().equals(Element.C)) {
900                                    bondType = BondType.HYDROPHOBIC;
901                                }
902                                //not really interested in 'undefined' types
903                                if (bondType != BondType.UNDEFINED) {
904                                    Bond bond = new Bond(dist, bondType, group, atomA, groupB, atomB);
905                                    bondList.add(bond);
906                                }
907                            }
908
909                        } catch (StructureException ex) {
910                            Logger.getLogger(StructureTools.class.getName()).log(Level.SEVERE, null, ex);
911                        }
912
913                    }
914                }
915            }
916
917
918            return bondList;
919        }
