048/**
049 * Implements a data structure for a guide tree used during progressive multiple sequence alignment.  Leaf
050 * {@link Node}s correspond to single {@link Sequence}s.  Internal {@link Node}s correspond to multiple sequence
051 * alignments.  The root {@link Node} corresponds to the full multiple sequence alignment.
052 *
053 * @author Mark Chapman
054 * @param <S> each {@link Sequence} in the tree is of type S
055 * @param <C> each element of a {@link Sequence} is a {@link Compound} of type C
056 */
057public class GuideTree<S extends Sequence<C>, C extends Compound> implements Iterable<GuideTreeNode<S, C>> {
058
059    private List<S> sequences;
060    private List<PairwiseSequenceScorer<S, C>> scorers;
061    private BasicSymmetricalDistanceMatrix distances;
062    private String newick;
063    private Node root;
064
065    /**
066     * Creates a guide tree for use during progressive multiple sequence alignment.
067     *
068     * @param sequences the {@link List} of {@link Sequence}s to align
069     * @param scorers list of sequence pair scorers, one for each pair of sequences given
070     */
071    public GuideTree(List<S> sequences, List<PairwiseSequenceScorer<S, C>> scorers) {
072        this.sequences = Collections.unmodifiableList(sequences);
073        this.scorers = Collections.unmodifiableList(scorers);
074        distances = new BasicSymmetricalDistanceMatrix(sequences.size());
075        for (int i = 0, n = 0; i < sequences.size(); i++) {
076            AccessionID id = sequences.get(i).getAccession();
077            distances.setIdentifier(i, (id == null) ? Integer.toString(i + 1) : id.getID());
078            for (int j = i+1; j < sequences.size(); j++) {
079                PairwiseSequenceScorer<S, C> scorer = scorers.get(n++);
080                distances.setValue(i, j, scorer.getDistance());
081            }
082        }
083        // TODO UPGMA and other hierarchical clustering routines
084        Phylogeny phylogeny = NeighborJoining.createInstance().execute(distances);
085        newick = phylogeny.toString();
086        root = new Node(phylogeny.getRoot(), null);
087    }
088
089    /**
090     * Returns a sequence pair score for all {@link Sequence} pairs in the given {@link List}.
091     *
092     * @return list of sequence pair scores
093     */
094    public int[] getAllPairsScores() {
095        int[] scores = new int[scorers.size()];
096        int n = 0;
097        for (PairwiseSequenceScorer<S, C> scorer : scorers) {
098            scores[n++] = scorer.getScore();
099        }
100        return scores;
101    }
102
103    /**
104     * Returns the distance matrix used to construct this guide tree.  The scores have been normalized.
105     *
106     * @return the distance matrix used to construct this guide tree
107     */
108    public double[][] getDistanceMatrix() {
109        double[][] matrix = new double[distances.getSize()][distances.getSize()];
110        for (int i = 0; i < matrix.length; i++) {
111            for (int j = i+1; j < matrix.length; j++) {
112                matrix[i][j] = matrix[j][i] = distances.getValue(i, j);
113            }
114        }
115        return matrix;
116    }
117
118    /**
119     * Returns the root {@link Node} which corresponds to the full multiple sequence alignment.
120     *
121     * @return the root node
122     */
123    public Node getRoot() {
124        return root;
125    }
126
127    /**
128     * Returns the similarity matrix used to construct this guide tree.  The scores have not been normalized.
129     *
130     * @return the similarity matrix used to construct this guide tree
131     */
132    public int[][] getScoreMatrix() {
133        int[][] matrix = new int[sequences.size()][sequences.size()];
134        for (int i = 0, n = 0; i < matrix.length; i++) {
135            matrix[i][i] = scorers.get(i).getMaxScore();
136            for (int j = i+1; j < matrix.length; j++) {
137                matrix[i][j] = matrix[j][i] = scorers.get(n++).getScore();
138            }
139        }
140        return matrix;
141    }
142
143    /**
144     * Returns the {@link Sequence}s which make up the leaves of this tree.
145     *
146     * @return the sequences which make up the leaves of this tree
147     */
148    public List<S> getSequences() {
149        return sequences;
150    }
151
152    // method for Iterable
153
154    /**
155     * Returns a post-order {@link Iterator} that traverses the tree from leaves to root.
156     */
157    @Override
158    public Iterator<GuideTreeNode<S, C>> iterator() {
159        return new PostOrderIterator();
160    }
161
162    // method from Object
163
164    @Override
165    public String toString() {
166        return newick;
167    }

169    /**
170     * Implements a data structure for the node in a guide tree used during progressive multiple sequence alignment.
171     */
172    public class Node implements GuideTreeNode<S, C> {
173
174        private GuideTreeNode<S, C> parent, child1, child2;
175        private double distance;
176        private String name;
177        private boolean isLeaf, isVisited;
178        private Profile<S, C> profile;
179        private Future<ProfilePair<S, C>> profileFuture;
180
181        private Node(PhylogenyNode node, Node parent) {
182            this.parent = parent;
183            distance = node.getDistanceToParent();
184            name = node.getNodeName();
185            if(isLeaf = node.isExternal()) {
186                profile = new SimpleProfile<S, C>(sequences.get(distances.getIndex(name)));
187            } else {
188                child1 = new Node(node.getChildNode1(), this);
189                child2 = new Node(node.getChildNode2(), this);
190            }
191        }
192
193        // methods for GuideTreeNode
194
195        @Override
196        public GuideTreeNode<S, C> getChild1() {
197            return child1;
198        }
199
200        @Override
201        public GuideTreeNode<S, C> getChild2() {
202            return child2;
203        }
204
205        @Override
206        public double getDistanceToParent() {
207            return distance;
208        }
209
210        @Override
211        public String getName() {
212            return name;
213        }
214
215        @Override
216        public Profile<S, C> getProfile() {
217            return profile;
218        }
219
220        @Override
221        public Future<ProfilePair<S, C>> getProfileFuture() {
222            return profileFuture;
223        }
224
225        @Override
226        public void setProfile(Profile<S, C> profile) {
227            this.profile = profile;
228            profileFuture = null;
229        }
230
231        @Override
232        public void setProfileFuture(Future<ProfilePair<S, C>> profileFuture) {
233            this.profileFuture = profileFuture;
234            profile = null;
235        }
236
237        // methods for TreeNode
238
239        @Override
240        public Enumeration<GuideTreeNode<S, C>> children() {
241            Vector<GuideTreeNode<S, C>> children = new Vector<GuideTreeNode<S, C>>();
242            children.add(getChild1());
243            children.add(getChild2());
244            return children.elements();
245        }
246
247        @Override
248        public boolean getAllowsChildren() {
249            return !isLeaf();
250        }
251
252        @Override
253        public GuideTreeNode<S, C> getChildAt(int childIndex) {
254            if (childIndex == 1) {
255                return getChild1();
256            } else if (childIndex == 2) {
257                return getChild2();
258            }
259            throw new IndexOutOfBoundsException();
260        }
261
262        @Override
263        public int getChildCount() {
264            return 2;
265        }
266
267        @Override
268        public int getIndex(TreeNode child) {
269            return getChildAt(1) == child ? 1 : (getChildAt(2) == child ? 2 : -1);
270        }
271
272        @Override
273        public GuideTreeNode<S, C> getParent() {
274            return parent;
275        }
276
277        @Override
278        public boolean isLeaf() {
279            return isLeaf;
280        }
281
282        // helper methods for iterator
283
284        private void clearVisited() {
285            isVisited = false;
286            if (!isLeaf()) {
287                ((Node) getChild1()).clearVisited();
288                ((Node) getChild2()).clearVisited();
289            }
290        }
291
292        private boolean isVisited() {
293            return isVisited;
294        }
295
296        private void visit() {
297            isVisited = true;
298        }
299
300    }
301
302    // helper class that defines the default post-order (leaves to root) traversal
303    private class PostOrderIterator implements Iterator<GuideTreeNode<S, C>> {
304
305        private Stack<Node> nodes;
306
307        private PostOrderIterator() {
308            getRoot().clearVisited();
309            nodes = new Stack<Node>();
310            nodes.push(getRoot());
311        }
312
313        // methods for Iterator
314
315        @Override
316        public boolean hasNext() {
317            return !nodes.isEmpty();
318        }
319
320        @Override
321        public GuideTreeNode<S, C> next() {
322            while (hasNext()) {
323                Node next = nodes.peek(), child1 = (Node) next.getChild1(), child2 = (Node) next.getChild2();
324                if (child1 != null && !child1.isVisited()) {
325                    nodes.push(child1);
326                } else if (child2 != null && !child2.isVisited()) {
327                    nodes.push(child2);
328                } else {
329                    next.visit();
330                    return nodes.pop();
331                }
332            }
333            return null;
334        }
335
336        @Override
337        public void remove() {
338            throw new UnsupportedOperationException();
339        }
340
341    }
