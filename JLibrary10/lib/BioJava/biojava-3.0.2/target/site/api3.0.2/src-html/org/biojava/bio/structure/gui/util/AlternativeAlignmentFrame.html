102        public void setAlternativeAlignments(AlternativeAlignment[] aligs) {
103                this.aligs = aligs;

130        private Object[][] getDataFromAligs(AlternativeAlignment[] aligs){
131
132
133                Object[][] data = new Object[aligs.length][columnNames.length];
134
135                for ( int i=0;i< aligs.length;i++){
136                        AlternativeAlignment alig = aligs[i];
137
138                        data[i][0] = new Integer(i+1);
139                        data[i][1] = new Integer(alig.getEqr());
140                        data[i][2] = new Double(alig.getScore());
141                        data[i][3] = new Double(alig.getRmsd());
142                        data[i][4] = new Integer(alig.getGaps());
143                        data[i][5] = new Integer(alig.getCluster());
144                        JButton maxb = new JButton("Distance Matrix");
145                        maxb.addMouseListener(new MatrixMouseListener(this,i));
146
147                        data[i][6] = maxb;
148                        
149
150                        //Action action1 = new MyButtonAction(t,this,i);
151                        JButton but = new JButton("Show in Jmol");
152                        but.addMouseListener(new MyButtonMouseListener(this,i));
153                        data[i][7] = but;
154
155                        
156                }
157                return data;
158        }

160        public void showDistanceMatrix(int position){
161                if ( position > aligs.length){
162                        return;
163                }
164                AlternativeAlignment alig = aligs[position];
165                logger.info("display distance matrix for alternative alignment " + (position +1));
166                
167                ScaleableMatrixPanel smp = new ScaleableMatrixPanel();
168                JFrame frame = new JFrame();
169                frame.setTitle("Alt. Alig [" + position+"] - Distance Matrix & path");
170                
171                frame.addWindowListener(new WindowAdapter(){
172                        public void windowClosing(WindowEvent e){
173                                JFrame f = (JFrame) e.getSource();
174                                f.setVisible(false);
175                                f.dispose();
176                        }
177
178                                        
179                        
180                });
181                                        
182                smp.setMatrix(alig.getDistanceMatrix());                
183                smp.setAlternativeAligs(new AlternativeAlignment[]{alig});
184                
185                frame.getContentPane().add(smp);
186
187                frame.pack();
188                frame.setVisible(true);
189
190        }

192        public void showAlternative(int position){
193                if ( position > aligs.length){
194                        return;
195                }
196                AlternativeAlignment alig = aligs[position];
197                logger.info("display alternative alignment " + (position +1));
198
199                // create the structure alignment object and tell the listeners ...
200
201
202//              Matrix m1 = Matrix.identity(3,3);
203                Matrix m2 = alig.getRotationMatrix();
204
205                String pdb1 = structure1.getPDBCode();
206                String pdb2 = structure2.getPDBCode();
207
208
209                Atom shift1 = new AtomImpl();
210                shift1.setCoords(new double[]{0,0,1});
211                Atom shift2 = alig.getShift();
212
213                Structure s3 = (Structure)structure2.clone();
214
215                Calc.rotate(s3,m2);
216                Calc.shift(s3,shift2);
217
218                BiojavaJmol jmol = new BiojavaJmol();
219                jmol.setTitle(pdb1 + " vs. " + pdb2);
220
221                Structure n = new StructureImpl();
222                
223                List<Chain> chains1 = structure1.getChains();
224                
225                n.addModel(chains1);
226                
227                List<Chain> chains3 = s3.getChains();
228                n.addModel(chains3);
229        
230                n.setNmr(true);
231                
232                jmol.setStructure(n);
233                String[] cmds = createRasmolScripts(alig);
234                jmol.evalString("model 0 ; select * ; wireframe off ; spacefill off; backbone 0.3;");
235                jmol.evalString(cmds[0]);
236                jmol.evalString(cmds[1]);               
237                
238                JFrame frame = new JFrame("Sequences for AlternativeAlignment ["+position+"]");
239                
240                SequenceDisplay seqdisp;
241                seqdisp =  new SequenceDisplay(structurePairAligner);
242                seqdisp.setStructure1(structure1);
243                seqdisp.setStructure2(structure2);
244        
245                seqdisp.setAlternativeAlignment(alig);
246                
247                frame.getContentPane().add(seqdisp);
248                
249                frame.pack();
250                frame.setVisible(true);
251                frame.addWindowListener(new WindowAdapter(){
252                        public void windowClosing(WindowEvent e){
253                                JFrame f = (JFrame) e.getSource();
254                                f.setVisible(false);
255                                f.dispose();
256                        }
257
258                                        
259                        
260                });
261                
262                seqdisp.updateDisplay();
263                
264                JmolAlignedPositionListener jmolBridge = new JmolAlignedPositionListener(jmol,structurePairAligner);
265                jmolBridge.setStructure1(structure1);           
266                jmolBridge.setStructure2(s3);
267                
268                seqdisp.addAlignmentPositionListener(jmolBridge);
269                
270        }

274        private String[] createRasmolScripts(AlternativeAlignment alig){
275                String[] scripts = new String[2];
276
277                Color col1 = Color.red;
278                Color col2 = Color.blue;
279
280                Color chaincol1 = new Color(col1.getRed()/2,col1.getGreen()/2,col1.getBlue()/2);
281                Color chaincol2 = new Color(col2.getRed()/2,col2.getGreen()/2,col2.getBlue()/2);
282
283                
284                
285                String cmd1 = "";
286                String cmd2 = "";
287
288                cmd1 += "select */"+1+"; ";
289                cmd1 += " color [" +chaincol1.getRed()+","+chaincol1.getGreen() +","+chaincol1.getBlue() +"];";
290
291                cmd2 += "select */"+2+"; ";
292                cmd2 += " color [" +chaincol2.getRed()+","+chaincol2.getGreen() +","+chaincol2.getBlue() +"];";
293
294                cmd1 += "select ";
295                cmd2 += "select ";
296
297                String[] pdb1s = alig.getPDBresnum1();
298                String[] pdb2s = alig.getPDBresnum2();
299
300
301                for ( int i =0 ; i< pdb1s.length;i++){
302
303                        String p1 = pdb1s[i];
304                        String p2 = pdb2s[i];
305
306                        cmd1 += p1 +"/1";
307                        cmd2 += p2 +"/2";
308
309                        if ( i <= pdb1s.length -2){
310                                cmd1 += ",";
311                                cmd2 += ",";
312                        }
313                }
314
315                cmd1 += "; color [" +col1.getRed()+","+col1.getGreen() +","+col1.getBlue() +"];";
316                cmd1 += " backbone 0.6;";   
317
318                cmd2 += "; color [" +col2.getRed()+","+col2.getGreen() +","+col2.getBlue() +"];";
319                cmd2 += " backbone 0.6;";   
320
321                //System.out.println(cmd1);
322                scripts[0] = cmd1;
323                scripts[1] = cmd2;
324
325                return scripts;
326        }
