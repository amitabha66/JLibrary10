092        public static Atom[] getAlignedAtoms1(AFPChain afpChain,Atom[] ca1){
093                List<Atom> atoms = new ArrayList<Atom>();
094
095                int blockNum = afpChain.getBlockNum();
096
097                int[] optLen = afpChain.getOptLen();
098                int[][][] optAln = afpChain.getOptAln();
099
100
101                for(int bk = 0; bk < blockNum; bk ++)       {
102
103
104                        for ( int i=0;i< optLen[bk];i++){
105                                int pos = optAln[bk][0][i];
106                                atoms.add(ca1[pos]);
107                        }
108
109                }
110                return (Atom[])atoms.toArray(new Atom[atoms.size()]);
111        }

152        public static void getAlign(AFPChain afpChain,Atom[] ca1,Atom[] ca2, boolean showSeq) {
153
154                char[] alnsymb = afpChain.getAlnsymb();
155                char[] alnseq1 = afpChain.getAlnseq1();
156                char[] alnseq2 = afpChain.getAlnseq2();
157
158                int     i, j, k, p1, p2, p1b, p2b, lmax;
159
160                int pro1Len = ca1.length;
161                int pro2Len = ca2.length;
162
163                p1b = p2b = 0;
164
165
166
167                if(alnsymb == null)     {
168                        alnseq1 = new char[pro1Len+pro2Len +1];
169                        alnseq2 = new char[pro1Len+pro2Len +1] ;
170                        alnsymb = new char[pro1Len+pro2Len+1];
171
172                        afpChain.setAlnseq1(alnseq1);
173                        afpChain.setAlnseq2(alnseq2);
174                        afpChain.setAlnsymb(alnsymb);
175                }
176
177
178
179                //              if ( afpChain.getAlgorithmName().equals(FatCatRigid.algorithmName)){
180                //                      showSeq = true;
181                //              }
182
183                int blockNum = afpChain.getBlockNum();
184
185                int[] optLen = afpChain.getOptLen();
186                int[][][] optAln = afpChain.getOptAln();
187
188                int alnbeg1 = afpChain.getAlnbeg1();
189                int alnbeg2 = afpChain.getAlnbeg2();
190                int alnLength = afpChain.getAlnLength();
191                int optLength = afpChain.getOptLength();
192
193                if ( optLen == null) {
194                        optLen = new int[blockNum];
195                        for (int oi =0 ; oi < blockNum ; oi++)
196                                optLen[oi] = 0;
197                }
198                int     len = 0;
199                for(i = 0; i < blockNum; i ++)  {            
200                        for(j = 0; j < optLen[i]; j ++) {
201                                p1 = optAln[i][0][j];
202                                p2 = optAln[i][1][j];
203
204                                // weird, could not find a residue in the Atom array. Did something change in the underlying data?
205                                if (( p1 == -1 ) || ( p2 == -1)) {
206                                        System.err.println("Could not get atom on position " + j );
207                                        continue;
208                                }
209                                if(len > 0)     {
210                                        lmax = (p1 - p1b - 1)>(p2 - p2b - 1)?(p1 - p1b - 1):(p2 - p2b - 1);
211                                        for(k = 0; k < lmax; k ++)      {
212                                                if(k >= (p1 - p1b - 1)) alnseq1[len] = '-';
213                                                else {
214                                                        char oneletter = getOneLetter(ca1[p1b+1+k].getGroup());
215                                                        alnseq1[len] = oneletter;
216                                                }
217                                                if(k >= (p2 - p2b - 1)) alnseq2[len] = '-';
218                                                else  {
219                                                        char oneletter = getOneLetter(ca2[p2b+1+k].getGroup());
220                                                        alnseq2[len] = oneletter;
221                                                }
222                                                alnsymb[len ++] = ' ';
223                                        }
224                                }
225                                else    {
226                                        alnbeg1 = p1; //the first position of sequence in alignment
227                                        alnbeg2 = p2;
228                                }
229                                
230                                if ( p1 < ca1.length && p2<ca2.length) {
231                                
232                                        alnseq1[len] = getOneLetter(ca1[p1].getGroup());              
233                                        alnseq2[len] = getOneLetter(ca2[p2].getGroup());
234                                } else {
235                                        alnseq1[len]='?';
236                                        alnseq2[len]='?';
237                                }
238                                if ( showSeq) {
239                                        if ( alnseq1[len] == alnseq2[len]){
240                                                alnsymb[len ++] = '|';
241                                        } else {
242                                                double score = aaScore(alnseq1[len], alnseq2[len]);
243
244                                                if ( score > 1)
245                                                        alnsymb[len ++] = ':';
246                                                else 
247                                                        alnsymb[len ++] = '.';
248                                        }
249                                } else {
250                                        String tmpS = String.format( "%d", i + 1);
251                                        alnsymb[len ++] = tmpS.charAt(0);
252                                }
253                                p1b = p1;
254                                p2b = p2;
255                        }
256                }
257                alnLength = len;
258
259
260                //              System.out.println(alnseq1);
261                //              System.out.println(alnsymb);
262                //              System.out.println(alnseq2);
263
264                afpChain.setOptAln(optAln);
265                afpChain.setOptLen(optLen);
266                afpChain.setAlnbeg1(alnbeg1);
267                afpChain.setAlnbeg2(alnbeg2);
268                afpChain.setAlnLength(alnLength);
269                afpChain.setGapLen(alnLength-optLength);
270
271
272
273
274        }

277
278                try {
279                        Character c = StructureTools.get1LetterCode(g.getPDBName());
280                        return c;
281                } catch (Exception e){
282                        return 'X';
283                }

286        public static int aaScore(char a, char b){
287                if (a == 'x')
288                        a= '-';
289                if (b == 'x')
290                        b='-';
291                if ( a == 'X')
292                        a = '-';
293                if ( b == 'X')
294                        b = '-';
295
296                int pos1 = aa1List.indexOf(a);
297                int pos2 = aa1List.indexOf(b);
298
299
300                // SEC an PYL amino acids are not supported as of yet...
301
302                if ( pos1 < 0) {
303                        System.err.println("unknown char " + a);
304                        return 0;
305                }
306                if ( pos2 < 0) {
307                        System.err.println("unknown char " + b);
308                        return 0;
309                }
310
311                return aaMatrix[pos1][pos2];
312        }

314        public static Map<String,Double> calcIdSimilarity(char[] seq1, char[] seq2, int alnLength){
315                double identity = 0.0;
316                double similarity = 0.0;
317                
318                if ( seq1 == null || seq2 == null){
319                        Map<String, Double> m = new HashMap<String, Double>();
320                        m.put("similarity", 0d);
321                        m.put("identity", 0d);
322                        return m;
323                }
324                
325                int     i;
326
327                for(i = 0; i < alnLength; i ++)        {
328                
329                                if(seq1[i] == seq2[i])  {
330
331                                        identity += 1.0;
332                                }
333                                else if(seq1[i] == '-' || seq1[i] == '*' || seq1[i] == '.' ||
334                                                seq2[i] == '-' || seq2[i] == '*' || seq2[i] == '.' )
335                                        continue;
336                                else if(aaScore(seq1[i], seq2[i]) > 0)  similarity += 1.0;           
337                }
338
339
340                similarity = (identity + similarity) / (double)alnLength;
341                identity = identity/(double)alnLength;
342
343                Map<String, Double> m = new HashMap<String, Double>();
344                m.put("similarity", similarity);
345                m.put("identity", identity);
346
347                //System.out.println(m);
348                return m;
349        }

352        public static Structure createArtificalStructure(AFPChain afpChain, Atom[] ca1,
353                        Atom[] ca2) throws Exception{
354
355
356                if ( afpChain.getNrEQR() < 1){
357                        return GuiWrapper.getAlignedStructure(ca1, ca2);
358                }
359
360                Group[] twistedGroups = GuiWrapper.prepareGroupsForDisplay(afpChain,ca1, ca2);
361
362                List<Atom> twistedAs = new ArrayList<Atom>();
363
364                for ( Group g: twistedGroups){
365                        if ( g == null )
366                                continue;
367                        if ( g.size() < 1)
368                                continue;
369                        Atom a = g.getAtom(0);
370                        twistedAs.add(a);
371                }
372                Atom[] twistedAtoms = (Atom[])twistedAs.toArray(new Atom[twistedAs.size()]);
373
374                List<Group> hetatms  = new ArrayList<Group>();
375                List<Group> nucs1    = new ArrayList<Group>();
376                Group g1 = ca1[0].getGroup();
377                Chain c1 = null;
378                if ( g1 != null) {
379                        c1 = g1.getChain();
380                        if ( c1 != null){
381                                hetatms = c1.getAtomGroups("hetatm");;
382                                nucs1  = c1.getAtomGroups("nucleotide");
383                        }
384                }
385                List<Group> hetatms2 = new ArrayList<Group>();
386                List<Group> nucs2    = new ArrayList<Group>();
387                Group g2 = ca2[0].getGroup();
388                Chain c2 = null;
389                if ( g2 != null){
390                        c2 = g2.getChain();
391                        if ( c2 != null){
392                                hetatms2 = c2.getAtomGroups("hetatm");
393                                nucs2 = c2.getAtomGroups("nucleotide");
394                        }
395                }
396                Atom[] arr1 = GuiWrapper.getAtomArray(ca1, hetatms, nucs1);
397                Atom[] arr2 = GuiWrapper.getAtomArray(twistedAtoms, hetatms2, nucs2);
398
399                Structure artificial = GuiWrapper.getAlignedStructure(arr1,arr2);
400                return artificial;
401        }
