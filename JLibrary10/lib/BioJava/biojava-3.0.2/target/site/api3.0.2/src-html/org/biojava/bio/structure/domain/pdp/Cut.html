012
013                int nclose = pdpMatrix.getNclose();
014
015                int[] iclose = pdpMatrix.getIclose();
016                int[] jclose = pdpMatrix.getJclose();
017
018                int[] contacts = new int[PDPParameters.MAXLEN];
019                double[] max_contacts = new double [PDPParameters.MAXLEN];
020                double[] contact_density = new double[PDPParameters.MAXLEN]; 
021                double average_density,x,y;
022
023                int endsf,endst;
024                int k,l,nc;
025                int size1,size2;
026                int size1t,size2t;
027                int size11,size22,size0;
028                int contactsd;
029                int iseg,jseg,kseg;
030                int from,to,from1,to1,from2,to2,lseg;
031
032
033                int site_min = -1;
034
035
036                // AP add sort here..
037                //qsort(dom.segment,dom.nseg,sizeof(struct Segment),segcmp);
038                // what is going on with the segments??
039
040                List<Segment> segments = dom.getSegments();
041
042                java.util.Collections.sort(segments, new SegmentComparator());
043
044                if ( verbose)
045                        System.out.println("  ---  Cut.cut " + dom + " ");
046                average_density = 0.0;
047                size0=0;
048                for(iseg=0;iseg<dom.nseg;iseg++) {
049                        contactsd=1;
050                        size1t=0;
051                        size2t=0;
052                        for(jseg=0;jseg<iseg;jseg++)
053                                size1t+=(dom.getSegmentAtPos(jseg).getFrom() - dom.getSegmentAtPos(jseg).getFrom() + 1);
054                        for(jseg=iseg+1;jseg<dom.nseg;jseg++)
055                                size2t+=(dom.getSegmentAtPos(jseg).getTo() - dom.getSegmentAtPos(jseg).getFrom() + 1);

058                                to1 = dom.getSegmentAtPos(jseg).getTo();
059                                for(int i=from1;i<to1;i++) {
060                                        for(kseg=iseg+1;kseg<dom.nseg;kseg++) {
061                                                from2 = dom.getSegmentAtPos(kseg).getFrom();
062                                                to2 = dom.getSegmentAtPos(kseg).getFrom();
063                                                for(int j=from2;j<to2;j++)
064                                                        if(Math.abs(i-j)>4) contactsd+=(int)(dist[i][j]);
065                                        }
066                                }
067                        }
068                        from = dom.getSegmentAtPos(iseg).getFrom();
069                        to = dom.getSegmentAtPos(iseg).getTo();
070                        for(k=from;k<to;k++) {
071                                contacts[k] = contactsd;
072                                /*
073        if(k==392) printf("init contacts = %d\n",contacts[k]);
074                                 */
075                                size11=size1t+(k-from+1);
076                                size22=size2t+(to-k);
077                                for(int i=from;i<=k;i++) {
078                                        for(kseg=iseg+1;kseg<dom.nseg;kseg++) {
079                                                from2 = dom.getSegmentAtPos(kseg).getFrom();
080                                                to2 = dom.getSegmentAtPos(kseg).getTo();
081                                                for(int j=from2;j<=to2;j++)
082                                                        if(Math.abs(i-j)>4) contacts[k]+=(int)(dist[i][j]);
083                                        }
084                                }
085                                /*
086        if(k==392) printf("[from,k]x]iseg,nseg[ = %d\n",contacts[k]);
087                                 */
088                                for(int i=from;i<=k;i++) {
089                                        for (int j=k+1;j<=to;j++) 
090                                                if(Math.abs(i-j)>4) contacts[k]+=(int)(dist[i][j]);
091                                }
092                                /*
093        if(k==392) printf("[from,k]x]k,to[ = %d\n",contacts[k]);
094                                 */
095                                for(int i=k+1;i<=to;i++) {
096                                        for(kseg=0;kseg<iseg;kseg++) {
097                                                from2 = dom.getSegmentAtPos(kseg).getFrom();
098                                                to2 = dom.getSegmentAtPos(kseg).getTo();
099                                                for(int j=from2;j<to2;j++)
100                                                        if(Math.abs(i-j)>4) contacts[k]+=(int)(dist[j][i]);
101                                        }
102                                }
103                                /*
104        if(k==392) printf("]k,to]x]0,iseg[ = %d\n",contacts[k]);
105                                 */
106                                size1=Math.min(size11,size22);
107                                size2=Math.max(size11,size22);
108                                x=Math.min(PDPParameters.MAXSIZE,size1);
109                                y=Math.min(PDPParameters.MAXSIZE,size2);
110                                if(x>150&&y>1.5*x) y=1.5*x;
111                                else if(y>2*x) y=2*x;
112                                /*
113                                 */
114                                x=Math.min(Math.pow(x,1.3/3)+PDPParameters.RG,Math.pow(x,1.1/3)+Math.pow(PDPParameters.TD,1.3/3)+PDPParameters.RG);
115                                y=Math.min(Math.pow(y,1.3/3)+PDPParameters.RG,Math.pow(y,1.1/3)+Math.pow(PDPParameters.TD,1.3/3)+PDPParameters.RG);
116                                /* max_ contacts depend on the size of domains */
117                                /* stella wanted comments at this point */
118                                /*
119                        max_contacts[k] = min(MAXCONT,x*y);
120                                 */
121                                max_contacts[k] = 10*x*y;
122                                if(size1>150) max_contacts[k] = 9*x*y;
123                                contact_density[k]=(double)contacts[k]/max_contacts[k];
124                                /*
125                        if(contact_density[k]>2.5) contact_density[k]=2.5;
126                                 */
127                                /*
128        if(first_cut) 
129                if(k==277 && !first_cut)
130                printf("k=%d s1=%d s2=%d x=%f y=%f mc=%d c=%d cd=%f\n",k,size1,size2,x,y,max_contacts[k],contacts[k],contact_density[k]);
131                                 */
132                                //if(verbose) System.out.println(String.format("%d      %d      %d      %f      %f      %f      %d      %f",k,size1,size2,x,y,max_contacts[k],contacts[k],contact_density[k]));
133                                /*
134                                 */
135                                if(from==0) endsf = PDPParameters.ENDSEND;
136                                else endsf = PDPParameters.ENDS;
137                                if(to==ca.length-1) endst = PDPParameters.ENDSEND;
138                                else endst = PDPParameters.ENDS;
139                                if((contact_density[k])<val.s_min&&k>from+endsf&&k<to-endst) {
140                                        val.s_min = (contact_density[k]);
141                                        site_min=k+1;
142                                }
143                                if(k>from+endsf&&k<to-endst) {
144                                        average_density+=contact_density[k];
145                                        size0++;
146                                }
147                        }
148                }
149                average_density/=size0;
150                if(verbose) System.out.println(String.format("  --- Trying to cut domain of size %d having %d segments and  average cont_density %f\n",dom.size,dom.nseg,average_density));
151
152                if ( verbose )
153                        for(kseg=0;kseg<dom.nseg;kseg++) 
154                                System.out.println(String.format("      --- segment %d from %d to %d",kseg,dom.getSegmentAtPos(kseg).getFrom(),dom.getSegmentAtPos(kseg).getTo()) + " av density: " + average_density);
155
156
157                if(val.first_cut) {
158                        /*
159                for(k=from+ENDS;k<to-ENDS;k++)
160                        printf("%d      %s      %d      %d      %d      %d      %f\n",k,protein.res[k].type,k-from,to-k,contacts[k],max_contacts[k],contact_density[k]);
161                         */
162                        val.AD = average_density;
163                        if(verbose)     System.out.println(String.format("  --- AD=%f", average_density));
164                        /*
165                         */
166                }
167                val.AD = average_density;
168
169                val.s_min/=val.AD;
170
171                if(verbose) System.out.println(String.format("  --- after single cut: s_min = %f site_min = %d",val.s_min,site_min));
172
173                ///
174                k=0;
175
176                /* check double cuts */
177                if ( verbose )
178                System.out.println("  --- checking double cuts up to: " + nclose);
179                nc=0;
180                for(l=0;l<nclose;l++) {
181
182                        /************ find iseg, jseg ****************/
183                        iseg=jseg=-1;
184                        for(kseg=0;kseg<dom.nseg;kseg++) {
185                                from=dom.getSegmentAtPos(kseg).getFrom();
186                                to=dom.getSegmentAtPos(kseg).getTo();
187                                if(from==0) endsf = PDPParameters.ENDSEND;
188                                else endsf = PDPParameters.ENDS;
189                                if(to==ca.length-1) endst = PDPParameters.ENDSEND;
190                                else endst = PDPParameters.ENDS;
191                                if(iclose[l]>from+endsf&&iclose[l]<to-endst)
192                                        iseg=kseg;
193                                if(jclose[l]>from+endsf&&jclose[l]<to-endst)
194                                        jseg=kseg;
195                        }
196                        if(iseg<0||jseg<0) continue;
197                        /*
198        printf("l = %d iclose[l] = %d jclose[l] = %d\n",l,iclose[l],jclose[l]);
199                         */
200                        /*********************************************/
201
202                        from=dom.getSegmentAtPos(iseg).getFrom();
203                        to=dom.getSegmentAtPos(iseg).getTo();
204                        from1=dom.getSegmentAtPos(jseg).getFrom();
205                        to1=dom.getSegmentAtPos(jseg).getTo();
206
207                        /************ count contacts *****************/
208                        contacts[nc] = 1;
209                        //no=0;
210
211                        /******* contacts between [0,iseg[ and ]iseg,jseg[ ********/
212                        for(kseg=0;kseg<iseg;kseg++)
213                                for(lseg=iseg+1;lseg<jseg;lseg++)
214                                        for( int i=dom.getSegmentAtPos(kseg).getFrom();i<dom.getSegmentAtPos(kseg).getTo();i++)
215                                                for(int j=dom.getSegmentAtPos(lseg).getFrom();j<dom.getSegmentAtPos(lseg).getTo();j++) {
216                                                        contacts[nc]+=(dist[i][j]);
217                                                }
218
219                        //System.out.println(String.format("[0,iseg[ - ]iseg,jseg[ : %d\n",contacts[nc]-no));
220
221                        //no=contacts[nc];
222                        /**********************************************************/
223                        /**********************************************************/
224                        /******* contacts between ]jseg,nseg[ and ]iseg,jseg[ ********/
225                        for(kseg=jseg+1;kseg<dom.nseg;kseg++)
226                                for(lseg=iseg+1;lseg<jseg;lseg++)
227                                        for(int i=dom.getSegmentAtPos(kseg).getFrom();i<dom.getSegmentAtPos(kseg).getTo();i++)
228                                                for(int j=dom.getSegmentAtPos(lseg).getFrom();j<dom.getSegmentAtPos(lseg).getTo();j++) {
229                                                        contacts[nc]+=(dist[j][i]);
230                                                }
231                        /*
232                printf("]jseg,nseg] - ]iseg,jseg[ : %d\n",contacts[nc]-no);
233                         */
234                        //no=contacts[nc];
235                        /*************************************************************/
236                        /*************************************************************/
237                        /**** contacts between [from,iclose] in iseg and ]iseg,jseg[ ****/
238                        if(iseg==jseg) {
239                                //System.out.println(" CONTACT:  " + from + " " + iclose[l] + " " + iseg + " " + jseg);
240                                for(int i=from;i<=iclose[l];i++) {
241                                        for (int j=iclose[l]+1;j<=jclose[l];j++) {
242                                                contacts[nc]+=(dist[i][j]);
243                                        }
244                                }
245                                for (int j=iclose[l]+1;j<jclose[l];j++) {
246                                        for(kseg=0;kseg<iseg;kseg++) 
247                                                for(int i=dom.getSegmentAtPos(kseg).getFrom();i<dom.getSegmentAtPos(kseg).getTo();i++) {
248                                                        contacts[nc]+=(dist[i][j]);
249                                                }
250                                        for(int i=jclose[l];i<to;i++) {
251                                                contacts[nc]+=(dist[j][i]);
252                                        }
253                                        for(kseg=iseg+1;kseg<dom.nseg;kseg++) 
254                                                for(int i=dom.getSegmentAtPos(kseg).getFrom();i<dom.getSegmentAtPos(kseg).getTo();i++) {
255                                                        contacts[nc]+=(dist[j][i]);
256                                                }
257                                }
258                                /*
259                printf("iclose==jclose : %d\n",contacts[nc]-no);
260                                 */
261                                //no=contacts[nc];
262                        }
263                        else {
264                                //System.out.println(" ISEG!=JSEG " + " " + from + " " + iclose[l]);
265                                for(int i=from;i<=iclose[l];i++) { 
266                                        for(kseg=iseg+1;kseg<jseg;kseg++) 
267                                                for(int j=dom.getSegmentAtPos(kseg).getFrom();j<dom.getSegmentAtPos(kseg).getTo();j++) {
268                                                        contacts[nc]+=(dist[i][j]);
269                                                }
270                                        for(int j=from1;j<jclose[l];j++) {
271                                                contacts[nc]+=(dist[i][j]);
272                                        }
273                                        for(int j=iclose[l]+1;j<to;j++) {
274                                                contacts[nc]+=(dist[i][j]);
275                                        }
276                                }
277                                for(int i=iclose[l]+1;i<to;i++) {
278                                        for(kseg=0;kseg<iseg;kseg++) 
279                                                for(int j=dom.getSegmentAtPos(kseg).getFrom();j<dom.getSegmentAtPos(kseg).getTo();j++) {
280                                                        contacts[nc]+=(dist[j][i]);
281                                                }
282                                        for(kseg=jseg+1;kseg<dom.nseg;kseg++) 
283                                                for(int j=dom.getSegmentAtPos(kseg).getFrom();j<dom.getSegmentAtPos(kseg).getTo();j++) {
284                                                        contacts[nc]+=(dist[i][j]);
285                                                }
286                                        for(int j=jclose[l];j<=to1;j++) {
287                                                contacts[nc]+=(dist[i][j]);
288                                        }
289                                }
290                                for (int i=from1;i<jclose[l];i++) {
291                                        for(kseg=0;kseg<iseg;kseg++) 
292                                                for(int j=dom.getSegmentAtPos(kseg).getFrom();j<dom.getSegmentAtPos(kseg).getTo();j++) {
293                                                        contacts[nc]+=(dist[j][i]);
294                                                }
295                                        for(kseg=jseg+1;kseg<dom.nseg;kseg++)  {
296                                                for(int j=dom.getSegmentAtPos(kseg).getFrom();j<dom.getSegmentAtPos(kseg).getTo();j++)
297                                                        contacts[nc]+=(dist[i][j]);
298                                        }
299                                        for(int j=jclose[l];j<to1;j++) {
300                                                contacts[nc]+=(dist[i][j]);
301                                        }
302                                }
303                                for(int i=jclose[l];i<to1;i++)
304                                        for(kseg=iseg+1;kseg<jseg;kseg++) 
305                                                for(int j=dom.getSegmentAtPos(kseg).getFrom();j<dom.getSegmentAtPos(kseg).getTo();j++) {
306                                                        /*
307                                                if(iclose[l]==33&&jclose[l]==69&&dist[i][j]) printf("%d %s %d %s %d\n",i,protein.res[i].type,j,protein.res[j].type,dist[i][j]);
308                                                         */
309                                                        contacts[nc]+=(dist[j][i]);
310                                                }
311                        }
312                        /*******************************************************************/
313                        /*******************************************************************/
314                        /*******************************************************************/
315                        size11=0;
316                        size22=0;
317                        for(kseg=0;kseg<iseg;kseg++) 
318                                size11+=(dom.getSegmentAtPos(kseg).getTo()-dom.getSegmentAtPos(kseg).getFrom()+1);
319                        for(kseg=jseg+1;kseg<dom.nseg;kseg++) 
320                                size11+=(dom.getSegmentAtPos(kseg).getTo()-dom.getSegmentAtPos(kseg).getFrom()+1);
321                        size11+=(iclose[l]-from+1);
322                        size11+=(to1-jclose[l]+1);
323                        /*
324        printf("size11 = %d from = %d to1 = %d \n",size11,from,to1);
325                         */
326                        for(kseg=iseg+1;kseg<jseg;kseg++) 
327                                size22+=(dom.getSegmentAtPos(kseg).getTo()-dom.getSegmentAtPos(kseg).getFrom()+1);
328                        if(iseg==jseg)
329                                size22+=(jclose[l]-iclose[l]);
330                        else {
331                                size22+=(jclose[l]-from1);
332                                size22+=(to-iclose[l]);
333                        }
334                        /*
335        printf("size22 = %d from1 = %d jclose %d iclose %d \n",size22,from1,jclose[l],iclose[l]);
336                         */
337                        size1=Math.min(size11,size22);
338                        size2=Math.max(size11,size22);
339                        x=Math.min(PDPParameters.MAXSIZE,size1);
340                        y=Math.max(PDPParameters.MAXSIZE,size2);
341                        if(y>2*x) y=2*x;
342                        /*
343                         */
344                        x=Math.min(Math.pow(x,1.3/3)+PDPParameters.RG,Math.pow(x,1.1/3)+Math.pow(PDPParameters.TD,1.3/3)+PDPParameters.RG);
345                        y=Math.min(Math.pow(y,1.3/3)+PDPParameters.RG,Math.pow(y,1.1/3)+Math.pow(PDPParameters.TD,1.3/3)+PDPParameters.RG);
346                        /*
347                max_contacts[nc] = min(MAXCONT,x*y);
348                         */
349                        max_contacts[nc] = x*y*10;
350                        if(size1>150) max_contacts[k] = 9*x*y;
351                        contact_density[nc]=(double)contacts[nc]/(double)max_contacts[nc];
352                        /*
353        if(first_cut)
354                         */
355                        //if ( verbose)
356                        //      System.out.println("double cut" + l + " " + ca[iclose[l]].getGroup().getType() + " " + iclose[l] + " " + jclose[l] + " " + contacts[nc]+
357                        //                      " " + max_contacts[nc] + " " + x+ " " + y + " " + size11 + " " + size22 + " " + contact_density[nc] + " " + contact_density[nc]/val.AD );
358                        //if(verbose) System.out.println(String.format(" double cut: %d %s %d %d c=%d mc=%d x=%f y=%f s1=%d s2=%d cd=%f cd/ad=%f\n",l,ca[iclose[l]].getGroup().getType(),iclose[l],jclose[l],contacts[nc],max_contacts[nc],x,y,size11,size22,contact_density[nc],contact_density[nc]/val.AD));
359                        if((contact_density[nc]/val.AD+PDPParameters.DBL)<val.s_min&&contact_density[nc]/val.AD+PDPParameters.DBL<PDPParameters.CUT_OFF_VALUE2) {
360                                /*
361        printf("________________\n");
362                                 */
363                                val.s_min = (contact_density[nc]/val.AD)+PDPParameters.DBL;
364                                site_min=iclose[l];
365                                val.site2=jclose[l];
366                        }
367
368                        nc++;
369                        if ( nc >= PDPParameters.MAXSIZE)
370                                nc = PDPParameters.MAXSIZE-1;
371                }
372                val.first_cut=false;
373                if(verbose) 
374                        System.out.println(String.format("  --- E ... at the end of cut: s_min %f CUTOFF %f site_min %d *site2 %d",val.s_min,PDPParameters.CUT_OFF_VALUE,site_min,val.site2));
375                if(val.s_min> PDPParameters.CUT_OFF_VALUE) return -1;
376
377                return(site_min);
378        }
