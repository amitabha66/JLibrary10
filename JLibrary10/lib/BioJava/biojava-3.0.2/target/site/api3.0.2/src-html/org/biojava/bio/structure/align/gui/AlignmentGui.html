348   private void calcAlignment() {
349
350      int pos = tabPane.getSelectedIndex();
351      StructurePairSelector tab = null;
352
353      if (pos == 0 ){
354         tab = tab1;         
355        
356      } else if (pos == 1){
357         tab = tab2;
358      
359      } else if (pos == 2){
360         tab = tab3;
361      }
362      
363      
364      try {
365         Structure s1 = tab.getStructure1();
366         Structure s2 = tab.getStructure2();
367
368         if ( s1 == null) {
369            System.err.println("please select structure 1");
370            return ;
371         }
372
373         if ( s2 == null) {
374            System.err.println("please select structure 2");
375            return;
376         }
377
378         String name1 = "custom1"; 
379         String name2 = "custom2";
380
381         if  ( pos == 0){
382            name1 = tab1.getName1();
383            name2 = tab1.getName2();
384         } else {
385            name1 = s1.getPDBCode();
386            name2 = s2.getPDBCode();
387         }
388
389
390         alicalc = new AlignmentCalc(this,s1,s2, name1, name2);
391
392
393         thread = new Thread(alicalc);
394         thread.start();
395         abortB.setEnabled(true);
396         progress.setIndeterminate(true);
397         ProgressThreadDrawer drawer = new ProgressThreadDrawer(progress);
398         drawer.start();
399      } catch (StructureException e){
400         JOptionPane.showMessageDialog(null,"Could not align structures. Exception: " + e.getMessage());
401      }
402
403   }

406   private void calcDBSearch() {
407
408      JTabbedPane tabPane = dbsearch.getTabPane();
409      System.out.println("run DB search " + tabPane.getSelectedIndex());
410
411      Structure s = null;
412
413
414      StructurePairSelector tab = null;
415      int pos = tabPane.getSelectedIndex();
416
417      if (pos == 0 ){
418
419         tab = dbsearch.getSelectPDBPanel();
420         
421   
422
423      } else if (pos == 1){
424
425         tab = dbsearch.getPDBUploadPanel();
426
427      }
428
429      try {
430
431         s = tab.getStructure1();
432
433         if ( s == null) {
434            System.err.println("please select structure 1");
435            return ;
436         }
437
438      } catch (Exception e){
439         e.printStackTrace();
440      }
441
442      String name1 = "custom1"; 
443
444
445      if  ( pos == 0) {
446
447         name1 = dbsearch.getSelectPDBPanel().getName1();
448
449      } else {
450
451         name1 = s.getPDBCode();
452
453      }
454
455      System.out.println("name1 in alig gui:" + name1);
456      String file = dbsearch.getOutFileLocation();
457      if ( file == null || file.equals("")){
458         JOptionPane.showMessageDialog(null,"Plrease select a directory to contain the DB search results.");
459         return;
460      }
461
462      UserConfiguration config = WebStartMain.getWebStartConfig();
463      
464      int totalNrCPUs = Runtime.getRuntime().availableProcessors();
465      
466      int useNrCPUs = 1;
467      if ( totalNrCPUs > 1){
468          Object[] options = new Integer[totalNrCPUs];
469          int posX = 0;
470          for ( int i = totalNrCPUs; i> 0 ; i--){
471                  options[posX] = i;
472                  posX++;
473          }
474          int n = JOptionPane.showOptionDialog(null,
475                          "How many would you like to use for the calculations?",
476                          "We detected " + totalNrCPUs + " processors on your system.",                           
477                          JOptionPane.OK_CANCEL_OPTION,
478                          JOptionPane.QUESTION_MESSAGE,
479                          null,
480                          options,
481                          options[0]);
482          
483          if ( n < 0)
484                  return;
485          useNrCPUs = (Integer) options[n];
486          System.out.println("will use " + useNrCPUs + " CPUs." );
487      }
488      
489      alicalc = new AlignmentCalcDB(this, s,  name1,config,file);
490      alicalc.setNrCPUs(useNrCPUs);
491      abortB.setEnabled(true);
492      progress.setIndeterminate(true);
493      ProgressThreadDrawer drawer = new ProgressThreadDrawer(progress);
494      drawer.start();
495
496      Thread t = new Thread(alicalc);
497      t.start();
498   }

540class ProgressThreadDrawer extends Thread {
541
542   JProgressBar progress;
543   static int interval = 300;
544
545   public ProgressThreadDrawer(JProgressBar progress) {
546      this.progress = progress;
547   }
548
549
550   public void run() {
551      progress.setVisible(true);
552      boolean finished = false;
553      while ( ! finished) {
554         try {
555            progress.repaint();
556            if ( ! progress.isIndeterminate() ){
557               finished =false;
558               break;
559            }
560
561            sleep(interval);
562         } catch (InterruptedException e){
563         }
564         progress.repaint();
565      }
566      progress.setVisible(false);
567      progress = null;
568   }
