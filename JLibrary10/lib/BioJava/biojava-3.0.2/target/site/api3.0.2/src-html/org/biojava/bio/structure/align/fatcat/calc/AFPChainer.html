050  /**
051  // Key function: chain (assembly) the AFPs
052  // a AFP (k) is defined as (i, j, k), with i and j are staring points
053  // AFP extension (eg. AFP(k-1) -> AFP(k) ) requiring
054  // AFP(k-1) < AFP(k)(refer AFP.h definition),
055  // ie i(k-1) < i(k) and j(k-1) < j(k)
056  // in the figure, only (2) AFP can extend to AFP(k)
057  // Key features: a coordination transformation is allowed in the AFP extension
058  //                      gap penalties are also considered
059  //
060  //                                   protein1
061  //                 ---------------------------
062  //                 |        \                |
063  //                 |         \(1)            |
064  //                 |     \    \              |
065  //                 |      \(2) \             |
066  //              p  |       \                 |
067  //              r  |   \                     |
068  //              o  |    \(3)  \(i,j, k)      |
069  //              t  |     \     \             |
070  //              e  |            \            |
071  //              i  |                         |
072  //              n  |                         |
073  //              2  ---------------------------
074  //                  schematic of AFP chaining
075  */
076  public static void doChainAfp(FatCatParameters params, AFPChain afpChain,Atom[] ca1, Atom[] ca2){
077     List<AFP> afpSet = afpChain.getAfpSet();
078     
079     afpChain.setConn(0d);
080     afpChain.setDVar(0d);
081     
082     int afpNum = afpSet.size();
083     if(afpNum <= 0) return;
084
085
086     int[] twi = afpChain.getTwi();
087     if(twi == null) {
088        twi = new int[afpNum];
089        afpChain.setTwi(twi);
090     }
091     
092     //transformation, calculated at DoChainAfp, be used in List extraction
093
094     //forward: calculate the score matrix
095     boolean isConnected = false;
096     int     i, j, j0,  n;
097     double  stmp;
098     
099     afpChain.setConn(0d);
100     afpChain.setDVar(0d);
101     
102     double[]  sco = new double[afpNum]; //the score ending at an AFP
103     int[] pre = new int[afpNum];    //the previous AFP
104     double  maxsco = 0;
105     int     maxafp = 0;
106     int[] list = new int[afpNum];
107
108     int maxGap = params.getMaxGap();
109     int fragLen = params.getFragLen();
110     int maxTra = params.getMaxTra();
111    
112     
113     Matrix disTable1 = getDisTable(maxGap + 2 * fragLen + 1,ca1);
114     Matrix disTable2 = getDisTable(maxGap + 2 * fragLen + 1,ca2);
115     
116     afpChain.setDisTable1(disTable1);
117     afpChain.setDisTable2(disTable2);
118
119     for(i = 0; i < afpNum; i ++)    {
120        sco[i] = afpSet.get(i).getScore(); //start from itself
121        pre[i] = -1;
122        twi[i] = 0;
123        if ( afpSet.get(i).getP1() < fragLen || afpSet.get(i).getP2() < fragLen)
124           n = 0;
125        else
126           n = getCompatibleAfps(i, list, params, afpChain); //get a compatible list
127        //printf("afp %d, compatible %d\n", i, n);
128        for(j0 = 0; j0 < n; j0 ++)      {
129           j = list[j0];
130           isConnected = afpPairConn(j, i, params,afpChain); //note: j, i
131           Double conn = afpChain.getConn();
132           int t = 0;
133           if ( isConnected)
134              t=1;
135           if(twi[j] + t > maxTra) continue;
136           //two many transformation are disfavored
137           stmp = sco[j] + afpSet.get(i).getScore() + conn;
138           if(stmp > sco[i])       { //considered all previous compatible AFPs
139              sco[i] = stmp;
140              twi[i] = twi[j] + t;
141              pre[i] = j;
142           }
143        }
144        if(maxsco < sco[i])     {
145           maxsco = sco[i];
146           maxafp = i;
147        }
148     }
149
150     int     currafp = maxafp;
151     if(debug)
152        System.out.println(String.format("maximum score %f, %d\n", maxsco, twi[currafp]));
153
154     //trace-back from maxafp (maxsco)
155     
156     afpChain.setAlignScore(maxsco);
157     afpChain.setAlignScoreUpdate(maxsco);
158     afpChain.setAfpChainTwiNum(0);
159     
160     traceBack(pre, currafp, twi[currafp],params,afpChain,ca1,ca2);
161
162
163  }

165  private static Matrix getDisTable(int maxlen, Atom[]ca)
166
167  {
168     int length = ca.length;
169     Matrix dis = new Matrix(length,length);
170
171     int     i, j;
172     for(i = 0; i < length; i ++)    {
173        dis.set(i,i,0);
174        for(j = i + 1;( j < length) && (j <= i + maxlen); j ++)     {
175           dis.set(i,j,0);
176
177           try {
178
179              double val = dis.get(i,j) + (Calc.getDistance(ca[i],ca[j])) * (Calc.getDistance(ca[i],ca[j]));
180              dis.set(i,j,val);
181           } catch (StructureException e){
182              e.printStackTrace();
183           }
184
185           dis.set(i,j,Math.sqrt(dis.get(i,j)));
186           dis.set(j,i,dis.get(i,j));
187        }
188     }
189     return dis;
190
191  }

193  /*
194
195  derive the compabitle AFP lists for AFP-chaining
196  this is important for speeding up the process
197  for a given AFP(i1,j1), there are three regions that could be the starting
198  point for the compabitle AFPs of AFP(i1,j1)
199  //                 a1        a2   a3
200  //               i1-G    i1-f-c i1-f+1 i1
201  //                 |          |   |   |
202  //              ----------------------------
203  //              | B               |   |
204  //b1  j1-G  -|  ---------------|   |
205  //              |  |          |   |   |
206  //              |  |     C    | 3 |   |
207  //              |  |          |   |   |
208  //b2 j1-f-c -|  |--------------|   |
209  //              |  |     2    | 1 |   |
210  //b3 j1-f+1 -|------------------   |
211  //              |                   A |
212  //          j1 -|---------------------\
213  //              |                      \ (AFP(i1,j1))
214  //              -----------------------------
215  //
216  f: the length of AFPs (we use segments of same length)
217  G: g + f, where g is the maximum allowed gaps
218  c: the maximum allowed cross-over in AFP-connection,
219     here we use c = f, and j1-f-c = j1-2f
220  incompatible region A: its AFPs overlap with given AFP(i1,j1)
221  incompatible region B: the gaps between its AFP with given AFP is larger than g
222  incompatible region C: its AFPs connect with given AFP but cross a given threshold.
223  compatible region 1: [i1-f-c,i1-f+1>,[j1-f-c,j1-f+1> or [a2,a3],[b2,b3]
224  compatible region 2: [i1-G,i1-f-c],[j1-f-c,j1-f] or [a1,a2],[b2,b3]
225  combine 1 and 2    : [i1-G,i1-f],[j1-f-c,j1-f]   or [a1,a3],[b2,b3]
226  compatible region 3: [i1-f-c,i1-f],[j1-G,j1-f-c] or [a2,a3],[b1,b2]
227  c->misCut
228  f->fragLen
229  G->fragLen+maxGap->maxGapFrag
230   *
231   *
232   */
233  private  static int getCompatibleAfps(int afp, int[] list, FatCatParameters params, AFPChain afpChain){
234
235     int     i, j, i1, j1, f, G, c, a1, a2, a3, b1, b2, b3, s1, s2;
236
237     int fragLen = params.getFragLen();
238     int maxGapFrag = params.getMaxGapFrag();
239     int misCut = params.getMisCut();
240     int maxTra = params.getMaxTra();
241     List<AFP> afpSet = afpChain.getAfpSet();
242     
243     f = fragLen;
244     G = maxGapFrag;
245     c = misCut;
246
247     i1 = afpSet.get(afp).getP1();
248     j1 = afpSet.get(afp).getP2();
249     a3 = i1 - f;
250     a2 = a3 - c;
251     a1 = i1 - G;
252     a2 = a2>0?a2:0;
253     a1 = a1>0?a1:0;
254
255     b3 = j1 - f;
256     b2 = b3 - c;
257     b1 = j1 - G;
258     b2 = (b2 > 0)?b2:0;
259     b1 = (b1 > 0)?b1:0;
260
261     int[][] afpAftIndex = afpChain.getAfpAftIndex();
262     int[][] afpBefIndex = afpChain.getAfpBefIndex();
263     int[] twi = afpChain.getTwi();
264     
265     
266     
267     int     n = 0;
268     //compatible region 1-2, [a1,a3][b2,b3]
269     for(i = a1; i <= a3; i ++)      {//i <= a3 instead of i < a3
270        s1 = afpAftIndex[i][b2]; //note afpAftIndex, not afpIndex
271        if(s1 < 0)      continue;//no AFP for the given i with j > b2
272        s2 = afpBefIndex[i][b3]; //afps is sorted by j given a i,it's sparse matrix
273        if(s2 < 0)      continue;//no AFP for the given i with j < b3
274        for(j = s1; j <= s2; j ++)      { //j <= s2 instead of j < s2
275           if(twi[j] <= maxTra)    {
276              list[n ++] = j;
277           }
278        }
279     }
280
281     //compatible region 3  [a2,a3][b1,b2]
282     for(i = a2; i <= a3; i ++)      {
283        s1 = afpAftIndex[i][b1];
284        if(s1 < 0)      continue;
285        s2 = afpBefIndex[i][b2]; //afps is sorted by j given a i
286        if(s2 < 0)      continue;
287        //note j < s2, as the cases of j == s2 is alread considered in previous region
288        for(j = s1; j < s2; j ++)       {
289           if(twi[j] <= maxTra)    {
290              list[n ++] = j;
291           }
292        }
293     }
294
295     return n;
296
297  }

300  /**
301  //Key function: calculate the connectivity of AFP pairs
302  //no compatibility criteria is executed
303  //note: afp1 is previous to afp2 in terms of the position
304     //this module must be optimized
305   *
306   * @param afp1
307   * @param afp2
308   * @return flag if they are connected
309   */
310  public static boolean afpPairConn(int afp1, int afp2,  FatCatParameters params, AFPChain afpChain)
311 
312  {
313     
314     Double conn = afpChain.getConn();
315     Double dvar = afpChain.getDVar();
316     
317     double misScore = params.getMisScore();
318     double maxPenalty = params.getMaxPenalty();
319     double disCut = params.getDisCut();
320     double gapExtend = params.getGapExtend();
321     double torsionPenalty = params.getTorsionPenalty();
322     double disSmooth = params.getDisSmooth();
323     
324     List<AFP> afpSet = afpChain.getAfpSet();
325     
326     int     m = calcGap(afpSet.get(afp2),afpSet.get(afp1));
327     int     g = calcMismatch(afpSet.get(afp2),afpSet.get(afp1));
328
329
330     double  gp = misScore * m;      //on average, penalty for a mismatch is misScore, no modification on score
331     if(g > 0)       {
332        gp += gapExtend * g;
333     }
334     if(gp < maxPenalty)     gp = maxPenalty; //penalty cut-off
335     //note: use < (smaller) instead of >, because maxPenalty is a negative number
336
337     double  d;
338     d = calAfpDis(afp1, afp2,params, afpChain);
339     //note: the 'dis' value is numerically equivalent to the 'rms' with exceptions
340
341     boolean     ch = false;
342     double  tp = 0.0;
343     if(d >= disCut) {
344        tp = torsionPenalty;
345        ch = true;
346     } //use the variation of the distances between AFPs
347     else  if(d > disCut - disSmooth)        {
348        double  wt = Math.sqrt((d - disCut + disSmooth) / disSmooth);
349        //using sqrt: penalty increase with dis more quicker than linear function
350        tp = torsionPenalty * wt;
351     }
352
353     dvar = d;
354     conn = tp + gp;
355
356     afpChain.setConn(conn);
357     afpChain.setDVar(dvar);
358     return ch;
359  }

361  /**
362   * return the gaps between this and afp
363   * requiring afp1 >  afp2
364   * ( operator % in C)
365   */
366
367  private static int  calcGap(AFP afp1 , AFP afp2)
368  {
369     int     g = (afp1.getP1() - afp2.getP1()) - (afp1.getP2() - afp2.getP2());
370     if(g < 0)       g = -g;
371     return g;
372  }

374  /**
375   * return the mis-matched between this and afp
376   *     requiring  AFP1 > afp2
377   *     (operator / in C)
378   * @param afp1
379   * @param afp2
380   * @return
381   */
382
383  //--------------------------------------------
384  private static int calcMismatch(AFP afp1, AFP afp2)
385  {
386     int     l1 = afp1.getP1() - afp2.getP1() - afp2.getFragLen();
387     int     l2 = afp1.getP2() - afp2.getP2() - afp2.getFragLen();
388     return (l1 > l2?l2:l1);
389  }

391  /**
392  //return the root mean square of the distance matrix between the residues
393  //from the segments that form the given AFP list
394  //this value can be a measurement (2) for the connectivity of the AFPs
395  //and its calculation is quicker than the measurement (1), rmsd
396  //currently only deal with AFP pair
397//
398//           |-d1--|
399//          |--d2---|
400//         |---d3----|
401  //-----------------------------------------------------------------------
402  //this module is optimized
403   *
404   * @param afp1
405   * @param afp2
406   * @return
407   */
408  private static double calAfpDis(int afp1, int afp2, FatCatParameters params, AFPChain afpChain)
409  {
410   
411     List<AFP> afpSet = afpChain.getAfpSet();
412     
413     Matrix disTable1 = afpChain.getDisTable1();
414     Matrix disTable2 = afpChain.getDisTable2();
415     
416     int fragLen = params.getFragLen();
417     double afpDisCut = params.getAfpDisCut();
418     double disCut = params.getDisCut();
419     double fragLenSq = params.getFragLenSq();
420     
421     int     i, j, ai, bi, aj, bj;
422     double  d;
423     double  rms = 0;
424     for(i = 0; i < fragLen; i ++)   {
425        ai = afpSet.get(afp1).getP1() + i;
426        bi = afpSet.get(afp1).getP2() + i;
427        for(j = 0; j < fragLen; j ++)   {
428           aj = afpSet.get(afp2).getP1() + j;
429           bj = afpSet.get(afp2).getP2() + j;
430           d = disTable1.get(aj,ai) - disTable2.get(bj,bi);
431           rms += d * d;
432           if(rms > afpDisCut)     { return (disCut); }
433        }
434     }
435     return (Math.sqrt(rms / fragLenSq));
436  }

439  /**
440   * derive the optimal chaining of AFPs by trace-back
441   */
442  private static void traceBack(int[] pre, int currafp0, int twist, FatCatParameters params, AFPChain afpChain,Atom[] ca1, Atom[]ca2)
443  {
444   
445     afpChain.setConn(0d);
446     afpChain.setDVar(0d);
447     
448     int minLen = afpChain.getMinLen();
449     List<AFP> afpSet = afpChain.getAfpSet();
450     
451     int afpChainLen = 0;
452     
453     //trace-back from currafp (maxsco)
454     int[]     afpchain    = new int[minLen];
455     int[]     afptwibin   = new int[minLen];
456     double[]  afptwilist  = new double[minLen];
457     int       currafp = currafp0;
458     int       s = 0;
459
460     afptwibin[s] = 0;
461     afpchain[s ++] = currafp;
462
463     boolean isConnected = false;
464     int     prevafp;
465    // int     alnlen = afpSet.get(afpchain[s]).getFragLen();
466
467     //Double  conn = new Double(0) ;
468     //Double dvar =  new Double(0);
469
470     while((prevafp = pre[currafp]) != -1)   {
471
472        isConnected = afpPairConn(prevafp, currafp, params,afpChain);
473
474        if ( isConnected )
475           afptwibin[s - 1] = 1;
476        else
477           afptwibin[s - 1] = 0;
478        Double dvar = afpChain.getDVar();
479        afptwilist[s - 1] = dvar;
480        //note s - 1: the transformation of prevafp-currafp is recorded in currafp
481        currafp = prevafp;
482       // alnlen += afpSet.get(currafp).getFragLen();
483        afpchain[s ++] = currafp;
484     }
485     
486     afpChainLen = s;
487     afpChain.setAfpChainLen(afpChainLen);
488
489     //first afp without transformation
490     if ( isConnected)
491        afptwibin[s - 1] = 1;
492     else
493        afptwibin[s - 1] = 0;
494
495     //if(debug)
496     //   System.out.println(String.format("including %d AFPs, %d residues\n", afpChainLen, alnlen));
497
498     //record the optimal alignment in afpChainList (afpChainLen)
499
500     int[] afpChainList = afpChain.getAfpChainList();
501     double[] afpChainTwiBin = afpChain.getAfpChainTwiBin();
502     double[] afpChainTwiList = afpChain.getAfpChainTwiList();
503     
504     if(afpChainList == null)        {
505        afpChainList   = new int[s];
506        afpChain.setAfpChainList(afpChainList);
507        afpChainTwiBin     = new double[s];
508        afpChain.setAfpChainTwiBin(afpChainTwiBin);
509        afpChainTwiList = new double[s];
510        afpChain.setAfpChainTwiList(afpChainTwiList);
511     }
512     
513     int afpChainTwiNum = afpChain.getAfpChainTwiNum();
514     
515     int     i;
516     for(i = 0; i < s; i ++) {
517        afpChainList[i]     = afpchain[s - 1 - i];
518        afpChainTwiBin[i]   = afptwibin[s - 1 - i];
519        afpChainTwiList[i]  = afptwilist[s - 1 - i];
520        afpChainTwiNum      += afptwibin[s - 1 - i];
521     }
522     
523     if(afpChainTwiNum != twist)     {
524        System.err.println(String.format("AFPChainer Warning: the twists number is not consistent %d %d\n", afpChainTwiNum, twist));
525     }
526
527     double alignScore = afpChain.getAlignScore();
528     
529     double  checkscore = afpSet.get(afpChainList[0]).getScore();
530     for(i = 1; i < afpChainLen; i ++)       {
531        isConnected = afpPairConn(afpChainList[i - 1], afpChainList[i], params,afpChain);
532        checkscore = checkscore + afpSet.get(afpChainList[i]).getScore() + afpChain.getConn();
533     }
534     if(Math.abs(checkscore - alignScore) > 1e-4)        {
535        System.err.println(String.format("AFPChainer Warning: fail in alignment score checking %.4f %.4f\n", alignScore, checkscore));
536     }
537
538     if ( debug )
539        System.out.println("traceBack:" + afpChainLen + " " + afpChainList.length);
540     double  rmsd = calAfpRmsd(afpChainLen, afpChainList,0, afpChain,ca1,ca2);
541     afpChain.setChainRmsd(rmsd);
542     if (debug )
543        System.out.println("Chain RMSD: " + rmsd);
544     int     b1 = 0;
545     int     bk = 0;
546     int     a, b;
547     afpChain.setChainLen( 0);
548     int chainLen       = afpChain.getChainLen();
549     int block2Afp[]    = afpChain.getBlock2Afp();
550     
551       
552     double[] blockRmsd = afpChain.getBlockRmsd();
553     int[] blockSize = afpChain.getBlockSize();
554     
555     block2Afp[0] = 0;
556     for(i = 0; i < afpChainLen; i ++)       {
557        a = afpChainList[i];
558        chainLen += afpSet.get(a).getFragLen();
559        if(i > 0)       {
560           b = afpChainList[i - 1];
561           int misLen = afpChain.getMisLen();
562           misLen += calcMismatch(afpSet.get(a),afpSet.get(b));
563           afpChain.setMisLen(misLen);
564           int gapLen = afpChain.getGapLen();
565           gapLen += calcGap(afpSet.get(a),afpSet.get(b));
566           afpChain.setGapLen(gapLen);
567
568        }
569
570        if(afpChainTwiBin[i] == 1)   {
571           if (debug)
572              System.out.println(" ** calAfpTmsd : afpChainWtiBin == 1 : i: "+i+" i-b1: " + (i-b1) + " b1: " + b1 + " afpChainList.len: " + afpChainList.length);
573         
574           //int len = afpChainList.length - b1 +1;
575           //int[] fakeList = new int[len];
576           //int pos = -1;
577           //for ( int fPos = b1 ; fPos< afpChainList.length ; fPos++){
578           //   pos++;
579           //   fakeList[pos] = afpChainList[fPos];
580           //}
581           if (debug )
582              System.err.println("calculation calAfpRmsd " + i + " " + b1 + " " );
583          
584          
585           rmsd = calAfpRmsd(i - b1, afpChainList,b1, afpChain, ca1, ca2);
586          
587           
588           blockRmsd[bk] = rmsd;
589           blockSize[bk] = i - b1;
590           b1 = i;
591           
592           //System.out.println("block2Afp.length:"+ block2Afp.length + " " + bk + " " + i + " " + afpChain.getMaxTra() );
