114    /**
115     * Make a copy of the specified feature. The mappings in the userMap() HashMap
116     * are copied, so each feature has independent user data. Note, however, that the
117     * actual objects in the HashMap are shared (not copied), so a change to such an object may
118     * affect multiple features.
119     *
120     * @param feature Feature to clone.
121     */
122    public Feature(Feature feature) {
123
124        mSeqname = feature.mSeqname;
125        mSource = feature.mSource;
126        mType = feature.mType;
127        mLocation = feature.mLocation;
128        mScore = feature.mScore;
129        mFrame = feature.mFrame;
130        mAttributes = feature.mAttributes;
131        initAttributeHashMap();
132        mUserMap = new HashMap<String, Object>(feature.mUserMap);
133    }

135    /**
136     * Construct a new Feature from raw data (usually a GFF row).
137     *
138     * @param seqname The sequence name field (field 1).
139     * @param source The source or method field (field 2).
140     * @param type The type of feature field (field 3).
141     * @param location The location of the feature. (calculated from GFF start, end and strand fields).
142     * @param score The score field (field 7).
143     * @param frame The frame or phase field (field 8).
144     * @param attributes A string of key/value pairs separated by semicolons (field 9).
145     */
146    public Feature(String seqname, String source, String type, Location location, Double score, int frame, String attributes) {
147
148        mSeqname = seqname;
149        mSource = source;
150        mType = type;
151        mLocation = location;
152        mScore = score;
153        mFrame = frame;
154        mAttributes = attributes;
155        initAttributeHashMap();
156        mUserMap = new HashMap<String, Object>();
157
158    }

160    /**
161     * Get HashMap of user data. Each Feature object has a Java HashMap object
162     * which can be used to annotate the Feature. JavaGene does not use or interpret
163     * the keys or values. The values can be any subtype of the Java Object class.
164     *<br><br>
165     * If a Feature is constructed from data fields, the initial HashMap has no mappings (is empty).
166     * If a Feature is constructed from another Feature, a copy of the mappings is made.
167     * Note that the Objects in the copied mapping are shared, even though the mapping itself
168     * is copied (not shared). Thus removing or adding a mapping to one Feature will not affect the
169     * other, but changing an Object which is part of an established mapping may affect both Features.
170     *
171     * @return The user HashMap.
172     */
173    public HashMap<String, Object> userData() {
174        return mUserMap;
175    }
176
177    HashMap<String,String> attributeHashMap = new HashMap<String,String>();
178    
179    private void initAttributeHashMap(){
180       String[] values = mAttributes.split(";");
181       for(String attribute : values){
182           attribute = attribute.trim();
183           int equalindex = attribute.indexOf("=");
184           String splitData = "=";
185           if(equalindex == -1) //gtf uses space and gff3 uses =
186               splitData = " ";
187           String[] data = attribute.split(splitData);
188           String value = "";
189           if(data.length >= 2 && data[1].indexOf('"') != -1){ // an attibute field could be empty
190               value = data[1].replaceAll('"' + "","").trim();
191           }else if(data.length >= 2){
192               value = data[1].trim();
193           }
194           attributeHashMap.put(data[0].trim(), value);
195       }
196
197    }
198    
199    /**
200     * Get value of specified attribute key. Returns null if the attribute key has no value (does not exist).
201     * Keys are case-sensitive. Assumes attributes are correctly formatted in GFF style.
202     * Known bug: a semicolon within a quoted value will cause parse failure.
203     *
204     * @param key The key.
205     * @return The corresponding value. Null if the key has no value defined.
206     */
207    public String getAttribute(String key) {
208        
209        return attributeHashMap.get(key);
210    }
211
212    public String getAttributeOld(String key) {
213        int start = 0;
214
215        int end = mAttributes.indexOf(';');
216        while (0 < end) {
217            //find the first word (up to space) in chunk,
218            // see if it is this key
219            int i = mAttributes.indexOf(' ', start);
220            if (0 < i && i < end) {
221                if (mAttributes.substring(start, i).equals(key)) {
222                    //remove quotes, if needed
223                    if (mAttributes.charAt(i + 1) == '\"' && mAttributes.charAt(end - 1) == '\"') {
224                        return mAttributes.substring(i + 2, end - 1);//return attribute
225                    } else {
226                        return mAttributes.substring(i + 1, end);       //return attribute
227                    }
228                }
229            }
230            start = end + 2;    //skip required semicolon and single space
231            end = mAttributes.indexOf(';', start);
232        }
233
234        return null;
235    }
236
237    public boolean hasAttribute(String key) {
238        return attributeHashMap.containsKey(key);
239    }
240
241    public boolean hasAttribute(String key, String value) {
242        String data = getAttribute(key);
243        if(data == null)
244            return false;
245        if(data.equals(value))
246            return true;
247        else
248            return false;
249    }
250
251    /**
252     * Get the first item (everything before first semicolon, if it has one)
253     * in the attribute field, which is assumed to
254     * be a group identifer. This is appropriate for GFF1 files and variants. It is not
255     * appropriate for GTF and GFF2 files, although they may use a named attribute key,
256     * such as "gene_id" or "transcript_id", for grouping.
257     *
258     * @return The group id. Everything before the first semicolon in the attributes string (minus trailing whitespace).
259     */
260    public String group() {
261        int i = mAttributes.indexOf(';');
262        return (i < 0) ? mAttributes.trim() : mAttributes.substring(0, i).trim();
263    }
264
265    /**
266     *
267     */
268    public String toString() {
269        String s = mSeqname + '\t';
270        s += mSource + '\t';
271        s += mType + '\t';
272        s += mLocation.start() + "\t";
273        s += mLocation.end() + "\t";
274        s += Double.toString(mScore) + "\t";
275
276        if (mFrame == -1) {
277            s += ".\t";
278        } else {
279            s += mFrame + "\t";
280        }
281
282        s += mAttributes;
283
284        return s;
285    }
