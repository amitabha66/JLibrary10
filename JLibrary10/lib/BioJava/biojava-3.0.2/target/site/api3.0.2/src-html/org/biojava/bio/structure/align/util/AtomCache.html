055 *  script that is frequently re-using the same PDB structures, the AtomCache keeps an 
056 *  in-memory cache of the files for quicker access. The cache is a soft-cache, this 
057 *  means it won't cause out of memory exceptions, but garbage collects the data if the 
058 *  Java virtual machine needs to free up space. The AtomCache is thread-safe.

100        /** Creates an instance of an AtomCache that is pointed to the a particular
101         * path in the file system.
102         * 
103         * @param pdbFilePath a directory in the file system to use as a location to cache files.
104         * @param isSplit a flag to indicate if the directory organisation is "split" as on the PDB ftp servers, or if all files are contained in one directory.
105         */
106        public AtomCache(String pdbFilePath, boolean isSplit){
107
108
109                if ( ! pdbFilePath.endsWith(FILE_SEPARATOR))
110                        pdbFilePath += FILE_SEPARATOR;
111
112                // we are caching the binary files that contain the PDBs gzipped
113                // that is the most memory efficient way of caching...
114                // set the input stream provider to caching mode
115                System.setProperty(InputStreamProvider.CACHE_PROPERTY, "true");
116
117                path = pdbFilePath;
118
119                System.setProperty(AbstractUserArgumentProcessor.PDB_DIR,path);
120                //this.cache = cache;
121                this.isSplit = isSplit;
122
123                autoFetch = true;
124                fetchFileEvenIfObsolete = false;
125                fetchCurrent = false;
126
127                currentlyLoading.clear();
128                params = new FileParsingParameters();
129
130                // we don't need this here
131                params.setAlignSeqRes(false);
132                // no secstruc either 
133                params.setParseSecStruc(false);
134                // 
135                
136
137                this.strictSCOP = true;
138
139                scopInstallation = null;
140        }

264        /** Returns the representation of a ScopDomain as a BioJava Structure object
265         * 
266         * @param domain a scop domain
267         * @return a Structure object.
268         * @throws IOException
269         * @throws StructureException
270         */
271
272        public Structure getStructureForDomain(ScopDomain domain) throws IOException, StructureException{
273
274
275                Structure s = null;
276
277                String pdbId = domain.getPdbId();
278
279                try {
280                        s = getStructure(pdbId);
281
282                } catch (StructureException ex){
283                        System.err.println("error getting Structure for " + pdbId);
284
285                        throw new StructureException(ex);
286                }
287
288
289                String range = "(";
290                int rangePos = 0;
291                for ( String r : domain.getRanges()) {
292                        rangePos++;
293                        range+= r;
294                        if ( ( domain.getRanges().size()> 1) && (rangePos < domain.getRanges().size())){
295                                range+=",";
296                        }
297
298                }
299                range+=")";
300                //System.out.println("getting range for "+ pdbId + " " + range);
301
302                Structure n = StructureTools.getSubRanges(s, range);
303
304                // get free ligands of first chain...
305                if ( n.getChains().size()> 0) {
306                        Chain c1 = n.getChains().get(0);
307                        for ( Chain c : s.getChains()) {
308                                if ( c1.getChainID().equals(c.getChainID())) {
309                                        List<Group> ligands = c.getAtomLigands();
310
311                                        for(Group g: ligands){
312                                                if ( ! c1.getAtomGroups().contains(g)) {
313                                                        c1.addGroup(g);
314                                                }
315                                        }
316                                }
317
318                        }
319                }
320                n.setName(domain.getScopId());
321                n.setPDBCode(domain.getScopId());
322
323                return n;
324        }

381        /** Request a Structure based on a <i>name</i>.
382         * 
383         * <pre>
384                Formal specification for how to specify the <i>name</i>:
385                
386                name     := pdbID
387                               | pdbID '.' chainID
388                               | pdbID '.' range
389                               | scopID
390                range         := '('? range (',' range)? ')'?
391                               | chainID
392                               | chainID '_' resNum '-' resNum
393                pdbID         := [0-9][a-zA-Z0-9]{3}
394                chainID       := [a-zA-Z0-9]
395                scopID        := 'd' pdbID [a-z_][0-9_]
396                resNum        := [-+]?[0-9]+[A-Za-z]?
397                
398                
399                Example structures:
400                1TIM     #whole structure
401                4HHB.C     #single chain
402                4GCR.A_1-83     #one domain, by residue number
403                3AA0.A,B     #two chains treated as one structure
404                d2bq6a1     #scop domain
405                </pre>
406         * 
407         * With the additional set of rules:
408         *  
409         *  <ul>
410         *  <li>If only a PDB code is provided, the whole structure will be return including ligands, but the first model only (for NMR).
411         *      <li>Chain IDs are case sensitive, PDB ids are not. To specify a particular chain write as: 4hhb.A or 4HHB.A </li>
412         *  <li>To specify a SCOP domain write a scopId e.g. d2bq6a1. Some flexibility can be allowed in SCOP domain names, see {@link #setStrictSCOP(boolean)}</li>
413         *  <li>URLs are accepted as well</li>
414         *  </ul>
415         *  
416         * @param name
417         * @return a Structure object, or null if name appears improperly formated (eg too short, etc)
418         * @throws IOException The PDB file cannot be cached due to IO errors
419         * @throws StructureException The name appeared valid but did not correspond to a structure.
420         *      Also thrown by some submethods upon errors, eg for poorly formatted subranges.
421         */
422        public Structure getStructure(String name) throws IOException, StructureException{
423
424                if ( name.length() < 4)
425                        throw new IllegalArgumentException("Can't interpred IDs that are shorter than 4 residues!");
426                
427                Structure n = null;
428
429                boolean useChainNr = false;
430                boolean useDomainInfo = false;
431                String range = null;
432                int chainNr = -1;
433
434                try {
435
436
437                        String pdbId   = null;
438                        String chainId = null;
439
440                        if ( name.length() == 4){
441
442                                pdbId = name; 
443
444                        } else if ( name.startsWith("d")){
445
446                                // return based on SCOP domain ID
447                                return getStructureFromSCOPDomain(name);
448
449                        } else if (name.length() == 6){
450                                // name is PDB.CHAINID style (e.g. 4hhb.A)
451                                
452                                pdbId = name.substring(0,4);
453                                if ( name.substring(4,5).equals(CHAIN_SPLIT_SYMBOL)) {
454                                        chainId = name.substring(5,6);
455                                } else if ( name.substring(4,5).equals(CHAIN_NR_SYMBOL)) {
456
457                                        useChainNr = true;      
458                                        chainNr = Integer.parseInt(name.substring(5,6));
459                                }
460                                
461                        } else if ( (name.length() > 6) &&  
462                                        (name.contains(CHAIN_NR_SYMBOL) || name.contains(UNDERSCORE)) && (! (name.startsWith("file:/") || name.startsWith("http:/")))) {
463                                
464                                // this is a name + range 
465                                
466                                pdbId = name.substring(0,4);
467                                // this ID has domain split information...
468                                useDomainInfo = true;
469                                range = name.substring(5);
470                                
471                        } else if ( name.startsWith("file:/") || name.startsWith("http:/") ) {
472                                
473                                // this is a URL
474                                try {
475                                        
476                                        URL url = new URL(name);
477                                        
478                                        return getStructureFromURL(url);
479                                        
480                                } catch (Exception e){
481                                        e.printStackTrace();
482                                        return null;
483                                }
484
485
486                        }
487
488                        //System.out.println("got: " + name + " " + pdbId + " " + chainId + " useChainNr:" + useChainNr + " " +chainNr + " useDomainInfo:" + useDomainInfo + " " + range);
489
490                        if (pdbId == null) {
491
492                                return null;
493                        }
494
495                        while ( checkLoading(pdbId) ){
496                                // waiting for loading to be finished...
497
498                                try {
499                                        Thread.sleep(100);
500                                } catch (InterruptedException e){
501                                        System.err.println(e.getMessage());
502                                }
503
504                        }
505
506
507                        //long start  = System.currentTimeMillis();
508
509                        Structure s;
510                        flagLoading(pdbId);
511                        try {
512                                PDBFileReader reader = new PDBFileReader();
513                                reader.setPath(path);
514                                reader.setPdbDirectorySplit(isSplit);
515                                reader.setAutoFetch(autoFetch);
516                                reader.setFetchFileEvenIfObsolete(fetchFileEvenIfObsolete);
517                                reader.setFetchCurrent(fetchCurrent);
518
519                                reader.setFileParsingParameters(params);
520
521                                s = reader.getStructureById(pdbId.toLowerCase());
522
523                        } catch (Exception e){
524                                flagLoadingFinished(pdbId);
525                                throw new StructureException(e.getMessage() + " while parsing " + pdbId,e);
526                        }
527                        flagLoadingFinished(pdbId);
528
529                        //long end  = System.currentTimeMillis();
530                        //System.out.println("time to load " + pdbId + " " + (end-start) + "\t  size :" + StructureTools.getNrAtoms(s) + "\t cached: " + cache.size());
531                        if ( chainId == null && chainNr < 0 && range == null) {                                                              
532                                // we only want the 1st model in this case
533                                n = StructureTools.getReducedStructure(s,-1);
534
535                        }
536                        else {
537
538                                if ( useChainNr) {
539                                        //System.out.println("using ChainNr");
540                                        n = StructureTools.getReducedStructure(s, chainNr);
541                                } else if ( useDomainInfo) {
542                                        //System.out.println("calling getSubRanges");
543                                        n = StructureTools.getSubRanges(s, range);
544                                } else  {
545                                        //System.out.println("reducing Chain Id " + chainId);
546                                        n = StructureTools.getReducedStructure(s, chainId);
547                                }
548                        }
549
550
551                } catch (Exception e){
552
553                        e.printStackTrace();
554
555                        throw new StructureException(e.getMessage() + " while parsing " + name,e);
556
557                }
558                
559                n.setName(name);
560                return n;
561
562
563        }

565        private Structure getStructureFromSCOPDomain(String name)
566                        throws IOException, StructureException {
567                // looks like a SCOP domain!
568                ScopDomain domain;
569                if( this.strictSCOP) {
570                        domain = getScopDomain(name);
571                } else {
572                        domain = guessScopDomain(name);
573                }
574                if ( domain != null){
575                        Structure s = getStructureForDomain(domain);
576                        return s;
577                }
578
579                if( !this.strictSCOP) {
580                        Matcher scopMatch = scopIDregex.matcher(name);
581                        if( scopMatch.matches() ) {
582                                String pdbID = scopMatch.group(1);
583                                String chainID = scopMatch.group(2);
584
585                                // None of the actual SCOP domains match. Guess that '_' means 'whole chain'
586                                if( !chainID.equals("_") ) {
587                                        //Add chain identifier
588                                        pdbID += "."+scopMatch.group(2);
589                                }
590                                // Fetch the structure by pdb id
591                                Structure struct = getStructure(pdbID);
592                                if(struct != null) {
593                                        System.err.println("Trying chain "+pdbID);
594                                }
595                                return struct;
596                        }
597                }
598
599                throw new StructureException("Unable to get structure for SCOP domain: "+name);
600        }

603        private Structure getStructureFromURL(URL url) throws IOException, StructureException {
604                // looks like a URL for a file was provided:
605                System.out.println("fetching structure from URL:" + url);
606                
607                String queryS = url.getQuery();
608                                
609                String chainId = null;
610                if ( queryS != null && (queryS.startsWith("chainId="))) {
611                        chainId = queryS.substring(8);
612                
613                        String fullu = url.toString();
614                        
615                        if (fullu.startsWith("file:") && fullu.endsWith("?"+queryS)) {
616                                // for windowze, drop the query part from the URL again
617                                // otherwise there will be a "file not found error" ...
618
619                                String newu = fullu.substring(0,(fullu.length()-(("?"+queryS).length())));
620                                //System.out.println(newu);
621                                url = new URL(newu);
622                        }
623                }
624                
625                
626                PDBFileReader reader = new PDBFileReader();
627                reader.setPath(path);
628                reader.setPdbDirectorySplit(isSplit);
629                reader.setAutoFetch(autoFetch);
630                reader.setFetchFileEvenIfObsolete(fetchFileEvenIfObsolete);
631                reader.setFetchCurrent(fetchCurrent);
632
633                reader.setFileParsingParameters(params);
634                
635                Structure s = reader.getStructure(url);
636                if ( chainId == null)
637                        return StructureTools.getReducedStructure(s,-1);
638                else 
639                        return StructureTools.getReducedStructure(s,chainId);
640        }

643        private static final Pattern scopIDregex = Pattern.compile("d(....)(.)(.)" );
644        /**
645         * <p>Guess a scop domain. If an exact match is found, return that.
646         * 
647         * <p>Otherwise, return the first scop domain found for the specified protein
648         * such that<ul>
649         *   <li>The chains match, or one of the chains is '_' or '.'.
650         *   <li>The domains match, or one of the domains is '_'.
651         * </ul>
652         *   
653         *   
654         * @param name
655         * @return
656         * @throws IOException
657         * @throws StructureException
658         */
659        private ScopDomain guessScopDomain(String name) throws IOException, StructureException {
660                List<ScopDomain> matches = new LinkedList<ScopDomain>();
661
662                // Try exact match first
663                ScopDomain domain = getScopDomain(name);
664                if ( domain != null){
665                        return domain;
666                }
667
668                // Didn't work. Guess it!
669                System.err.println("Warning, could not find SCOP domain: " + name);
670
671                Matcher scopMatch = scopIDregex.matcher(name);
672                if( scopMatch.matches() ) {
673                        String pdbID = scopMatch.group(1);
674                        String chainID = scopMatch.group(2);
675                        String domainID = scopMatch.group(3);
676
677                        if ( scopInstallation == null) {
678                                scopInstallation = new ScopInstallation(path);
679                        }
680
681                        for( ScopDomain potentialSCOP : scopInstallation.getDomainsForPDB(pdbID) ) {
682                                Matcher potMatch = scopIDregex.matcher(potentialSCOP.getScopId());
683                                if(potMatch.matches()) {
684                                        if( chainID.equals(potMatch.group(2)) ||
685                                                        chainID.equals("_") || chainID.equals(".") ||
686                                                        potMatch.group(2).equals("_") || potMatch.group(2).equals(".") ) {
687                                                if( domainID.equals(potMatch.group(3)) || domainID.equals("_") || potMatch.group(3).equals("_") ) {
688                                                        // Match, or near match
689                                                        matches.add(potentialSCOP);
690                                                }
691                                        }
692                                }
693                        }
694                }
695
696                Iterator<ScopDomain> match = matches.iterator();
697                if( match.hasNext() ) {
698                        ScopDomain bestMatch = match.next();
699                        System.err.print("Trying domain "+bestMatch.getScopId()+".");
700                        if( match.hasNext() ) {
701                                System.err.print(" Other possibilities: ");
702                                while(match.hasNext()) {
703                                        System.err.print(match.next().getScopId() + " ");
704                                }
705                        }
706                        System.err.println();
707                        return bestMatch;
708                } else {
709                        return null;
710                }
711        }

736        public ScopDatabase getScopInstallation() {

772        /** 
773         * Loads the biological assembly for a given PDB ID and bioAssemblyId. 
774         * If a bioAssemblyId > 0 is specified, the corresponding biological assembly file will be loaded. Note, the
775         * number of available biological unit files varies. Many entries don't have a biological assembly specified (i.e. NMR structures),
776         * many entries have only one biological assembly (bioAssemblyId=1), and a few structures have multiple biological assemblies.
777         * Set bioAssemblyFallback to true, to download the original PDB file in cases that a biological assembly file is not available.
778         * 
779         * @param pdbId the PDB ID
780         * @param bioAssemblyId the ID of the biological assembly
781         * @param bioAssemblyFallback if true, try reading original PDB file in case the biological assembly file is not available
782         * @return a structure object
783         * @throws IOException 
784         * @throws StructureException 
785         * @author Peter Rose
786         * @since 3.2
787         */
788        public Structure getBiologicalAssembly(String pdbId, int bioAssemblyId, boolean bioAssemblyFallback) throws StructureException, IOException {
789                Structure s;
790                if (bioAssemblyId < 1) {
791                        throw new StructureException("bioAssemblyID must be greater than zero: " + pdbId + 
792                                        " bioAssemblyId " + bioAssemblyId);
793                }
794                PDBFileReader reader = new PDBFileReader();
795                reader.setPath(path);
796                reader.setPdbDirectorySplit(isSplit);
797                reader.setAutoFetch(autoFetch);
798                reader.setFetchFileEvenIfObsolete(fetchFileEvenIfObsolete);
799                reader.setFetchCurrent(fetchCurrent);
800                reader.setFileParsingParameters(params);
801                reader.setBioAssemblyId(bioAssemblyId);
802                reader.setBioAssemblyFallback(bioAssemblyFallback);
803                s = reader.getStructureById(pdbId.toLowerCase());
804                s.setPDBCode(pdbId);
805                return s;
806        }
