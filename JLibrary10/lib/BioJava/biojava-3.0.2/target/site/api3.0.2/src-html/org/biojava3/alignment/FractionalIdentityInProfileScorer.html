024package org.biojava3.alignment;
025
026import org.biojava3.alignment.template.AlignedSequence;
027import org.biojava3.alignment.template.PairInProfileScorer;
028import org.biojava3.alignment.template.Profile;
029import org.biojava3.core.sequence.template.Compound;
030import org.biojava3.core.sequence.template.Sequence;
031
032/**
033 * Implements an algorithm which computes a score for a sequence alignment pair picked from an alignment
034 * {@link Profile}.  The reported score is the number of alignment columns which have identical {@link Compound}s.
035 *
036 * @author Mark Chapman
037 * @param <S> each {@link Sequence} of the alignment pair is of type S
038 * @param <C> each element of an {@link AlignedSequence} is a {@link Compound} of type C
039 */
040public class FractionalIdentityInProfileScorer<S extends Sequence<C>, C extends Compound>
041        extends FractionalIdentityScorer<S, C> implements PairInProfileScorer<S, C> {
042
043    private Profile<S, C> profile;
044
045    /**
046     * Creates a fractional identity scorer for an aligned pair of sequences in the given alignment profile.
047     *
048     * @param profile alignment profile containing pair of sequences
049     * @param query index in the profile of the first sequence of the pair
050     * @param target index in the profile of the second sequence of the pair
051     */
052    public FractionalIdentityInProfileScorer(Profile<S, C> profile, int query, int target) {
053        super(new SimpleSequencePair<S, C>(profile.getAlignedSequence(query), profile.getAlignedSequence(target)));
054        this.profile = profile;
055    }
056
057    @Override
058    public Profile<S, C> getProfile() {
059        return profile;
060    }
061
062}
