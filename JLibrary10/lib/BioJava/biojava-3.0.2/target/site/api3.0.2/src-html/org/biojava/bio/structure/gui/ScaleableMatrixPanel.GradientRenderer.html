161        public ScaleableMatrixPanel(){
162
163                mPanel   = new JMatrixPanel();
164                Box vBox = Box.createVerticalBox();
165                
166                Box gradientBox = Box.createHorizontalBox();
167                vBox.add(gradientBox);
168                gradientBox.add(new JLabel("Coloring:"));
169                gradients = createGradients(); //sets gradients
170                //Set first gradient
171                this.setCellColor(gradients.values().iterator().next());
172                
173                coloring = new JComboBox(gradients.keySet().toArray(new String[] {}));
174                coloring.setRenderer(new GradientRenderer());
175                coloring.addActionListener(this);
176                coloring.setMaximumSize(new Dimension(1000,30));
177                gradientBox.add(coloring);
178                
179                int RES_MIN  = 0*SLIDER_STEPS;
180                int RES_MAX  = 8*SLIDER_STEPS;
181                int RES_INIT = 1*SLIDER_STEPS;
182
183                slider = new JSlider(JSlider.HORIZONTAL, RES_MIN,RES_MAX,RES_INIT);
184                slider.setInverted(false);
185                slider.setPaintTicks(true);
186                //slider.setMinorTickSpacing(1);
187                slider.setMajorTickSpacing(SLIDER_STEPS);
188                slider.setSnapToTicks(false);
189                slider.setPaintLabels(false);
190                slider.setPaintTrack(true);
191                //slider.setLabelTable(slider.createStandardLabels(SLIDER_STEPS));
192                slider.addChangeListener(this);
193                slider.setPreferredSize(new Dimension(100,slider.getPreferredSize().height));
194
195                vBox.add(slider);
196
197                scroll = new JScrollPane(mPanel);
198                scroll.getHorizontalScrollBar().setUnitIncrement(60);
199                scroll.getVerticalScrollBar().setUnitIncrement(60);
200                scroll.getHorizontalScrollBar().setBlockIncrement(60);
201                scroll.getVerticalScrollBar().setBlockIncrement(60);
202                vBox.add(scroll);
203                this.setPreferredSize(new Dimension(400,400));
204                this.add(vBox);
205
206
207                mPanel.setLayout(new BoxLayout(mPanel,BoxLayout.Y_AXIS));
208                this.setLayout(new BoxLayout(this,BoxLayout.Y_AXIS));
209
210        }

215        protected static Map<String,ContinuousColorMapper> createGradients() {
216                SortedMap<String,ContinuousColorMapper> gradients = new TreeMap<String,ContinuousColorMapper>();
217                
218                int i = 0; //prepend number, since sorted alphabetically
219                ColorSpace hsv = HSVColorSpace.getHSVColorSpace();
220                LinearColorInterpolator interp;
221                GradientMapper gradient;
222                
223                /*
224                DefaultMatrixMapper defaultMapper = new DefaultMatrixMapper(10., .9f);
225                gradients.put((++i)+". Default", defaultMapper);
226                defaultMapper = new DefaultMatrixMapper(5., .9f);
227                gradients.put((++i)+". Sensitive", defaultMapper);
228                
229                
230                gradients.put((++i)+". Rainbow", GradientMapper.getGradientMapper(GradientMapper.RAINBOW_INTENSITY_GRADIENT, 0, 10));
231                gradients.put((++i)+". Rainbow", GradientMapper.getGradientMapper(GradientMapper.RAINBOW_INTENSITY_GRADIENT, 10, 0));
232                */
233                
234                // Mimic DefaultMapper
235                interp = new LinearColorInterpolator(hsv);
236                interp.setInterpolationDirection(0, InterpolationDirection.INNER);
237                gradient = new GradientMapper(Color.green, Color.black, hsv);
238                gradient.put( 0., new Color(hsv,new float[] {1f, .9f, 1f},1f));
239                gradient.put(10., new Color(hsv,new float[] {0f, .9f, 0f},1f));
240                gradient.setInterpolator(interp);
241                
242                gradients.put((++i)+". Default", gradient);
243                
244                // Sensitive DefaultMapper
245                interp = new LinearColorInterpolator(hsv);
246                interp.setInterpolationDirection(0, InterpolationDirection.INNER);
247                gradient = new GradientMapper(Color.green, Color.black, hsv);
248                gradient.put( 0., new Color(hsv,new float[] {1f, .9f, 1f},1f));
249                gradient.put( 5., new Color(hsv,new float[] {0f, .9f, 0f},1f));
250                gradient.setInterpolator(interp);
251                
252                gradients.put((++i)+". Sensitive", gradient);   
253                
254                // color [0,1]import java.awt.Color;
255
256                interp = new LinearColorInterpolator(hsv);
257                interp.setInterpolationDirection(0, InterpolationDirection.INNER);
258                gradient = new GradientMapper(Color.green, Color.black, hsv);
259                gradient.put( 0., new Color(hsv,new float[] {1f, .9f, 1f},1f));
260                gradient.put( 1., new Color(hsv,new float[] {.2f, .9f, 1f},1f));
261                gradient.put( 1+1e-6, Color.white);
262                gradient.put(5., Color.black);
263                gradient.setInterpolator(interp);
264                
265                gradients.put((++i)+". Emphasize low", gradient);
266                
267                
268                interp = new LinearColorInterpolator(hsv);
269                interp.setInterpolationDirection(0, InterpolationDirection.INNER);
270                gradient = new GradientMapper(Color.green, Color.black, hsv);
271                gradient.put( 0., new Color(hsv,new float[] {0f, .9f, 0f},1f));
272                gradient.put( 100., new Color(hsv,new float[] {1f, .9f, 1f},1f));
273                gradient.setInterpolator(interp);
274                
275                gradients.put((++i)+". 0 to 100", gradient);    
276                
277                // log color
278                interp = new LinearColorInterpolator(hsv);
279                interp.setInterpolationDirection(0, InterpolationDirection.INNER);
280                gradient = new GradientMapper(Color.red, Color.black, hsv);
281                gradient.put( 0., new Color(hsv,new float[] {1f, .9f, 1f},1f));
282                gradient.put( 10., new Color(hsv,new float[] {0f, .9f, 0f},1f));
283                gradient.setInterpolator(interp);
284                
285                ContinuousColorMapper logGradient = new LogColorMapper(gradient,2);
286                gradients.put((++i)+". Logorithmic", logGradient);      
287                
288                // sqrt color
289                interp = new LinearColorInterpolator(hsv);
290                interp.setInterpolationDirection(0, InterpolationDirection.INNER);
291                gradient = new GradientMapper(Color.red, Color.black, hsv);
292                gradient.put( 0., new Color(hsv,new float[] {1f, .9f, 1f},1f));
293                gradient.put( 4., new Color(hsv,new float[] {0f, .9f, 0f},1f));
294                gradient.setInterpolator(interp);
295                
296                ContinuousColorMapper sqrtGradient = new SqrtColorMapper(gradient);
297                gradients.put((++i)+". Square Root", sqrtGradient);     
298                
299                return gradients;
300        }

370        /**
371         * A renderer for the the gradient dropdown menu at the top of scaleableMatrixPanes.
372         * Knows how to draw a gradient and nicely label it.
373         * 
374         * @author Spencer Bliven
375         *
376         */
377        protected class GradientRenderer extends JPanel
378        implements ListCellRenderer {
379
380                private static final long serialVersionUID = -2000575579184232365L;
381                private int min,max;
382                JLabel title;
383                JPanel gradientContainer;
384                
385                public GradientRenderer() {
386                        this.setPreferredSize(new Dimension(100,25));
387                        this.setLayout(new BorderLayout());
388                        this.min = -1;
389                        this.max = 10;
390                        
391                        JPanel gradientBounds = new JPanel();
392                        gradientBounds.setLayout(new BorderLayout());
393                        //gradientBounds.setBorder(BorderFactory.createLineBorder(Color.GRAY));
394                        gradientBounds.add(new JLabel(Integer.toString(min)),BorderLayout.WEST);
395                        
396                        gradientContainer = new JPanel();
397                        gradientContainer.setLayout(new BorderLayout());
398                        gradientContainer.setOpaque(false);
399                        gradientContainer.add(new JLabel("<No gradient>"),BorderLayout.CENTER);
400                        //gradientContainer.setMinimumSize(new Dimension(50,20));
401                        //gradientContainer.setBorder(BorderFactory.createLineBorder(Color.DARK_GRAY));
402                        //gradientBounds.setPreferredSize(new Dimension(200,25));
403                        gradientBounds.add(gradientContainer,BorderLayout.CENTER);
404                        gradientBounds.setOpaque(false);
405                        
406                        gradientBounds.add(new JLabel(Integer.toString(max)),BorderLayout.EAST);
407                        
408                        this.add(gradientBounds, BorderLayout.EAST);
409                        
410                        this.title = new JLabel("No gradient");
411                        //this.title.setOpaque(true);
412                        this.title.setHorizontalAlignment(JLabel.CENTER);
413                        this.title.setVerticalAlignment(JLabel.CENTER);
414                        this.add(this.title,BorderLayout.CENTER);
415                }
416
417                /*
418                 * This method finds the image and text corresponding
419                 * to the selected value and returns the label, set up
420                 * to display the text and image.
421                 */
422                public Component getListCellRendererComponent(
423                                JList list,
424                                Object value,
425                                int index,
426                                boolean isSelected,
427                                boolean cellHasFocus) {
428                        //Get the selected index. (The index param isn't
429                        //always valid, so just use the value.)
430                        String gradientLabel = (String)value;
431                        
432                        if (isSelected) {
433                                setBackground(list.getSelectionBackground());
434                                setForeground(list.getSelectionForeground());
435                        } else {
436                                setBackground(list.getBackground());
437                                setForeground(list.getForeground());
438                        }
439                        
440                        //Set the icon and text.  If icon was null, say so.
441                        GradientPanel gradPanel = new GradientPanel(gradients.get(gradientLabel),min,max);
442                        gradPanel.setPreferredSize(new Dimension(100,20));
443                        //gradPanel.setBorder(BorderFactory.createLineBorder(Color.cyan));
444                        gradientContainer.removeAll();
445                        gradientContainer.add(gradPanel,BorderLayout.CENTER);
446
447                        title.setText(gradientLabel);
448
449                        this.validate();
450                        
451                        return this;
452                }
453        }

455        /**
456         * @param e
457         * @see java.awt.event.ActionListener#actionPerformed(java.awt.event.ActionEvent)
458         */
459        public void actionPerformed(ActionEvent e) {
460        JComboBox cb = (JComboBox)e.getSource(); // == coloring
461        String gradientName = (String)cb.getSelectedItem();
462        ContinuousColorMapper gradient = gradients.get(gradientName);
463        assert(gradient != null);
464        this.setCellColor(gradient);
465        this.repaint();
466        }
