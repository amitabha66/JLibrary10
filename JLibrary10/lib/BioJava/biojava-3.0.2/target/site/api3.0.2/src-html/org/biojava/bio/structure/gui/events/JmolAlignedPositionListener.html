034public class JmolAlignedPositionListener implements AlignmentPositionListener{
035
036        BiojavaJmol parent;
037        Atom[] ca1;
038        Atom[] ca2;
039        StructurePairAligner structurePairAligner;
040        
041        public JmolAlignedPositionListener(BiojavaJmol parent, StructurePairAligner alig){
042                this.parent = parent;
043                structurePairAligner = alig;
044        }
045        
046
047
048        public void setStructure1(Structure structure1) {
049                
050                ca1 = structurePairAligner.getAlignmentAtoms(structure1);
051        }
052
053        
054
055        public void setStructure2(Structure structure2) {
056                
057                ca2 = structurePairAligner.getAlignmentAtoms(structure2);
058        }
059
060        public void mouseOverPosition(AlignedPosition p) {
061                
062                //System.out.println("mouseoverposition " + p);
063                
064                int p1 = p.getPos1();
065                int p2 = p.getPos2();
066                String s = "select ";
067                
068                if ((p1 > ca1.length) || (p2 > ca2.length)){
069                        System.err.println("requsting atom out of bounds! " );
070                        return;
071                }
072                
073                String pdbpos1 ="";
074                String pdbpos2 = "";
075                
076                if ( p1 >-1) {
077                        Atom a = ca1[p1];
078                        Group parent = a.getGroup();
079                        Chain c = parent.getChain();
080                        pdbpos1 = parent.getResidueNumber().toString();
081                        //System.out.printlng"chainid 1 is >"+c.getName()+"<");
082                        if (! c.getChainID().equals( " ")) {
083                                pdbpos1 += ":" + c.getChainID();
084                        } 
085                                
086                        
087                        //System.out.println("1:" + parent);
088                        s += pdbpos1 +"/1";
089                }
090                
091                if ( p2 >-1) {
092                        Atom a = ca2[p2];
093                        Group parent = a.getGroup();
094                        Chain c = parent.getChain();
095                        pdbpos2 = parent.getResidueNumber().toString();
096                        //System.out.println("2:" + parent);
097                        //System.out.println("chainid 2 is >"+c.getName()+"<");
098                        if (! c.getChainID().equals( " ")) {
099                                pdbpos2 += ":" + c.getChainID();
100                        }
101                        if ( p1 > -1)
102                                s +=",";
103                        s += pdbpos2 +"/2";
104                }
105                s+="; set display selected;";
106                //System.out.println(s);
107                parent.evalString(s);
108                
109                
110        }
111

113                mouseOverPosition(p);

115        }
