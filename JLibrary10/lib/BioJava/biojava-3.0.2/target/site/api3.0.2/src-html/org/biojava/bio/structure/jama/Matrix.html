013/**
014   Jama = Java Matrix class.
015<P>
016   The Java Matrix Class provides the fundamental operations of numerical
017   linear algebra.  Various constructors create Matrices from two dimensional
018   arrays of double precision floating point numbers.  Various "gets" and
019   "sets" provide access to submatrices and matrix elements.  Several methods 
020   implement basic matrix arithmetic, including matrix addition and
021   multiplication, matrix norms, and element-by-element array operations.
022   Methods for reading and printing matrices are also included.  All the
023   operations in this version of the Matrix Class involve real matrices.
024   Complex matrices may be handled in a future version.
025<P>
026   Five fundamental matrix decompositions, which consist of pairs or triples
027   of matrices, permutation vectors, and the like, produce results in five
028   decomposition classes.  These decompositions are accessed by the Matrix
029   class to compute solutions of simultaneous linear equations, determinants,
030   inverses and other matrix functions.  The five decompositions are:
031<P><UL>
032   <LI>Cholesky Decomposition of symmetric, positive definite matrices.
033   <LI>LU Decomposition of rectangular matrices.
034   <LI>QR Decomposition of rectangular matrices.
035   <LI>Singular Value Decomposition of rectangular matrices.
036   <LI>Eigenvalue Decomposition of both symmetric and nonsymmetric square matrices.
037</UL>
038<DL>
039<DT><B>Example of use:</B></DT>
040<P>
041<DD>Solve a linear system A x = b and compute the residual norm, ||b - A x||.
042<P><PRE>
043      double[][] vals = {{1.,2.,3},{4.,5.,6.},{7.,8.,10.}};
044      Matrix A = new Matrix(vals);
045      Matrix b = Matrix.random(3,1);
046      Matrix x = A.solve(b);
047      Matrix r = A.times(x).minus(b);
048      double rnorm = r.normInf();
049</PRE></DD>
050</DL>
051
052@author The MathWorks, Inc. and the National Institute of Standards and Technology.
053@version 5 August 1998
054*/

137   /** Construct a matrix from a one-dimensional packed array
138   @param vals One-dimensional array of doubles, packed by columns (ala Fortran).
139   @param m    Number of rows.
140   @exception  IllegalArgumentException Array length must be a multiple of m.
141   */
142
143   public Matrix (double vals[], int m) {
144      this.m = m;
145      n = (m != 0 ? vals.length/m : 0);
146      if (m*n != vals.length) {
147         throw new IllegalArgumentException("Array length must be a multiple of m.");
148      }
149      A = new double[m][n];
150      for (int i = 0; i < m; i++) {
151         for (int j = 0; j < n; j++) {
152            A[i][j] = vals[i+j*m];
153         }
154      }
155   }

161   /** Construct a matrix from a copy of a 2-D array.
162   @param A    Two-dimensional array of doubles.
163   @exception  IllegalArgumentException All rows must have the same length
164   @return a Matrix
165   */

167   public static Matrix constructWithCopy(double[][] A) {
168      int m = A.length;
169      int n = A[0].length;
170      Matrix X = new Matrix(m,n);
171      double[][] C = X.getArray();
172      for (int i = 0; i < m; i++) {
173         if (A[i].length != n) {
174            throw new IllegalArgumentException
175               ("All rows must have the same length.");
176         }
177         for (int j = 0; j < n; j++) {
178            C[i][j] = A[i][j];
179         }
180      }
181      return X;
182   }

740   /** Linear algebraic matrix multiplication, A * B
741   @param B    another matrix
742   @return     Matrix product, A * B
743   @exception  IllegalArgumentException Matrix inner dimensions must agree.
744   */
745
746   public Matrix times (Matrix B) {
747      if (B.m != n) {
748         throw new IllegalArgumentException("Matrix inner dimensions must agree.");
749      }
750      Matrix X = new Matrix(m,B.n);
751      double[][] C = X.getArray();
752      double[] Bcolj = new double[n];
753      for (int j = 0; j < B.n; j++) {
754         for (int k = 0; k < n; k++) {
755            Bcolj[k] = B.A[k][j];
756         }
757         for (int i = 0; i < m; i++) {
758            double[] Arowi = A[i];
759            double s = 0;
760            for (int k = 0; k < n; k++) {
761               s += Arowi[k]*Bcolj[k];
762            }
763            C[i][j] = s;
764         }
765      }
766      return X;
767   }

958   // DecimalFormat is a little disappointing coming from Fortran or C's printf.
959   // Since it doesn't pad on the left, the elements will come out different
960   // widths.  Consequently, we'll pass the desired column width in as an
961   // argument and do the extra padding ourselves.

963   /** Print the matrix to the output stream.  Line the elements up in columns.
964     * Use the format object, and right justify within columns of width
965     * characters.
966     * Note that is the matrix is to be read back in, you probably will want
967     * to use a NumberFormat that is set to US Locale.
968   @param output the output stream.
969   @param format A formatting object to format the matrix elements 
970   @param width  Column width.
971   @see java.text.DecimalFormat#setDecimalFormatSymbols
972   */
973
974   public void print (PrintWriter output, NumberFormat format, int width) {
975      output.println();  // start on new line.
976      for (int i = 0; i < m; i++) {
977         for (int j = 0; j < n; j++) {
978            String s = format.format(A[i][j]); // format the number
979            int padding = Math.max(1,width-s.length()); // At _least_ 1 space
980            for (int k = 0; k < padding; k++)
981               output.print(' ');
982            output.print(s);
983         }
984         output.println();
985      }
986      //output.println();   // end with blank line.
987   }

989   /** Read a matrix from a stream.  The format is the same the print method,
990     * so printed matrices can be read back in (provided they were printed using
991     * US Locale).  Elements are separated by
992     * whitespace, all the elements for each row appear on a single line,
993     * the last row is followed by a blank line.
994   @param input the input stream.
995   @return a Matrix
996   @throws java.io.IOException
997   */
998
999   @SuppressWarnings({ "rawtypes", "unchecked" })
1000public static Matrix read (BufferedReader input) throws java.io.IOException {
1001      StreamTokenizer tokenizer= new StreamTokenizer(input);
1002
1003      // Although StreamTokenizer will parse numbers, it doesn't recognize
1004      // scientific notation (E or D); however, Double.valueOf does.
1005      // The strategy here is to disable StreamTokenizer's number parsing.
1006      // We'll only get whitespace delimited words, EOL's and EOF's.
1007      // These words should all be numbers, for Double.valueOf to parse.
1008
1009      tokenizer.resetSyntax();
1010      tokenizer.wordChars(0,255);
1011      tokenizer.whitespaceChars(0, ' ');
1012      tokenizer.eolIsSignificant(true);
1013      java.util.Vector v = new java.util.Vector();
1014
1015      // Ignore initial empty lines
1016      while (tokenizer.nextToken() == StreamTokenizer.TT_EOL);
1017      if (tokenizer.ttype == StreamTokenizer.TT_EOF)
1018        throw new java.io.IOException("Unexpected EOF on matrix read.");
1019      do {
1020         v.addElement(Double.valueOf(tokenizer.sval)); // Read & store 1st row.
1021      } while (tokenizer.nextToken() == StreamTokenizer.TT_WORD);
1022
1023      int n = v.size();  // Now we've got the number of columns!
1024      double row[] = new double[n];
1025      for (int j=0; j<n; j++)  // extract the elements of the 1st row.
1026         row[j]=((Double)v.elementAt(j)).doubleValue();
1027      v.removeAllElements();
1028      v.addElement(row);  // Start storing rows instead of columns.
1029      while (tokenizer.nextToken() == StreamTokenizer.TT_WORD) {
1030         // While non-empty lines
1031         v.addElement(row = new double[n]);
1032         int j = 0;
1033         do {
1034            if (j >= n) throw new java.io.IOException
1035               ("Row " + v.size() + " is too long.");
1036            row[j++] = Double.valueOf(tokenizer.sval).doubleValue();
1037         } while (tokenizer.nextToken() == StreamTokenizer.TT_WORD);
1038         if (j < n) throw new java.io.IOException
1039            ("Row " + v.size() + " is too short.");
1040      }
1041      int m = v.size();  // Now we've got the number of rows.
1042      double[][] A = new double[m][];
1043      v.copyInto(A);  // copy the rows out of the vector
1044      return new Matrix(A);
1045   }
