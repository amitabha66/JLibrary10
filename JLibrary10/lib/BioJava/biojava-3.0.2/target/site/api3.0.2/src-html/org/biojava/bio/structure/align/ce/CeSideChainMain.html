012   public CeSideChainMain(){
013      super();
014      
015      if ( params == null) {
016         CeSideChainUserArgumentProcessor proc = new CeSideChainUserArgumentProcessor();
017         params = (CeParameters) proc.getParameters();
018      }
019   }

021   public static void main(String[] args){
022
023      CeSideChainMain ce = new CeSideChainMain();
024      if (args.length  == 0 ) {                 
025         System.out.println(ce.printHelp());
026         return;                        
027      }
028
029      if ( args.length == 1){
030         if (args[0].equalsIgnoreCase("-h") || args[0].equalsIgnoreCase("-help")|| args[0].equalsIgnoreCase("--help")){
031            System.out.println(ce.printHelp());                                                         
032            return;
033         }
034
035      }
036
037      CeSideChainUserArgumentProcessor processor = new CeSideChainUserArgumentProcessor();
038      processor.process(args);
039
040   }
