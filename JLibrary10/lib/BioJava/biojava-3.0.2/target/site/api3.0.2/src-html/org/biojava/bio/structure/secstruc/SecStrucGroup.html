020package org.biojava.bio.structure.secstruc;
021
022import org.biojava.bio.structure.Atom;
023import org.biojava.bio.structure.Group;
024import org.biojava.bio.structure.HetatomImpl;
025
026public class SecStrucGroup
027extends HetatomImpl
028implements Group {
029
030    /**
031    *
032    */
033   private static final long serialVersionUID = 313490286720467714L;
034    Atom N;
035    Atom CA;
036    Atom C;
037    Atom O;
038    Atom H;
039    Group original;
040
041    public SecStrucGroup(){
042        super();
043    }
044
045    public String toString(){
046
047        StringBuffer str = new StringBuffer("SecStrucGroup ");
048        str.append(residueNumber);
049        str.append(" ");
050        str.append(pdb_name);
051        str.append(" ");
052        str.append(pdb_flag);
053        if (pdb_flag) {
054            str.append(" atoms: ");
055            str.append(atoms.size());
056        }
057
058
059        return str.toString() ;
060
061    }
062
063
064    public Group getOriginal()
065   {
066      return original;
067   }
068
069   public void setOriginal(Group original)
070   {
071      this.original = original;
072   }
073
074   public Atom getC() {
075        return C;
076    }
077    public void setC(Atom c) {
078        addAtom(c);
079        C = c;
080    }
081    public Atom getCA() {
082        return CA;
083    }
084    public void setCA(Atom ca) {
085        addAtom(ca);
086        CA = ca;
087    }
088    public Atom getH() {
089        return H;
090    }
091    public void setH(Atom h) {
092        addAtom(h);
093        H = h;
094    }
095    public Atom getN() {
096        return N;
097    }
098    public void setN(Atom n) {
099        addAtom(n);
100        N = n;
101    }
102    public Atom getO() {
103        return O;
104    }
105    public void setO(Atom o) {
106        addAtom(o);
107        O = o;
108    }
109
110
111
112}
