061        public static String toScoresList(AFPChain afpChain){
062
063                // see sippl On distance and similarity in fold space 2008 bioinformatics
064
065                StringWriter writer = new StringWriter();
066
067                if ( afpChain.getAlgorithmName().startsWith("CE")) {
068                        writer.append("Z-score " );
069                        writer.append(String.format("%.2f", afpChain.getProbability()));
070                        writer.append(newline);
071                }
072
073
074                writer.append("Sab (nr. equivalent residues): " );
075                writer.append(afpChain.getNrEQR()+"");
076                writer.append(newline);
077
078                writer.append("Dab (distance between folds a,b): ");
079                int dab = afpChain.getCa1Length()+afpChain.getCa2Length() - 2 * afpChain.getNrEQR();
080                writer.append(dab+"");
081                writer.append(newline);
082
083                writer.append("sab (relative similarity): ");
084                double sab = 2 * afpChain.getNrEQR() / (double)( afpChain.getCa1Length() + afpChain.getCa2Length());
085                writer.append(sab+"");
086                writer.append(newline);
087
088                writer.append("cab (coverage a): ");
089                double cab = afpChain.getNrEQR() / (double) afpChain.getCa1Length();
090                writer.append(cab+"");
091                writer.append(newline);
092
093                writer.append("cba (coverage b): ");
094                double cba = afpChain.getNrEQR() / (double) afpChain.getCa2Length();
095                writer.append(cba+"");
096                writer.append(newline);
097
098                writer.append("seq similarity: ");
099                writer.append(afpChain.getSimilarity()+"");
100                writer.append(newline);
101
102                writer.append("TM-score: ");
103                writer.append(afpChain.getTMScore()+"");
104                writer.append(newline);
105
106                return writer.toString();
107        }

109        public static String toFatCatCore(
110                        AFPChain afpChain, 
111                        Atom[] ca1, 
112                        Atom[] ca2, 
113                        boolean printLegend, boolean longHeader, boolean showHTML, boolean showAlignmentBlock){
114
115                if(!afpChain.isSequentialAlignment()) {
116                        //TODO find some topology-independent output format
117                        return "Can't display circular permutations";
118                }
119
120                String name1 = afpChain.getName1();
121                String name2 = afpChain.getName2();
122                int ca1Length = afpChain.getCa1Length();
123                int ca2Length = afpChain.getCa2Length();
124
125                int blockNum = afpChain.getBlockNum();
126                int totalLenIni = afpChain.getTotalLenIni();
127                double totalRmsdIni = afpChain.getTotalRmsdIni();
128                int optLength = afpChain.getOptLength();
129                double totalRmsdOpt = afpChain.getTotalRmsdOpt();
130                double chainRmsd = afpChain.getChainRmsd();
131                double alignScore = afpChain.getAlignScore();
132                int alnLength = afpChain.getAlnLength();
133                int gapLen = afpChain.getGapLen();
134                List<AFP> afpSet = afpChain.getAfpSet();
135
136                double similarity = afpChain.getSimilarity();
137                double identity = afpChain.getIdentity();
138
139                if (similarity <0  || identity < 0){
140                        afpChain.calcSimilarity();
141                        similarity = afpChain.getSimilarity();
142                        identity = afpChain.getIdentity();
143                }
144
145
146
147                String algorithmName = afpChain.getAlgorithmName();
148                //String version = afpChain.getVersion();
149
150                double probability = afpChain.getProbability();
151
152
153                int afpNum = afpSet.size();
154
155                int[] blockGap = afpChain.getBlockGap();
156
157
158                double[] blockScore = afpChain.getBlockScore();
159                double[] blockRmsd = afpChain.getBlockRmsd();
160                int[] blockSize = afpChain.getBlockSize();
161
162
163                int alnbeg1 = afpChain.getAlnbeg1();
164                int alnbeg2 = afpChain.getAlnbeg2();
165
166                char[] alnseq1 = afpChain.getAlnseq1();
167                char[] alnseq2 = afpChain.getAlnseq2();
168                char[] alnsymb = afpChain.getAlnsymb();
169
170                // == end of extractation of data values from afpChain 
171                ////////////////////////////////
172
173                StringBuffer txt = new StringBuffer();
174
175                if ( longHeader) {
176                        txt.append(String.format("Align %s.pdb %d with %s.pdb %d", name1, ca1Length, name2, ca2Length));
177                }
178                else {
179                        txt.append(String.format("Align %s.pdb Length1: %d with %s.pdb Length2: %d", name1, ca1Length, name2, ca2Length));
180                }
181                txt.append(newline);
182                if ( afpChain.isShortAlign()){
183                        txt.append("Short match");
184                        return txt.toString();
185                }
186                //txt.append(String.format("raw-score: %.2f norm.-score: %.2f ", alignScore, normAlignScore));
187
188                if ( longHeader ) {
189                        txt.append(String.format( "Twists %d ini-len %d ini-rmsd %.2f opt-equ %d opt-rmsd %.2f chain-rmsd %.2f Score %.2f align-len %d gaps %d (%.2f%%)",
190                                        blockNum - 1, totalLenIni, totalRmsdIni, optLength, totalRmsdOpt, chainRmsd, alignScore, 
191                                        alnLength, gapLen, (100.0 * (double)gapLen/(double)alnLength)) );
192                        txt.append(newline);
193
194                }  else {
195
196                        if ( ! longHeader)
197                                printScore(txt,algorithmName,probability,longHeader);
198
199                        printScoresInLines(afpChain, blockNum, optLength, totalRmsdOpt, alignScore, alnLength, gapLen, identity, similarity,txt);
200                }
201
202
203                //txt.append(String.format("P-value %.2e Afp-num %d Identity %.2f%% Similarity %.2f%% norm.-score: %.2f"+newline, probability, afpNum, identity * 100, similarity * 100, normAlignScore));
204
205                if ( longHeader) {
206                        printScore(txt,algorithmName,probability,longHeader);
207
208                        txt.append(String.format("Afp-num %d Identity %.2f%% Similarity %.2f%%", afpNum, identity * 100, similarity * 100));
209                        txt.append(newline);
210                } 
211
212                int i;
213                double gap;
214
215                if ( longHeader ){
216                        int fragLen = 8 ; // FatCatParameters.DEFAULT_FRAGLEN;
217                        for(i = 0; i < blockNum; i ++)  {
218                                gap = (double)blockGap[i] /( (double)blockGap[i] + fragLen * blockSize[i]);
219                                txt.append(String.format( "Block %2d afp %2d score %5.2f rmsd %5.2f gap %d (%.2f%%)",
220                                                i, blockSize[i], blockScore[i], blockRmsd[i], blockGap[i], gap));
221                                txt.append(newline);
222                        }
223                }
224
225                int     linelen = 70;
226                String a;
227                String b;
228                String c;
229
230
231                int     t = 0;
232                int     ap = alnbeg1;
233                int     bp = alnbeg2;
234                int     k, len;
235
236                //System.out.println(alnseq1.length + " " + alnseq1.toString());
237                
238                while((alnLength - t) > 0)      {
239                        if(alnLength - t > linelen)     len = linelen;
240                        else    len = alnLength - t;
241
242                        if ( ap >= ca1.length)
243                                break;
244                        if ( bp >= ca2.length)
245                                break;
246
247                        String pdb1 = ca1[ap].getGroup().getResidueNumber().toString();
248                        String pdb2 = ca2[bp].getGroup().getResidueNumber().toString();
249                        
250
251                        //System.err.println("t,len:"+t+":"+len);
252                        String lseq1 = new String(alnseq1).substring(t,t+len); 
253                        String lseq2 = new String(alnseq2).substring(t,t+len); 
254                        String lsymb = new String(alnsymb).substring(t,t+len); 
255
256                        //System.err.println("B:" + b);
257
258
259                        // check conservation and color accordingly, if requested by user.
260                        if ( showHTML ) {
261                                a = "";
262                                b = "";
263                                c = "";
264
265                                //      <span class=\"m\">|</span> ... Structurally equivalent and identical residues 
266                                //  <span class=\"sm\">:</span> ... Structurally equivalent and similar residues  
267                                //  <span class=\"qg\">.</span> ... Structurally equivalent, but not similar residues. 
268
269                                for (int pos = 0 ; pos < lseq1.length() ; pos ++){
270                                        char c1 = lseq1.charAt(pos);
271                                        char c2 = lseq2.charAt(pos);
272                                        char cl = lsymb.charAt(pos);
273                                        int block = -1 ;
274                                        if ( cl != ' ') {
275                                                try { 
276                                                        block = Integer.parseInt(cl+"");
277                                                } catch (Exception e){
278                                                        //
279                                                }
280                                        }
281                                        if ( cl != ' ' ){
282                                                
283                                                if ( showAlignmentBlock && block > -1 ) {
284                                                        a += "<span class=\"alignmentBlock1"+block+"\">" + c1 + "</span>";
285                                                        b += "<span class=\"alignmentBlock2"+block+"\">" + c2 + "</span>";
286                                                        c += "<span class=\"m\">" + cl + "</span>";
287                                                } else {
288                                                        a += getPrefix(c1, c2, 0, block, false).toString() + c1 + "</span>";
289                                                        b += getPrefix(c1, c2, 1, block, false).toString() + c2 + "</span>";
290                                                        c += "<span class=\"m\">" + cl + "</span>";
291                                                }
292
293                                        } else if ( c1 != '-' && c2 != '-') {
294
295                                                a += "<span class=\"sm\">" + c1 + "</span>";
296                                                b += "<span class=\"sm\">" + c2 + "</span>";
297                                                c += "<span class=\"sm\">" + cl + "</span>";
298
299
300                                        } else {
301
302                                                a += "<span class=\"qg\">" + c1 + "</span>";
303                                                b += "<span class=\"qg\">" + c2 + "</span>";
304                                                c += "<span class=\"qg\">" + cl + "</span>";
305
306                                        }
307
308                                        if(c1 != '-') ap ++;
309                                        if(c2 != '-') bp ++;
310                                }
311
312
313                        } else {
314
315                                a = lseq1;
316                                b = lseq2;
317                                c = lsymb;
318                        }
319
320                        txt.append(newline);
321                        if ( longHeader )
322                                txt.append(String.format("%14s", " "));
323                        else 
324                                txt.append(String.format("%14s", " "));
325
326                        if (  longHeader ) {
327                                for(k = 10; k <= len; k += 10)
328                                        txt.append("    .    :");
329                                if(k <= len + 5) txt.append("    .");
330
331                        } else {
332
333                                for(k = 10; k <= len; k += 10)
334                                        txt.append("----+----|");
335                                if(k <= len + 5) txt.append("----+");
336
337
338                        }
339
340                        
341
342                        txt.append(newline);
343                        txt.append(String.format("Chain 1:%5s %s"+newline +"%14s%s"+newline+"Chain 2:%5s %s",
344                                        pdb1, a, " ", c, pdb2, b));
345
346                        txt.append(newline);
347
348                        if ( ! showHTML){
349                                for(k = 0; k < len; k ++)       {
350                                        if(a.charAt(k) != '-') ap ++;
351                                        if(b.charAt(k) != '-') bp ++;
352                                }
353                        }
354                        t += len;
355
356
357
358                }
359                txt.append(newline);
360                if ( printLegend ){
361                        if ( algorithmName.equalsIgnoreCase(CeMain.algorithmName) || 
362                                        algorithmName.equalsIgnoreCase(SmithWaterman3Daligner.algorithmName)){
363                                txt.append("Note: positions are from PDB; | means alignment of identical amino acids, : of similar amino acids ");
364
365                        } else {
366                                txt.append("Note: positions are from PDB; the numbers between alignments are block index");
367                        }
368                        txt.append(newline);
369
370                }
371                return txt.toString();
372
373        }

375        private static void printScoresInLines(AFPChain afpChain, int blockNum, int optLength, double totalRmsdOpt, double alignScore,
376                        int alnLength, int gapLen, double identity, double similarity, StringBuffer txt)
377        {
378                if ( blockNum - 1 > 0) {
379                        txt.append(String.format( "Twists %d ", blockNum -1 ));
380                        txt.append(newline);
381                }
382
383                txt.append(String.format("Equ: %d ", optLength));
384                txt.append(newline);
385                txt.append(String.format("RMSD: %.2f ", totalRmsdOpt));
386                txt.append(newline);
387                txt.append(String.format("Score: %.2f ", alignScore));
388                txt.append(newline);
389                txt.append(String.format("Align-len: %d ", alnLength));
390                txt.append(newline);
391                txt.append(String.format("Gaps: %d (%.2f%%)",
392                                gapLen, (100.0 * (double)gapLen/(double)alnLength)) );
393                txt.append(newline);
394                if ( afpChain.getTMScore() >= 0) {
395                        txt.append(String.format("TM-score: %.2f",afpChain.getTMScore()));
396                        txt.append(newline);
397                }
398                txt.append(newline);
399                txt.append(String.format("Identity: %.2f%% ", identity * 100 ));
400                txt.append(newline);
401                txt.append(String.format("Similarity: %.2f%%", similarity * 100));
402                txt.append(newline);
403        }

628        /**
629         * Prints the alignment in the simplest form: a list of aligned residues.
630         * Format is one line per residue pair, tab delimited:
631         * <ul><li>1. PDB number. Includes insertion code</li>
632         * <li>1. Chain.</li>
633         * <li>1. Amino Acid. Three letter code.</li>
634         * <li>2. PDB number.</li>

762        private static CharSequence getPrefix(char oneletter1, char oneletter2,
763                        int i, int blockNr, boolean showAlignmentBlock) {
764
765                if ( oneletter1 == '-' || oneletter2 == '-' ) {
766                        // a gap in the alignment. 
767                        // label as mismatch
768                        return "<span class=\"qg\">";
769                }
770
771                // an aligned position
772                
773                if ( showAlignmentBlock && blockNr > -1){
774                        return "<span class=\"alignmentBlock"+(i+1)+(blockNr+1)+"\">";
775                }
776                
777                // we return the "default" sequence alignment view...
778                
779                if ( oneletter1 == oneletter2)
780                        return "<span class=\"m\">";
781
782                double score = AFPAlignmentDisplay.aaScore(oneletter1,oneletter2);
783                if ( score > 0 )
784                        return "<span class=\"sm\">";
785
786                // not similar
787                return "<span class=\"qg\">";
788        }

790        private static void formatPosition(int pos1, Atom[] ca, int len, StringWriter header1, StringWriter header2)
791        {
792                int linePos = len % LINELENGTH;
793
794                if ( header1.getBuffer().length() < linePos) {
795                        // fill up the buffer, we are probably shortly after the start...
796                        for ( int i = header1.getBuffer().length() ; i< linePos ; i++){
797                                header1.append(" ");
798                        }
799                }
800
801
802
803                Atom a = ca[pos1];
804                Group g = a.getGroup();
805
806                ResidueNumber residueNumber = g.getResidueNumber();
807                pos1 = residueNumber.getSeqNum();
808                boolean hasInsertionCode = false;                
809                if ( residueNumber.getInsCode() != null) {
810                        hasInsertionCode = true;
811                } 
812
813                if ( (pos1 %10  == 0) && ( ! hasInsertionCode)) {
814                        CharSequence display = getPDBPos(a);
815
816                        boolean ignoreH1 = false; 
817
818                        // make sure we don't have a problem with the left boundary...
819                        if ( header1.getBuffer().length()-1 > linePos) {
820                                ignoreH1 = true;
821                                System.out.println("Ignore h1: " + len + " " + header1.getBuffer().length() + " linePos: " + linePos +"  >" + header1.toString() +"<");
822                        }
823                        //System.out.println(len + " p1:" + tmp + " = " + pos1 + " " + " " + display + " " + ignoreH1);
824                        if ( ! ignoreH1) {
825                                header1.append(String.format("%-10s",display ));
826                                header2.append("|");
827                        } else {
828                                header2.append("|");
829                        }
830
831                } else if ( hasInsertionCode){
832                        Character insCode = g.getResidueNumber().getInsCode();
833                        if ( insCode != null)
834                                header2.append(insCode);
835                        else {
836                                header2.append("!");
837                        }
838                } else if ( ((pos1) %5 ) == 0 && len > 5) {
839                        header2.append(".");
840                } else {
841                        if ( len > 0)
842                                header2.append(" ");
843                }
844
845        }
