006package org.biojava3.genome.parsers.gff;
007
008import java.util.ArrayList;
009import java.util.LinkedHashMap;
010
011/**
012 *
013 * @author Scooter Willis <willishf at gmail dot com>
014 */
015public class FeatureHelper {
016
017    /**
018     * Build a list of individual features to allow easy indexing and to avoid iterating through large genome gff3 files
019     * The index for the returned HashMap is the value of the attribute used to build the index
020     * @param attribute
021     * @param list
022     * @return
023     */
024    static public LinkedHashMap<String,FeatureList> buildFeatureAtrributeIndex(String attribute,FeatureList list){
025
026        LinkedHashMap<String,FeatureList> featureHashMap = new LinkedHashMap<String,FeatureList>();
027        FeatureList featureList = list.selectByAttribute(attribute);
028        for(FeatureI feature : featureList){
029            String value = feature.getAttribute(attribute);
030            FeatureList features = featureHashMap.get(value);
031            if(features == null){
032                features = new FeatureList();
033                featureHashMap.put(value, features);
034            }
035            features.add(feature);
036        }
037
038        return featureHashMap;
039    }
040
041}
