023package org.biojava.bio.structure.align.gui.jmol;
024
025import java.util.logging.Logger;
026import java.util.regex.Matcher;
027import java.util.regex.Pattern;
028
029import org.biojava.bio.structure.align.gui.jmol.AtomInfo;
030
031public class AtomInfoParser {
032
033    static Logger    logger      = Logger.getLogger("org.biojava.spice");
034    
035    static Pattern pattern;
036    
037    static {
038        
039        String numberPattern = "\\[(.*)\\]([0-9^a-zA-Z]+)(:[a-zA-Z]*)?\\.([a-zA-Z]+)(/[0-9]*)?";
040        pattern = Pattern.compile(numberPattern);
041        
042    }
043    
044    public AtomInfoParser() {
045        super();
046
047        
048        
049    }
050    
051    public static void main(String[]args){
052        String s1 = "[GLY]371:A.CA #2811";
053        String s2 = "[ASP]1^A:A.CA/2 #2";
054        System.out.println(s1 + " got: " + AtomInfoParser.parse(s1));
055        System.out.println(s2 + " got: " + AtomInfoParser.parse(s2));
056    }
057    
058    
059    /** parses e.g. 
060     *  [MET]361:A.CA/1 #2843
061     *  [GLY]339:A.CA #2573
062     *  [ASN]44.CA #704
063     *  
064     * @param jmolAtomInfo
065     * @return an AtomInfo
066     */
067    public static AtomInfo parse(String jmolAtomInfo){
068       
069        
070        Matcher matcher = pattern.matcher(jmolAtomInfo);
071           
072        boolean found = matcher.find();
073        if ( ! found) {
074            logger.info("AtomInfoParser: could not parse the atomInfo string " + jmolAtomInfo);
075            return new AtomInfo();
076        }
077        String residueName   = matcher.group(1);        
078        String residueNumber = matcher.group(2);
079        String chainId       = matcher.group(3);
080        String atomName      = matcher.group(4);
081        String modelNumber   = matcher.group(5);
082        
083   
084        
085        //System.out.println(jmolAtomInfo +" | " +  residueName + " number:" + residueNumber + " chain:" + chainId + 
086        //              " atomName:" + atomName + " modelNumber:" + modelNumber );
087        
088        AtomInfo info = new AtomInfo();
089        
090        info.setAtomName(atomName);
091        info.setResidueName(residueName);
092        info.setResidueNumber(residueNumber.replaceAll("\\^",""));
093        
094        String ci = " ";
095        if (chainId != null)
096            ci = chainId.substring(1,chainId.length());        
097        info.setChainId(ci);        
098        
099        int mn = 1;
100        if ( modelNumber != null)
101            mn = Integer.parseInt(modelNumber.substring(1,modelNumber.length()));            
102        info.setModelNumber(mn);
103       
104        
105        return info;
106    }
107
108}
