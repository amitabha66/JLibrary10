171   /** initiate new group, either Hetatom, Nucleotide, or AminoAcid */
172   private Group getNewGroup(String recordName,Character aminoCode1, long seq_id,String groupCode3) {
173
174      if ( params.isLoadChemCompInfo() ){
175         Group g = ChemCompGroupFactory.getGroupFromChemCompDictionary(groupCode3);
176         if ( g != null) {
177            if ( g instanceof AminoAcidImpl) {
178               AminoAcidImpl aa = (AminoAcidImpl) g;
179               aa.setId(seq_id);
180            } else if ( g instanceof NucleotideImpl) {
181               NucleotideImpl nuc =  (NucleotideImpl) g;
182               nuc.setId(seq_id);
183            } else if ( g instanceof HetatomImpl) {
184               HetatomImpl het = (HetatomImpl)g;
185               het.setId(seq_id);
186            }
187            return g;
188         }
189
190      }
191
192
193      Group group;
194      if ( recordName.equals("ATOM") ) {
195         if (aminoCode1 == null)  {
196            // it is a nucleotide
197            NucleotideImpl nu = new NucleotideImpl();
198            group = nu;
199            nu.setId(seq_id);
200
201         } else if (aminoCode1 == StructureTools.UNKNOWN_GROUP_LABEL){
202            HetatomImpl h = new HetatomImpl();
203            h.setId(seq_id);
204            group = h;
205
206         } else {
207            AminoAcidImpl aa = new AminoAcidImpl() ;
208            aa.setAminoType(aminoCode1);
209            aa.setId(seq_id);
210            group = aa ;
211         }
212      }
213      else {
214         HetatomImpl h = new HetatomImpl();
215         h.setId(seq_id);
216         group = h;
217      }
218      //System.out.println("new group type: "+ group.getType() );
219      return  group ;
220   }
221   /** test if the chain is already known (is in current_model
222    * ArrayList) and if yes, returns the chain
223    * if no -> returns null
224    */
225   private Chain isKnownChain(String chainID, List<Chain> chains){
226
227      for (int i = 0; i< chains.size();i++){
228         Chain testchain =  chains.get(i);
229         //System.out.println("comparing chainID >"+chainID+"< against testchain " + i+" >" +testchain.getName()+"<");
230         if (chainID.equals(testchain.getChainID())) {
231            //System.out.println("chain "+ chainID+" already known ...");
232            return testchain;
233         }
234      }
235
236      return null;
237   }

240   /** during mmcif parsing the full atom name string gets truncated, fix this...
241    *
242    * @param name
243    * @return
244    */
245   private String fixFullAtomName(String name){
246
247      if (name.equals("N")){
248         return " N  ";
249      }
250      if (name.equals("CA")){
251         return " CA ";
252      }
253      if (name.equals("C")){
254         return " C  ";
255      }
256      if (name.equals("O")){
257         return " O  ";
258      }
259      if (name.equals("CB")){
260         return " CB ";
261      }
262      if (name.equals("CG"))
263         return " CG ";
264
265      if (name.length() == 2)
266         return " " + name + " ";
267
268      if (name.length() == 1)
269         return " " + name + "  ";
270
271      if (name.length() == 3)
272         return " " + name ;
273
274      return name;
275   }

296      String groupCode3    = atom.getLabel_comp_id();

302         aminoCode1 = StructureTools.get1LetterCode(groupCode3);
303         if ( aminoCode1.equals(StructureTools.UNKNOWN_GROUP_LABEL)) 
304            aminoCode1 = null;
305      }
306      String insCodeS = atom.getPdbx_PDB_ins_code();
307      Character insCode = null;
308      if (!  insCodeS.equals("?")) {
309         insCode = insCodeS.charAt(0);
310      }
311      // we store the internal seq id in the Atom._id field
312      // this is not a PDB file field but we need this to internally assign the insertion codes later
313      // from the pdbx_poly_seq entries..
314
315      long seq_id = -1;

327
328      if (! current_nmr_model.equals(nmrModel)){

331         // add previous data
332         if ( current_chain != null ) {
333            current_chain.addGroup(current_group);
334         }

337         structure.setNmr(true);

357         // end up old chain...

359
360         // see if old chain is known ...

362         testchain = isKnownChain(current_chain.getChainID(),current_model);

364         //System.out.println("trying to re-using known chain " + current_chain.getName() + " " + chain_id);

366            //System.out.println("re-using known chain " + current_chain.getName() + " " + chain_id);

368         } else {

373         if ( testchain == null) {
374            //System.out.println("unknown chain. creating new chain.");

376            current_chain = new ChainImpl();

380            current_chain = testchain;
381         }
382
383         if ( ! current_model.contains(current_chain))
384            current_model.add(current_chain);
385
386      } 
387
388
389      ResidueNumber residueNumber = new ResidueNumber(chain_id,residueNrInt, insCode);
390      
391      if (current_group == null) {

406         current_group = getNewGroup(recordName,aminoCode1,seq_id, groupCode3);

408        // current_group.setPDBCode(residueNumber);
409         current_group.setResidueNumber(residueNumber);

411            current_group.setPDBName(groupCode3);

414         }

432

488   /** convert a MMCif AtomSite object to a BioJava Atom object
489    *
490    * @param atom the mmmcif AtomSite record
491    * @return an Atom
492    */
493   private Atom convertAtom(AtomSite atom){
494
495
496      Atom a = new AtomImpl();
497
498      a.setPDBserial(Integer.parseInt(atom.getId()));
499      a.setName(atom.getLabel_atom_id());
500      a.setFullName(fixFullAtomName(atom.getLabel_atom_id()));
501
502
503      double x = Double.parseDouble (atom.getCartn_x());
504      double y = Double.parseDouble (atom.getCartn_y());
505      double z = Double.parseDouble (atom.getCartn_z());
506      a.setX(x);
507      a.setY(y);
508      a.setZ(z);
509
510      double occupancy = Double.parseDouble(atom.getOccupancy());
511      a.setOccupancy(occupancy);
512
513      double temp = Double.parseDouble(atom.getB_iso_or_equiv());
514      a.setTempFactor(temp);
515
516      String alt = atom.getLabel_alt_id();
517      if (( alt != null ) && ( alt.length() > 0) && (! alt.equals("."))){
518         a.setAltLoc(new Character(alt.charAt(0)));
519      } else {
520         a.setAltLoc(new Character(' '));
521      }
522      return a;
523
524   }

527   private Group getCorrectAltLocGroup( Character altLoc,
528                        String recordName, Character aminoCode1, String groupCode3, long seq_id) {
529
530                // see if we know this altLoc already;
531                List<Atom> atoms = current_group.getAtoms();
532                if ( atoms.size() > 0) {
533                        Atom a1 = atoms.get(0);
534                        // we are just adding atoms to the current group
535                        // probably there is a second group following later...
536                        if (a1.getAltLoc().equals(altLoc)) {
537
538                                return current_group;
539                        }
540                }
541
542                List<Group> altLocs = current_group.getAltLocs();
543                for ( Group altLocG : altLocs ){
544                        atoms = altLocG.getAtoms();
545                        if ( atoms.size() > 0) {
546                                for ( Atom a1 : atoms) {
547                                        if (a1.getAltLoc().equals( altLoc)) {
548
549                                                return altLocG;
550                                        }
551                                }
552                        }
553                }
554
555                // no matching altLoc group found.
556                // build it up.
557
558                if ( groupCode3.equals(current_group.getPDBName())) {
559                        if ( current_group.getAtoms().size() == 0)
560                                return current_group;
561                        //System.out.println("cloning current group");
562                        Group altLocG = (Group) current_group.clone();
563                        current_group.addAltLoc(altLocG);
564                        return altLocG; 
565                }
566
567
568                Group altLocG = getNewGroup(recordName,aminoCode1,seq_id,groupCode3);
569
570                try {
571                        altLocG.setPDBName(groupCode3);
572                } catch (PDBParseException e) {
573                        e.printStackTrace();
574                }
575                altLocG.setResidueNumber(current_group.getResidueNumber());
576                current_group.addAltLoc(altLocG);
577                return altLocG;
578        }

604   public void documentEnd() {
605
606      // a problem occurred earlier so current_chain = null ...
607      // most likely the buffered reader did not provide data ...
608      if ( current_chain != null ) {
609
610         current_chain.addGroup(current_group);
611         if (isKnownChain(current_chain.getChainID(),current_model) == null) {
612            current_model.add(current_chain);
613         }
614      } else {
615         if ( DEBUG){
616            System.err.println("current chain is null at end of document.");
617         }
618      }
619
620      structure.addModel(current_model);
621
622      // Goal is to reproduce the PDB files exactly:
623      // What has to be done is to use the auth_mon_id for the assignment. For this
624
625      // map entities to Chains and Compound objects...
626
627
628      for (StructAsym asym : structAsyms) {
629         if ( DEBUG )
630            System.out.println("entity " + asym.getEntity_id() + " matches asym id:" + asym.getId() );
631
632         Chain s = getEntityChain(asym.getEntity_id());
633         Chain seqres = (Chain)s.clone();
634         seqres.setChainID(asym.getId());
635
636         seqResChains.add(seqres);
637         if ( DEBUG )
638            System.out.println(" seqres: " + asym.getId() + " " + seqres + "<") ;
639
640      }
641
642
643      if ( params.isAlignSeqRes() ){
644
645         SeqRes2AtomAligner aligner = new SeqRes2AtomAligner();
646         aligner.align(structure,seqResChains);
647      }
648
649
650      //TODO: add support for these:
651
652      //structure.setConnections(connects);
653      //structure.setCompounds(compounds);
654      //linkChains2Compound(structure);
655
656      // mismatching Author assigned chain IDS and PDB internal chain ids:
657      // fix the chain IDS in the current model:
658
659      Set<String> asymIds = asymStrandId.keySet();
660
661      for (int i =0; i< structure.nrModels() ; i++){
662         List<Chain>model = structure.getModel(i);
663
664         List<Chain> pdbChains = new ArrayList<Chain>();
665
666         for (Chain chain : model) {
667            for (String asym : asymIds) {
668               if ( chain.getChainID().equals(asym)){
669                  if (DEBUG)
670                     System.out.println("renaming " + asym  + " to : " + asymStrandId.get(asym));
671
672                  chain.setChainID(asymStrandId.get(asym));
673
674                  Chain known =  isKnownChain(chain.getChainID(), pdbChains);
675                  if ( known == null ){
676                     pdbChains.add(chain);
677                  } else {
678                     // and now we join the 2 chains together again, because in cif files the data can be split up...
679                     for ( Group g : chain.getAtomGroups()){
680                        known.addGroup(g);
681                     }
682                  }
683
684                  break;
685               }
686            }
687         }
688         structure.setModel(i,pdbChains);
689      }
690

695   }

1085   /** finds the residue in the internal representation and fixes the residue number and insertion code
1086    *
1087    * @param ppss
1088    */
1089   private void replaceGroupSeqPos(PdbxPolySeqScheme ppss){
1090
1091      if (ppss.getAuth_seq_num().equals("?"))
1092         return;
1093      // at this stage we are still using the internal asym ids...
1094      List<Chain> matchinChains = getChainsFromAllModels(ppss.getAsym_id());
1095
1096      long sid = Long.parseLong(ppss.getSeq_id());
1097      for (Chain c: matchinChains){
1098         Group target = null;
1099         for (Group g: c.getAtomGroups()){
1100                 
1101            if ( g instanceof AminoAcidImpl){
1102               AminoAcidImpl aa = (AminoAcidImpl)g;
1103               if (aa.getId() == sid ) {
1104                  // found it:
1105                  target = g;
1106                  break;
1107               }
1108            }
1109            else if ( g instanceof NucleotideImpl) {
1110               NucleotideImpl n = (NucleotideImpl)g;
1111               if ( n.getId() == sid) {
1112                  target = g;
1113                  break;
1114               }
1115            } else if ( g instanceof HetatomImpl){
1116                HetatomImpl h = (HetatomImpl)g;
1117                if ( h.getId() == sid){
1118                        target =h;
1119                        break;
1120                }
1121            }
1122         }
1123         if (target == null){
1124            logger.info("could not find group at seq. position " + 
1125         ppss.getSeq_id() + " in internal chain " + c.getChainID() + ". " + ppss);
1126            continue;
1127         }
1128
1129         if (! target.getPDBName().equals(ppss.getMon_id())){
1130            logger.info("could not match PdbxPolySeqScheme to chain:" + ppss);
1131            continue;
1132         }
1133
1134         // fix the residue number to the one used in the PDB files...
1135         Integer pdbResNum = Integer.parseInt(ppss.getAuth_seq_num());
1136         // check the insertion code...
1137         String insCodeS = ppss.getPdb_ins_code();
1138         Character insCode = null;
1139         if ( ( insCodeS != null) && (! insCodeS.equals(".")) && insCodeS.length()>0){
1140            //pdbResNum += insCode
1141                 insCode = insCodeS.charAt(0);
1142         }
1143
1144         ResidueNumber residueNumber = new ResidueNumber(null, pdbResNum,insCode);
1145         target.setResidueNumber(residueNumber);
1146      }
1147   }
