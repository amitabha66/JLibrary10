027    /**
028     * Output gff3 format for a DNA Sequence
029     * @param fileName
030     * @param chromosomeSequence
031     * @throws Exception
032     */
033    public void write(OutputStream outputStream, LinkedHashMap<String, ChromosomeSequence> chromosomeSequenceList) throws Exception {
034
035        outputStream.write("##gff-version 3\n".getBytes());
036        for (String key : chromosomeSequenceList.keySet()) {
037            ChromosomeSequence chromosomeSequence = chromosomeSequenceList.get(key);
038            String gff3line = "";
039   //         if(source.length() == 0){
040   //             Collection<GeneSequence> genes = chromosomeSequence.getGeneSequences().values();
041   //             for(GeneSequence gene : genes){
042   //                 source = gene.getSource();
043   //                 break;
044   //             }
045   //         }
046   //         gff3line = key + "\t" + source + "\t" + "size" + "\t" + "1" + "\t" + chromosomeSequence.getBioEnd() + "\t.\t.\t.\tName=" + key + "\r\n";
047   //         outputStream.write(gff3line.getBytes());
048
049            for (GeneSequence geneSequence : chromosomeSequence.getGeneSequences().values()) {
050                gff3line = key + "\t" + geneSequence.getSource() + "\t" + "gene" + "\t" + geneSequence.getBioBegin() + "\t" + geneSequence.getBioEnd() + "\t";
051                Double score = geneSequence.getSequenceScore();
052                if (score == null) {
053                    gff3line = gff3line + ".\t";
054                } else {
055                    gff3line = gff3line + score + "\t";
056                }
057                gff3line = gff3line + geneSequence.getStrand().getStringRepresentation() + "\t";
058                gff3line = gff3line + ".\t";
059                gff3line = gff3line + "ID=" + geneSequence.getAccession().getID() + ";Name=" + geneSequence.getAccession().getID();
060                gff3line = gff3line + getGFF3Note(geneSequence.getNotesList());
061                gff3line = gff3line + "\n";
062                outputStream.write(gff3line.getBytes());
063
064                int transcriptIndex = 0;
065                for (TranscriptSequence transcriptSequence : geneSequence.getTranscripts().values()) {
066                    transcriptIndex++;
067
068                    gff3line = key + "\t" + transcriptSequence.getSource() + "\t" + "mRNA" + "\t" + transcriptSequence.getBioBegin() + "\t" + transcriptSequence.getBioEnd() + "\t";
069                    score = transcriptSequence.getSequenceScore();
070                    if (score == null) {
071                        gff3line = gff3line + ".\t";
072                    } else {
073                        gff3line = gff3line + score + "\t";
074                    }
075                    gff3line = gff3line + transcriptSequence.getStrand().getStringRepresentation() + "\t";
076                    gff3line = gff3line + ".\t";
077                    String id = geneSequence.getAccession().getID() + "." + transcriptIndex;
078                    gff3line = gff3line + "ID=" + id + ";Parent=" + geneSequence.getAccession().getID() + ";Name=" + id;
079                    gff3line = gff3line + getGFF3Note(transcriptSequence.getNotesList());
080
081                    gff3line = gff3line + "\n";
082                    outputStream.write(gff3line.getBytes());
083
084                    String transcriptParentName = geneSequence.getAccession().getID() + "." + transcriptIndex;
085                    ArrayList<CDSSequence> cdsSequenceList = new ArrayList(transcriptSequence.getCDSSequences().values());
086                    Collections.sort(cdsSequenceList, new SequenceComparator());
087                    for (CDSSequence cdsSequence : cdsSequenceList) {
088                        gff3line = key + "\t" + cdsSequence.getSource() + "\t" + "CDS" + "\t" + cdsSequence.getBioBegin() + "\t" + cdsSequence.getBioEnd() + "\t";
089                        score = cdsSequence.getSequenceScore();
090                        if (score == null) {
091                            gff3line = gff3line + ".\t";
092                        } else {
093                            gff3line = gff3line + score + "\t";
094                        }
095                        gff3line = gff3line + cdsSequence.getStrand().getStringRepresentation() + "\t";
096                        gff3line = gff3line + cdsSequence.getPhase() + "\t";
097                        gff3line = gff3line + "ID=" + cdsSequence.getAccession().getID() + ";Parent=" + transcriptParentName;
098                        gff3line = gff3line + getGFF3Note(cdsSequence.getNotesList());
099
100                        gff3line = gff3line + "\n";
101                        outputStream.write(gff3line.getBytes());
102                    }
103
104                }
105            }
106
107        }
108
109
110    }

112    private String getGFF3Note(ArrayList<String> notesList) {
113        String notes = "";
114
115        if (notesList.size() > 0) {
116            notes = ";Note=";
117            int noteindex = 1;
118            for (String note : notesList) {
119                notes = notes + note;
120                if (noteindex < notesList.size() - 1) {
121                    notes = notes + " ";
122                }
123            } 
124
125        }
126        return notes;
127    }
