024package org.biojava.bio.structure;
025
026import java.util.Map;
027
028
029/**
030 * <p>
031 * A {@link Group} that represents an AminoAcid.
032 * </p>
033 * 
034 * <p>
035 * In PDB files information on AminoAcids can be observed in the SEQRES and in the ATOM records.
036 * Since frequently coordinates for some of the amino acids are missing, during parsing of the PDB
037 * files the SEQRES and the ATOM records are aligned and a whenever possible the AminoAcid objects 
038 * are combined. Access to the SEQRES and ATOM sequence is possible through the {@link Chain} object.
039 * It is possible to distinguish between SEQRES and ATOM derived AminoAcids by {@link #getRecordType()}.
040 * </p>
041 * 
042 *<p>
043 *  AminoAcid inherits most from {@link HetatomImpl }.  Adds a few AminoAcid
044 *  specific methods.
045 *  </p>
046 *  
047 * @author Andreas Prlic
048 * @since 1.4
049 * @version %I% %G%
050 * 
051 */
052public interface AminoAcid extends Group {
053        
054        /** Field to distribguish AminoAcids that have been created from SEQRES records and ATOM records.
055         * 
056         */
057        public static final String ATOMRECORD = "ATOM";
058        
059        /** Field to distribguish AminoAcids that have been created from SEQRES records and ATOM records.
060         * 
061         */
062        public static final String SEQRESRECORD = "SEQRES";
063        
064    /**
065     * Specifies the secondary structure as a Map.
066     *
067     * @param secstr  a Map object specifying the sec struc 
068     * @see #getSecStruc
069     */
070    public void setSecStruc(Map<String,String> secstr) ;
071    
072    /** get secondary structure data .
073     *
074     * @return a Map object representing the sec struc value
075     *
076     * @see #setSecStruc
077     */
078    public Map<String,String> getSecStruc() ;
079
080    /** get N atom.
081     *
082     * @return an Atom object
083     * @throws StructureException ...
084    */
085    public Atom getN()  throws StructureException ;
086
087    /** get CA atom.
088     * @return an Atom object
089     * @throws StructureException ...
090     */
091    public Atom getCA() throws StructureException ;
092
093    /** get C atom.
094     * @return an Atom object
095     * @throws StructureException ...
096     */
097    public Atom getC()  throws StructureException ;
098
099    /** get O atom.
100     * @return an Atom object
101     * @throws StructureException ...
102     */
103    public Atom getO()  throws StructureException ;
104
105    /** get CB atom.
106     * @return an Atom object
107     * @throws StructureException ...
108     */
109    public Atom getCB() throws StructureException ;
110
111    
112
113
114    /** returns the name of the AA, in single letter code.
115     *
116     * @return a Character object representing the amino type value
117     * @see #setAminoType
118     */
119    public  Character getAminoType() ;
120
121    /** set the name of the AA, in single letter code .
122     *
123     * @param aa  a Character object specifying the amino type value
124     * @see #getAminoType
125     */
126    public void setAminoType(Character aa) ;
127
128        /** Allows to distinguish between amino acids that are provided
129                as ATOM records and a SEQRES records.
130         * @param recordName either ATOMRECORD or SEQRESRECORD
131         * @see #getRecordType()
132        */
133        public void setRecordType(String recordName);
134
135        /** Allows to distinguish between amino acids that are provided
136                as ATOM records and a SEQRES records.
137         * @return the origin of this amino acid (ATOM or SEQRES records)
138         * @see #setRecordType(String)
139        */
140    public String getRecordType();
141
142    /** string representation. */
143    public String toString() ;
144
145}
