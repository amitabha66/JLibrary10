030import org.biojava.bio.structure.Atom;
031import org.biojava.bio.structure.StructureException;
032import org.biojava.bio.structure.align.StructureAlignment;

034import org.biojava.bio.structure.align.fatcat.calc.FatCatParameters;
035
036import org.biojava.bio.structure.align.model.AFPChain;
037
038
039public class FatCatFlexible  extends FatCat implements StructureAlignment{
040
041
042
043        public static String algorithmName = "jFatCat_flexible";
044
045        FatCatParameters params;
046
047        public FatCatFlexible(){
048                super();
049                params = new FatCatParameters();
050        }
051
052        public AFPChain align(Atom[] ca1, Atom[] ca2) throws StructureException {
053
054                AFPChain afpChain = alignFlexible(ca1, ca2, params);
055                afpChain.setAlgorithmName(algorithmName);
056                afpChain.setVersion(VERSION+"");
057                return afpChain;
058        }
059
060        public AFPChain align(Atom[] ca1, Atom[] ca2, Object param)
061        throws StructureException {
062
063                if ( ! (param instanceof FatCatParameters)){
064                        throw new IllegalArgumentException("FatCat algorithm needs FatCatParameters object as argument.");
065                }
066
067                params = (FatCatParameters) param;              
068
069                AFPChain afpChain= alignFlexible(ca1, ca2, params);
070                afpChain.setAlgorithmName(algorithmName);
071                afpChain.setVersion(VERSION+"");
072                return afpChain;
073        }
074
075        public String getAlgorithmName() {
076
077                return algorithmName;
078        }
079
080        public String getVersion(){
081                return VERSION+"";
082        }
083
084        public ConfigStrucAligParams getParameters() {
085
086                return params;
087        }
