050        public static final  void extractAFPChains(FatCatParameters params, AFPChain afpChain,Atom[] ca1,Atom[] ca2){
051
052
053
054                List<AFP> afpSet = new ArrayList<AFP>();
055                afpChain.setAfpSet(afpSet);
056
057                if ( debug )
058                        System.err.println("nr of atoms ca1: " + ca1.length + " ca2: " +  ca2.length);
059
060
061
062                int     p1, p2;
063                int n0, n, n1, n2;
064                double  filter1;
065                double rmsd = 0;
066                //double[] r = new double[9]; // rotation matrix
067                //double[] t = new double[3]; // shift vector
068
069                Matrix r = new Matrix(3,3);
070                Atom   t = new AtomImpl();
071
072
073                int sparse = params.getSparse();
074                int maxTra = params.getMaxTra();
075                int fragLen = params.getFragLen();
076                double disFilter = params.getDisFilter();
077                double rmsdCut = params.getRmsdCut();
078                double badRmsd = params.getBadRmsd();
079                double fragScore = params.getFragScore();
080
081                int     add = sparse + 1; //if add > 1, use sparse sampling
082                n0 = n = n1 = n2 = 0;
083
084                int minLen = 0;
085
086                int prot1Length = ca1.length;
087                int prot2Length = ca2.length;
088
089                if(prot1Length < prot2Length)
090                        minLen = prot1Length;
091                else
092                        minLen = prot2Length;
093                afpChain.setMinLen(minLen);
094
095                afpChain.setBlockResList(new int[maxTra+1][2][minLen]);
096                afpChain.setFocusRes1(new int[minLen]);
097                afpChain.setFocusRes2(new int[minLen]);
098
099                for(p1 = 0; p1 < prot1Length - fragLen; p1 += add )    {
100                        for(p2 = 0; p2 < prot2Length - fragLen; p2 += add)     {
101                                n0 ++;
102                                filter1 = getEnd2EndDistance(ca1, ca2, p1, p1 + fragLen - 1, p2, p2 + fragLen - 1);
103                                //difference bewteen end-to-end distances
104                                if(filter1 > disFilter) { n1 ++; continue; }
105                                boolean filter2 = filterTerminal(ca1,ca2, p1, p1 + fragLen - 1, p2, p2 + fragLen - 1, fragLen, minLen);
106                                if(filter2)     {
107                                        n2 ++;
108                                        continue;
109
110                                } //be cautious to use this filter !!
111
112                                // here FATCAT does a a jacobi transformation
113                                //rmsd = kearsay(fragLen, ca1[p1], ca2[p2], r, t);
114                                // we use the BioJava SVD instead...
115
116                                //
117                                rmsd = getRmsd(ca1,ca2,fragLen, p1,p2,r,t);
118
119                                //printf("afp %d: p1 %d p2 %d rmsd %f end-to-end dis %f\n", afpSet.size(), p1, p2, rmsd, filter1);
120
121                                if(rmsd < rmsdCut)      {
122                                        AFP     afptmp = new AFP();
123                                        afptmp.setP1(p1);
124                                        afptmp.setP2(p2);
125                                        afptmp.setFragLen(fragLen);
126                                        afptmp.setRmsd(rmsd);
127                                        afptmp.setM(r);
128                                        afptmp.setT(t.getCoords());
129                                        afptmp.setScore(scoreAfp(afptmp,badRmsd,fragScore));
130                                        afpSet.add(afptmp);
131                                        n ++;
132                                }
133                        }
134                }
135
136                int afpNum = afpSet.size();
137
138                if(debug) {
139                        String msg = String.format("possible AFP-pairs %d, remain %d after filter 1 remove %d; filter 2 remove %d\n",
140                                        n0, afpNum, n1, n2);
141                        System.err.println(msg);
142                }
143
144
145        }
146
147        /**
148         * filter 1 for AFP extration: the distance of end-to-end
149         * @param p1b
150         * @param p1e
151         * @param p2b
152         * @param p2e
153         * @return
154         */
155        private static final double getEnd2EndDistance(Atom[] ca1, Atom[] ca2, int p1b, int p1e, int p2b, int p2e)
156        {
157
158                double min = 99;
159                try {
160                        double dist1 = Calc.getDistance(ca1[p1b], ca1[p1e]);
161                        double dist2 = Calc.getDistance(ca2[p2b], ca2[p2e]);
162                        min = dist1 - dist2;
163
164                } catch (Exception e){
165                        e.printStackTrace();
166                }
167                return Math.abs(min);
168        }
169
170        /**
171         * filter 2 for AFP extration: the context
172         * @param p1b
173         * @param p1e
174         * @param p2b
175         * @param p2e
176         * @return
177         */
178
179        private static final  boolean filterTerminal(Atom[] ca1, Atom[] ca2, int p1b, int p1e, int p2b, int p2e, int fragLen, int minLen)
180        {
181                int     d1 = (p1b < p2b)?p1b:p2b;
182                int     d2 = (ca1.length - p1e) < (ca2.length - p2e)?(ca1.length - p1e):(ca2.length - p2e);
183                int     d3 = d1 + d2 + fragLen; //maximum alignment length from current AFP
184
185
186                /// DO NOT DO Math.round() this will give different results to FATCAT....
187                int     d4 = (int)(0.3 * minLen);
188
189                if(d3 < d4)     {
190                        //if (debug){
191                        //   System.out.println("filterTerminal: " + d3 + " " + d4 +" " + p1b + " " + p1e + " " + p2b + " " +  p2e + " " + fragLen + " " + minLen );
192                        //}
193                        return true;
194                }
195
196                return false;
197        }

199        private static final double getRmsd(Atom[] ca1, Atom[] ca2, int fragLen, int p1, int p2, Matrix m, Atom t) {
200
201
202                double rmsd = 99.9;
203                try {
204                        Atom[] catmp1 = getFragment(ca1, p1, fragLen,false);
205                        Atom[] catmp2 = getFragment(ca2, p2, fragLen,true); // clone the atoms for fragment 2 so we can manipulate them...
206
207                        if ( catmp1 == null) {
208                                System.err.println("could not get fragment for ca1 " + p1 + " " + fragLen );
209                                return rmsd;
210
211                        }
212
213                        if ( catmp2 == null) {
214                                System.err.println("could not get fragment for ca2 " + p2 + " " + fragLen );
215                                return rmsd;
216
217                        }
218
219                        SVDSuperimposer svd = new SVDSuperimposer(catmp1, catmp2);
220
221                        m = svd.getRotation();
222                        t = svd.getTranslation();
223
224                        for (Atom a : catmp2){
225                                Calc.rotate(a,m);
226                                Calc.shift(a,t);
227
228                        }
229
230                        rmsd = SVDSuperimposer.getRMS(catmp1,catmp2);
231
232                } catch (Exception e){
233                        e.printStackTrace();
234                }
235
236                return rmsd;
237        }

239        /** get a continue subset of Atoms based by the starting position and the length
240         *
241         * @param caall
242         * @param pos ... the start position
243         * @param fragmentLength .. the length of the subset to extract.
244         * @param clone: returns a copy of the atom (in case the coordinate get manipulated...)
245         * @return an Atom[] array
246         */
247        private static final Atom[] getFragment(Atom[] caall, int pos, int fragmentLength , boolean clone){
248
249                if ( pos+fragmentLength > caall.length)
250                        return null;
251
252                Atom[] tmp = new Atom[fragmentLength];
253
254                for (int i=0;i< fragmentLength;i++){
255                        if (clone){
256                                tmp[i] = (Atom)caall[i+pos].clone();
257                        } else {
258                                tmp[i] = (Atom)caall[i+pos];
259                        }
260                }
261                return tmp;
262
263        }

266        /**
267         * Assign score to each AFP
268         */
269
270        private static final double scoreAfp(AFP afp, double badRmsd, double fragScore)
271        {
272                //longer AFP with low rmsd is better
273                double  s, w;
274                //s = (rmsdCut - afptmp.rmsd) * afptmp.len; //the same scroing strategy as that in the post-processing
275                w = afp.getRmsd() / badRmsd;
276                w = w * w;
277                s = fragScore * (1.0 - w);
278                return s;
279        }

281        //------------------------------------------------------------------
282        //Sort the AFPs in increase of their diagonals(i,j)
283        //------------------------------------------------------------------
284        public static final  void sortAfps(AFPChain afpChain, Atom[] ca1, Atom[] ca2)
285        {
286
287
288                List<AFP> afpSet = afpChain.getAfpSet();
289
290                if ( debug)
291                        System.err.println("entering sortAfps");
292
293                int pro1Len = ca1.length;
294                int pro2Len = ca2.length;
295
296                afpChain.setAfpIndex(      new int[pro1Len][pro2Len]); //the index of (i,j) pair in AFP list, otherwise -1
297                afpChain.setAfpAftIndex(   new int[pro1Len][pro2Len]);  //the index of AFP (i,j*) nearest to (i,j), j*<j. if a AFP exits for (i,j), it equals to afpIndex
298                afpChain.setAfpBefIndex(   new int[pro1Len][pro2Len]); //the index of AFP (i,j*) nearest to (i,j), j*>j. if a AFP exits for (i,j), it equals to afpIndex
299
300                int[][] afpIndex       = afpChain.getAfpIndex();
301                int[][] afpAftIndex    = afpChain.getAfpAftIndex();
302                int[][] afpBefIndex    = afpChain.getAfpBefIndex();
303
304                for(int i = 0; i < pro1Len; i ++)   {
305                        for(int j = 0; j < pro2Len; j ++)   {
306
307                                afpIndex[i][j] = afpAftIndex[i][j] = afpBefIndex[i][j] = -1;
308                        }
309                }
310
311                //index the AFP for easy extraction of compatible AFPs
312                int afpNum = afpSet.size();
313
314                int     b0 = 0;
315                for(int a = 0; a < afpNum; a ++)    {
316                        if(a == afpNum - 1 || afpSet.get(a).getP1() != afpSet.get(a+1).getP1())   {
317                                int i = afpSet.get(a).getP1();
318                                for(int b = b0; b <= a; b ++)       {
319                                        int j = afpSet.get(b).getP2();
320                                        afpIndex[i][j]=b ;
321                                        afpBefIndex[i][j]=b;
322                                        afpAftIndex[i][j]=b;
323                                        if(afpSet.get(b).getP1() != i)    {
324                                                System.err.println(String.format("Warning: wrong afp index %d %d\n", i, afpSet.get(b).getP1()));
325                                                return;
326                                        }
327                                }
328                                for(int k = 1; k < pro2Len; k ++)   {
329                                        if( afpBefIndex[i][k] == -1){
330                                                afpBefIndex[i][k] = afpBefIndex[i][k-1];
331                                        }
332                                }
333                                for(int k = pro2Len - 2; k >= 0; k --)      {
334                                        if(afpAftIndex[i][k] == -1) {
335                                                afpAftIndex[i][k] =  afpAftIndex[i][k+1];
336                                        }
337                                }
338                                b0 = a + 1;
339                        }
340                }
341
342                if ( debug)
343                        System.err.println("done sortAfps");
344
345
346        }
