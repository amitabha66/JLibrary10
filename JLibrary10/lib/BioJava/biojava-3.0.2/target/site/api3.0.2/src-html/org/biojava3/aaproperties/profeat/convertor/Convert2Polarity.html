001package org.biojava3.aaproperties.profeat.convertor;
002
003public class Convert2Polarity extends Convertor{
004        /**
005         * Class for the conversion of protein sequence into polarity
006         */
007        
008        @Override
009        public char convert(char c){
010                switch(c){
011                case 'L':case 'I':case 'F':case 'W':case 'C':case 'M':case 'V':case 'Y':
012                        return group1;//Polarity value 4.9-6.2
013                case 'P':case 'A':case 'T':case 'G':case 'S':
014                        return group2;//Polarity value 8.0-9.2
015                case 'H':case 'Q':case 'R':case 'K':case 'N':case 'E':case 'D':
016                        return group3;//Polarity value 10.4-13.0
017                default:
018                        return unknownGroup;//Non-standard AA
019                }
020        }
021        private static String[] subCategory = {"Value_4.9-6.2", "Value_8.0-9.2", "Value_10.4-13.0"};
022        @Override
023        public String[] getGrouping(){return subCategory;}
024        @Override
025        public String getAttribute(){return "Polarity";}
026}
