067    /**
068     * calculate distance between two atoms.
069     *
070     * @param a  an Atom object
071     * @param b  an Atom object
072     * @return a double
073     * @throws StructureException ...
074     */
075    public static final double getDistance(Atom a, Atom b) 
076    throws StructureException
077    {
078        double x = a.getX() - b.getX();
079        double y = a.getY() - b.getY();
080        double z = a.getZ() - b.getZ();
081        
082        double s  = x * x  + y * y + z * z;
083        
084        double dist = Math.sqrt(s);
085        
086        return dist ;
087    }

090    /**
091     * Will calculate the *square* of distances between two atoms. This will be
092     * faster as it will not perform the final square root to get the actual
093     * distance. Use this if doing large numbers of distance comparisons - it is
094     * marginally faster than getDistance().
095     *
096     * @param a  an Atom object
097     * @param b  an Atom object
098     * @return a double
099     * @throws StructureException ...
100     */
101    public static double getDistanceFast(Atom a, Atom b)
102    throws StructureException
103    {
104        double x = a.getX() - b.getX();
105        double y = a.getY() - b.getY();
106        double z = a.getZ() - b.getZ();
107
108        double distSquared  = x * x  + y * y + z * z;
109
110        return distSquared ;
111    }

189    /** Vector product .
190     *
191     * @param a  an Atom object
192     * @param b  an Atom object
193     * @return an Atom object
194     */
195    public static final Atom vectorProduct(Atom a , Atom b){
196        double[] coords = new double[3];
197        
198        coords[0] = a.getY() * b.getZ() - a.getZ() * b.getY();
199        coords[1] = a.getZ() * b.getX() - a.getX() * b.getZ();
200        coords[2] = a.getX() * b.getY() - a.getY() * b.getX();
201        
202        Atom c = new AtomImpl();
203        c.setCoords(coords);
204        return c ;
205        
206    }

230    /** angle.
231     *
232     * @param a  an Atom object
233     * @param b  an Atom object
234     * @return a double
235     */
236    public static final double angle(Atom a, Atom b){
237        
238        double skalar;
239        double angle;
240        
241        skalar = skalarProduct(a,b);
242        
243        angle = skalar/( amount(a) * amount (b) );
244        angle = Math.acos(angle);
245        angle = angle * RADIAN ; 
246        
247        return angle;
248    }

270    /** torsion angle 
271     * = angle between the normal vectors of the 
272     * two plains a-b-c and b-c-d.
273     *
274     * @param a  an Atom object
275     * @param b  an Atom object
276     * @param c  an Atom object
277     * @param d  an Atom object
278     * @return a double
279     * @throws StructureException ...
280     */
281    
282    public static final double torsionAngle(Atom a, Atom b, Atom c, Atom d)
283    throws StructureException
284    {
285        
286        Atom ab = substract(a,b);
287        Atom cb = substract(c,b);
288        Atom bc = substract(b,c);
289        Atom dc = substract(d,c);
290        
291        Atom abc = vectorProduct(ab,cb);        
292        Atom bcd = vectorProduct(bc,dc);
293        
294        
295        double angl = angle(abc,bcd) ;
296        
297        /* calc the sign: */
298        Atom vecprod = vectorProduct(abc,bcd);  
299        double val = skalarProduct(cb,vecprod);
300        if (val<0.0) angl = -angl ;
301        
302        return angl;
303    }

306     *
307     * @param a  an AminoAcid object
308     * @param b  an AminoAcid object
309     * @return a double
310     * @throws StructureException ...
311     */
312    public static final double getPhi(AminoAcid a, AminoAcid b)
313    throws StructureException
314    {
315        
316        if ( ! isConnected(a,b)){
317            throw new StructureException("can not calc Phi - AminoAcids are not connected!") ;
318        } 
319        
320        Atom a_C  = a.getC();
321        Atom b_N  = b.getN();
322        Atom b_CA = b.getCA();
323        Atom b_C  = b.getC();
324        
325        double phi = torsionAngle(a_C,b_N,b_CA,b_C);
326        return phi ;
327    }

353    /** test if two amino acids are connected, i.e.
354     * if the distance from C to N < 2,5 Angstrom.
355     *
356     * @param a  an AminoAcid object
357     * @param b  an AminoAcid object
358     * @return true if ...
359     * @throws StructureException ...
360     */    
361    public static final boolean isConnected(AminoAcid a, AminoAcid b)
362    throws StructureException
363    {
364        Atom C = a.getC();
365        Atom N = b.getN();
366        
367        // one could also check if the CA atoms are < 4 A...
368        double distance = getDistance(C,N);
369        if ( distance < 2.5) { 
370            return true ;
371        } else {
372            return false ;
373        }
374    }

378    /** rotate a single atom aroud a rotation matrix.
379     * matrix must be a 3x3 matrix.
380     * @param atom atom to be rotated
381     * @param m a rotation matrix represented as a double[3][3] array
382     */
383    public static final void rotate(Atom atom, double[][] m){
384
385        double x = atom.getX();
386        double y = atom.getY() ;
387        double z = atom.getZ();
388        
389        double nx = m[0][0] * x + m[0][1] * y +  m[0][2] * z ;
390        double ny = m[1][0] * x + m[1][1] * y +  m[1][2] * z ;
391        double nz = m[2][0] * x + m[2][1] * y +  m[2][2] * z ;
392        
393        double[] coords = new double[3] ;
394        coords[0] = nx ;
395        coords[1] = ny ;
396        coords[2] = nz ;
397        
398        atom.setCoords(coords);
399    }

616    /** Returns the center  of mass of the set of atoms.
617     * @param atomSet a set of Atoms
618     * @return an Atom representing the Centroid of the set of atoms
619     */
620    public static final Atom getCentroid(Atom[] atomSet){
621        
622        double[] coords = new double[3];
623        
624        coords[0] = 0;
625        coords[1] = 0;
626        coords[2] = 0 ;
627        
628        for (int i =0 ; i < atomSet.length; i++){
629            Atom a = atomSet[i];
630            coords[0] += a.getX();
631            coords[1] += a.getY();
632            coords[2] += a.getZ();
633        }
634        
635        int n = atomSet.length;
636        coords[0] = coords[0] / n;
637        coords[1] = coords[1] / n;
638        coords[2] = coords[2] / n;
639        
640        Atom vec = new AtomImpl();
641        vec.setCoords(coords);
642        return vec;
643        
644    }

646    public static  Atom centerOfMass(Atom[] points) {
647                Atom center = new AtomImpl();
648
649                float totalMass = 0.0f;
650                for (int i = 0, n = points.length; i < n; i++) {
651                        Atom a = points[i];
652                        float mass = a.getElement().getAtomicMass();
653                        totalMass += mass;
654                        center = scaleAdd(mass, a, center);
655                }
656                
657                center = scale(center, 1.0f/totalMass);
658                return center;
659        }

661    private static Atom scale(Atom a, float s) {
662                double x = a.getX();
663                double y = a.getY();
664                double z = a.getZ();
665                
666                x *= s;
667                y *= s;
668                z *= s;
669                
670                //Atom b = new AtomImpl();
671                a.setX(x);
672                a.setY(y);
673                a.setZ(z);
674
675                return a;
676        }

686                t2.setX(x);
687                t2.setY(y);
688                t2.setZ(z);

743

746        Atom[] newAtoms = new AtomImpl[atomSet.length];

749            Atom a = atomSet[i];

751            newAtoms[i] = n ;
752        }
753        return newAtoms;

759    /** creates a virtual C-beta atom. this might be needed when working with GLY

761     * thanks to Peter Lackner for a python template of this method.
762     * @param amino the amino acid for which a "virtual" CB atom should be calculated 
763     * @return a "virtual" CB atom
764     * @throws StructureException

766    public static final Atom createVirtualCBAtom(AminoAcid amino) 
767    throws StructureException{
768        
769        AminoAcid  ala = StandardAminoAcid.getAminoAcid("ALA");
770        Atom aN  = ala.getN();
771        Atom aCA = ala.getCA();
772        Atom aC  = ala.getC();
773        Atom aCB = ala.getCB();
774        
775        
776        Atom[] arr1 = new Atom[3];
777        arr1[0] = aN;
778        arr1[1] = aCA;
779        arr1[2] = aC;
780        
781        Atom[] arr2 = new Atom[3];
782        arr2[0] = amino.getN();
783        arr2[1] = amino.getCA();
784        arr2[2] = amino.getC();
785        
786        // ok now we got the two arrays, do a SVD:
787        
788        SVDSuperimposer svd = new SVDSuperimposer(arr2,arr1);
789        
790        Matrix rotMatrix = svd.getRotation();
791        Atom tranMatrix = svd.getTranslation();
792                    
793        Calc.rotate(aCB,rotMatrix);
794
795        Atom virtualCB = Calc.add(aCB,tranMatrix);
796        virtualCB.setName("CB");
797        virtualCB.setFullName(" CB ");
798        
799        return virtualCB;
800    }

803    /**Gget euler angles for a matrix given in ZYZ convention.
804     * (as e.g. used by Jmol)
805     * 
806     * @param m the rotation matrix
807     * @return the euler values for a rotation around Z, Y, Z in degrees...
808     */
809    public static final double[] getZYZEuler(Matrix m) {
810        double m22 = m.get(2,2);
811        double rY = (float) Math.acos(m22) * degreesPerRadian;
812        double rZ1, rZ2;
813        if (m22 > .999d || m22 < -.999d) {
814          rZ1 = (double) Math.atan2(m.get(1,0),  m.get(1,1)) * degreesPerRadian;
815          rZ2 = 0;
816        } else {
817          rZ1 = (double) Math.atan2(m.get(2,1), -m.get(2,0)) * degreesPerRadian;
818          rZ2 = (double) Math.atan2(m.get(1,2),  m.get(0,2)) * degreesPerRadian;
819        }
820        return new double[] {rZ1,rY,rZ2};
821    }

824    /** Convert a rotation Matrix to Euler angles.
825     *   This conversion uses conventions as described on page:
826     *   http://www.euclideanspace.com/maths/geometry/rotations/euler/index.htm
827     *   Coordinate System: right hand
828     *   Positive angle: right hand
829     *   Order of euler angles: heading first, then attitude, then bank
830     *   
831     * @param m the rotation matrix
832     * @return a array of three doubles containing the three euler angles in radians
833     */   
834    public static final double[] getXYZEuler(Matrix m){
835        double heading, attitude, bank;
836        
837        // Assuming the angles are in radians.
838        if (m.get(1,0) > 0.998) { // singularity at north pole
839            heading = Math.atan2(m.get(0,2),m.get(2,2));
840            attitude = Math.PI/2;
841            bank = 0;
842            
843        } else if  (m.get(1,0) < -0.998) { // singularity at south pole
844            heading = Math.atan2(m.get(0,2),m.get(2,2));
845            attitude = -Math.PI/2;
846            bank = 0;
847            
848        } else {
849            heading = Math.atan2(-m.get(2,0),m.get(0,0));
850            bank = Math.atan2(-m.get(1,2),m.get(1,1));
851            attitude = Math.asin(m.get(1,0));
852        }
853        return new double[] { heading, attitude, bank };
854    }

858    /** This conversion uses NASA standard aeroplane conventions as described on page:
859    *   http://www.euclideanspace.com/maths/geometry/rotations/euler/index.htm
860    *   Coordinate System: right hand
861    *   Positive angle: right hand
862    *   Order of euler angles: heading first, then attitude, then bank.
863    *   matrix row column ordering:
864    *   [m00 m01 m02]
865    *   [m10 m11 m12]
866    *   [m20 m21 m22]
867     * @param heading in radians
868     * @param attitude  in radians
869     * @param bank  in radians
870     * @return the rotation matrix */
871    public static final  Matrix matrixFromEuler(double heading, double attitude, double bank) {
872        // Assuming the angles are in radians.
873        double ch = Math.cos(heading);
874        double sh = Math.sin(heading);
875        double ca = Math.cos(attitude);
876        double sa = Math.sin(attitude);
877        double cb = Math.cos(bank);
878        double sb = Math.sin(bank);
879
880        Matrix m = new Matrix(3,3);
881        m.set(0,0, ch * ca);
882        m.set(0,1, sh*sb - ch*sa*cb);
883        m.set(0,2, ch*sa*sb + sh*cb);
884        m.set(1,0, sa);
885        m.set(1,1, ca*cb);
886        m.set(1,2, -ca*sb);
887        m.set(2,0, -sh*ca);
888        m.set(2,1, sh*sa*cb + ch*sb);
889        m.set(2,2, -sh*sa*sb + ch*cb);
890        
891        return m;
892    }
