075        public static void main(String[] args){
076                try {
077
078                        PDBFileReader pdbr = new PDBFileReader();
079
080                        pdbr.setAutoFetch(true);
081                        pdbr.setPath("/tmp/");
082
083                        String pdbCode = "5pti";
084
085                        Structure struc = pdbr.getStructureById(pdbCode);
086
087                        BiojavaJmol jmolPanel = new BiojavaJmol();
088
089                        jmolPanel.setStructure(struc);
090
091                        // send some RASMOL style commands to Jmol
092                        jmolPanel.evalString("select * ; color chain;");
093                        jmolPanel.evalString("select *; spacefill off; wireframe off; backbone 0.4;  ");
094                        jmolPanel.evalString("save STATE state_1");
095                } catch (Exception e){
096                        e.printStackTrace();
097                }
098        }

103        public BiojavaJmol() {          
104
105                frame = new JFrame();
106
107                JMenuBar menu = MenuCreator.initMenu();
108
109                frame.setJMenuBar(menu);
110
111                frame.addWindowListener( new WindowAdapter() {
112                        public void windowClosing(WindowEvent e) {
113                                frame.dispose();
114                                //System.exit(0);
115                        }
116                });
117
118                Container contentPane = frame.getContentPane();
119
120                Box vBox = Box.createVerticalBox();
121
122                jmolPanel = new JmolPanel();
123        
124                jmolPanel.setPreferredSize(new Dimension(500,500));
125                vBox.add(jmolPanel);
126
127
128                JTextField field = new JTextField();
129
130                field.setMaximumSize(new Dimension(Short.MAX_VALUE,30));   
131                field.setText("enter RASMOL like command...");
132                org.biojava.bio.structure.align.gui.jmol.RasmolCommandListener listener = new org.biojava.bio.structure.align.gui.jmol.RasmolCommandListener(jmolPanel,field) ;
133
134                field.addActionListener(listener);
135                field.addMouseListener(listener);
136                field.addKeyListener(listener);
137                vBox.add(field);
138
139
140                /// COMBO BOXES 
141                Box hBox1 = Box.createHorizontalBox();
142
143

145                JComboBox style = new JComboBox(styles);
146
147                hBox1.add(new JLabel("Style"));
148                hBox1.add(style);
149                vBox.add(hBox1);
150                
151
152                style.addActionListener(jmolPanel);
153
154                String[] colorModes = new String[] { "Secondary Structure", "By Chain", "Rainbow", "By Element", "By Amino Acid", "Hydrophobicity" };
155                JComboBox colors = new JComboBox(colorModes);
156                colors.addActionListener(jmolPanel);
157                hBox1.add(Box.createGlue());
158                hBox1.add(new JLabel("Color"));
159                hBox1.add(colors);
160                
161                // Check boxes
162                Box hBox2 = Box.createHorizontalBox();
163                
164                
165                JButton resetDisplay = new JButton("Reset Display");
166                
167                resetDisplay.addActionListener(new ActionListener() {
168                        
169                        public void actionPerformed(ActionEvent e) {
170                                System.out.println("reset!!");
171                                jmolPanel.executeCmd("restore STATE state_1");
172                                
173                        }
174                });
175                
176                hBox2.add(resetDisplay); hBox2.add(Box.createGlue());
177                
178                JCheckBox toggleSelection = new JCheckBox("Show Selection");
179                toggleSelection.addItemListener(
180                            new ItemListener() {
181                                        
182                                        public void itemStateChanged(ItemEvent e) {
183                                                  boolean showSelection = (e.getStateChange() == ItemEvent.SELECTED);
184                                                  
185                                                  if (showSelection){
186                                                          jmolPanel.executeCmd("set display selected");
187                                                  } else {
188                                                          jmolPanel.executeCmd("set display off");
189                                                  }
190                                                
191                                        }
192                                }
193                        );
194                
195                
196                
197                hBox2.add(toggleSelection);
198                
199                hBox2.add(Box.createGlue());
200                vBox.add(hBox2);
201                
202        
203                // finish up
204                contentPane.add(vBox);
205                frame.pack();
206                frame.setVisible(true); 
207        
208        
209        }

213        /** returns true if Jmol can be found in the classpath, otherwise false.
214         * 
215         * @return true/false depending if Jmol can be found
216         */
217        public static boolean jmolInClassPath(){
218                try {
219                        Class.forName(viewer);          
220                } catch (ClassNotFoundException e){
221                        e.printStackTrace();                    
222                        return false;
223                }
224                return true;
225        }

228                if ( jmolPanel == null ){
229                        System.err.println("please install Jmol first");
230                        return;
231                }
232                jmolPanel.evalString(rasmolScript);
233        }
