301        private void parserTest(String pdbId, String residId) throws IOException, StructureException {
302                Set<ProteinModification> mods;
303                if (residId==null) {
304                        mods = ProteinModificationRegistry.allModifications();
305                } else {
306                        mods = ProteinModificationRegistry.getByResidId(residId);
307                }
308                
309                parserTest(pdbId, mods);
310        }
311        
312        private void parserTest(String pdbId, Set<ProteinModification> mods) throws IOException, StructureException {     
313                Structure struc = TmpAtomCache.cache.getStructure(pdbId);
314
315                ProteinModificationIdentifier parser = new ProteinModificationIdentifier();
316                boolean recordUnidentifiable = false;
317                parser.setRecordUnidentifiableCompounds(recordUnidentifiable);
318//              parser.setbondLengthTolerance(2);
319                
320                assertFalse(mods.isEmpty());
321
322                parser.identify(struc, mods);
323
324                assertFalse(pdbId + " is not false" , 
325                                parser.getIdentifiedModifiedCompound().isEmpty());
326                
327                boolean print = false;
328                if (print)
329                        printResult(pdbId, parser, recordUnidentifiable);
330        }
331        
332        private void printResult(String pdbId, ProteinModificationIdentifier parser, boolean recordUnidentifiable) {
333                StringBuilder sb = new StringBuilder();
334
335                sb.append("===");
336                sb.append(pdbId);
337                sb.append("===\n");
338                
339                Set<ModifiedCompound> mcs = parser.getIdentifiedModifiedCompound();
340                
341                int i=0;
342                for (ModifiedCompound mc : mcs) {
343                        sb.append("Modification #");
344                        sb.append(++i);
345                        sb.append(":\n");
346                        sb.append(mc);
347                        sb.append('\n');
348                }
349                
350                if (recordUnidentifiable) {
351                        Set<StructureGroup> unidentifiedModifiedResidues = parser.getUnidentifiableModifiedResidues();
352                        i = 0;
353                        for (StructureGroup group : unidentifiedModifiedResidues) {
354                                sb.append("Unidenfied modified residue #");
355                                sb.append(++i);
356                                sb.append(":\n");
357                                sb.append(group);
358                                sb.append('\n');
359                        }
360        
361                        Set<StructureAtomLinkage> unidentifiedLinkages = parser.getUnidentifiableAtomLinkages();
362                        i = 0;
363                        for (StructureAtomLinkage link : unidentifiedLinkages) {
364                                sb.append("Unidenfied linkage #");
365                                sb.append(++i);
366                                sb.append(":\n");
367                                sb.append(link);
368                                sb.append('\n');
369                        }
370                }
371                
372                System.out.println(sb.toString());
373        }
