099        public void testGetSubRanges() throws StructureException {
100                String range;
101                Structure substr;
102                Chain chain;
103
104                // normal substructures
105                range = "A:3-7";
106                substr = StructureTools.getSubRanges(structure2, range);
107                assertEquals("Wrong number of chains in "+range, 1, substr.size());
108
109                chain = substr.getChain(0);
110
111                assertEquals("Did not find the expected number of residues in "+range, 5, chain.getAtomLength() );
112
113                // full chains
114                range = "A";
115                substr = StructureTools.getSubRanges(structure2, range);
116                assertEquals("Wrong number of chains in "+range, 1, substr.size());
117
118                chain = substr.getChain(0);
119
120                assertEquals("Did not find the expected number of residues in "+range, 411, chain.getAtomLength() );
121                //assertEquals("subrange doesn't equal original chain A.", structure2.getChainByPDB("A"), chain);
122
123                // full chains
124                range = "A:";
125                substr = StructureTools.getSubRanges(structure2, range);
126                assertEquals("Wrong number of chains in "+range, 1, substr.size());
127
128                chain = substr.getChain(0);
129
130                assertEquals("Did not find the expected number of residues in "+range, 411, chain.getAtomLength() );
131                //assertEquals("subrange doesn't equal original chain A.", structure2.getChainByPDB("A"), chain);
132
133                // combined ranges
134                range = "A:3-7,B:8-12";
135                substr = StructureTools.getSubRanges(structure2, range);
136                assertEquals("Wrong number of chains in "+range, 2, substr.size());
137
138                chain = substr.getChain(0);
139                assertEquals("Did not find the expected number of residues in first chain of "+range, 5, chain.getAtomLength() );
140
141                chain = substr.getChain(1);
142                assertEquals("Did not find the expected number of residues in second chain of "+range, 5, chain.getAtomLength() );
143
144                // combined ranges
145                range = "A,B:8-12";
146                substr = StructureTools.getSubRanges(structure2, range);
147                assertEquals("Wrong number of chains in "+range, 2, substr.size());
148
149                chain = substr.getChain(0);
150                assertEquals("Did not find the expected number of residues in first chain of "+range, 411, chain.getAtomLength() );
151
152                chain = substr.getChain(1);
153                assertEquals("Did not find the expected number of residues in second chain of "+range, 5, chain.getAtomLength() );
154
155                // parentheses
156                range = "(A:3-7)";
157                substr = StructureTools.getSubRanges(structure2, range);
158                assertEquals("Wrong number of chains in "+range, 1, substr.size());
159
160                chain = substr.getChain(0);
161                assertEquals("Did not find the expected number of residues in "+range, 5, chain.getAtomLength() );
162
163                // some negative tests
164                try {
165                        range = "7-10";
166                        substr = StructureTools.getSubRanges(structure2, range);
167                        fail("Illegal range '"+range+"'. Should throw StructureException");
168                } catch(StructureException ex) {} //expected 
169                try {
170                        range = "A7-10";
171                        substr = StructureTools.getSubRanges(structure2, range);
172                        fail("Illegal range '"+range+"'. Should throw StructureException");
173                } catch(StructureException ex) {} //expected 
174        }

176        public void testRevisedConvention() throws IOException, StructureException{
177
178                AtomCache cache = TmpAtomCache.cache;
179
180                
181                String name11 = "4hhb.A";
182                Structure s = cache.getStructure(name11);
183                assertTrue(s.getChains().size() == 1);
184
185
186                String name12 = "4hhb.A:";
187                s = cache.getStructure(name12);
188                assertTrue(s.getChains().size() == 1);
189
190                String name13 = "4hhb.A_";
191                s = cache.getStructure(name13);
192                assertTrue(s.getChains().size() == 1);
193                
194                String name9 = "4hhb.C_1-83";
195                String chainId = "C";
196                s = cache.getStructure(name9);
197
198                assertTrue(s.getChains().size() == 1);
199                Chain c = s.getChainByPDB(chainId);
200                assertEquals(c.getChainID(),chainId);
201                Atom[] ca = StructureTools.getAtomCAArray(s);
202                assertEquals(83,ca.length);
203
204                String name10 = "4hhb.C_1-83,A_1-10";
205                s = cache.getStructure(name10);
206                assertTrue(s.getChains().size() == 2);
207                ca = StructureTools.getAtomCAArray(s);
208                assertEquals(93, ca.length);
209
210                
211        }

277

282
283                chain = substr.getChain(0);

286
