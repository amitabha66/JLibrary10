001package org.biojava3.alignment;
002
003import java.net.URL;
004import java.util.ArrayList;
005import java.util.List;
006
007import org.biojava3.alignment.template.Profile;
008import org.biojava3.core.sequence.ProteinSequence;
009import org.biojava3.core.sequence.compound.AminoAcidCompound;
010import org.biojava3.core.sequence.io.FastaReaderHelper;
011import org.biojava3.core.util.ConcurrencyTools;
012
013public class CookbookMSA {
014
015    public static void main(String[] args) {
016        String[] ids = new String[] {"Q21691", "Q21495", "O48771"};
017        try {
018            multipleSequenceAlignment(ids);
019        } catch (Exception e){
020            e.printStackTrace();
021        }
022    }
023
024    private static void multipleSequenceAlignment(String[] ids) throws Exception {
025        List<ProteinSequence> lst = new ArrayList<ProteinSequence>();
026        for (String id : ids) {
027            lst.add(getSequenceForId(id));
028        }
029        Profile<ProteinSequence, AminoAcidCompound> profile = Alignments.getMultipleSequenceAlignment(lst);
030        System.out.printf("Clustalw:%n%s%n", profile);
031        ConcurrencyTools.shutdown();
032    }
033
034    private static ProteinSequence getSequenceForId(String uniProtId) throws Exception {
035        URL uniprotFasta = new URL(String.format("http://www.uniprot.org/uniprot/%s.fasta", uniProtId));
036        ProteinSequence seq = FastaReaderHelper.readFastaProteinSequence(uniprotFasta.openStream()).get(uniProtId);
037        System.out.printf("id : %s %s%n%s%n", uniProtId, seq, seq.getOriginalHeader());
038        return seq;
039    }
040
041}
