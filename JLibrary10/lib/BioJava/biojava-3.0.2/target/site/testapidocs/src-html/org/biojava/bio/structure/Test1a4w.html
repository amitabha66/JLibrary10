109        private void testStructure(Structure structure){
110                List<Chain> chains = structure.getChains();
111                assertEquals("1a4w should have 3 chains. " , 3 , chains.size());
112
113                Chain a = chains.get(0);
114                assertEquals("1a4w first chain should be L. " , a.getChainID(), "L");
115
116                Chain b = chains.get(1);
117                assertEquals("1a4w second chain should be H. " , b.getChainID(), "H");
118
119                Chain c = chains.get(2);
120                assertEquals("1a4w third chain should be I. " , c.getChainID(), "I");
121
122                //System.out.println(structure);
123                assertTrue("chain " + a.getChainID() + " length should be 26. was: " + a.getAtomGroups(GroupType.AMINOACID).size(), ( a.getAtomGroups(GroupType.AMINOACID).size() == 26 ) );
124
125                assertTrue("chain " + a.getChainID() + " seqres length should be 36. was: " + a.getSeqResLength(), a.getSeqResLength() == 36);
126
127                assertTrue("chain " + b.getChainID() + " length should be 248. was: " + b.getAtomGroups(GroupType.AMINOACID).size(), ( b.getAtomGroups(GroupType.AMINOACID).size() == 248 ) );
128
129                assertTrue("chain " + b.getChainID() + " seqres length should be 259. was: " + b.getSeqResLength(), b.getSeqResLength() == 259);
130
131                assertTrue("chain " + c.getChainID() + " length should be 9. was: " + c.getAtomGroups(GroupType.AMINOACID).size(), ( c.getAtomGroups(GroupType.AMINOACID).size() == 9 ) );
132
133                assertTrue("chain " + c.getChainID() + " seqres length should be 12. was: " + c.getSeqResLength(), c.getSeqResLength() == 12);
134
135                assertEquals("chain " + c.getChainID() + " seqres sequences is not correct!", "NGDFEEIPEEYL", c.getSeqResSequence());
136        }

138        private void testEqualChains(Chain a,Chain b){
139
140                assertEquals("length of seqres " + a.getChainID() + " and "+b.getChainID()+" should be same. " , a.getSeqResLength(), b.getSeqResLength() );
141                assertEquals("length of atom "   + a.getChainID() + " and "+b.getChainID()+" should be same. " , a.getAtomGroups(GroupType.AMINOACID).size(), b.getAtomGroups(GroupType.AMINOACID).size());
142                assertEquals("sequences should be identical. " , a.getAtomSequence(),   b.getAtomSequence());
143                assertEquals("sequences should be identical. " , a.getSeqResSequence(), b.getSeqResSequence());
144        }

146        public void testChemComps(){
147                try {
148
149                        Structure s = TmpAtomCache.cache.getStructure("1a4w");
150
151                        assertTrue(s.getChains().size() == 3);
152
153                        Chain c2 = s.getChain(1);
154                        assertTrue(c2.getChainID().equals("H"));
155
156                        List<Group> ligands = c2.getAtomLigands();
157
158
159                        boolean noWater = true;
160                        boolean darPresent = false;
161                
162
163                        for ( Group g : ligands){
164                                String pdbName = g.getPDBName();
165                                if ( pdbName.equals("QWE"))
166                                        darPresent = true;
167                        
168                                else if ( pdbName.equals("H2O"))
169                                        noWater = false;
170                        }
171
172                        assertTrue("Found water in ligands list!", noWater );
173
174                        assertTrue("Did not find QWE in ligands list!", darPresent);
175                        
176                        //System.out.println("LIGANDS:" + ligands);
177                        assertEquals("Did not find the correct nr of ligands in chain! " , 3,ligands.size());
178                } catch (Exception e){
179                        e.printStackTrace();
180                        fail(e.getMessage());
181                }
182
183        }
