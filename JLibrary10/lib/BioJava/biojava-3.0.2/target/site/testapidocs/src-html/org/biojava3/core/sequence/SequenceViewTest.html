001package org.biojava3.core.sequence;
002
003import static org.junit.Assert.assertEquals;
004
005import org.biojava3.core.sequence.compound.NucleotideCompound;
006import org.biojava3.core.sequence.template.CompoundSet;
007import org.biojava3.core.sequence.template.SequenceView;
008import org.junit.Test;
009
010public class SequenceViewTest {
011
012  @Test
013  public void testGetCompoundAt() {
014    SequenceView<NucleotideCompound> s = new DNASequence("ATGC").getSubSequence(1, 4);
015    assertEquals("Compound @ 1", s.getCompoundAt(1).toString(), "A");
016    assertEquals("Compound @ 3", s.getCompoundAt(3).toString(), "G");
017    assertEquals("Compound @ 3", s.getSubSequence(2,3).getCompoundAt(1).toString(), "T");
018  }
019
020  @Test
021  public void testLastIndexOf() {
022    SequenceView<NucleotideCompound> s = new DNASequence("ATGC").getSubSequence(1, 4);
023    CompoundSet<NucleotideCompound> cs = s.getCompoundSet();
024    assertEquals("Last index of ", 4, s.getLastIndexOf(cs.getCompoundForString("C")));
025    
026    s = new DNASequence("GAAAAAAAAG").getSubSequence(1, 10);
027    assertEquals("Last index of G is 10", 10,
028        s.getLastIndexOf(cs.getCompoundForString("G")));
029    assertEquals("Last index of G is 5", 6,
030        s.getSubSequence(5, 10).getLastIndexOf(cs.getCompoundForString("G")));
031  }
032
033  @Test
034  public void testInverse() {
035    SequenceView<NucleotideCompound> s = new DNASequence("ATGC").getSubSequence(2, 3).getInverse();
036    assertEquals("Reversed complementing view", s.getSequenceAsString(), "CA");
037  }
038}
