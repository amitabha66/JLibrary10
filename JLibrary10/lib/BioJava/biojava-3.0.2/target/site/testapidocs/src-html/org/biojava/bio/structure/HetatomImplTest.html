507    @Test
508    public void testGetResidueNumberUsage() {
509//        System.out.println("testGetResidueNumberUsage");
510        List<Group> resNumgroups = new ArrayList<Group>();
511
512        for (int i = 0; i < bigTestNumber; i++) {
513            ResidueNumber resNum = new ResidueNumber("A", i, ' ');
514            HetatomImpl hetAtom = new HetatomImpl();
515            hetAtom.setResidueNumber(resNum);
516            resNumgroups.add(hetAtom);
517        }
518
519        List<Integer> integers = new ArrayList<Integer>();
520
521        for (Group group : resNumgroups) {
522            ResidueNumber resnum = group.getResidueNumber();
523            integers.add(resnum.getSeqNum());
524        }
525        assertEquals(bigTestNumber, integers.size());
526    }
