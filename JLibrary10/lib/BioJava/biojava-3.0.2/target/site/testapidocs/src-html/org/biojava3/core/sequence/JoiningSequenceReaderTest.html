001package org.biojava3.core.sequence;
002
003import static org.junit.Assert.assertEquals;
004
005import org.biojava3.core.sequence.compound.NucleotideCompound;
006import org.biojava3.core.sequence.storage.JoiningSequenceReader;
007import org.biojava3.core.sequence.template.SequenceMixin;
008import org.junit.Test;
009
010public class JoiningSequenceReaderTest {
011
012  @SuppressWarnings("unchecked")
013  @Test
014  public void canScan() {
015    JoiningSequenceReader<NucleotideCompound> seq =
016      new JoiningSequenceReader<NucleotideCompound>(
017          new DNASequence("AAAA"),
018          new DNASequence("GGG"),
019          new JoiningSequenceReader<NucleotideCompound>(new DNASequence("A"), new DNASequence("C")),
020          new DNASequence("TT"),
021          new DNASequence("C")
022    );
023
024    String expected = "AAAAGGGACTTC";
025
026    StringBuilder builderByIndex = new StringBuilder();
027    for(int i = 1; i <= seq.getLength(); i++) {
028      builderByIndex.append(seq.getCompoundAt(i));
029    }
030
031    StringBuilder builderByIterator = SequenceMixin.toStringBuilder(seq);
032
033    assertEquals("Index builder", expected, builderByIndex.toString());
034    assertEquals("Iterator builder", expected, builderByIterator.toString());
035  }
036
037  @SuppressWarnings("unchecked")
038  @Test
039  public void empty() {
040    JoiningSequenceReader<NucleotideCompound> seq =
041      new JoiningSequenceReader<NucleotideCompound>(
042          new DNASequence(""),
043          new DNASequence(""),
044          new DNASequence("A"),
045          new DNASequence("")
046      );
047    assertEquals("Testing empty sequences", "A", seq.getSequenceAsString());
048  }
049}
